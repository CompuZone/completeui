  2 /**
  3  * Defines a column of the list
  4  * @class Describes the appearance and functioning of a column in the combo list.
  5  * @constructor
  6  * @param {HTMLElement} oNewListColumn The HTML object that represents ListColumnDefinition
  7  * @see nitobi.combo.Combo
  8  */
  9 nitobi.combo.ListColumnDefinition = function(oNewListColumn)
 10 {
 11 	// in case we're passing in a user created oNewListColumn object vs. a HTML tag
 12 	if(!oNewListColumn.getAttribute)
 13 		oNewListColumn.getAttribute=function(a){return this[a];};
 14 
 15 	var DEFAULTWIDTH="50px";
 16 	var DEFAULTCSSCLASSNAME="ntb-combobox-list-column-definition";
 17 	var DEFAULTCOLUMNTYPE="text";
 18 	var DEFAULTIMAGEHANDLERURL="";
 19 	var DEFAULTALIGN="left";
 20 	var DEFAULTTEXTCOLOR="#000";
 21 
 22 
 23 	var textcolor=(oNewListColumn ? oNewListColumn.getAttribute("TextColor") : null);
 24 	((null == textcolor) || ("" == textcolor))
 25 		? this.SetTextColor(DEFAULTTEXTCOLOR)
 26 		: this.SetTextColor(textcolor);
 27 
 28 	var align=(oNewListColumn ? oNewListColumn.getAttribute("Align") : null);
 29 	((null == align) || ("" == align))
 30 		? this.SetAlign(DEFAULTALIGN)
 31 		: this.SetAlign(align);
 32 
 33 	var width=(oNewListColumn ? oNewListColumn.getAttribute("Width") : null);
 34 	((null == width) || ("" == width))
 35 		? this.SetWidth(DEFAULTWIDTH)
 36 		: this.SetWidth(width);
 37 
 38 	var ihu=(oNewListColumn ? oNewListColumn.getAttribute("ImageHandlerURL") : null);
 39 	((null == ihu) || ("" == ihu))
 40 		? this.SetImageHandlerURL(DEFAULTIMAGEHANDLERURL)
 41 		: this.SetImageHandlerURL(ihu);
 42 
 43 	var ct=(oNewListColumn ? oNewListColumn.getAttribute("ColumnType") : null);
 44 	((null == ct) || ("" == ct))
 45 		? this.SetColumnType(DEFAULTCOLUMNTYPE)
 46 		: this.SetColumnType(ct.toLowerCase());
 47 
 48 //		if ((this.GetColumnType() == "image") && ((null == ihu) || ("" == ihu)))
 49 //			alert("EBA Combo Error: You cannot have an image column without specifying the ImageHandlerURL");
 50 	/**
 51 	 * @private
 52 	 */
 53 	this.ImageUrlFromData = ((this.GetColumnType() == "image") && ((null == ihu) || ("" == ihu)));
 54 
 55 	var ccn=(oNewListColumn ? oNewListColumn.getAttribute("CSSClassName") : null);
 56 	((null == ccn) || ("" == ccn))
 57 		? this.SetCSSClassName(DEFAULTCSSCLASSNAME)
 58 		: this.SetCSSClassName(ccn);
 59 
 60 	var hp=(oNewListColumn ? oNewListColumn.getAttribute("HTMLPrefix") : null);
 61 	((null == hp) || ("" == hp))
 62 		? this.SetHTMLPrefix("")
 63 		: this.SetHTMLPrefix(hp);
 64 
 65 	var hs=(oNewListColumn ? oNewListColumn.getAttribute("HTMLSuffix") : null);
 66 	((null == hs) || ("" == hs))
 67 		? this.SetHTMLSuffix("")
 68 		: this.SetHTMLSuffix(hs);
 69 
 70 	var hl=(oNewListColumn ? oNewListColumn.getAttribute("HeaderLabel") : null);
 71 	((null == hl) || ("" == hl))
 72 		? this.SetHeaderLabel("")
 73 		: this.SetHeaderLabel(hl);
 74 
 75 	var dfi=(oNewListColumn ? oNewListColumn.getAttribute("DataFieldIndex") : null);
 76 	((null == dfi) || ("" == dfi))
 77 		? this.SetDataFieldIndex(0)
 78 		: this.SetDataFieldIndex(dfi);
 79 }

 93 /**
 94  * Sets the alignment of text in this column as one of left, right, or center.
 95  * If this property is changed after the Combo has loaded its records, future records will be
 96  * rendered according to the new ListColumnDefinition, but already rendered records will not change. 
 97  * If you want to re-render the list, you will have to call the {@link nitobi.combo.List#Clear} and the {@link nitobi.combo.XmlDatasource#Clear} methods.
 98  * @param {String} Align Alignment of text in this column. It can be one of left, right, or center.
 99  * @see #GetAlign
100  */
101 nitobi.combo.ListColumnDefinition.prototype.SetAlign = function(Align){
102 	Align = Align.toLowerCase();
103 	if("right"!=Align && "left"!=Align && "center"!=Align)
104 		Align="left";
105 	this.m_Align = Align;
106 }
