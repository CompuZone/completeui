192 /**
193  * Renders the fisheye.  If you add or remove items to the fisheye control,
194  * you will need to call this method.
195  * <p>
196  * <b>Example</b>
197  * </p>
198  * <div class="code">
199  * <pre><code>
200  * function add()
201  * {
202  * 	var fisheye = nitobi.getComponent('fisheye1');
203  * 	var item = new nitobi.fisheye.MenuItem();
204  * 	item.setLabel("New item");
205  * 	item.setImageSource("images/info.png");
206  * 	fisheye.add(item);
207  * 	fisheye.render();
208  * }
209  * </code></pre>
210  * </div>
211  */
212 nitobi.fisheye.FishEye.prototype.render = function()
213 {
214 	if (this.renderTimes == 0)
215 	{
216 		nitobi.fisheye.FishEye.base.render.call(this);
217 		this.renderContainers();
218 		this.renderItems();
219 		this.updateMenuPosition();
220 		this.reDrawItems();
221 		
222 		if (nitobi.fisheye.FishEye.isMouseAttached == false)
223 		{
224 			nitobi.html.attachEvent(document.body, "mousemove", handleMouse);	
225 			nitobi.html.attachEvent(window, "onresize", nitobi.fisheye.FishEye.handleResize);
226 			nitobi.fisheye.FishEye.isMouseAttached = true;
227 		}
228 		this.renderTimes++;
229 	}
230 	else
231 	{
232 		while (this.MasterContainer.hasChildNodes())
233 		{
234 			var node = this.MasterContainer;
235 			node.removeChild(node.childNodes[0]);
236 		}
237 		// TODO: I don't think minWidth needs to be a member variable.
238 		this.minWidth = 0;
239 		this.renderItems();
240 		this.updateMenuPosition();
241 		this.reDrawItems();
242 		
243 		this.renderTimes++;
244 	}
245 	this.labelObj.style.width = "50px";
246 }

316 nitobi.fisheye.FishEye.prototype.handleBounce = function(menuItem)
317 {
318 	// TODO: Move into nitobi.fisheye.FishEye.MenuItem
319 	var item = this.get(menuItem);
320 	var ffs = this;
321 	item.bounceIt += 0.045;
322 	if (item.bounceIt > 1.0)
323 		item.bounceIt -= 1.0;
324 	
325 	item.yoffset = Math.sin(item.bounceIt*3.1415926)*(this.growPercent*this.iconWidth*0.13);

327 	item.bounceTimer = setTimeout(function(){ffs.handleBounce(menuItem)}, 30);

330 /**
331  * Causes a menu item to visually bounce, indicating that it has been activated. This will self terminate after a few seconds.
332  * @param {int} menuItem The ordinal item number of the object you want to bounce. Begins at 0.
333  * @param {int} bounceTime How long you would like the bounce to happen for. In miliseconds (1000 = 1 second).
334  */
335 nitobi.fisheye.FishEye.prototype.bounceItem = function(menuItem, bounceTime)
336 {
337 	var ffd = this;
338 	var lastItem = this.get(this.lastBounce);
339 	var item = this.get(menuItem);
340 	lastItem.bounceIt = 0;
341 	lastItem.yoffset = 0;
342 	clearTimeout(lastItem.bounceTimer);
343 	item.bounceIt = 0;
344 	this.lastBounce = menuItem;
345 	clearTimeout(item.bounceTimer);
346 	clearTimeout(this.bounceKiller);
347 	clearTimeout(this.iteratetimer);	
348 	this.bounceKiller = setTimeout(function(){clearTimeout(item.bounceTimer); ffd.get(menuItem).yoffset = 0;}, bounceTime);
349 	this.handleBounce(menuItem);
350 }

636 					this.positionLabel(mo.imgLabel, this.x + cx + (nw/2)-lof, this.y + nh + cy);
637 				}
638 			}			
639 		}
640 	}
641 	
642 	var mox = 0;
643 	
644 	for (f = 0; f < ol; f++)
645 	{
646 		mo = this.get(f);
647 		mox = mo.img.style.left.replace('px','');
648 		mo.img.style.left = (parseFloat(mox)-lof) + 'px';
649 	}	
650 	this.currentxoffset = xoffset;
651 
652 	var ls = parseInt(this.MasterContainer.style.left.replace('px', '')) + parseInt(this.get(0).img.style.left.replace('px', ''));
653 	this.BGContainer.style.left = (ls - this.containerPadding) + 'px';
654 	this.BGContainer.style.top = (parseInt(this.MasterContainer.style.top.replace('px', ''))-this.containerPadding) + 'px';
655 	this.BGContainer.style.height = (parseInt(this.MasterContainer.offsetHeight)+this.containerPadding+this.containerPadding) + "px";
656 	this.BGContainer.style.width = (parseInt(this.get(this.getLength()-1).img.style.left.replace('px', '')) + (parseInt(this.get(this.getLength()-1).img.style.width.replace('px', ''))) - parseInt(this.get(0).img.style.left.replace('px', '')) + this.containerPadding + this.containerPadding) + "px";
657 
658 	if ((doagain) && ((this.useIterator) || (!this.foundActive))) 
659 	{
660 		this.iteratetimer = setTimeout(function(){fds.reDrawItems()}, 40);
661 	} 
662 	else 
663 	{
664 		this.startedIKillTimer = false;
665 		clearTimeout(this.disableIterator);
666 		this.restartIterator = setTimeout(function(){clearTimeout(fds.disableIterator); fds.startedIKillTimer = false; fds.useIterator = true;}, 420);
667 	}
668 }

735 /**
736  * Handles the mousemove event.
737  * @private
738  */
739 function handleMouse(event)
740 {
741 
742 	var sP = nitobi.html.getScroll();
743 	var doRedraw = false;
744 	var mouseX, mouseY;
745 	var cancelZoom = false;
746 	var highZoom = 0;
747 	/*if (navigator.userAgent.toLowerCase().indexOf("msie") != -1) { // grab the x-y pos.s if browser is IE
748 	mouseX = event.clientX + sP.scrollLeft;
749 	mouseY = event.clientY + sP.scrollTop;
750 	} else { */
751 	//mouseX =event.clientX
752 	//mouseY =event.clientY;
753 	mouseX = event.clientX + sP.left ;
754 	mouseY = event.clientY + sP.top;	
755 	//} 
756 
757 	//*** USE NITOBI EVENT STUFF HERE.
758 
759 	var t,f,w,h,x,y,o,ol,p,q,dist,calcval;
760 	// For each fisheye menu
761 	var listLength = fisheyeList.length;
762 	for (t = 0; t < listLength; t++) 
763 	{
764 		// get the fisheye object
765 		o = fisheyeList[t];
766 		if (o.loaded) 
767 		{
768 
769 			w = o.iconWidth;
770 			h = o.iconHeight;
771 			// ol is how many icons in this particular fisheye menu
772 			ol = o.getLength();
773 			o.highval = 0;
774 			o.highindex = -1;
775 			o.foundActive = false;
776 			cancelZoom = false;
777 			highZoom = 0;
778 			for (f = 0; f < ol; f++) 
779 			{
780 				var item = o.get(f);
781 				if (item.mysize > 0.01)
782 				{	
783 					doRedraw = true;
784 				}
785 				x = o.x + item.myx + (w/2);
786 				y = o.y + item.myy + (h/2);	
787 				
788 				if (o.eD == 2)	
789 				{		
790 					p = Math.abs(x - mouseX);	
791 				}		
792 				if (o.eD == 0)			
793 				{
794 					p = Math.abs(x - mouseX + o.iconWidth/2 + (o.currentxoffset * (f/ol)));
795 				}
796 				if (o.eD == 1)			
797 				{
798 					p = Math.abs(x - mouseX - (o.iconWidth/2) - (o.currentxoffset * ((ol - f)/ol)));			
799 				}
800 				//p = Math.abs((x)-mouseX);
801 				q = Math.abs(y - mouseY);
802 				//dist = Math.sqrt((p*p)+(q*q));
803 				//dist = Math.sqrt((p*p));	
804 				dist = p;
805 				if (q > o.yBoundary) {
806 					q = 1000;
807 				}
808 				//console.log(dist);
809 				if ((p < (w * o.rangeSensitivity)) && (q < (h * 1.5)))
810 				{
811 					doRedraw = true;
812 					o.mouseX = mouseX;
813 					o.mouseY = mouseY;
814 					o.foundActive = true;
815 
816 					calcval = 1 - (dist/(w * o.rangeSensitivity));
817 					//o.menuObjects[f].distance = 0;
818 					//o.menuObjects[f].mysize = calcval;
819 					item.mysize = calcval;
820 					if (o.highval < calcval)
821 					{
822 						o.highval = calcval;
823 						o.highindex = f;
824 						highZoom = q;
825 					}
826 				} 
827 				else 
828 				{
829 					item.mysize = 0;
830 				}
831 			}	
832 		
833 			if (highZoom > (o.iconWidth * 0.8)) 
834 			{
835 				o.foundActive = false;
836 				doRedraw = false;
837 				cancelZoom = true; 
838 				for (f = 0; f < ol; f++) 
839 				{
840 					o.foundActive = false;
841 					doRedraw = true;	
842 					o.highval = 0;
843 					// The following lines were commented out by Mike because
844 					// I wasn't sure why these assignments were necessary
845 					// Their inclusion caused the fisheye to not work properly
846 					//o.highindex = -1;					
847 					//item.mysize = 0;
848 				}
849 			}
850 		
851 			if (!o.foundActive)
852 			{
853 				o.useIterator = true;
854 			}
855 			if ((doRedraw) || (o.highindex > -1))
856 			{
857 				o.reDrawItems();
858 			}
859 		}		
860 	}
