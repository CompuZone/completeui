 56 nitobi.grid.Viewport.prototype.makeLastBlock = function(low,high) {
 57 	if (this.lastEmptyBlock == null && this.grid && this.region>2 && this.region<5 && this.container) {
 58 		if (this.container.lastChild) {
 59 //			alert(low+":"+high)
 60 			low=Math.max(low,this.container.lastChild.bottom);
 61 		}
 62 		this.lastEmptyBlock=this.renderEmptyBlock(low,high);
 63 	}

 96 nitobi.grid.Viewport.prototype.clear = function(Surface,Placeholder,Center,Viewport)
 97 {
 98 	var uid = this.grid.uid;
 99 	if (this.surface && Surface)
100 		this.surface.innerHTML = '<div id="gridvpcontainer_'+this.region+'_'+uid+'"></div>';

103 	if (this.surface && Center)
104 		this.surface.innerHTML = '<div id="gridvpcontainer_'+this.region+'_'+uid+'"></div>';
105 	this.surface = nitobi.html.getFirstChild(this.element);
106 	this.container = nitobi.html.getFirstChild(this.surface);
107 	if (this.grid && this.region>2 && this.region<5) { //This is a jinky way of detecting when to use empty blocks
108 		this.lastEmptyBlock = null;
109 	}
110 	
111 	// After we clear, we need to ensure there is an empty block
112 	// otherwise the Grid won't be able to determine whether to
113 	// get more data from the server.
114 	this.makeLastBlock(0, this.grid.getRowsPerPage() * 5);
115 }

159 nitobi.grid.Viewport.prototype.findBlock=function(row) {
160 	var blk=this.container.childNodes;
161 	for (var i=0;i<blk.length;i++) {
162 		if (row>=blk[i].top && row<=blk[i].bottom) {
163 			return blk[i];
164 		}
165 	}

171 nitobi.grid.Viewport.prototype.findBlockAtCoord=function(top) {
172 	var blk=this.container.childNodes;
173 	for (var i=0;i<blk.length;i++) {
174 		var rt = blk[i].offsetTop;
175 		var rb = rt+blk[i].offsetHeight;
176 		if (top>=rt && top<=rb) {
177 			return blk[i];
178 		}
179 	}

182 /**
183  *
184  */
185 nitobi.grid.Viewport.prototype.getBlocks = function(startRow, endRow)
186 {
187 	var blocks = [];
188 	var startBlock = this.findBlock(startRow);
189 	var endBlock = startBlock;
190 	blocks.push(startBlock);
191 	while (endRow > endBlock.bottom)
192 	{
193 		var nextSibling = endBlock.nextSibling;
194 		if (nextSibling != null)
195 			endBlock = nextSibling;
196 		else
197 			break;
198 		blocks.push(endBlock);
199 	}
200 	return blocks;
201 }

206 nitobi.grid.Viewport.prototype.clearBlocks = function(startRow, endRow)
207 {
208 	// TODO: should split up the first and last block rather than destroy it entirely.
209 	var blocks = this.getBlocks(startRow, endRow);
210 	var len = blocks.length;
211 	var top = blocks[0].top;
212 	var bottom = blocks[len-1].bottom;
213 	var nextSibling = blocks[len-1].nextSibling;
214 	for (var i=0; i<len; i++)
215 	{
216 		blocks[i].parentNode.removeChild(blocks[i]);
217 	}
218 	this.renderEmptyBlock(top, bottom, nextSibling);
219 	return {"top":top,"bottom":bottom};
220 }

222 /**
223  *
224  */
225 nitobi.grid.Viewport.prototype.renderInsideEmptyBlock=function(top, bottom, Empty) {
226 	if (Empty==null) {
227 		return this.renderBlock(top,bottom);
228 	}
229 	
230 	// Case - Completely Replace
231 	if (top==Empty.top && bottom>=Empty.bottom) {
232 		var newBlock = this.renderBlock(top,bottom,Empty)
233 		this.container.replaceChild(newBlock,Empty);
234 		if (Empty.bottom<Empty.top)
235 			throw "Render error";
236 		return newBlock;
237 	}
238 
239 	// Case - Insert at Beginning - (Move top and insert before)
240 	if (top==Empty.top && bottom<Empty.bottom) {
241 		Empty.top = bottom+1;
242 		Empty.style.height = (this.rowHeight*(Empty.bottom-Empty.top+1))+"px";
243 		Empty.rows = Empty.bottom-Empty.top+1
244 		if (Empty.bottom<Empty.top)
245 			throw "Render error";
246 		return this.renderBlock(top,bottom,Empty);
247 	}
248 
249 	// Case - Insert at End - (Move bottom and insert after)
250 	if (top>Empty.top && bottom>=Empty.bottom) {
251 		Empty.bottom = top-1;
252 		Empty.style.height = (this.rowHeight*(Empty.bottom-Empty.top+1))+"px";
253 		if (Empty.bottom<Empty.top)
254 			throw "Render error";
255 		return this.renderBlock(top,bottom,Empty.nextSibling);
256 	}
257 
258 	// Case - Insert in the Middle (Move end insert after followed by a new empty block)
259 	if (top>Empty.top && bottom<Empty.bottom) {
260 		// Original emptyblock becomes the end emptyblock and a new start emptyblock is created
261 		var startBlock=this.renderEmptyBlock(Empty.top,top-1,Empty);
262 		Empty.top = bottom+1;
263 		Empty.style.height = (this.rowHeight*(Empty.bottom-Empty.top+1))+"px";
264 		if (Empty.bottom<Empty.top)
265 			throw "Render error";
266 		return this.renderBlock(top,bottom,Empty);
267 	}
268 
269 	throw "Could not insert "+top+"-"+bottom+Empty.outerHTML;
270 }

275 nitobi.grid.Viewport.prototype.renderEmptyBlock=function(top, bottom,nextSibling) {
276 	var o = this.renderBlock(top,bottom,nextSibling)
277 	o.setAttribute('id','eba_grid_emptyblock_'+this.region+'_'+top+'_'+bottom+'_'+this.grid.uid);
278 	if(top==0 && bottom ==99) {
279 		crash
280 	}
281 	o.setAttribute('rendered','false');
282 	o.style.height = Math.max(((bottom - top + 1)*this.rowHeight),0)+"px";
283 	return o;
284 }

286 /**
287  *
288  */
289 nitobi.grid.Viewport.prototype.renderBlock=function(top, bottom,nextSibling) {
290 //	if (bottom<top) return null;
291 	var o = document.createElement("div");
292 	// This will be reset by the calling function if called from renderEmptyBlock
293 	o.setAttribute('id','eba_grid_block_'+this.region+'_'+top+'_'+bottom+'_'+this.grid.uid);
294 	o.top=top;
295 	o.bottom=bottom;
296 	o.left = this.startColumn;
297 	o.right = this.startColumn + this.columns;
298 	o.rows=bottom-top+1;
299 	o.columns = this.columns;
300 	if (nextSibling) {
301 		this.container.insertBefore(o,nextSibling);
302 	} else {
303 		this.container.insertBefore(o,null);
304 	}
305 //	if (o.bottom<o.top) {
306 //	crash
307 //	}
308 	return o;
309 }
