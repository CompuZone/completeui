  3 /**
  4  * Creates the calendar portion of the DatePicker.
  5  * @class The calendar portion of the DatePicker allows users to choose a date visually from a rendered calendar.
  6  * @constructor
  7  * @param {Node} element
  8  * @extends nitobi.ui.Element
  9  * @see nitobi.calendar.DatePicker
 10  */
 11 nitobi.calendar.Calendar = function(element) 
 12 {
 13 	nitobi.calendar.Calendar.baseConstructor.call(this, element);
 14 	
 15 	/**
 16 	 * The date that is currently selected in the Calendar.  Not necessarily the same
 17 	 * as the selectedDate of the DatePicker.
 18 	 * @private
 19 	 */
 20 	this.selectedDate;
 21 
 22 	/**
 23 	 * The DatePicker class has a custom renderer.
 24 	 * @type nitobi.calendar.Renderer()
 25 	 * @private
 26 	 */
 27 	this.renderer = new nitobi.calendar.CalRenderer();
 28 	
 29 	/**
 30 	 * Fires when the Calendar is hidden.
 31 	 * @type nitobi.base.Event
 32 	 */
 33 	this.onHide = new nitobi.base.Event();
 34 	this.eventMap["hide"] = this.onHide;
 35 	
 36 	/**
 37 	 * Fires when the Calendar is shown.
 38 	 * @type nitobi.base.Event
 39 	 */
 40 	this.onShow = new nitobi.base.Event();
 41 	this.eventMap["show"] = this.onShow;
 42 	
 43 	/**
 44 	 * Fires when a date in the Calendar is clicked.
 45 	 * @type nitobi.base.Event
 46 	 */
 47 	this.onDateClicked = new nitobi.base.Event();
 48 	this.eventMap["dateclicked"] = this.onDateClicked;
 49 	
 50 	/**
 51 	 * Fires when the month is changed, either by the user clicking the "next" button
 52 	 * or via the quick nav panel.
 53 	 * @type nitobi.base.Event
 54 	 */
 55 	this.onMonthChanged = new nitobi.base.Event();
 56 	this.eventMap["monthchanged"] = this.onMonthChanged;
 57 	
 58 	/**
 59 	 * Fires when the year is changed, either by the user clicking the "next" button
 60 	 * or via the quick nav panel.
 61 	 * @type nitobi.base.Event
 62 	 */
 63 	this.onYearChanged = new nitobi.base.Event();
 64 	this.eventMap["yearchanged"] = this.onYearChanged;
 65 	
 66 	/**
 67 	 * Fires when the Calendar is done rendering.
 68 	 * @type nitobi.base.Event
 69 	 */
 70 	this.onRenderComplete = new nitobi.base.Event();
 71 	
 72 	this.onSetVisible.subscribe(this.handleToggle, this);
 73 	
 74 	/**
 75 	 * The effect class to use when showing the Calendar.
 76 	 * Default: {@link nitobi.effects.ShadeDown}
 77 	 * @see nitobi.effects.families
 78 	 * @type Class
 79 	 */

 81 	/**
 82 	 * The effect class to use when hiding the Calendar.
 83 	 * Default: {@link nitobi.effects.ShadeUp}
 84 	 * @see nitobi.effects.families
 85 	 * @type Class
 86 	 */
 87 	this.hideEffect = (this.isEffectEnabled()?nitobi.effects.families["shade"].hide:null);
 88 	
 89 	/**
 90 	 * Use this to keep track of events we've attached to html elements so we can properly 
 91 	 * detach them.
 92 	 * @private
 93 	 */
 94 	this.htmlEvents = {'body': [], 'nav': [], 'navconfirm': [], 'navcancel': [], 'navpanel': [], 'nextmonth': [], 'prevmonth': []};
 95 	
 96 	this.subscribeDeclarationEvents();
 97 }

104 /**
105  * Renders the Calendar component only.
106  */
107 nitobi.calendar.Calendar.prototype.render = function()
108 {
109 	this.detachEvents();
110 	
111 	this.setContainer(this.getHtmlNode());
112 	nitobi.calendar.Calendar.base.render.call(this);
113 	
114 	this.selectedDate = this.getParentObject().getSelectedDate();
115 	var he = this.htmlEvents;
116 	var H = nitobi.html;
117 	
118 	// Attach Body events
119 	var calBody = this.getHtmlNode("body");
120 	H.attachEvent(calBody,"click", this.handleBodyClick, this);
121 	H.attachEvent(calBody, "mousedown", this.handleMouseDown, this);
122 	
123 	he.body.push({type: "click", handler: this.handleBodyClick});
124 	he.body.push({type: "mousedown", handle: this.handleMouseDown});
125 	
126 	// Attach quick nav panel events
127 	var nav = this.getHtmlNode("nav");
128 	var navConfirm = this.getHtmlNode("navconfirm");
129 	var navCancel = this.getHtmlNode("navcancel");
130 	H.attachEvent(nav, "click", this.showNav, this);
131 	H.attachEvent(navCancel, "click", this.handleNavCancel, this);
132 	H.attachEvent(navConfirm, "click", this.handleNavConfirm, this);
133 	H.attachEvent(this.getHtmlNode("navpanel"), "keypress", this.handleNavKey, this);
134 	
135 	he.nav.push({type: "click", handler: this.showNav});
136 	he.navcancel.push({type: "click", handler: this.handleNavCancel});
137 	he.navconfirm.push({type: "click", handler: this.handleNavConfirm});
138 	he.navpanel.push({type: "keypress", handler: this.handleNavKey});
139 	
140 	// Attach next/prev events
141 	H.attachEvent(this.getHtmlNode("nextmonth"),"click",this.nextMonth,this);
142 	H.attachEvent(this.getHtmlNode("prevmonth"),"click",this.prevMonth,this);
143 	
144 	he.nextmonth.push({type: "click", handler: this.nextMonth});
145 	he.prevmonth.push({type: "click", handler: this.prevMonth});
146 	
147 	var calNode = this.getHtmlNode();
148 	var shim = this.getHtmlNode("shim");
149 	var Css = nitobi.html.Css;
150 	// If we are in IE 6 or FF 2 on Mac, we need an iframe backer so that scrollbars underneath
151 	// the Calendar are hidden properly.
152 	if (shim)
153 	{
154 		var hidden = Css.hasClass(calNode, "nitobi-hide");
155 		if (hidden)
156 		{
157 			Css.removeClass(calNode, "nitobi-hide");
158 			calNode.style.top = "-1000px";
159 		}
160 		var width = calNode.offsetWidth;
161 		var height = calNode.offsetHeight;
162 		shim.style.height = height + "px";
163 		shim.style.width = width - 1 + "px";
164 		if (hidden)
165 		{
166 			Css.addClass(calNode, "nitobi-hide");
167 			calNode.style.top = "";
168 		}
169 	}
170 	
171 	this.onRenderComplete.notify(new nitobi.ui.ElementEventArgs(this, this.onRenderComplete));
172 }

202 /**
203  * Handles the case where the user clicks on a date in the Calendar.
204  * @private
205  */
206 nitobi.calendar.Calendar.prototype.handleBodyClick = function(event)
207 {
208 	var activeDate = this.findActiveDate(event.srcElement);
209 	if (!activeDate || nitobi.html.Css.hasClass(activeDate, "ntb-calendar-lastmonth") || nitobi.html.Css.hasClass(activeDate, "ntb-calendar-nextmonth"))
210 		return;
211 	var datePicker = this.getParentObject();
212 	
213 	var day = activeDate.getAttribute("ebadate");
214 	var month = activeDate.getAttribute("ebamonth");
215 	var year = activeDate.getAttribute("ebayear");
216 	var date = new Date(year, month, day);
217 
218 	var eventsManager = datePicker.getEventsManager();
219 	if (eventsManager.isDisabled(date))
220 		return;
221 
222 	datePicker._setSelectedDate(date);
223 	this.onDateClicked.notify(new nitobi.ui.ElementEventArgs(this, this.onDateClicked));
224 	this.toggle();
225 }

275 /**
276  * Returns the html element in the Calendar corresponding to a given date.  Returns null
277  * if that date is not currently rendered.
278  * @type HTMLElement
279  * @private
280  */
281 nitobi.calendar.Calendar.prototype.findDateElement = function(date)
282 {
283 	var table = this.getHtmlNode(date.getMonth() + "." + date.getFullYear());
284 	var dm = nitobi.base.DateMath;
285 	if (table)
286 	{
287 		var startDate = dm.getMonthStart(dm.clone(date));
288 		startDate = dm.subtract(startDate,'d',startDate.getDay());
289 		var days = dm.getNumberOfDays(startDate, date) - 1;
290 		if (days >=0 && days < 42)
291 		{
292 			var row = 1 + Math.floor(days / 7);
293 			var col = days % 7;
294 			var dateCell = nitobi.html.getFirstChild(table.rows[row].cells[col])
295 			return dateCell;
296 		}
297 	}
298 	return null;
299 }

301 /**
302  * Shows the quick nav panel.
303  */
304 nitobi.calendar.Calendar.prototype.showNav = function()
305 {
306 	var datePicker = this.getParentObject();
307 	var startDate = datePicker.getStartDate();
308 	var monthsSelect = this.getHtmlNode("months");
309 	monthsSelect.selectedIndex = startDate.getMonth();
310 	this.getHtmlNode("year").value = startDate.getFullYear();
311 	this.getHtmlNode("warning").style.display = "none";
312 	var overlay = this.getHtmlNode("overlay");
313 	var panel = this.getHtmlNode("navpanel");
314 	
315 	var effect = new nitobi.effects.BlindDown(panel, {duration: 0.3});
316 	var alignTarget = this.getHtmlNode("nav");
317 	this.fitOverlay();
318 	overlay.style.display = "block";
319 	var D = nitobi.drawing;
320 	D.align(panel, alignTarget, D.align.ALIGNMIDDLEHORIZ);
321 	D.align(panel, this.getHtmlNode("body"), D.align.ALIGNTOP);
322 	D.align(overlay, this.getHtmlNode("body"), D.align.ALIGNTOP | D.align.ALIGNLEFT);
323 	effect.callback = 
324 		function() 
325 		{
326 			monthsSelect.focus();
327 		};
328 	effect.start();
329 }

365 /**
366  * @private
367  */
368 nitobi.calendar.Calendar.prototype.handleNavConfirm = function(event)
369 {
370 	var datePicker = this.getParentObject();
371 	
372 	var monthsList = this.getHtmlNode("months");
373 	var month = monthsList.options[monthsList.selectedIndex].value;
374 	var year = this.getHtmlNode("year").value;
375 	
376 	if (isNaN(year))
377 	{
378 		var warning = this.getHtmlNode("warning");
379 		warning.style.display = "block";
380 		warning.innerHTML = datePicker.getNavInvalidYearText();
381 		return;
382 	}
383 	year = parseInt(year);
384 	var newDate = new Date(year, month, 1);
385 	if (datePicker.isOutOfRange(newDate))
386 	{
387 		var warning = this.getHtmlNode("warning");
388 		warning.style.display = "block";
389 		warning.innerHTML = datePicker.getNavOutOfRangeText();
390 		return;
391 	}
392 	var startDate = datePicker.getStartDate();
393 	var monthChanged = false;
394 	var yearChanged = false;
395 	if (year != startDate.getFullYear()) yearChanged = true;
396 	if (parseInt(month) != startDate.getMonth()) monthChanged = true;
397 	datePicker.setStartDate(newDate);
398 	var callback = nitobi.lang.close(this, this.render);
399 	this.onRenderComplete.subscribeOnce(nitobi.lang.close(this, 
400 		function() {
401 			if (monthChanged) this.onMonthChanged.notify(new nitobi.ui.ElementEventArgs(this, this.onMonthChanged));
402 			if (yearChanged) this.onYearChanged.notify(new nitobi.ui.ElementEventArgs(this, this.onYearChanged));
403 		}
404 	));
405 	this.hideNav(callback);
406 }

417 /**
418  * @private
419  */
420 nitobi.calendar.Calendar.prototype.findActiveDate = function(element)
421 {
422 	var breakOut = 5;
423 	for (var i = 0; i < breakOut && element.getAttribute; i++) 
424 	{
425 		var t = element.getAttribute('ebatype');
426 		if (t == 'date') return element;
427 		element = element.parentNode;
428 	}
429 	return null;
430 }

441 /**
442  * Moves the start month of the Calendar forward by one month.
443  */
444 nitobi.calendar.Calendar.prototype.nextMonth = function()
445 {
446 	var datePicker = this.getParentObject();
447 	if (!datePicker.disNext)
448 	{
449 		var totalMonths = this.getMonthColumns() * this.getMonthRows();
450 		this.changeMonth(totalMonths);
451 	}
452 }
453 
454 /**
455  * Moves the start month of the Calendar back by one month.
456  */
457 nitobi.calendar.Calendar.prototype.prevMonth = function()
458 {
459 	if (!this.getParentObject().disPrev)
460 	{
461 		var totalMonths = this.getMonthColumns() * this.getMonthRows();
462 		this.changeMonth(0 - totalMonths);
463 	}
464 }

466 /**
467  * Changes the start month of the Calendar.
468  * @param {Number} unit The number of months so change the start month by.  Negative numbers goes back in time,
469  * positive number go forward.
470  * @private
471  */
472 nitobi.calendar.Calendar.prototype.changeMonth = function(unit)
473 {
474 	var datePicker = this.getParentObject();
475 	var date = datePicker.getStartDate();
476 	var dm = nitobi.base.DateMath;
477 	date = dm._add(dm.clone(date), 'm', unit);
478 	var startDate = datePicker.getStartDate();
479 	var yearChanged = false;
480 	if (startDate.getFullYear() != date.getFullYear())
481 		yearChanged = true;
482 	datePicker.setStartDate(date);
483 	this.render();
484 	
485 	this.onMonthChanged.notify(new nitobi.ui.ElementEventArgs(this, this.onMonthChanged));
486 	if (yearChanged) this.onYearChanged.notify(new nitobi.ui.ElementEventArgs(this, this.onYearChanged));
487 }
