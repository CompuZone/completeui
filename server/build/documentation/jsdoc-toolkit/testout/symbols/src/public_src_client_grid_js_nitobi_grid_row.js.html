 52 /**
 53  * Returns the row HTML element for the given Grid and row indices.
 54  * @param {nitobi.grid} grid The Grid to which the row belongs.
 55  * @param {Number} row The row index.
 56  * @return {nitobi.grid.Row}
 57  */
 58 nitobi.grid.Row.getRowElement = function(grid, row)
 59 {
 60 	return nitobi.grid.Row.getRowElements(grid,row).mid;
 61 };
 62 /**
 63  * @private
 64  */
 65 nitobi.grid.Row.getRowElements = function(grid, row)
 66 {
 67 	// TODO: refactor this offset stuff into a method in grid.
 68 	
 69 	var midCol = grid.getFrozenLeftColumnCount();
 70 	if (!midCol)
 71 		return {left: null, mid: $("row_"+row+"_"+grid.uid)};
 72 
 73 	var C = nitobi.grid.Cell;
 74 	var rows = {};
 75 	rows.left = C.getCellElement(grid, row, 0).parentNode;
 76 	var cell = C.getCellElement(grid, row, midCol)
 77 	rows.mid = cell ? cell.parentNode : null;
 78 	return rows;
 79 }
