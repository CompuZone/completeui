  3 /**
  4  * Constructs a SelectionEventArgs object.
  5  * @class When you subscribe to Grid events through the declaration, you
  6  * can optionally pass information about the event to the function
  7  * registered to handle it.  You do this by using the eventArgs keyword.
  8  * Used for Copy and Paste operations.
  9  * @constructor
 10  * @param {nitobi.grid.Grid} source The object which is firing the event.
 11  * @param {String} data The data that was copied in HTML table format.
 12  * @param {Object} coords The top left and bottom right coords, which are nitobi.drawing.Point objects. {"top":POINT,"bottom":POINT}.
 13  */
 14 nitobi.grid.SelectionEventArgs = function(source, data, coords)
 15 {
 16 	/**
 17 	 * @private
 18 	 */
 19 	this.source = source;
 20 	/**
 21 	 * @private
 22 	 */
 23 	this.coords = coords
 24 	/**
 25 	 * @private
 26 	 */
 27 	this.data = data;
 28 }
 29 
 30 /**

 37 }
 38 
 39 /**
 40  * @private
 41  */

 47 /**
 48  * @private
 49  */
