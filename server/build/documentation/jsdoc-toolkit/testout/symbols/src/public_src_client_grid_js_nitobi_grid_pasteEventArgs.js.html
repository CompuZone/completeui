  3 /**
  4  * Constructs a PasteEventArgs object.
  5  * @class When you subscribe to Grid events through the declaration, you

  7  * registered to handle it.  You do this by using the eventArgs keyword.
  8  * @constructor
  9  * @param {nitobi.grid.Grid} source The object which is firing the event. 
 10  * @private
 11  */
 12 nitobi.grid.PasteEventArgs = function(source)
 13 {
 14 	/**
 15 	 * @private
 16 	 */
 17 	this.grid = source;
 18 	/**
 19 	 * @private
 20 	 */
 21 	this.event = nitobi.html.Event;
 22 }
 23 
 24 /**
 25  * Gets the Grid that fired the event.
 26  * @type nitobi.grid.Grid
 27  */
 28 nitobi.grid.PasteEventArgs.prototype.getSource = function()

 31 }
 32 /**
 33  * Gets the native browser Event object that is associated with the event.
