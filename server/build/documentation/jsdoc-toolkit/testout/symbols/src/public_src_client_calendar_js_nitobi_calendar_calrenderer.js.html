  3 /**
  4  * Creates a a renderer for the Calendar class.
  5  * @class This class provides the renderToString function necessary to render 
  6  * the DatePicker.  The renderer expects its datePicker to return itself (<code>this</code>) 
  7  * from the <code>getState</code> function.
  8  * @constructor
  9  * @extends nitobi.html.IRenderer 
 10  */
 11 nitobi.calendar.CalRenderer = function()
 12 {
 13 	nitobi.html.IRenderer.call(this);
 14 };
 15 
 16 nitobi.lang.implement(nitobi.calendar.CalRenderer, nitobi.html.IRenderer);
 17 
 18 /**
 19  * Transform the given data with the template.
 20  * @param {XMLDocument|String} data the data to transform.
 21  * @return The result of the transformation as a string.
 22  * @type String
 23  */
 24 nitobi.calendar.CalRenderer.prototype.renderToString = function(calendar)
 25 {
 26 	var datePicker = calendar.getParentObject();
 27 	var eventsManager = datePicker.getEventsManager();
 28 	
 29 	var dm = nitobi.base.DateMath;
 30 	var sb = new nitobi.lang.StringBuilder();
 31 	var id = calendar.getId();
 32 	
 33 	var monthColumns = calendar.getMonthColumns();
 34 	var monthRows = calendar.getMonthRows();
 35 	
 36 	var isMultiMonth = monthColumns > 1 || monthRows > 1;
 37 	
 38 	// Reset time so it is easier to compare date values directly
 39 	var startDate = dm.resetTime(dm.clone(datePicker.getStartDate()));
 40 	var selectedDate = datePicker.getSelectedDate();
 41 	if (selectedDate != null)
 42 		selectedDate = dm.resetTime(datePicker.getSelectedDate());
 43 	var today = dm.resetTime(new Date());
 44 	var minDate = datePicker.getMinDate();
 45 	var maxDate = datePicker.getMaxDate();
 46 	
 47 	var lastMonth = dm.subtract(dm.clone(startDate), 'd', 1);
 48 	var nextMonth = dm.add(dm.clone(startDate), 'm', monthColumns * monthRows);
 49 	
 50 	datePicker.disPrev = (minDate && dm.before(lastMonth, minDate)?true:false);
 51 	datePicker.disNext = (maxDate && dm.after(nextMonth, maxDate)?true:false);
 52 	
 53 	var monthNames = datePicker.getLongMonthNames();
 54 	var dayNames = datePicker.getLongDayNames();
 55 	var weekDays = datePicker.getMinDayNames();
 56 	
 57 	var monthTooltip = datePicker.getQuickNavTooltip();
 58 	
 59 	// Enable shim only for FF2 on Mac and IE6
 60 	var enableShim = (((nitobi.browser.MOZ && !document.getElementsByClassName && navigator.platform.indexOf("Mac") >= 0) || nitobi.browser.IE6) && datePicker.isShimEnabled())?true:false;
 61 	if (enableShim)
 62 		sb.append("<iframe id=\"" + id + ".shim\" style='position:absolute;top:0px;z-index:19999;'><!-- dummy --></iframe>");
 63 	sb.append("<div id=\"" + id + ".calendar\" style=\"" + (enableShim?"position:relative;z-index:20000;":"") + "\">");
 64 	
 65 	sb.append("<table cellspacing=\"0\" cellpadding=\"0\" border=\"0\"><tbody>");
 66 		if (isMultiMonth)
 67 		{
 68 			sb.append("<tr id=\"" + id + ".header\"><td>");
 69 			var startMonth = monthNames[startDate.getMonth()];
 70 			var startYear = startDate.getFullYear();
 71 			var endDate = dm.add(dm.clone(startDate), 'm', (monthColumns * monthRows) - 1);
 72 			var endMonth = monthNames[endDate.getMonth()];
 73 			var endYear = endDate.getFullYear();
 74 			sb.append("<div class=\"ntb-calendar-header\">");
 75 				sb.append("<table cellspacing=\"0\" cellpadding=\"0\" border=\"0\" style=\"height:100%;width:100%;\"><tbody>");
 76 				sb.append("<tr><td><a id=\"" + id + ".prevmonth\" onclick=\"return false;\" href=\"#\" class=\"ntb-calendar-prev" + (datePicker.disPrev?" ntb-calendar-prevdis":"") + "\"></a</td>");
 77 				sb.append("<td style=\"width:70%;\"><span class=\"ntb-calendar-title\" title=\"" + monthTooltip + "\" id=\"" + id + ".nav\">" + startMonth + " " + startYear + " - " + endMonth + " " + endYear + "</span></td>");
 78 				sb.append("<td><a id=\"" + id + ".nextmonth\" onclick=\"return false;\" href=\"#\" class=\"ntb-calendar-next" + (datePicker.disNext?" ntb-calendar-nextdis":"") + "\"></a></td></tr>");
 79 				sb.append("</tbody></table></div></td></tr>")
 80 		}
 81 		
 82 		sb.append("<tr id=\"" + id + ".body\"><td>");
 83 		sb.append("<table cellspacing=\"0\" cellpadding=\"0\" border=\"0\"><tbody>");
 84 		for (var i = 0; i < monthRows; i++)
 85 		{
 86 			sb.append("<tr>");
 87 			for (var j = 0; j < monthColumns; j++)
 88 			{
 89 				var calDate = dm.subtract(dm.clone(startDate), 'd', startDate.getDay());
 90 				var currentMonth = startDate.getMonth();
 91 				var currentYear = startDate.getFullYear();
 92 				
 93 				sb.append("<td>");
 94 				sb.append("<div class=\"ntb-calendar\">");
 95 					sb.append("<div><table cellspacing=\"0\" cellpadding=\"0\" border=\"0\" style=\"width:100%;\"><tbody>");
 96 						sb.append("<tr class=\"ntb-calendar-monthheader\">");
 97 							if (!isMultiMonth)
 98 								sb.append("<td><a id=\"" + id + ".prevmonth\" onclick=\"return false;\" href=\"#\" class=\"ntb-calendar-prev" + (datePicker.disPrev?" ntb-calendar-prevdis":"") + "\"></a></td>");
 99 							sb.append("<td style=\"width:70%;\"><span title=\"" + monthTooltip + "\" " + (!isMultiMonth?"id=\"" + id + ".nav\"":"") + "><a onclick=\"return false;\" href=\"#\" style=\"" + (isMultiMonth?"cursor:default;":"") + "\" class=\"ntb-calendar-month\">" + monthNames[currentMonth] + "</a>");
100 							sb.append("<a onclick=\"return false;\" href=\"#\" style=\"" + (isMultiMonth?"cursor:default;":"") + "\" class=\"ntb-calendar-year\">" + " " + currentYear + "</a></span></td>");
101 							if (!isMultiMonth)
102 								sb.append("<td><a id=\"" + id + ".nextmonth\" onclick=\"return false;\" href=\"#\" class=\"ntb-calendar-next" + (datePicker.disNext?" ntb-calendar-nextdis":"") + "\"></a></td>");
103 					sb.append("</tbody></table></div>");
104 					sb.append("<div><table id=\"" + id + "." + currentMonth + "." + currentYear + "\" cellspacing=\"0\" cellpadding=\"0\" border=\"0\" style=\"width: 100%;\"><tbody>");
105 					sb.append("<tr>");
106 					for (var k = 0; k < 7; k++)
107 					{
108 						sb.append("<th class=\"ntb-calendar-dayheader\">" + weekDays[k] + "</th>");
109 					}
110 					sb.append("</tr>");
111 					for (var m = 0; m < 6; m++)
112 					{
113 						sb.append("<tr>");
114 						for (var n = 0; n < 7; n++)
115 						{
116 							sb.append("<td>");
117 							sb.append("<a ebatype=\"date\" ebamonth=\"" + calDate.getMonth() + "\" ebadate=\"" + calDate.getDate() + "\" ebayear=\"" + calDate.getFullYear() + "\" title=\"" + dayNames[calDate.getDay()] + ", " + monthNames[calDate.getMonth()] + " " + calDate.getDate() + ", " + calDate.getFullYear() + "\" href=\"#\" onclick=\"return false;\" style=\"display:block;text-decoration:none;\" class=\"");
118 							if (selectedDate && calDate.valueOf() == selectedDate.valueOf() && calDate.getMonth() == startDate.getMonth())
119 								sb.append("ntb-calendar-currentday ");
120 							
121 							if (calDate.getMonth() < startDate.getMonth() || (minDate && calDate.valueOf() < minDate.valueOf())) sb.append("ntb-calendar-lastmonth ");
122 							else if (calDate.getMonth() > startDate.getMonth() || (maxDate && calDate.valueOf() > maxDate.valueOf())) sb.append("ntb-calendar-nextmonth ");
123 							else if (calDate.getMonth() == startDate.getMonth()) sb.append("ntb-calendar-thismonth ");
124 							
125 							if (eventsManager && eventsManager.isDisabled(calDate) && calDate.getMonth() == startDate.getMonth())
126 							{	
127 								sb.append("ntb-calendar-disabled ");
128 							}
129 							else if (eventsManager && eventsManager.isEvent(calDate) && calDate.getMonth() == startDate.getMonth())
130 							{
131 								sb.append("ntb-calendar-event ");	
132 							}
133 							
134 							if (today.valueOf() == calDate.valueOf())
135 								sb.append("ntb-calendar-today");
136 							
137 							sb.append(" ntb-calendar-day\">"+calDate.getDate()+"</a></td>");
138 							
139 							calDate = dm.add(calDate, 'd', 1);
140 						}
141 						sb.append("</tr>"); 
142 					}
143 					sb.append("</tbody></table></div></div></td>");
144 				
145 				startDate = dm.resetTime(dm.add(startDate, 'm', 1));
146 			}
147 			sb.append("</tr>");
148 		}
149 		sb.append("</tbody></table></td></tr></tbody></table></div></div>");
150 	sb.append("</tbody><colgroup span=\"7\" style=\"width:17%\"></colgroup></table></div>");
151 	sb.append("<div id=\"" + id + ".overlay\" class=\"ntb-calendar-overlay\" style=\"" + (enableShim?"z-index:20001;":"") + "top:0px;left:0px;display:none;position:absolute;background-color:gray;filter:alpha(opacity=40);-moz-opacity:.50;opacity:.50;\"></div>");
152 	sb.append(this.renderNavPanel(calendar));
153 	sb.append("</div></div>");
154 	return sb.toString();
155 }

157 /**
158  * Renders the quick nav panel.
159  * @param {nitobi.calendar.Calendar} calendar The Calendar object that the nav panel is bound to.
160  */
161 nitobi.calendar.CalRenderer.prototype.renderNavPanel = function(calendar)
162 {
163 	var sb = new nitobi.lang.StringBuilder();
164 	var datePicker = calendar.getParentObject();
165 
166 	var monthNames = datePicker.getLongMonthNames();
167 	var id = calendar.getId();
168 	var enableShim = (nitobi.browser.MOZ && !nitobi.browser.MOZ3) || (nitobi.browser.IE6 && !nitobi.browser.IE7)?true:false;
169 	
170 	sb.append("<div id=\"" + id + ".navpanel\" style=\"" + (enableShim?"z-index:20002;":"") + "position:absolute;top:0px;left:0px;overflow:hidden;\" class=\"ntb-calendar-navcontainer nitobi-hide\">");
171 		sb.append("<div class=\"ntb-calendar-monthcontainer\">");
172 			sb.append("<label style=\"display:block;\" for=\"" + id + ".months\">Choose Month</label>");
173 			sb.append("<select id=\"" + id + ".months\" class=\"ntb-calendar-navms\" style=\"\" tabindex=\"1\">");
174 			for (var i = 0; i < monthNames.length; i++)
175 			{
176 				sb.append("<option value=\"" + i + "\">" + monthNames[i] + "</option>");
177 			}
178 			sb.append("</select>");
179 		sb.append("</div>");
180 		sb.append("<div class=\"ntb-calendar-yearcontainer\">");
181 			sb.append("<label style=\"display:block;\" for=\"" + id + ".year\">Enter Year</label>");
182 			sb.append("<input size=\"4\" maxlength=\"4\" id=\"" + id + ".year\" class=\"ntb-calendar-navinput\" style=\"-moz-user-select: normal;\" tabindex=\"2\"/>");
183 		sb.append("</div>");
184 		sb.append("<div class=\"ntb-calendar-controls\">");
185 			sb.append("<button id=\"" + id + ".navconfirm\" type=\"button\">" + datePicker.getNavConfirmText() + "</button>");
186 			sb.append("<button id=\"" + id + ".navcancel\" type=\"button\">" + datePicker.getNavCancelText() + "</button>");
187 		sb.append("</div>");
188 		sb.append("<div id=\"" + id + ".warning\" style=\"display:none;\" class=\"ntb-calendar-navwarning\">You must enter a valid year.</div>");
189 	sb.append("</div>");
190 	return sb.toString();
191 }
