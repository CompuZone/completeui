1 /**
  2  * @class Fixed-height scrollable grid without paging 
  3  * <p>
  4  * <b>Fixed Size</b><br/> 
  5  * 	Height of grid specified explicitly.<br/> 
  6  * 	Vertical scrollbar allows user to scroll through rendered rows.  
  7  * 	If there are not enough rows to fill the grid, a blank area will be present, 
  8  *  the grid will not shrink.
  9  * </p>
 10  * <p> 
 11  * <b>Client-side Data</b><br/> 
 12  * 	Data in the client-side datasource constitutes all the available records.  
 13  * 	The grid renders all records present in the client-side datasource that match the specified filter 
 14  *  criteria.
 15  * 	Sorting is performed client-side.<br/>
 16  * 	All grid rows are rendered in a single render operation.<br/>
 17  * 	Data is loaded once.<br/> 
 18  * 	Allows for static client-side XML and loadXML().<br/>
 19  * </p>
 20  * <p>
 21  * <b>No Paging</b><br/> 
 22  * 	Paging operations have no effect 
 23  * </p>
 24  * @constructor
 25  * @param {String} uid The unique ID of the Grid.
 26  */
 27 nitobi.grid.GridNonpaging = function(uid) {
 28 	nitobi.grid.GridNonpaging.baseConstructor.call(this);
 29 	this.mode = "nonpaging";
 30 }
 31 nitobi.lang.extend(nitobi.grid.GridNonpaging, nitobi.grid.TreeGrid);
 32 
 33 nitobi.grid.GridNonpaging.prototype.createChildren=function() {
 34 	var args = arguments;
 35 	nitobi.grid.GridNonpaging.base.createChildren.call(this,args);
 36 	nitobi.grid.GridNonpaging.base.createToolbars.call(this,nitobi.ui.Toolbars.VisibleToolbars.STANDARD);	
 37 }
 38 
 39 //get all data if getHandler specified (no paging stuff)
 40 nitobi.grid.GridNonpaging.prototype.bind=function() 
 41 {
 42 	nitobi.grid.GridStandard.base.bind.call(this);
 43 	
 44 	if (this.getGetHandler()!='') 
 45 	{
 46 		this.ensureConnected();
 47 		this.datatable.get(0, null, this, this.getComplete);
 48 	} else 
 49 	{
 50 		// TODO: if we have created the datasource and loaded it with data BEFORE connecting to it
 51 		// the rowcount changed events etc will not have propagated to the Grid and so we will
 52 		// have an incorrect row count.
 53 		this.finalizeRowCount(this.datatable.getRemoteRowCount());
 54 		this.bindComplete();
 55 	}
 56 }
 57 nitobi.grid.GridNonpaging.prototype.getComplete=function(evtArgs)
 58 {
 59   	nitobi.grid.GridNonpaging.base.getComplete.call(this, evtArgs);
 60 
 61 	// TODO: This is not needed since the connected datatable will fire all the
 62 	// right events on the Grid when it gets the data back.
 63 	this.finalizeRowCount(evtArgs.numRowsReturned);
 64 
 65 	// TODO: This is not needed since the connected datatable will fire all the
 66 	// right events on the Grid when it gets the data back.
 67 	this.defineColumnsFinalize();
 68 
 69   	this.bindComplete();
 70 }
 71 
 72 nitobi.grid.GridNonpaging.prototype.renderMiddle= function()
 73 {
 74 	nitobi.grid.GridNonpaging.base.renderMiddle.call(this,arguments);
 75 	var startRow = this.getfreezetop();
 76 	endRow = this.getRowCount();
 77 	this.Scroller.view.midcenter.renderGap(startRow, endRow, false);
 78 }
