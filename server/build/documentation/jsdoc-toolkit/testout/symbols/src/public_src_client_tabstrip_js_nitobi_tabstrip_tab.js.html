  3 /**
  4  * Creates a tab.

  6  * @constructor
  7  * @param {XmlNode} node Optional. If you want to create a tab and deserialize it from the node.

 28 	this.eventMap["mouseout"] = this.onMouseOut;
 29 	/**
 30 	 * Fired on mouse over. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 31 	 * @see nitobi.ui.ElementEventArgs
 32 	 * @type nitobi.base.Event
 33 	 */
 34 	this.onMouseOver = new nitobi.base.Event("mouseover");
 35 	this.eventMap["mouseover"] = this.onMouseOver;

 39 	 * @type nitobi.base.Event
 40 	 */
 41 	this.onFocus = new nitobi.base.Event("focus");
 42 	this.eventMap["focus"] = this.onFocus;
 43 	/**
 44 	 * Fired on blur. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 45 	 * @see nitobi.ui.ElementEventArgs
 46 	 * @type nitobi.base.Event
 47 	 */
 48 	this.onBlur = new nitobi.base.Event("blur");
 49 	this.eventMap["blur"] = this.onBlur;
 50 	
 51 	/**
 52 	 * Fired when the tab is activated. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 53 	 * @see nitobi.ui.ElementEventArgs
 54 	 * @type nitobi.base.Event
 55 	 */
 56 	this.onActivate = new nitobi.base.Event("activate");
 57 	this.eventMap["activate"] = this.onActivate;
 58 	/**
 59 	 * Fired when the tab is deactivated. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 60 	 * @see nitobi.ui.ElementEventArgs

 62 	 */

 64 	this.eventMap["deactivate"] = this.onDeactivate;

 73 	/**
 74 	 * @private
 75 	 */
 76 	this.callback=null;
 77 	/**
 78 	 * @private
 79 	 */	
 80 	this.contentLoaded = false;
 81 	this.subscribeDeclarationEvents();
 82 	this.onCreated.notify(new nitobi.ui.ElementEventArgs(this));

216 /**
217  * @private
218  */
219 nitobi.tabstrip.Tab.prototype.handleEventNotify = function(eventArgs)
220 {
221 	var event = eventArgs.htmlEvent;
222 	var idInfo = nitobi.ui.Element.parseId(eventArgs.targetId);
223 	var returnResult = true;
224 	switch(event.type)
225 	{
226 		case("load"):
227 		{
228 			this.handleOnLoad();
229 			break;
230 		}
231 		case("click"):

233 			returnResult = this.isEnabled();
234 			break;
235 		}
236 	}
237 	return returnResult;

241 /**
242  * Loads content into the tab. If the container type is an IFrame, then the source
243  * is treated as an accessible URL. If the container type is not an iframe, the source
244  * is first treated as an element on the page, or the id of an element on the page.  If none
245  * of these conditions are true, the source is treated as an URL from which content is to be loaded. 
246  * Note: in the last case, the URL must be on the same domain as the tabstrip; browsers prohibit cross domain AJAX.
247  * If there are Nitobi components in the contents of the tab, the time of their initialization can
248  * be controlled.  If you use the onloaddemandenabled attribute, initialization of Nitobi components will 
249  * be automatically delayed until the tab is first loaded.   You can handle
250  * errors to this call with the usual try,catch procedures. As with all asynchronous calls and events
251  * you can also subscribe to {@link nitobi.error.onError} to handle errors when the load completes.
252  * @see nitobi.tabstrip.Tab#setSource 
253  * @see nitobi.tabstrip.Tab#getSource
254  * @param {String} value A source (either an html element or a url). Optional. The object's source
255  * property is used if this argument is not supplied.
256  */
257 nitobi.tabstrip.Tab.prototype.load = function(value)
258 {
259 	this.nodelist = new Array();
260 	this.nodelist[0] = this.getHtmlNode("leftpulse");
261 	nitobi.html.setOpacity(this.nodelist[0],0);
262 	this.nodelist[1] = this.getHtmlNode("bodypulse");
263 	this.nodelist[2] = this.getHtmlNode("rightpulse");
264 	var box = nitobi.html.getBox(this.getHtmlNode("labeltable"));
265 	this.nodelist[1].style.width = box.width + "px";
266 	
267 	if (value == null)
268 	{
269 		value = this.getSource();
270 	}
271 	if (value == null)
272 	{
273 		return;
274 	}
275 	this.setActivityIndicatorVisible(true);
276 	var iframeNode = this.getIframeHtmlNode();
277 	var el = $(value);
278 	if (iframeNode != null)
279 	{
280 		// IFrame target requested.
281 		try
282 		{
283 			iframeNode.src = value;
284 		}
285 		catch(err)
286 		{
287 			nitobi.lang.throwError("Could not load iframe with src " + value);
288 		}
289 	}
290 	else if (el != null)
291 	{
292 		// Another node on the page targeted.
293 		var node = this.getNodeFrameHtmlNode();
294 		node.appendChild(el);
295 		nitobi.component.loadComponentsFromNode(node);
296 		nitobi.html.Css.removeClass(el, "ntb-tab-domnode");
297 		this.handleOnLoad();
298 	} 
299 	else
300 	{
301 		try
302 		{
303 			// Load an html fragment from the server.
304 			var nodeframe = this.getNodeFrameHtmlNode();
305 			this.setActivityIndicatorVisible(true);
306 			var pool = nitobi.ajax.HttpRequestPool.getInstance();
307 			this.callback = pool.reserve();
308 			this.callback.handler = value;
309 			this.callback.context = this;
310 			this.callback.params = this.callback;
311 			this.callback.onGetComplete.subscribe(this.handleOnLoad,this);
312 			this.callback.responseType = "text";
313 			this.callback.get();
314 		} 
315 		catch(err)
316 		{
317 			nitobi.lang.throwError("The HTTP request for tab could not be performed. Is the website accessible by client script? Cross domain scriping is not permitted. Use IFrame for this purpose.",err);
318 		}
319 	}
320 }

322 /**
323  * @private
324  */
325 nitobi.tabstrip.Tab.prototype.handleOnLoad = function(eventArgs)
326 {
327 	try
328 	{
329 		if (eventArgs != null && eventArgs.params != null)
330 		{
331 			var node = this.getNodeFrameHtmlNode();
332 			if (nitobi.ajax.HttpRequest.isError(eventArgs.status))
333 			{
334 				node.innerHTML = '<div style="margin-left:20px;margin-right:20px"><h1 style="font-family:arial;font-size:14pt;">Error</h1><p style="font-family:tahoma;font-size:10pt;">The tab could not be opened because the location of the tab content could not be found.</p><ul style="font-family:tahoma;font-size:10pt;"><li>The server may be busy or not responding</li><li>The address of the tab content may be incorrect.</li><li>The address may be that of an HTML Element that was not on the page</li></ul><p style="font-family:tahoma;font-size:10pt;">Try again later. If the problem persists, contact your local administrator.</p><p style="font-family:tahoma;font-size:10pt;">The faulty source was ' + this.getSource() + '. The server return code was <b>'+eventArgs.status+' ('+eventArgs.statusText+').</b> The server response follows:</p><hr/><p>'+eventArgs.response+'</p></div>';
335 				nitobi.error.onError.notify(new nitobi.error.ErrorEventArgs(this,nitobi.error.HttpRequestError + "\n\n OR \n\n " + nitobi.error.NoHtmlNode,nitobi.tabstrip.Tab.profile.className));
336 			}
337 			else
338 			{
339 				node.innerHTML = eventArgs.response;
340 				if (this.isScriptEvaluationEnabled())
341 				{
342 					nitobi.html.evalScriptBlocks(node);
343 				}
344 				nitobi.component.loadComponentsFromNode(node);
345 			}
346 			var pool = nitobi.ajax.HttpRequestPool.getInstance();
347 			pool.release(eventArgs.params); 
348 		}
349 		this.setContentLoaded(true);
350 		this.setActivityIndicatorVisible(false);
351 		this.onLoad.notify(new nitobi.ui.ElementEventArgs(this));
352 	}
353 	catch(err)
354 	{
355 		
356 		nitobi.error.onError.notify(new nitobi.error.ErrorEventArgs(this,"The tab encountered an error while trying to parse the response from load. There may be an error in the onLoad event.",nitobi.tabstrip.Tab.profile.className));
357 	}
358 }

384 /**
385  * Sets the activity indicator according to the current state of the load operation. This change takes 
386  * effect immediately; you do not have to call render.
387  */
388 nitobi.tabstrip.Tab.prototype.autoSetActivityIndicator = function()
389 {
390 	if (this.getContainerType() == "iframe")
391 	{
392 		var iframeNode = this.getIframeHtmlNode();
393 		if (iframeNode != null)
394 		{
395 			// readyState property is IE only.
396 			if (nitobi.browser.IE)
397 			{
398 				this.setActivityIndicatorVisible(iframeNode.readyState != "complete");
399 			}
400 			else
401 			{
402 				if (this.contentLoaded == true) 
403 				{
404 					this.setActivityIndicatorVisible(false);
405 				} 
406 				else if (this.getLoadOnDemandEnabled() == true)
407 				{
408 					this.setActivityIndicatorVisible(false);
409 				} 
410 			}				
411 		} else if (this.callback != null)
412 		{
413 			this.setActivityIndicatorVisible(this.callback.readyState != 4);
414 		}
415 	}		
416 }

682 /**
683  * The type of container for the tab body. It can be iframe, or if the string is empty, it is a domnode.
684  * Once the tab has been rendered, this is immutable. To change it, remove the container, by removing
685  * its HTML node (with standard Javascript DHTML), and re-render.  Alternatively, you can remove the 
686  * tab, and create a new one with a different container type.
687  * @param {String} value The container type. Either 'iframe' or ''.
688  */
689 nitobi.tabstrip.Tab.prototype.setContainerType = function(value)
690 {
691 	if (value != '' && value != 'iframe')
692 	{
693 		nitobi.lang.throwError(nitobi.error.BadArg + " Valid values are 'iframe' or ''");
694 	}
695 	this.setAttribute("containertype",value);
696 }
