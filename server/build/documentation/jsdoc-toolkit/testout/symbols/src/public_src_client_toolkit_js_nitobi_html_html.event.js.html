1 /**
  2  * <I>This class has no constructor.</I>
  3  * @class nitobi.html.Event is a cross-browser, globally accessible event object which is set when DOM events are fired through the EventManager.
  4  * @constructor
  5  */
  6 nitobi.html.Event = function(){
  7 	/**
  8 	 * The DOM element on which the event was fired. This is equivalent to the <code>target</code> property in 
  9 	 * the Firefox and Safari browsers.
 10 	 */
 11 	this.srcElement = null;
 12 	/**
 13 	 * The DOM element from which activation or the mouse pointer is exiting during the event. This is useful for mouseout type events.
 14 	 */
 15 	this.fromElement = null;
 16 	/**
 17 	 * The DOM element from which activation or the mouse pointer is entering during the event. This is useful for mouseout type events.
 18 	 */
 19 	this.toElement = null;
 20 	/**
 21 	 * The DOM element to which the event handler was attached.
 22 	 */
 23 	this.eventSrc = null;
 24 };

 47 /**
 48  * Attaches multiple events to a single element
 49  * @param {HtmlElement} element The HTML element to which the events should be attached.
 50  * @param {Struct} events Structs of the events with type and handler properties.
 51  * @param {Object} context The context in which the handlers should be executed.
 52  * @type {Array}
 53  * @return Returns an array of GUIDs that can be used to detach the event handlers.
 54  */
 55 nitobi.html.attachEvents = function(element, events, context)
 56 {
 57 	var guids = [];
 58 	for (var i=0; i<events.length; i++)
 59 	{
 60 		var e = events[i];
 61 		guids.push(nitobi.html.attachEvent(element, e.type, e.handler, context, e.capture || false));
 62 	}
 63 	return guids;
 64 }

 68 /**
 69  * Attaches the specified handler to the DOM element for the specified event type.
 70  * @param {HtmlElement} element Element on which the event is to be registered.
 71  * @param {String} type Event type such as 'click', 'mouseover' etc. Note that the 'on' prefix is dropped.
 72  * @param {Object} context Object that the callback function will be called on. Optional.
 73  * @param {Function} handler Function pointer to the function that will handle the event.
 74  * @param {Boolean} capture Specifies if event capturing should be used. Optional.
 75  * @type {String}
 76  * @return The GUID of the attached event handler.
 77  */
 78 nitobi.html.attachEvent = function(element, type, handler, context, capture, override)
 79 {
 80 	if (type == "anyclick")
 81 	{
 82 		if (nitobi.browser.IE)
 83 		{
 84 			nitobi.html.attachEvent(element, "dblclick", handler, context, capture, override);
 85 		}
 86 		type = "click";
 87 	} 
 88 	if (!(handler instanceof Function))
 89 	{
 90 		nitobi.lang.throwError("Event handler needs to be a Function");
 91 	}
 92 
 93 	//	Allow people to pass in a node id or an actual DOM node.
 94 	element = $(element);
 95 
 96 	// Divert unload attachments to custom unload registration
 97 	if (type.toLowerCase() == 'unload' && override != true)
 98 	{
 99 		var funcRef = handler;
100 		if (context != null)
101 		{
102 			funcRef = function() {handler.call(context)};
103 		}
104 		return this.addUnload(funcRef);
105 	}
106 
107 	//	Increment our unique id to keep it unique
108 	var handlerGuid = this.handlerId++;
109 	var elementGuid = this.elementId++;
110 
111 	//	Check if the handler already has a guid or not
112 	//	The guid is used again when we detach the handler for an event
113 	if (typeof(handler.ebaguid) != "undefined")
114 	{
115 		handlerGuid = handler.ebaguid;
116 	}
117 	else
118 	{
119 		handler.ebaguid = handlerGuid;
120 	}
121 
122 	//	Check the custom ebaguid property on the DOM Element object
123 	//	The ebaguid on the element is used to keep track of all elements that have had events attached
124 	if (typeof(element.ebaguid) == "undefined")
125 	{
126 		element.ebaguid = elementGuid;
127 		//	add a reference to the element in the elements hash - this could also just use array.push()
128 		nitobi.html.elements[elementGuid] = element;
129 	}
130 
131 	//	The custom eba_events property on the DOM Element object contains
132 	//	all the events that have been registered on this element.
133 	if (typeof(element.eba_events) == "undefined")
134 	{
135 		element.eba_events = {};
136 	}
137 
138 	//	The 'eba_events' property on the DOM node is a hash of the event type
139 	//	ie store all the handlers defined for event type 'mouseover' in another hash
140 	if (element.eba_events[type] == null)
141 	{
142 		element.eba_events[type] = {};
143 
144 		//	Browser checking for IE / nitobi.browser.MOZ
145 		if (element.attachEvent)
146 		{
147 			// This maintains a reference to the closure we use as the event handler so we can clean it up later.
148 			element['eba_event_'+type] = function () {nitobi.html.notify.call(element, window.event)};
149 			//	Detach will need to be called to get rid of this closure...
150 			element.attachEvent('on'+type, element['eba_event_'+type]);
151 
152 			if (capture && element.setCapture != null) element.setCapture(true);
153 		}
154 		else if (element.addEventListener)
155 		{
156 			// No need for this code really - only used for detachment purposes later.
157 			// If there are no more handlers attached for a certain event we clean things up by deleting this function reference.
158 			element['eba_event_'+type] = function () {nitobi.html.notify.call(element, arguments[0])};
159 			element.addEventListener(type, element['eba_event_'+type], capture);
160 		}
161 	}
162 
163 	//	Once we get here the event has been hooked up 
164 	//	or it is already hooked up - either way we need to
165 	//	add the handler to the list of handlers to fire
166 	element.eba_events[type][handlerGuid] = {handler: handler, context: context};
167 
168 	return handlerGuid;
169 }

171 /**
172  * @private
173  * The method that is used to handle all the DOM events and dispatch them accordingly.
174  */
175 nitobi.html.notify = function(e)
176 {
177 	if (!nitobi.browser.IE)
178 	{
179 		e.srcElement = e.target;
180 		e.fromElement = e.relatedTarget;
181 		e.toElement = e.relatedTarget;
182 	}
183 	var element = this;
184 	e.eventSrc = element
185 	nitobi.html.Event = e;
186 
187 	for (var handlerGuid in element.eba_events[e.type])
188 	{
189 		var event_ = element.eba_events[e.type][handlerGuid];
190 		if (typeof(event_.context) == "object")
191 		{
192 			//	Call the handler in the context of the object located in context.
193 			event_.handler.call(event_.context, e, element);
194 		}
195 		else
196 		{
197 			event_.handler.call(element, e, element);
198 		}
199 	}
200 }

202 /**
203  * Detaches multiple events to a single element
204  * 
205  * @param {HTMLElement} element The HTML element to which the events should be detached from.
206  * @param {Object} events Structs of the events with type and handler properties.
207  */
208 nitobi.html.detachEvents = function(element, events)
209 {
210 	for (var i=0; i<events.length; i++)
211 	{
212 		var e = events[i];
213 		nitobi.html.detachEvent(element, e.type, e.handler);
214 	}
215 }

217 /**
218  * Removes an event handler from an HTML element.
219  * @param {HTMLElement} element The HTML element to remove the handler from.
220  * @param {String} type The event type string such as "mouseover".
221  * @param {Function | String} handler The event handler function reference or the GUID of the handler returned from the attachEvent function.
222  * @see nitobi.html#attachEvent
223  */
224 nitobi.html.detachEvent = function(element, type, handler)
225 {
226 	//	Allow either a node id or an actual DOM node.
227 	element = $(element);
228 
229 	// Allow either a Function with a guid or just the guid
230 	var handlerGuid = handler; 
231 	if (handler instanceof Function)
232 		handlerGuid = handler.ebaguid;
233 
234 	// If it is for an unload event it is special.
235 	if (type == "unload")
236 	{
237 		this.unload.splice(ebaguid,1);
238 	}
239 
240 	//	Check if the event type and handler combination are defined
241 	if (element != null && element.eba_events != null && element.eba_events[type] != null && element.eba_events[type][handlerGuid] != null)
242 	{
243 
244 		var handlers = element.eba_events[type];
245 		// Remove it from the list of handlers
246 		handlers[handlerGuid] = null;
247 		delete handlers[handlerGuid];
248 
249 		//	What actually needs to be done here is ..
250 		//	if this is the last handler for a certain event type on an element to be removed
251 		//	then we need to detach it...
252 		// element['eba_event_'+type]
253 
254 		if (nitobi.collections.isHashEmpty(handlers))
255 		{
256 			this.m_detach(element, type, element['eba_event_'+type]);
257      		element['eba_event_'+type] = null;
258      		element.eba_events[type] = null;
259      		handlers = null;
260      		if (element.nodeType == 1)
261          		element.removeAttribute('eba_event_'+type);
262 		}
263 	}
264 	return true;
265 }

267 /**
268  * @private
269  * Does the event detachment leg work for both detachAllEvents and detachEvent.
270  */
271 nitobi.html.m_detach = function(element, type, handler)
272 {
273 	if (handler != null && handler instanceof Function)
274 	{
275 		//	Detach in IE
276 		if (element.detachEvent)
277 		{
278 			element.detachEvent('on' + type, handler)
279 		}
280 		//	Remove in Firefox
281 		else if (element.removeEventListener)
282 		{
283 			element.removeEventListener(type, handler, false);
284 		}
285 		element['on' + type] = null;
286 	
287 		if (type == "unload")
288 		{
289 			//	Here we are doing unload so call all the things that are manually registered for unload
290 			for (var i=0; i<this.unload.length; i++)
291 			{
292 				this.unload[i].call(this);
293 				this.unload[i] = null;
294 			}
295 		}
296 	}
297 }

299 /**
300  * Detaches all the events that have been attached using the <CODE>nitobi.html.attachEvent</CODE> method.
301  */
302 nitobi.html.detachAllEvents = function(evt)
303 {
304 	//	TODO: this needs to be fixed not to use the looping through the array and use the hash instead
305 	for (var i=0; i<nitobi.html.elements.length; i++)
306 	{
307 		if (typeof(nitobi.html.elements[i]) != "undefined")
308 		{
309 			for (var eventType in nitobi.html.elements[i].eba_events)
310 			{
311 				//	Need to detach the nitobi.html.event.notify method from the element.
312 				nitobi.html.m_detach(nitobi.html.elements[i], eventType, nitobi.html.elements[i]['eba_event_'+eventType]);
313 
314 				if (typeof(nitobi.html.elements[i]) != "undefined" && nitobi.html.elements[i].eba_events[eventType] != null)
315 				{
316 					for (var handlerGuid in nitobi.html.elements[i].eba_events[eventType])
317 					{
318 						nitobi.html.elements[i].eba_events[eventType][handlerGuid] = null;
319 					}
320 				}
321 				nitobi.html.elements[i]['eba_event_'+eventType] = null;
322 			}
323 		}
324 	}
325 	nitobi.html.elements = null;
326 }

328 /**
329  * Used to register unload events since using the standard attachEvent can be problematic for unload events.
330  */
331 nitobi.html.addUnload = function(funcRef)
332 {
333 	this.unload.push(funcRef);
334 	return this.unload.length-1;

347 /**
348  * Stops the event bubbling.
349  * @param {nitobi.html.Event} The event that should be stopped.
350  */
351 nitobi.html.stopPropagation = function(evt)
352 {
353 	if (evt == null)
354 		return;
355 
356 	if (nitobi.browser.IE)
357 		evt.cancelBubble = true;
358 	else
359 		evt.stopPropagation();
360 }

382 /**
383  * The coordinates of the event in the format of <CODE>{"x":0,"y":0}</CODE>.
384  * @param {nitobi.html.Event} That event object of the event for which the coordinates are required.
385  * @type Object
386  */
387 nitobi.html.getEventCoords = function(evt)
388 {
389     var coords = {'x':evt.clientX,'y':evt.clientY};
390     if (nitobi.browser.IE) 
391     {
392     	// TODO: document.body if not standards mode.
393         coords.x += document.documentElement.scrollLeft + document.body.scrollLeft;
394         coords.y += document.documentElement.scrollTop + document.body.scrollTop;
395     }
396     else
397     {
398         coords.x += window.scrollX;
399         coords.y += window.scrollY;
400     }
401 
402     return coords;
403 }
