  8 /**
  9  * Creates an Element.
 10  * @class An abstract class that represents an element.
 11  * @constructor
 12  * @param {String|XMLNode|HTMLElement} The id of an element, or the node representing an element, or an HTML declaration for the element.
 13  * @extends nitobi.Object
 14  * @implements nitobi.base.ISerializable
 15  * @implements nitobi.ui.IStyleable
 16  */
 17 nitobi.ui.Element = function(id)
 18 {
 19 	nitobi.ui.Element.baseConstructor.call(this);
 20 	// ISerializable is initialized below so that we don't have to re-initialize it with another node.
 21 	nitobi.ui.IStyleable.call(this);
 22 	
 23 	if (id != null)
 24 	{
 25 		if (nitobi.lang.typeOf(id) == nitobi.lang.type.XMLNODE)
 26 		{
 27 			// Assume that id is an XmlNode.
 28 			nitobi.base.ISerializable.call(this,id);
 29 		}
 30 		else if ($(id) != null)
 31 		{
 32 			// Assume a decl.
 33 			var decl = new nitobi.base.Declaration();
 34 			var xmlNode = decl.loadHtml($(id));
 35 
 36 			var oldContainer = $(id);
 37 			var parentNode = oldContainer.parentNode;
 38 			var wrapper = parentNode.ownerDocument.createElement('ntb:component');
 39 			parentNode.insertBefore(wrapper, oldContainer);
 40 			parentNode.removeChild(oldContainer);
 41 			this.setContainer(wrapper);
 42 
 43 			nitobi.base.ISerializable.call(this,xmlNode);
 44 						
 45 		} else
 46 		{
 47 			nitobi.base.ISerializable.call(this);
 48 			// Assume string.
 49 			this.setId(id);
 50 		}
 51 	}
 52 	else
 53 	{
 54 		nitobi.base.ISerializable.call(this);
 55 	}
 56 	
 57 	/**
 58 	 * @ignore
 59 	 */
 60 	this.eventMap = {};
 61 	
 62 	/**
 63 	 * Fired when the object is created. {@link nitobi.ui.ElementEventArgs} are passed to the listener.
 64 	 * @see nitobi.ui.ElementEventArgs
 65 	 * @type nitobi.base.Event
 66 	 */
 67 	this.onCreated = new nitobi.base.Event("created");
 68 	this.eventMap["created"] = this.onCreated;
 69 	
 70 	
 71 	// API Events
 72 	/**
 73 	 * Fired on before render. {@link nitobi.ui.ElementEventArgs} are passed to the listener.
 74 	 * @type nitobi.base.Event
 75 	 */
 76 	this.onBeforeRender = new nitobi.base.Event("beforerender");
 77 	this.eventMap["beforerender"] = this.onBeforeRender;
 78 	
 79 	/**
 80 	 * Fired on render. {@link nitobi.ui.ElementEventArgs} are passed to the listener.
 81 	 * @see nitobi.ui.ElementEventArgs
 82 	 * @type nitobi.base.Event
 83 	 */
 84 	this.onRender = new nitobi.base.Event("render");
 85 	this.eventMap["render"] = this.onRender;
 86 	
 87 	/**
 88 	 * Fired on before set visible. {@link nitobi.ui.ElementEventArgs} are passed to the listener.
 89 	 * @see nitobi.ui.ElementEventArgs
 90 	 * @type nitobi.base.Event
 91 	 */
 92 	this.onBeforeSetVisible = new nitobi.base.Event("beforesetvisible");
 93 	this.eventMap["beforesetvisible"] = this.onBeforeSetVisible;
 94 	/**
 95 	 * Fired on set visible. {@link nitobi.ui.ElementEventArgs} are passed to the listener.
 96 	 * @see nitobi.ui.ElementEventArgs
 97 	 * @type nitobi.base.Event
 98 	 */
 99 	this.onSetVisible = new nitobi.base.Event("setvisible");
100 	this.eventMap["setvisible"] = this.onSetVisible;
101 	
102 	/**
103 	 * Fires before the element propagates an event 
104 	 * down the tree. If false is returned, then the propagation does
105 	 * not occur. If true is returned, then the default
106 	 * propagation occurs. EventNotificationEventArgs are passed to the event handler.
107 	 * @see nitobi.ui.EventNotificationEventArgs
108 	 * @private
109 	 * @type nitobi.base.Event;
110 	 */
111 	this.onBeforePropagate = new nitobi.base.Event("beforepropagate");
112 	
113 	/**
114 	 * Fired when the event needs handling by this object. EventNotificationEventArgs are passed to the event handler.
115 	 * @see nitobi.ui.EventNotificationEventArgs
116 	 * @private
117 	 * @type nitobi.base.Event;
118 	 */
119 	this.onEventNotify = new nitobi.base.Event("eventnotify");
120 	
121 	/**
122 	 * Fired before the object is notified. If this event returns false, then this.onEventNotify is not called. EventNotificationEventArgs are passed to the event handler.
123 	 * @see nitobi.ui.EventNotificationEventArgs
124 	 * @type nitobi.base.Event;
125 	 * @private
126 	 */
127 	this.onBeforeEventNotify = new nitobi.base.Event("beforeeventnotify");
128 		
129 	/**
130 	 * Fires before the element propagates an event 
131 	 * down the tree. If false is returned, then the propagation does
132 	 * not occur. If true is returned, then the default
133 	 * propagation occurs. This differs from onBeforeProgagate because
134 	 * at this point the childId is available. EventNotificationEventArgs are passed to the event handler.
135 	 * @see nitobi.ui.EventNotificationEventArgs
136 	 * @private
137 	 * @type nitobi.base.Event;
138 	 */
139 	this.onBeforePropagateToChild = new nitobi.base.Event("beforepropogatetochild");
140 
141 	
142 	this.subscribeDeclarationEvents();
143 	
144 	this.setEnabled(true);
145 	
146 	/**
147 	 * The renderer.
148 	 * @type nitobi.html.XslRenderer
149 	 */
150 	this.renderer = new nitobi.html.XslRenderer();
151 	
152 };

199 /**
200  * Each element has an id; sub elements have a name that is appended
201  * to the id; this function retrieves both the "id" and the "localName" 
202  * in a map.
203  * @param {String} id The full id of the sub element.
204  * @type Map
205  */
206 nitobi.ui.Element.parseId = function(id)
207 {
208 	var ids = id.split(".");
209 	if (ids.length <= 2)
210 		return {localName:ids[1],id:ids[0]};
211 	return {localName:ids.pop(),id:ids.join('.')};
212 }

222 /**
223  * A catch all event notifier. If the child class does not override this function,
224  * then element will just fire the event, whatever it is.
225  * @private
226  */
227 nitobi.ui.Element.prototype.notify = function(event, id, propagationList, cancelBubble)
228 {
229 	try
230 	{
231 		event = nitobi.html.getEvent(event);
232 		if (cancelBubble !== false)
233 		{
234 			nitobi.html.cancelEvent(event);
235 		}
236 		var targetId = nitobi.ui.Element.parseId(id).id;
237 		
238 		// If notify is called on this element, and the target doesn't exist here
239 		// then just return;
240 		if (!this.isDescendantExists(targetId))
241 		{
242 			return false;
243 		}
244 		var propagate = !(targetId == this.getId());
245 		var eventArgs = new nitobi.ui.ElementEventArgs(this,null,id);
246 		var eventNotificationArgs = new nitobi.ui.EventNotificationEventArgs(this,null,id,event);
247 		propagate = propagate && this.onBeforePropagate.notify(eventNotificationArgs);
248 		var result=true;
249 		if (propagate)
250 		{
251 			if (propagationList == null)
252 			{
253 				propagationList = this.getPathToLeaf(targetId);								
254 			}
255 			var fireNotify = this.onBeforeEventNotify.notify(eventNotificationArgs);
256 			var eventResult = (fireNotify ? this.onEventNotify.notify(eventNotificationArgs) : true);
257 			var nextId = propagationList.pop().getAttribute("id");
258 			var nextObject = this.getObjectById(nextId);
259 			var result = this.onBeforePropagateToChild.notify(eventNotificationArgs);
260 			if (nextObject.notify && result && eventResult)
261 			{
262 				result = nextObject.notify(event,id,propagationList,cancelBubble);
263 			}
264 		}
265 		else
266 		{
267 			result = this.onEventNotify.notify(eventNotificationArgs);
268 		}
269 		
270 		// Notify the nitobi event object for this event, if one exists.
271 		// e.g. if the event is onclick, then lookup to see if an Event object exists (e.g. this.onClick = new nitobi.base.Event())
272 		// and if it does, fire it.
273 		var objEvent = this.eventMap[event.type];
274 		if (objEvent != null && result)
275 		{
276 			objEvent.notify(this.getEventArgs(event, id));
277 		}
278 		
279 		return result;
280 	}
281 	catch(err)
282 	{
283 		nitobi.lang.throwError(nitobi.error.Unexpected + " Element.notify encountered a problem.",err);
284 	}
285 }

299 /**
300  * Subscribes any events found in the decl to the corresponding event object.
301  * @private
302  */
303 nitobi.ui.Element.prototype.subscribeDeclarationEvents = function()
304 {
305 	for (var name in this.eventMap)
306 	{
307 		var ev = this.getAttribute("on" + name);
308 		if (ev != null && ev != "")
309 		{
310 			this.eventMap[name].subscribe(ev,this,name);
311 		}
312 	}

315 /**
316  * Returns the HTML node associated with this element. If name is specified,
317  * the HTML node searched for is assumed to be a sub-element of this object's node.
318  * @param {String} name The name of the sub element. Optional.
319  * @type HTMLElement
320  */
321 nitobi.ui.Element.prototype.getHtmlNode = function(name)
322 {
323 	var id = this.getId();
324 	id = (name != null ? id + "." + name : id); 
325 	var node = nitobi.ui.Element.htmlNodeCache[name];
326 	if (node==null)
327 	{
328 		node = $(id);
329 		nitobi.ui.Element.htmlNodeCache[id] = node;		
330 	}
331 	return node;
332 };

364 /**
365  * Returns <code>true</code> if the component has not been hidden. This value does not take into 
366  * account the visibility of any parent objects.
367  * @type Boolean
368  */
369 nitobi.ui.Element.prototype.isVisible = function()
370 {
371 	var node = this.getHtmlNode();
372 	return node && !nitobi.html.Css.hasClass(node, NTB_CSS_HIDE);
373 };
374 /**
375  * Show or hide the component using the given effect.  <code>effectClass</code> is the class 
376  * of effect to use - <code>setVisible()</code> will instantiate a new effect object of this class. 
377  * @param {Boolean} visible if <code>true</code>, show the component, otherwise hide
378  * @param {Class} effectClass the class of effect to use (if any) to show the element (Optional)
379  * @param {Function} callback a function to evaluate after the element is shown (Optional)
380  */
381 nitobi.ui.Element.prototype.setVisible = function(visible, effectClass, callback, params)
382 {
383 
384 	var element = this.getHtmlNode();
385 	if (element && this.isVisible() != visible && this.onBeforeSetVisible.notify({source: this, event: this.onBeforeSetVisible, args: arguments}) !== false)
386 	{
387 		if (this.effect)
388 		{
389 			this.effect.end();
390 		}
391 		if (visible)
392 		{
393 			if (effectClass)
394 			{
395 				var effect = new effectClass(element, params);
396 				effect.callback = nitobi.lang.close(this, this.handleSetVisible, [callback]);
397 				this.effect = effect;
398 				effect.onFinish.subscribeOnce(nitobi.lang.close(this, function(){this.effect = null}));
399 				effect.start();
400 			}
401 			else
402 			{
403 				nitobi.html.Css.removeClass(element, NTB_CSS_HIDE);
404 				this.handleSetVisible(callback);
405 			}
406 		}
407 		else
408 		{
409 			if (effectClass)
410 			{
411 				var effect = new effectClass(element, params);
412 				effect.callback = nitobi.lang.close(this, this.handleSetVisible, [callback]);
413 				this.effect = effect;
414 				effect.onFinish.subscribeOnce(nitobi.lang.close(this, function(){this.effect = null}));
415 				effect.start();
416 			}
417 			else
418 			{
419 				nitobi.html.Css.addClass(this.getHtmlNode(), NTB_CSS_HIDE);
420 				this.handleSetVisible(callback);
421 			}
422 		}
423 	}
424 };

453 /**
454  * Render the element.  Without arguments this will render the component in place of the in-page
455  * XML declaration.
456  * @param {HTMLElement} container a container whose inner HTML will be replaced 
457  * by the rendered component (Optional)
458  * @param {XMLDocument} state an XML document representing the state of the component (Optional)
459  */
460 nitobi.ui.Element.prototype.render = function(container, state)
461 {
462 	this.flushHtmlNodeCache();
463 	state = state || this.getState();
464 	container = $(container) || this.getContainer();
465 	
466 	if (container == null)
467 	{
468 		var container = document.createElement("span");
469 		document.body.appendChild(container);
470 		this.setContainer(container);
471 	}
472 	this.htmlNode = this.renderer.renderIn(container,state)[0];
473 	this.htmlNode.jsObject = this;
474 };
