1 /**
  2  * Constructs a Selection object.
  3  * @class A Selection represents the cells of a Grid that are currently
  4  * selected.  This class can be used to programatically select a region of
  5  * cells.
  6  * @constructor
  7  */
  8 nitobi.grid.Selection = function(owner, dragFillEnabled)
  9 {
 10 	nitobi.grid.Selection.baseConstructor.call(this,owner);
 11 	
 12 	/**
 13 	 * @private
 14 	 */
 15 	this.owner = owner;
 16     var t = new Date();
 17     /**
 18      * Specifies whether the selection is in the active selecting state or not.
 19      * @private
 20      */
 21 	this.selecting = false;
 22 	/**
 23 	 * Specifies whether the selection is being expanded or not. During expansion 
 24 	 * the expanding and selection flags will be true.
 25 	 * @private
 26 	 */
 27 	this.expanding = false;
 28 	/**
 29 	 * @private
 30 	 */
 31 	this.resizingRow = false;
 32 	/**
 33 	 * @private
 34 	 */
 35 	this.created=false;
 36 	/**
 37 	 * @private
 38 	 */
 39 	this.freezeTop = this.owner.getfreezetop();
 40 	/**
 41 	 * @private
 42 	 */
 43 	this.freezeLeft = this.owner.getFrozenLeftColumnCount();
 44 	/**
 45 	 * @private
 46 	 */
 47 	this.rowHeight = 23;
 48 
 49 	/**
 50 	 * Event that fires when an existing selection has been expanded using the mouse.
 51 	 */
 52 	this.onAfterExpand = new nitobi.base.Event();
 53 
 54 	/**
 55 	 * Event that fires when a selection expansion starts.
 56 	 */
 57 	this.onBeforeExpand = new nitobi.base.Event();
 58 
 59 	/**
 60 	 * Event that fires when a mouseup event fires on the selection.
 61 	 */
 62 	this.onMouseUp = new nitobi.base.Event();
 63 
 64 	/**
 65 	 * The bottom right cell of the selection when an expansion is started.
 66 	 * @private
 67 	 */
 68 	this.expandEndCell = null;
 69 
 70 	/**
 71 	 * The top left cell of the selection when an expansion is started.
 72 	 * @private
 73 	 */
 74 	this.expandStartCell = null;
 75 	
 76 	/**
 77 	 * @private
 78 	 */
 79 	this.dragFillEnabled = dragFillEnabled || false;

135 			{type:'click', handler:this.handleSelectionClick},
136 			{type:'dblclick', handler:this.handleDblClick}];

139 		H.attachEvents(this.boxl, this.events, this);
140 
141 		var sv = this.owner.scroller.surface.view;
142 		sv.midcenter.surface.appendChild(this.box);
143 		sv.midleft.surface.appendChild(this.boxl);

154 nitobi.grid.Selection.prototype.createBox = function (id)
155 {
156 	// boxBorder is the outer container for the selections
157 	var boxBorder;
158 	var cell;
159 	// In IE quirksmode and IE6 use a div for the selection rather than a table ...
160 	if (nitobi.browser.IE) {
161 		cell = boxBorder = document.createElement("div");
162 	} else {
163 		boxBorder = nitobi.html.createTable({"cellpadding":0,"cellspacing":0,"border":0}, {"backgroundColor":"transparent"});
164 		cell = boxBorder.rows[0].cells[0];
165 	}
166 	boxBorder.className = "ntb-grid-selection ntb-grid-selection-border";
167 	boxBorder.setAttribute("id", "ntb-grid-selection-"+id);
168 	var boxBackground = nitobi.html.createElement("div", {"id":id, "class":"ntb-grid-selection-background"});
169 	cell.appendChild(boxBackground);
170 	return boxBorder;
171 }

210 nitobi.grid.Selection.prototype.handleGrabbyMouseDown = function(evt)
211 {
212 	this.selecting = true;
213 	this.setExpanding(true, "vert");
214 
215 	var topLeft = this.getTopLeftCell();
216 	var btmRight = this.getBottomRightCell();
217 
218 	this.expandStartCell = topLeft;
219 	this.expandEndCell = btmRight;
220 
221 	var scrollSurface = this.owner.getScrollSurface();
222 	// TODO: this will be more complicated with multiple selections available
223 	this.expandStartCoords = this.box.getBoundingClientRect(scrollSurface.scrollTop+document.body.scrollTop, scrollSurface.scrollLeft+document.body.scrollLeft);
224 
225 	this.expandStartHeight = Math.abs(topLeft.getRow() - btmRight.getRow()) + 1;
226 	this.expandStartWidth = Math.abs(topLeft.getColumn() - btmRight.getColumn()) + 1;
227 	this.expandStartTopRow = topLeft.getRow();
228 	this.expandStartBottomRow = btmRight.getRow();
229 	this.expandStartLeftColumn = topLeft.getColumn();
230 	this.expandStartRightColumn = btmRight.getColumn();
231 
232 	var Cell = nitobi.grid.Cell;
233 	// When we start expanding the top left needs to be made into the start cell
234 	if (Cell.getRowNumber(this.startCell) > Cell.getRowNumber(this.endCell))
235 	{
236 		var tmpCell = this.startCell;
237 		this.startCell = this.endCell;
238 		this.endCell = tmpCell;
239 	}
240 
241 	this.onBeforeExpand.notify(this);
242 
243 // This is commented out for Safari
244 //	nitobi.html.cancelEvent(evt);
245 //	return false;

271 /**
272  * Expands the area of the selection to the location of the mouse. This could be limited to either vertical or horizontal expansion like Excel.
273  * @private 
274  */
275 nitobi.grid.Selection.prototype.expand = function(cell, dir) {
276 	this.setExpanding(true, dir);
277 
278 	var Cell = nitobi.grid.Cell;
279 	var endCell;
280 
281 	// Theses are the variables that keep track of the original coords of the selection when expansion started
282 	var expandStartTopRow = this.expandStartTopRow, expandStartLeftColumn = this.expandStartLeftColumn;
283 	var expandStartBottomRow = this.expandStartBottomRow, expandStartRightColumn = this.expandStartRightColumn;
284 
285 	// Get the end column and row numbers
286 	var endRow = Cell.getRowNumber(this.endCell), endColumn = Cell.getColumnNumber(this.endCell);
287 	// Get the start column and row numbers
288 	var startRow = Cell.getRowNumber(this.startCell), startColumn = Cell.getColumnNumber(this.startCell);
289 
290 	// Get the new end column
291 	var newEndColumn = Cell.getColumnNumber(cell);
292 	// Get the new end row
293 	var newEndRow = Cell.getRowNumber(cell);
294 	// Get the surface path
295 	var surfacePath = Cell.getSurfacePath(cell);
296 
297 	var newStartColumn = startColumn, newStartRow = startRow;
298 	var o = this.owner;
299 
300 	if (dir == "horiz") {
301 		// We are expanding horizontal
302 		if (startColumn < endColumn & newEndColumn < startColumn) {
303 			// We are expanding to the left back on ourselves so the start and end cells need to be switched.
304 			this.changeEndCellWithDomNode(o.getCellElement(expandStartBottomRow, newEndColumn, surfacePath));
305 			this.changeStartCellWithDomNode(o.getCellElement(expandStartTopRow, expandStartRightColumn, surfacepath));
306 		} else if (startColumn > endColumn && newEndColumn > startColumn) {
307 			// We are expanding to the right back on ourselves so the start and end cells need to be switched.
308 			this.changeEndCellWithDomNode(o.getCellElement(expandStartBottomRow, newEndColumn, surfacePath));
309 			this.changeStartCellWithDomNode(o.getCellElement(expandStartTopRow, expandStartLeftColumn, surfacePath));
310 		} else {
311 			// We are expanding away from the start cell so choose the row furthest from the start row
312 			this.changeEndCellWithDomNode(o.getCellElement((startRow==expandStartBottomRow?expandStartTopRow:expandStartBottomRow), newEndColumn, surfacePath));
313 		}
314 	} else {
315 		// We are expanding vertical
316 		if (startRow < endRow & newEndRow < startRow) {
317 			// We are expanding to the top back on ourselves so the start and end cells need to be switched.
318 			this.changeEndCellWithDomNode(o.getCellElement(newEndRow, expandStartRightColumn, surfacePath));
319 			this.changeStartCellWithDomNode(o.getCellElement(expandStartBottomRow, expandStartLeftColumn, surfacePath));
320 		} else if (startRow > endRow && newEndRow > startRow) {
321 			// We are expanding to the bottom back on ourselves so the start and end cells need to be switched.
322 			this.changeEndCellWithDomNode(o.getCellElement(newEndRow, expandStartRightColumn, surfacePath));
323 			this.changeStartCellWithDomNode(o.getCellElement(expandStartTopRow, expandStartLeftColumn, surfacePath));
324 		} else {
325 			// We are expanding away from the start cell so choose the column furthest from the start column
326 			this.changeEndCellWithDomNode(o.getCellElement(newEndRow, (startColumn==expandStartRightColumn?expandStartLeftColumn:expandStartRightColumn), surfacePath));
327 		}
328 	}
329 
330 	this.alignBoxes();
331 }

333 /**
334  * Shrinks the area of the selection down to the location of the mouse.
335  * @private
336  */
337 nitobi.grid.Selection.prototype.shrink = function(evt)
338 {
339 	// This is for Firefox where the selection expand border is a bit bigger than the selection so it can cause 
340 	// shrink to be called instead of expand because the expand border is 1px over the next cell where expand should be fired
341 	if (nitobi.html.Css.hasClass(evt.srcElement, "ntb-grid-selection-border") || nitobi.html.Css.hasClass(evt.srcElement, "ntb-grid-selection-grabby")) return;
342 
343 	//	First make sure that the start and end cell are not the same - ie we have a selection
344 	//	Also check that we are in "selecting" mode
345 	if(this.endCell != this.startCell && this.selecting)
346 	{
347 		var scrollSurface = this.owner.getScrollSurface();
348 
349 		var Cell = nitobi.grid.Cell;
350 		//	Get the end column and row numbers
351 		var endRow = Cell.getRowNumber(this.endCell), endColumn = Cell.getColumnNumber(this.endCell);
352 		//	Get the start column and row numbers
353 		var startRow = Cell.getRowNumber(this.startCell), startColumn = Cell.getColumnNumber(this.startCell);
354 
355 		//	Get the event coords that cause this function to be called
356 		var eventCoords = nitobi.html.getEventCoords(evt);
357 		var evtY = eventCoords.y, evtX = eventCoords.x;
358 		if (nitobi.browser.IE || document.compatMode == "BackCompat") {
359 			// TODO: How did this get past us for so long? I guess we never really scrolled in IE?
360 			evtY = evt.clientY, evtX = evt.clientX;
361 		}
362 
363 		// Get the bounding rect of the endCell in our selection
364 		var endCellRect = this.endCell.getBoundingClientRect(scrollSurface.scrollTop+document.body.scrollTop, scrollSurface.scrollLeft+document.body.scrollLeft);
365 		var endCellRectTop = endCellRect.top, endCellRectLeft = endCellRect.left;
366 
367 		//	Check if we are shrinking from the bottom up
368 		if (endRow > startRow && evtY < endCellRectTop)
369 			// The 4 here is a hack for the selection expansion when we are expanding up.
370 			endRow = endRow - Math.floor(((endCellRectTop-4) - evtY)/this.rowHeight)-1;
371 		//	of if we are shrinking from the top down
372 		else if (evtY > endCellRect.bottom)
373 			endRow = endRow + Math.floor((evtY - endCellRectTop)/this.rowHeight);
374 
375 		//	Check if we are shrinking from right to left
376 		if (endColumn > startColumn && evtX < endCellRectLeft)
377 			endColumn --;
378 		//	or if we are shrinking from left to right
379 		else if (evtX > endCellRect.right)
380 			endColumn ++;
381 
382 		// At this point we have got the new end and start cells from the normal selection process.
383 
384 		if (this.expanding) {
385 			// If the coords of the event are inside the original selection just forget about it...
386 
387 			// This is the top left row and column of the expansion selection
388 			var expandStartCellRow = this.expandStartCell.getRow(), expandStartCellColumn = this.expandStartCell.getColumn();
389 			// This is the bottom right row and column of the expansion selection
390 			var expandEndCellRow = this.expandEndCell.getRow(), expandEndCellColumn = this.expandEndCell.getColumn(); 
391 
392 			if (endColumn >= this.expandStartLeftColumn && endColumn <= this.expandStartRightColumn) {
393 				// The endColumn is inside of the original selection so we need to limit the selection to at least that width
394 
395 				if (endColumn >= startColumn && endColumn < expandEndCellColumn) {
396 					// When we have selected from left to right (ie startColum < endColumn) and then contract back into the 
397 					// selection, force the selection width to the original selection
398 					endColumn = expandEndCellColumn;
399 				} else if (endColumn <= startColumn && endColumn > expandStartCellColumn) {
400 					// When we have selected from right to left (ie endColumn < startColum) and then contract back into the 
401 					// selection, force the selection width to the original selection
402 					endColumn = expandStartCellColumn;
403 				}
404 
405 				// When the end column is moving back past the start of the original selection we can change the end Column
406 				
407 				if (endColumn >= startColumn && endColumn <= this.expandStartRightColumn) {
408 					// The we are shrinking from the left to right
409 					endColumn = this.expandStartRightColumn;
410 				}
411 			}
412 
413 			if (endRow >= this.expandStartTopRow && endRow <= this.expandStartBottomRow) {
414 				// The endRow is inside of the original selection so we need to limit the selection to at least that height
415 
416 				// TODO: There are some edge cases here with the <= >= cases ... 
417 				if (startRow < endRow && endRow <= expandEndCellRow) {
418 					// When we have selected from top to bottom (ie startRow < endRow) and then contract back into the 
419 					// selection, force the selection height to the original selection
420 					endRow = expandEndCellRow;
421 				} else if (startRow > endRow && endRow >= expandStartCellRow) {
422 					// When we have selected from bottom to top (ie endRow < startRow) and then contract back into the 
423 					// selection, force the selection height to the original selection
424 					endRow = expandStartCellRow;
425 				} else if (startRow == endRow) {
426 					// We have to either set the endRow to the original top or bottom row
427 					endRow = (startRow == expandStartCellRow?expandEndCellRow:expandStartCellRow)					 
428 				}
429 			}
430 		}
431 
432 		var surfacePath = nitobi.grid.Cell.getSurfacePath(this.endCell);
433 		var newEndCell = this.owner.getCellElement(endRow, endColumn, surfacePath);
434 		var newStartCell = this.owner.getCellElement(startRow, startColumn, surfacePath);
435 
436 		if (newEndCell != null &&newEndCell != this.endCell || newStartCell != null && newStartCell != this.startCell)
437 		{
438 			this.changeEndCellWithDomNode(newEndCell);
439 			this.changeStartCellWithDomNode(newStartCell);
440 			//	Align the selection boxes to the new selection start and end points
441 			this.alignBoxes();
442 			//	TODO: need to take into account the reduction in size of the selection
443 			//	and scroll the grid over the left so that we can keep scrolling over
444 			this.owner.ensureCellInView(newEndCell);
445 		}
446 	}
447 };

459 /**
460  * Collapses the selection down to one cell 
461  * @param cell {HtmlDomNode} (Optional) The HTML DOM node of the cell to collapse to.  If omitted, the initial start cell of the selection is used.
462  */
463 nitobi.grid.Selection.prototype.collapse = function(cell)
464 {
465  	if (!cell)
466  	{
467  		cell = this.startCell;
468  	}
469  	if (!cell)
470  	{
471  		return;
472  	}
473 
474 	var surfacePath = nitobi.grid.Cell.getSurfacePath(cell);
475 	this.setRangeWithDomNodes(cell, cell, surfacePath);
476 
477 	// TODO: this needs to be cleaned up ...
478 	if ((this.box==null) || (this.box.parentNode==null) || (this.boxl==null) || (this.boxl.parentNode==null))
479 	{
480 		this.created=false;
481 		this.createBoxes();
482 	}
483 
484 	this.alignBoxes();
485 
486 	this.selecting = false;
487 };

530 /**
531  * @private
532  */
533 nitobi.grid.Selection.prototype.alignBoxes = function()
534 {
535  	var endCell = this.endCell || this.startCell;
536  	var sc = this.getCoords();
537  	var topRow = sc.top.y;
538  	var topCol = sc.top.x;
539   	var bottomRow = sc.bottom.y;
540  	var bottomCol = sc.bottom.x;
541 	var standards = nitobi.lang.isStandards();
542 	// This is take into account the border width in Standards mode
543 	// Mozilla is nice cause the tables just align perfectly since they are magic but the tables in IE cause other problems...
544  	// TODO: check on different style grids since it used to say IE && standards on the line below ...
545  	var ox = oy = (nitobi.browser.IE?-1:0);
546  	var ow = oh = (nitobi.browser.IE && standards?-1:1);
547 	// TODO: these numbers depend on the border etc of the selection i think ...
548  	if (nitobi.browser.SAFARI) {
549 		oy = ox = -1;
550 		if (standards)
551 			oh = ow = -1;
552 	}
553  	// IE6
554  		// Standards
555  			// ow = -1;
556  			// oh = -1;
557  	// IE6
558  		// Quirks
559  			// perfect
560  	// IE7
561  		// Standards
562  			// perfect
563  	// IE7
564  		// quirks
565  			// perfect
566  	// FF
567  		// Standards
568  			// perfect - though the edit box is a bit weird
569  	// FF
570  		// Quirks
571  			// perfect
572 
573 	if (bottomCol >= this.freezeLeft && bottomRow >= this.freezeTop) {
574 		var e = this.box;
575 		e.style.display="block";
576 		this.align(e,this.startCell,endCell,0x11101000,oh,ow,oy,ox);
577 		// Show the selection expander grabby in the mid center viewport
578 		if (this.dragFillEnabled) (e.rows != null?e.rows[0].cells[0]:e).appendChild(this.boxexpanderGrabby);
579 	} else {
580 		this.box.style.display="none";
581 	}
582 	if (bottomCol < this.freezeLeft || topCol < this.freezeLeft) {
583 		var e = this.boxl;
584 		e.style.display="block";
585 		this.align(e,this.startCell,endCell,0x11101000,oh,ow,oy,ox);
586 		if (this.box.style.display == "none") {
587 			// There is only left cells selected so show the selection expander grabby in the mid left viewport
588 			if (this.dragFillEnabled) (e.rows != null?e.rows[0].cells[0]:e).appendChild(this.boxexpanderGrabby);
589 		}

671  */
672 /**
673  * @private
674  */
675 nitobi.grid.Selection.prototype.handleSelectionClick = function(evt)
676 {
677 	if (!this.selected())
678 	{
679 		if (NTB_SINGLECLICK == null)
680 		{
681 			if (nitobi.browser.IE)
682 				evt = nitobi.lang.copy(evt);
683 			NTB_SINGLECLICK = window.setTimeout(nitobi.lang.close(this, this.edit, [evt]), 150);
684 		}
685 	} else {
686 		// Otherwise collapse and go into selection mode ...
687 		// Get the coords of the click and relate that to an underlying cell ...
688 		this.collapse();
689 		this.owner.focus();
690 	}

695  * @param {Event} evt The browser event object.

699 	{
700 		window.clearTimeout(NTB_SINGLECLICK);
701 		NTB_SINGLECLICK = null;
702 		if (this.owner.handleDblClick(evt))
703 			this.edit(evt);
704 	}
705 	else
706 		this.collapse();
