117 nitobi.form.ListBox.prototype.deactivate = function(ok)
118 {
119 	//	Need to check the blur value since deactivate can be called twice by
120 	//	the onblur handler on the textbox or by hitting enter which then causes the blur to occur ...
121 	if (this.keypress)
122 	{
123 		this.keypress = false;
124 		return;
125 	}
126 	
127 	if (nitobi.form.ListBox.base.deactivate.apply(this, arguments) == false) return;
128 
129 	if (this.onChange.notify(this) == false) return;
130 
131 	var c = this.control;
132 
133 	var text = "", value = "";
134 	if (ok || ok == null) {
135 		//	get the tet and the value that is selected ...
136 		text = c.options[c.selectedIndex].text;
137 		value = c.options[c.selectedIndex].value;
138 	} else {
139 		value = this.cell.getValue();
140 		var len = c.options.length;
141 		for (var i=0; i<len; i++)
142 		{
143 			if (c.options[i].value == value)
144 				text = c.options[i].text;
145 		}
146 	}
147 
148 	this.typedString = null;
149 
150 	return this.afterDeactivate(nitobi.html.encode(text), value);
151 }

156 nitobi.form.ListBox.prototype.handleKey = function(evt)
157 {
158 	var k = evt.keyCode;
159 
160 	this.keypress = false;
161 
162 	var K = nitobi.form.Keys;
163 
164 	switch (k) {
165 		case K.DOWN:
166 			// If we are at the end of the list the keypress down will not fire onchange
167 			if (this.control.selectedIndex < this.control.options.length - 1)
168 				this.keypress = true;
169 			break;
170 		case K.UP:
171 			// If we are at the start of the list the keypress going up will not fire onchange
172 			if (this.control.selectedIndex > 0)
173 				this.keypress = true;
174 			break;
175 		case K.ENTER:
176 		case K.TAB:
177 		case K.ESC:
178 			return nitobi.form.ListBox.base.handleKey.call(this, evt);
179 		default:
180 			// Prevent the editor from bluring in IE in particular
181 			nitobi.html.cancelEvent(evt);
182 			// in general let them enter the value and have the list select it.
183 	        return this.searchComplete(String.fromCharCode(k));
184 	}
185 }

190 nitobi.form.ListBox.prototype.searchComplete = function(keyVal, matchString)
191 {
192     //var keyVal = String.fromCharCode(keyCode); // Convert ASCII Code to a string
193 
194     if( typeof(matchString) != 'undefined' && matchString != '' )
195     {
196         this.typedString = matchString;
197         this.maxLinearSearch = 500;
198     }
199     else
200     {
201         this.typedString = this.typedString + keyVal; // Add to previously typed characters
202     }
203 
204     var c = this.control;
205     var elementCnt  = c.options.length;    // Calculate length of array -1
206 
207     if ( elementCnt > this.maxLinearSearch ) //make sure it's worthwhile doing a binary search
208     {
209         var result = this.searchBinary(this.typedString, 0, (elementCnt -1));
210         if ( result )
211         {
212             //once match found, search backawards to first match in list
213             for ( i = result; i > 0; i-- )
214             {
215                 if (c.options[i].text.toLowerCase().substr(0,this.typedString.length) != this.typedString.toLowerCase())
216                 {
217                     c.selectedIndex = i+1; // Make the relevant OPTION selected
218                     break; //and STOP!
219                 }
220             }
221         }
222     }
223     else //it's not worthwhile doing a binary search because the number of items is too small, so search linearly
224     {
225         for (i = 1; i < elementCnt; i++)
226         {
227             if (c.options[i].text.toLowerCase().substr(0,this.typedString.length) == this.typedString.toLowerCase())
228             {
229                 c.selectedIndex = i;
230                 break;
231     }
232         }
233     }
234 
235     clearTimeout(this.timerid); // Clear the timeout
236 
237     var _this = this;
238     this.timerid = setTimeout(function(){_this.typedString = "";},1000); // Set a new timeout to reset the key press string
239 
240     return false; // to prevent IE from doing its own highlight switching
241 }

243 /**
244  * @ignore
245  */
246 nitobi.form.ListBox.prototype.searchBinary = function(matchString, low, high)
247 {
248     /* Termination check */
249     if (low > high)
250     {
251         return null;
252     }
253 
254     var c = this.control;
255 
256     var mid = Math.floor((high+low)/2);
257     var selectText = c.options[mid].text.toLowerCase().substr(0,matchString.length);
258     var matchText = matchString.toLowerCase();
259 
260     if ( matchText == selectText )
261     {
262         return mid;
263     }
264     else if ( matchText < selectText )
265     {
266         return this.searchBinary(matchString, low, (mid -1));
267     }
268     else if ( matchText > selectText )
269     {
270         return this.searchBinary(matchString, (mid + 1), high);
271     }
272     else
273     {
274         return null;
275     }
276 
277 }
