 10 /**
 11  * Supports adding more complex list functions to a class. 
 12  * @class An interface that provides for advanced array/list functions.
 13  * @constructor
 14  */
 15 nitobi.collections.IEnumerable = function() 
 16 {
 17 	/**
 18 		@ignore
 19 	*/
 20 	this.list = new Array();
 21 	
 22 	/**
 23 		@ignore
 24 	*/
 25 	this.length=0;
 26 }

 28 /**
 29 	Adds a item to the end of the list.
 30 	@param {Object} obj An object or primitive to add to the list.
 31  */
 32 nitobi.collections.IEnumerable.prototype.add = function(obj)
 33 {
 34 	this.list[this.getLength()] = obj;
 35 	this.length++
 36 }
 37 
 38 /**
 39  * Inserts an item at the specified index.  The object at that index and subsequent objects
 40  * will be moved one place to the right (one index higher).
 41  * @param {Number} index the index of the object once it is inserted
 42  * @param {Object} obj an object or primitive to insert in the list
 43  */
 44 nitobi.collections.IEnumerable.prototype.insert = function(index, obj)
 45 {
 46 	this.list.splice(index, 0, obj);
 47 	this.length++;
 48 };

 50 /**
 51  * Converts something that looks like a javascript Array into a javascript Array.
 52  * @type Array
 53  * @param {Object} obj An objects that supports [].
 54  * @param {Number} start Where to start the copy.
 55  */
 56 nitobi.collections.IEnumerable.createNewArray = function(obj, start)
 57 {
 58 	var length;
 59 	start = start || 0;
 60 	if (obj.count)
 61 	{
 62 		length = obj.count;
 63 	}
 64 	if (obj.length)
 65 	{
 66 		length = obj.length;
 67 	}
 68 	var x = new Array(length-start);
 69 	for (var i = start; i < length; i++)
 70 	{
 71 		x[i - start] = obj[i];
 72 	}
 73 	return x;
 74 }

 76 /**
 77 	Returns an item from the list.
 78 	@param {Number} index The index of the item you want. Indexed from zero, and throws an error if the index is out of bounds.
 79 	@returns {Object} The object you wanted.
 80  */
 81 nitobi.collections.IEnumerable.prototype.get = function(index)
 82 {
 83 	if (index < 0 || index >= this.getLength())
 84 	{
 85 		nitobi.lang.throwError(nitobi.error.OutOfBounds);
 86 	}
 87 	return this.list[index];
 88 }

 90 /**
 91 	Sets an item in the list.
 92 	@param {Number} index The index for where you want to add the item. Throws an error if out of bounds.
 93 	@param {Object} value The object you want to store.
 94  */
 95 nitobi.collections.IEnumerable.prototype.set = function(index,value)
 96 {
 97 	if (index < 0 || index >= this.getLength())
 98 	{
 99 		nitobi.lang.throwError(nitobi.error.OutOfBounds);
100 	}
101 	this.list[index] = value;
102 }

104 /**
105 	Finds the index of an item in the list.  The first match is returned if 
106 	duplicates exist.
107 	@param {Object} obj The item whose index you want to find.
108 	@returns {Number} The index of the item. -1 if not found.
109 */
110 nitobi.collections.IEnumerable.prototype.indexOf = function(obj)
111 {
112 	for (var i = 0; i < this.getLength(); i++)
113 	{
114 		if (this.list[i] === obj)
115 		{
116 			return i;
117 		}
118 	}
119 	return -1;
120 }

122 /**
123 	Removes an item from the list. An error is thrown if the item does not exists
124 	or the index is out of bounds.
125 	@param {Number/Object} value An index or object you want removed from the list.
126 */
127 nitobi.collections.IEnumerable.prototype.remove = function(value)
128 {
129 	var i;
130 	if (typeof(value) != "number")
131 	{
132 		i = this.indexOf(value);
133 	}
134 	else
135 	{
136 		i = value;
137 	}
138 	if (-1 == i || i < 0 || i >= this.getLength())
139 	{
140 		nitobi.lang.throwError(nitobi.error.OutOfBounds);
141 	}
142 	this.list[i] = null;
143 	this.list.splice(i,1);
144 	this.length--;
145 }

147 /**
148 	Returns the number of items in the list.
149 	@returns {Number} The number of items.
150 */
151 nitobi.collections.IEnumerable.prototype.getLength = function()
152 {
153 	return this.length;
154 }

156 /**
157  * Applies the function <CODE>func</CODE> to each element in the collection, in order.
158  * @param {Function} func A function that will be evaluated on each element in the collection.
159  */
160 nitobi.collections.IEnumerable.prototype.each = function(func)
161 {
162 	var l = this.length;
163 	var list = this.list;
164 	for (var i = 0; i < l; i++)
165 	{
166 		func(list[i]);
167 	}
168 };
