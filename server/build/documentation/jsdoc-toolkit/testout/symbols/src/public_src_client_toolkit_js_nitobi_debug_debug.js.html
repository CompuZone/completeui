213  * @ignore
214  * @private
215  */
216 nitobi.debug.errorReport = function(errorMessage, errorCode, errorSeverity) 
217 {
218 	// if empty set to debug
219 	errorSeverity = (errorSeverity)?errorSeverity:NTB_DEBUG;
220 
221 	if(NTB_DEBUG == errorSeverity && !NTB_DEBUG_MODE)
222 	{
223 		// maybe set a lastdebug anyway
224 		return;
225 	}	
226 
227 	 var errorString  =   errorMessage          +
228 	                      "\nerror code    ["  +
229 	                      errorCode             +
230 	                      "]\nerror Severity["  +
231 	                      errorSeverity         +
232 	                      "]";
233 
234 	LastError = errorString;
235 	
236 	
237 	//this should be made into an object with each spreadsheet creating an instance of error reporter
238 	if(eval(NTB_ON_ERROR || "true"))
239 	{	
240 		switch(errorCode)
241 		{		
242 			case NTB_EM_ATTRIBUTE_ERROR:
243 				confirm(errorMessage);
244 			break;
245 			case NTB_XHR_RESPONSE_ERROR:
246 				confirm(errorMessage);
247 			break;
248 			default:
249 				window.status = errorMessage;	
250 			break;
251 		}		
252 	}
253 	if(NTB_THROW == errorSeverity)
254 	{
255 		throw(errorString);
256 	}
257 }
