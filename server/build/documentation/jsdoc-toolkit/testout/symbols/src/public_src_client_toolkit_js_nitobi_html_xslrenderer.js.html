  3 /**
  4  * Creates an XslRenderer with an optional template.
  5  * @class XslRenderer
  6  * Renders before, after, or to the contents of a given dom node.  The renderer maintains an 
  7  * XSLProcessor for a template and is provided with an XMLDocument when rendering.
  8  * @constructor 
  9  * @param {XSLProcessor|XMLDocument|String} template an XSL Processor to use as the template 
 10  */
 11 nitobi.html.XslRenderer = function(template)
 12 {
 13 	nitobi.html.IRenderer.call(this, template);
 14 };
 15 
 16 nitobi.lang.implement(nitobi.html.XslRenderer, nitobi.html.IRenderer);
 17 
 18 
 19 /**
 20  * Sets the renderer's template.
 21  * @param {XSLProcessor|XMLDocument|String} template an XSL Processor to use as the template 
 22  */
 23 nitobi.html.XslRenderer.prototype.setTemplate = function(template)
 24 {
 25 	if (typeof(template) === 'string')
 26 	{
 27 		template = nitobi.xml.createXslProcessor(template);
 28 	}
 29 	this.template = template;
 30 };
 31 
 32 /**
 33  * Transform the given data with the template.
 34  * @param {XMLDocument|String} data the data to transform.
 35  * @return The result of the transformation as a string.
 36  * @type String
 37  */
 38 nitobi.html.XslRenderer.prototype.renderToString = function(data)
 39 {
 40 	if (typeof(data) === 'string')
 41 	{
 42 		data = nitobi.xml.createXmlDoc(data);
 43 	}
 44 	if (nitobi.lang.typeOf(data) === nitobi.lang.type.XMLNODE)
 45 	{
 46 		data = nitobi.xml.createXmlDoc(nitobi.xml.serialize(data));
 47 	}
 48 	var template = this.getTemplate();
 49 	var params = this.getParameters();
 50 	for (var p in params)
 51 	{
 52 		template.addParameter(p, params[p], '');
 53 	}
 54 	var s = nitobi.xml.transformToString(data,template,'xml');
 55 	for (var p in params)
 56 	{
 57 		template.addParameter(p, '', '');
 58 	}
 59 	return s;
 60 		
 61 };
