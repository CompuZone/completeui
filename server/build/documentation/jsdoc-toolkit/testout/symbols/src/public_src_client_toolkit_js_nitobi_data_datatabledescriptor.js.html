1 nitobi.lang.defineNs("nitobi.data");
  2 /**
  3  * This class provides information about the size and nature of the 
  4  * datasource. It does not provide any data.  It's single purpose is
  5  * to perform reflection on the table datasource.
  6  * @param {Boolean} estimateRowCount If true, the descriptor will try to leap ahead to find the end of the table.
  7  */
  8 nitobi.data.DataTableDescriptor = function(table, tableProjectionUpdatedEvent, estimateRowCount)
  9 {
 10 	/**
 11 	 * Items that are destroyed when this object is destroyed.
 12 	 * @private
 13 	 */
 14 	this.disposal = [];
 15 	
 16 	/**
 17 	 * The number of rows thought to be in the database. 
 18 	 * This is equal to, or larger than, the last known row.
 19 	 * @private
 20 	 */
 21 	this.estimatedRowCount=0;
 22 	/**
 23 	 * The amount by which the projected database size is adjusted.
 24 	 * @private
 25 	 */
 26 	this.leapMultiplier=2;
 27 	/**
 28 	 * If true, the descriptor will try to leap ahead to find the end of the table.
 29 	 */
 30 	this.estimateRowCount = (estimateRowCount == null ? true : estimateRowCount);
 31 	/**
 32 	 * The index of the greatest last known row in the db.
 33 	 * @private
 34 	 */
 35 	this.lastKnownRow=0;
 36 	/**
 37 	 * Indicates whether or not the end of table is known.
 38 	 * @private
 39 	 */
 40 	this.isAtEndOfTable = false;
 41 	
 42 	/**
 43 	 * The table that this descriptor describes.
 44 	 * @private
 45 	 */
 46 	this.table = table;
 47 	
 48 	/**
 49 	 * The Edge of (un)known space. This will be set if you do a leap and no results are returned
 50 	 * then the lowestEmptyRow becomes the lowest index of that get request.
 51 	 * @private
 52 	 */
 53 	this.lowestEmptyRow = 0;
 54 	
 55 	/**
 56 	 * Fires when the descriptor finds new information
 57 	 * about the datasource and updates its knowledge.
 58 	 */ /* How does closelater and context work here?*/
 59 	this.tableProjectionUpdatedEvent = tableProjectionUpdatedEvent;
 60 	this.disposal.push(this.tableProjectionUpdatedEvent);
 61 }

 68 /**
 69  * Starts the peek process whereby the descriptor asks for a single
 70  * record farther into the table in an attempt to find the end of the
 71  * table.
 72  * Peek actually uses DataTable.get to do its thing and so update
 73  * on the Descriptor is called from DataTable which will stop the peek
 74  * action (see update logic).
 75  */
 76 nitobi.data.DataTableDescriptor.prototype.startPeek = function()
 77 {
 78 	this.enablePeek = true;
 79 	this.peek();
 80 }

 82 nitobi.data.DataTableDescriptor.prototype.peek = function()
 83 {
 84 	var indexToPeekAt;
 85 	if (this.lowestEmptyRow > 0)
 86 	{
 87 		var blankRows = this.lowestEmptyRow - this.lastKnownRow;
 88 		// Peek at the halfway point between the lastKnownRow and the last row we peeked at unsuccessfully
 89 		indexToPeekAt = this.lastKnownRow + Math.round(blankRows/2);
 90 	}
 91 	else
 92 	{
 93 		// If this is the first time we are peeking (ie lowestEmptyRow is 0)
 94 		// then increase the peek position to some estimated amount
 95 		indexToPeekAt = (this.estimatedRowCount * this.leapMultiplier);
 96 	}
 97 	// Do a get through the datatable at the peek index for a single row
 98 	this.table.get(Math.round(indexToPeekAt), 1, this, this.peekComplete);
 99 	//alert(this.startPeek);
100

109 	{

123 nitobi.data.DataTableDescriptor.prototype.peekComplete = function(evtArgs)
124 {
125 	if (this.enablePeek)
126 	{
127 		window.setTimeout(nitobi.lang.close(this,this.peek),1000);
128 	}
129 }

142 /**
143  * Recomputes the estimated row count based on a multiplier and an offset.
144  * @param {Number} multiplier A number by which to multiply the estimated row count.
145  * @param {Number} adjustment A number by which to increase the estimated row count.
146  */
147 nitobi.data.DataTableDescriptor.prototype.leap = function(multiplier,adjustment)
148 {
149 	if (this.lowestEmptyRow > 0)
150 	{
151 		var blankRows = this.lowestEmptyRow - this.lastKnownRow;
152 		this.estimatedRowCount = this.lastKnownRow + Math.round(blankRows/2);
153 	}
154 	else if (multiplier == null || adjustment == null)
155 	{
156 		this.estimatedRowCount = 0;
157 	}
158 	else if (this.estimateRowCount)
159 	{
160 		this.estimatedRowCount = (this.estimatedRowCount * multiplier) + adjustment;
161 	}
162 
163 //	window.status+="["+this.estimatedRowCount+"]";
164 //	alert("LEAP ERC: " + this.estimatedRowCount);
165 	this.fireProjectionUpdatedEvent();	
166 }

168 /**
169  * Updates the descriptor's knowledge about the table.
170  * @param {nitobi.components.grid.OnGetCompleteEventArgs} evtArgs A hash consisting of a lastPage boolean, pageSize, firstRow, lastRow, and startXi.
171  * @param {Boolean} needLeap A boolean that can be used to force the estimated row count to be adjusted.
172  */
173 
174 nitobi.data.DataTableDescriptor.prototype.update = function(evtArgs, needLeap)
175 {
176 	// There a few cases we can have here.
177 	// We need to check the eventArgs for the lastPage parameter and the numRowsReturned
178 
179 //alert("update"+this.estimatedRowCount+" : "+evtArgs.pageSize)
180 	if (null == needLeap)
181 	{
182 		needLeap = false;
183 	}
184 	if (this.isAtEndOfTable && !needLeap)
185 	{
186 		return false;
187 	}
188 
189 	// Check if the database is empty.
190 	var emptyDb = (evtArgs!=null && evtArgs.numRowsReturned == 0 && evtArgs.startXi == 0);
191 
192 	// Check if the data returned doesnt match what we wanted - ie we got to the end of the data.
193 	// What about the case where the lastRow is == lastRowReturned and this IS the last row of data ... 
194 	var lastPage = (evtArgs!=null && evtArgs.lastRow != evtArgs.lastRowReturned); // || !this.estimateRowCount;
195 
196 	if (null == evtArgs)
197 	{
198 		evtArgs = {lastPage:false,pageSize:1,firstRow:0, lastRow:0,startXi:0};
199 	}
200 
201 	var foundLastPage = (emptyDb) || (lastPage) || (this.isAtEndOfTable) || ((this.lastKnownRow == this.estimatedRowCount -1) && (this.estimatedRowCount == this.lowestEmptyRow));
202 
203 //	alert(this.lastKnownRow+":"+this.lowestEmptyRow+" - "+evtArgs.pageSize+" - "+foundLastPage+"PgSize:"+evtArgs.pageSize)
204   	if (evtArgs.pageSize == 0 && !foundLastPage)
205   	{
206   		// We've let the user page into a completely empty area and 
207   		// we don't know the last page.
208 		// Set the highest row we will ever try to get
209 		this.lowestEmptyRow = this.lowestEmptyRow > 0 ? Math.min(evtArgs.startXi, this.lowestEmptyRow) : evtArgs.startXi;
210 //		alert("LOWEST EMPTY ROW: " + this.lowestEmptyRow);
211 
212 		this.leap();
213 
214 /*		// Step back a small amount to see the page just loaded.
215 		this.leap(1,-1*blankRows/2);
216 		// From now on, we'll need to start leaping smaller so that
217 		// we don't expand the whitespace.
218   		this.leapMultiplier = 1 + ((this.leapMultiplier - 1 )/2);*/
219   		return true;
220   	}
221 
222 	// Does this codeneed to be moved here?
223 	/*if (!scrollAtTop && evtArgs.pageSize == 0 && this.owner.getPagingMode().toLowerCase() != "standard")
224 	{
225 		return
226 	}*/
227 //	this.lastKnownRow = Math.max(evtArgs.lastRow,this.lastKnownRow);
228 	this.lastKnownRow = Math.max(evtArgs.lastRowReturned, this.lastKnownRow);
229 //  	alert(("UPDATE is setting the last known row to: "+this.lastKnownRow))
230 
231   	if (foundLastPage && !needLeap)
232 	{
233 		if (evtArgs.lastRowReturned>=0) // ie we know the exact last row (we are not in no-mans-land)
234 		{
235 			this.estimatedRowCount = evtArgs.lastRowReturned+1;//evtArgs.startXi + evtArgs.pageSize;
236 
237 			 // This is the first time that we've hit the end of the table.
238 			this.isAtEndOfTable = true;
239 		}
240 		else // ie we have either got no data or we way past the end of our data
241 		{
242 			if (emptyDb) // this.lastKnownRow < 0- ie we have never had nor will get any data
243 			{
244 				this.estimatedRowCount = 0;
245 
246 				 // This is the first time that we've hit the end of the table.
247 				this.isAtEndOfTable = true;
248 			}
249 			else // ie we are in no-mans-land so back off a bit
250 			{
251 				this.estimatedRowCount = this.lastKnownRow + Math.ceil((evtArgs.lastRow-this.lastKnownRow)/2);
252 			}
253 		}
254 
255 		// TODO: this needs to be merged with the setRowCountKnown stuff in DataTable.
256 		this.fireProjectionUpdatedEvent();
257 		this.stopPeek();
258 		return true;
259 	}
260 
261 	if (!this.estimateRowCount)
262 	{
263 		this.estimatedRowCount = this.lastKnownRow+1;
264 	}
265 	if (this.estimatedRowCount==0)
266 	{
267 		// First page we've ever seen.
268 		// Estimate the remote row count to be the lastRow we received 
269 		// (plus 1 cause rows are zero based and rowCounts are 1 based) and double it.
270 		this.estimatedRowCount = (evtArgs.lastRow+1) * (this.estimateRowCount ? 2 : 1);
271 	}
272 
273 	// Uses evtArgs.last+1 since it is a row index compared to a rowCount
274   	if ((this.estimatedRowCount > (evtArgs.lastRow+1) && !needLeap) || !this.estimateRowCount)
275   	{
276   		// We got a page for which there is space.
277 
278   		return false;
279   	}
280 	if (!this.isAtEndOfTable)
281 	{
282 		this.leap(this.leapMultiplier,0);
283 		return true;
284 	}
285 	return false;
286 }

292 /**
293  * Resets the descriptor back to its initial state. In this state,
294  * the descriptor knows nothing about the table.
295  */
296 nitobi.data.DataTableDescriptor.prototype.reset = function()
297 {
298 	this.estimatedRowCount=0;
299 	this.leapMultiplier=2;
300 	this.lastKnownRow=0;
301 	this.isAtEndOfTable = false;
302 	this.lowestEmptyRow = 0;
303 	this.fireProjectionUpdatedEvent();
304 }
305 
306 nitobi.data.DataTableDescriptor.prototype.fireProjectionUpdatedEvent = function(evtArgs)
307 {
308 	if (this.tableProjectionUpdatedEvent != null)
309 	{
310 		this.tableProjectionUpdatedEvent(evtArgs);
311 	} 
312 }
