 10 /**
 11  * Sets the parameter key to value in the given url, and returns the new url.
 12  * @param {String} url the URL to modify, can be relative or absolute
 13  * @param {String} key the parameter name
 14  * @param {String} value the new parameter value 
 15  * @type String
 16  */
 17 nitobi.html.Url.setParameter = function(url, key, value)
 18 {
 19 	var reg = new RegExp('(\\?|&)('+encodeURIComponent(key)+')=(.*?)(&|$)');
 20 	if (url.match(reg))
 21 	{
 22 		return url.replace(reg, '$1$2='+encodeURIComponent(value)+'$4');		
 23 	}
 24 	if (url.match(/\?/))
 25 	{
 26 		url = url + '&';
 27 	}
 28 	else
 29 	{
 30 		url = url + '?';
 31 	}
 32 	return url + encodeURIComponent(key) + '=' + encodeURIComponent(value);
 33 };

 35 /**
 36  * Removes the parameter key from url's parameter list and returns the new URL.
 37  * @param {String} url the URL to modify
 38  * @param {String} key the parameter to remove
 39  * @type String
 40  */
 41 nitobi.html.Url.removeParameter = function(url, key)
 42 {
 43 	var reg = new RegExp('(\\?|&)('+encodeURIComponent(key)+')=(.*?)(&|$)');
 44 	return url.replace(reg, 
 45 		function(str,p1,p2,p3,p4,offset,s)
 46 		{
 47 			if (((p1) == '?') && (p4 != '&'))
 48 			{
 49 				return "";
 50 			}
 51 			else
 52 			{
 53 				return p1;
 54 			} 
 55 		}
 56 	);
 57 };

 59 /**
 60  * Returns the path of a URL.  The path will have a slash on the end, or will be the empty string.  
 61  * In that way,
 62  * <CODE><PRE> 
 63  * var url = 'http://nitobi.com/index.html'
 64  * normalize(url) === normalize(normalize(url)) === 'http://nitobi.com/';
 65  * </PRE></CODE>.
 66  * @private
 67  * @param {String} url The URL to normalize.
 68  * @param {String} file An optional file URL to append to the normalized url.  If the file URL
 69  * is absolute, the function returns just the file URL.
 70  * @returns {String} The normalized url.
 71  */
 72 nitobi.html.Url.normalize = function(url, file)
 73 {
 74 	if (file) {
 75 		if (file.indexOf('http://') == 0 || file.indexOf('https://') == 0 || file.indexOf('/') == 0) {
 76 			return file;
 77 		}
 78 	}
 79 	var href = (url.match(/.*\//) || "") + "";
 80 	if (file)
 81 	{
 82 		return href + file;
 83 	}
 84 	return href;
 85 }

 87 /**
 88  * Returns the url with a random parameter added.  This is useful for preventing 
 89  * a URL from being saved in browser cache.
 90  * @param url the URL to modify
 91  * @type String
 92  */
 93 nitobi.html.Url.randomize = function(url)
 94 {
 95 	return nitobi.html.Url.setParameter(url, 'ntb-random',(new Date).getTime());
 96 };
