1 /**
  2  * Creates a new Cell object
  3  * @class The Cell object represents a single row/column combination in the Grid. 
  4  * It can be used to get or set values of cells in the Grid, or to manipulate
  5  * the look of a particular cell.
  6  * <p>
  7  * A cell is represented in two ways:  as a rendered html element and as
  8  * an xml node.  The Cell object provides a means to affect either
  9  * representation
 10  * </p>
 11  * @constructor
 12  * @param {nitobi.grid.Grid} grid
 13  * @param {Object} row This argument can either be a row index or a HtmlElement. 
 14  * If the row index is used the third argument is also required.
 15  * @param {Number} column Index of the column of the Cell. This is optional, however, it is 
 16  * required if the row argument is specified as a row index and not a Cell HTML element.
 17  * @see nitobi.grid.Grid#getCellObject
 18  */
 19 nitobi.grid.Cell = function(grid, row, column)
 20 {
 21 	// Row should not be null
 22 	if (row == null || grid == null)
 23 		return null;
 24 
 25 	/**
 26 	 * @private
 27 	 */
 28 	this.grid=grid;
 29 
 30 	/**
 31 	 * @private
 32 	 */
 33 	var DomNode = null;
 34 	if (typeof(row) == "object")
 35 	{
 36 		var cell = row;
 37 		row = Number(cell.getAttribute('xi'));
 38 		column = cell.getAttribute('col');
 39 		DomNode = cell;
 40 	}
 41 	else
 42 	{
 43 		DomNode = this.grid.getCellElement(row, column);
 44 	}
 45 
 46 	/**
 47 	 * @private
 48 	 */
 49 	 this.DomNode = DomNode;
 50 
 51 
 52 	// For backwards compatibility have the lower case but they should be upper for getRow etc...
 53 	/**
 54 	 * @private
 55 	 */
 56 	this.row=Number(row);
 57 	/**
 58 	 * @private
 59 	 */
 60 	this.Row=this.row;
 61 
 62 	/**
 63 	 * @private
 64 	 */
 65 	this.column=Number(column);
 66 	/**
 67 	 * @private
 68 	 */
 69 	this.Column=this.column;
 70 
 71 	/**
 72 	 * @private
 73 	 */
 74 	this.dataIndex = this.Row;
 75 }

148 /**
149  * This sets the value of a cell which results in the value being set in 
150  * the datasource (its xml node) and the view of the data being updated
151  * (the html rendered for the cell).  If the cell 
152  * belongs to a number or date column, for example, the value will have 
153  * the appropriate mask applied to it.
154  * <p>
155  * <b>Example</b><br>
156  * <pre><code class="ntb javascript">
157  * function setCellValue(row, col, value)
158  * {
159  * 	var grid = nitobi.getComponent('grid1');
160  * 	var cell = grid.getCellObject(row, col);
161  * 	// Note the second parameter is optional
162  * 	cell.setValue(value);
163  * }
164  * </code></pre>
165  * </p>
166  * @param {String} value The value which will be put in the database.
167  * @param {String} display The value which will be displayed in the Grid. 
168  * This is optional.
169  */
170 nitobi.grid.Cell.prototype.setValue = function(value, display) 
171 {
172 	if (value == this.getValue())
173 		return;
174 
175 	//	First thing to do is validate the input for the given column type.
176 
177 	var colObj = this.getColumnObject();
178 
179 	//	Second thing to do is to actually format the data for presentation in the HTML
180 
181 	var domValue = '';
182 	switch (colObj.getType())
183 	{
184 		case 'PASSWORD':
185 			for (var i=0; i<value.length; i++)
186 			{
187 				domValue += '*';
188 			}
189 			break;
190 		case 'NUMBER':
191 			if (this.numberXsl == null)
192 				this.numberXsl = nitobi.form.numberXslProc;
193 
194 			if (value == "")
195 				value = colObj.getEditor().defaultValue || 0;
196 
197 			if (this.DomNode != null)
198 			{
199 				if (value < 0)
200 					nitobi.html.Css.addClass(this.DomNode, "ntb-cell-negativenumber");
201 				else
202 					nitobi.html.Css.removeClass(this.DomNode, "ntb-cell-negativenumber");
203 			}
204 
205 			var mask = colObj.getMask();
206 			var nmask = colObj.getNegativeMask();
207 			var maskedValue = value;
208 			if (value < 0 && nmask != "")
209 			{
210 				mask = nmask;
211 				// if there is a negative mask let it handle the negative sign.
212 				maskedValue = (value+"").replace("-","");
213 			}
214 
215 			this.numberXsl.addParameter("number", maskedValue, "");
216 			this.numberXsl.addParameter("mask", mask, "");
217 			this.numberXsl.addParameter("group", colObj.getGroupingSeparator(), "");
218 			this.numberXsl.addParameter("decimal", colObj.getDecimalSeparator(), "");
219 
220 			// Now reformat the number according to the mask and add the mask signs again.
221 			domValue = nitobi.xml.transformToString(nitobi.xml.Empty, this.numberXsl);
222 
223 			// Check if the mask was applied
224 
225 			if("" == domValue && value != "")
226 			{
227 				domValue = nitobi.html.getFirstChild(this.DomNode).innerHTML;
228 				value = this.getValue();
229 				// ebaErrorReport("[" + this.control.value + "] is not a valid date. Only a valid date such as '2006-12-05' can be entered in this cell.","",EBA_ERROR);
230 			}
231 
232 			break;
233 		case 'DATE':
234 			if (this.dateXsl == null)
235 				this.dateXsl = nitobi.form.dateXslProc.stylesheet;
236 
237 			var d = new Date();
238 			var xmlDoc = nitobi.xml.createXmlDoc('<root><date>'+value+'</date><year>'+(d.getFullYear())+'</year><mask>'+this.columnObject.getMask()+'</mask></root>');
239 
240 			domValue = nitobi.xml.transformToString(xmlDoc, this.dateXsl);
241 
242 			// Check if the mask was applied
243 			if("" == domValue)
244 			{
245 				domValue = nitobi.html.getFirstChild(this.DomNode).innerHTML;
246 				value = this.getValue();
247 				// ebaErrorReport("[" + this.control.value + "] is not a valid date. Only a valid date such as '2006-12-05' can be entered in this cell.","",EBA_ERROR);
248 			}
249 			/* TODO: this was merged from Schenker but is messing things up now...
250 			else
251             {
252                 value = domValue;
253             }
254             */
255 
256 			break;
257 		case 'TEXTAREA':
258 			domValue = nitobi.html.encode(value);
259 			break;
260 		case 'LOOKUP':
261 			var colObjModel = colObj.getModel();
262 			// TODO: these getAttribute calls need to be converted to proper API methods once we separate Editors and Columns
263 			var sDatasourceId = colObjModel.getAttribute('DatasourceId');
264 			var dataTable = this.grid.data.getTable(sDatasourceId);
265 			var displayFields = colObjModel.getAttribute('DisplayFields');
266 			var valueField = colObjModel.getAttribute('ValueField');
267 			var xmlNode = dataTable.xmlDoc.selectSingleNode('//'+nitobi.xml.nsPrefix+'e[@'+valueField+'=\''+value+'\']/@'+displayFields);
268 			if (xmlNode != null)
269 				domValue = xmlNode.nodeValue;
270 			else
271 				domValue = value;
272 			break;
273 		case 'CHECKBOX':
274 			var colObjModel = colObj.getModel();
275 			var sDatasourceId = colObjModel.getAttribute('DatasourceId');
276 			var dataTable = this.grid.data.getTable(sDatasourceId);
277 			var displayFields = colObjModel.getAttribute('DisplayFields');
278 			var valueField = colObjModel.getAttribute('ValueField');
279 			var checkedValue = colObjModel.getAttribute('CheckedValue');
280 			if (checkedValue == '' || checkedValue == null)
281 				checkedValue = 0;
282 
283 			var displayValue = dataTable.xmlDoc.selectSingleNode('//'+nitobi.xml.nsPrefix+'e[@'+valueField+'=\''+value+'\']/@'+displayFields).nodeValue;
284 
285 			var checkstate = (value == checkedValue)?"checked":"unchecked";
286 			domValue = '<div style="overflow:hidden;"><div class="ntb-checkbox ntb-checkbox-'+checkstate+'" checked="'+value+'"> </div><div class="ntb-checkbox-text">'+nitobi.html.encode(displayValue)+'</div></div>';
287 
288 			break;
289 		case 'LISTBOX':
290 			var colObjModel = colObj.getModel();
291 			var sDatasourceId = colObjModel.getAttribute('DatasourceId');
292 			var dataTable = this.grid.data.getTable(sDatasourceId);
293 			var displayFields = colObjModel.getAttribute('DisplayFields');
294 			var valueField = colObjModel.getAttribute('ValueField');
295 			domValue = dataTable.xmlDoc.selectSingleNode('//'+nitobi.xml.nsPrefix+'e[@'+valueField+'=\''+value+'\']/@'+displayFields).nodeValue;
296 			break;
297 		case 'IMAGE':
298 			domValue = nitobi.html.getFirstChild(this.DomNode).innerHTML;
299 			if (nitobi.lang.typeOf(value) == nitobi.lang.type.HTMLNODE)
300 				domValue = '<img border="0" src="'+value.getAttribute('src')+'" />';
301 			else if (typeof(value) == "string")
302 				domValue = '<img border="0" src="'+value+'" />';
303 //			else
304 //				throw "Invalid value for an Image cell. Expected either an <img> element or image src location.";
305 			break;
306 		default:
307 			domValue = value;
308 	}
309 	
310 	domValue = domValue || " ";
311 
312 	if (this.DomNode != null)
313 	{
314 		var elem = nitobi.html.getFirstChild(this.DomNode);
315 		elem.innerHTML = domValue || " "; 
316 		elem.setAttribute("title", value);
317 		this.DomNode.setAttribute("value", value);
318 	}
319 
320 	this.grid.datatable.updateRecord(this.dataIndex, colObj.getColumnName(), value);
321 }

323 /**
324  * Gets the value of the cell from its xml node.  
325  * @return {String}
326  */
327 nitobi.grid.Cell.prototype.getValue = function()
328 {
329 	// TODO: Need to return the correct JS type
330 	var colObj = this.getColumnObject();
331 	var val = this.GETDATA();
332 	switch (colObj.getType())
333 	{
334 		case 'NUMBER':
335 			val = parseFloat(val);
336 			break;
337 //		case 'DATE':
338 //			break;
339 		default:
340 	}
341 
342 	return val;
343 }
