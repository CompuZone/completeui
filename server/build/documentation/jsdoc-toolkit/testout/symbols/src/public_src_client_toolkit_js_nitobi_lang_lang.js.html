 51 /**
 52  * Extends the <code>subClass</code> with the functionality included in the <code>baseClass</code>. 
 53  * This function should be called immediately after the subClass constructor is defined.
 54  * <div class="code">
 55  * <pre><code class="html">
 56  * example.Cat = function() {}
 57  * example.Cat.prototype.run = function() {}
 58  * 
 59  * example.Lion = function() {
 60  *   example.Lion.baseConstructor.call(this);
 61  * }
 62  * 
 63  * nitobi.lang.extend(example.Lion, example.Cat);
 64  * </code></pre>
 65  * </div>
 66  * @param subClass {object} The class that will inherit from the base.
 67  * @param baseClass {object} The class that will be inherited from.
 68  */
 69 nitobi.lang.extend = function(subClass, baseClass) {
 70    function inheritance() {};
 71    inheritance.prototype = baseClass.prototype;
 72 
 73    subClass.prototype = new inheritance();
 74    subClass.prototype.constructor = subClass;
 75    subClass.baseConstructor = baseClass;
 76    if (baseClass.base)
 77    {
 78 	   baseClass.prototype.base = baseClass.base;
 79    }
 80    subClass.base = baseClass.prototype;
 81 }

 83 /**
 84  * Copies all functions from the interface to the class - commonly referred to as a mixin. 
 85  * This is very similar to <code>extend</code>. 
 86  * <div class="code">
 87  * <pre><code class="html">
 88  * example.Cat = function() {}
 89  * example.Cat.prototype.run = function() {}
 90  * 
 91  * example.Lion = function() {
 92  *   example.Lion.baseConstructor.call(this);
 93  * }
 94  * 
 95  * nitobi.lang.implement(example.Lion, example.Cat);
 96  * </code></pre>
 97  * </div>
 98  * @param {Function} class_ the class to copy functions onto
 99  * @param {Function} interface_ the interface to copy from
100  * @see #extend
101  */
102 nitobi.lang.implement = function(class_, interface_)
103 {
104 	for (var member in interface_.prototype)
105 	{
106 		if (typeof(class_.prototype[member]) == "undefined" || class_.prototype[member] == null)
107 		{
108 			class_.prototype[member] = interface_.prototype[member];
109 		}
110 	}
111 }

543  * Returns the base-26 style character string for the given number.  This will be valid up from 'a' to 'zz' or (26^2-1)
544  * @param {Number} num the number to use in the conversion
545  * @type String
546  * @private
547  */
548 nitobi.lang.numToAlpha = function(num)
549 {
550 	if (typeof(nitobi.lang.numAlphaCache[num]) === 'string')
551 	{
552 		return nitobi.lang.numAlphaCache[num];
553 	}
554 	var ck1 = num % 26;
555 	var ck2 = Math.floor(num / 26);
556 	var alpha = (ck2>0?String.fromCharCode(96+ck2):"")+String.fromCharCode(97+ck1);
557 	nitobi.lang.alphaNumCache[alpha] = num;
558 	nitobi.lang.numAlphaCache[num] = alpha;
559 	return alpha;
560 };

563  * Returns the numeric value (base-10) for the given base-26 style character string.
564  * @param {String} alpha the base-26 character string ('a'-'zz')
565  * @type Number
566  * @private
567  */
568 nitobi.lang.alphaToNum = function(alpha)
569 {
570 	if (typeof(nitobi.lang.alphaNumCache[alpha]) === 'number')
571 	{
572 		return nitobi.lang.alphaNumCache[alpha];
573 	}
574 	var j = 0;
575 	var num = 0;
576 	for (var i = alpha.length-1; i >= 0; i--)

578 		num += (alpha.charCodeAt(i)-96) * Math.pow(26,j++);
579 	}
580 	num = num-1;
581 	nitobi.lang.alphaNumCache[alpha] = num;
582 	nitobi.lang.numAlphaCache[num] = alpha;
583 	return num;
584 };
