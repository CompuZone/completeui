  3 /**
  4  * Creates the EventsManager used by the DatePicker to manage all events dates and disabled dates.
  5  * @class The EventsManager allows the DatePicker to easily manage which dates have related event information
  6  * and which dates are disabled.  When the DatePicker is rendered, it uses the EventsManager to issue and XHR to the
  7  * server before rendering.
  8  * @constructor
  9  * @param {String} url The url that supplies the EventsManager with its event and disabled date information.
 10  * @see nitobi.calendar.DatePicker
 11  */
 12 nitobi.calendar.EventsManager = function(url)
 13 {
 14 	this.connector = new nitobi.data.UrlConnector(url);
 15 	this.onDataReady = new nitobi.base.Event();
 16 	
 17 	this.dates = {events: {}, disabled: {}};
 18 	this.eventsCache = {};
 19 	this.disabledCache = {};
 20 }
 21 
 22 /**
 23  * Returns true if the date has event information.
 24  * @param {Date} date The date to check for event information.
 25  */
 26 nitobi.calendar.EventsManager.prototype.isEvent = function(date)
 27 {
 28 	return (this.eventsCache[date.valueOf()]?true:false);
 29 }
 30 
 31 /**
 32  * Returns true if the date is disabled.
 33  * @param {Date} date The date to check if it's disabled.
 34  */
 35 nitobi.calendar.EventsManager.prototype.isDisabled = function(date)
 36 {
 37 	return (this.disabledCache[date.valueOf()]?true:false);
 38 }
 39 
 40 /**
 41  * Issues the XHR to the server.  If the EventsManager isn't connected to a url,
 42  * it simply notifies its onDataReady event.
 43  * @private
 44  */
 45 nitobi.calendar.EventsManager.prototype.getFromServer = function()
 46 {
 47 	if (this.connector.url != null)
 48 		this.connector.get({}, nitobi.lang.close(this, this.getComplete));
 49 	else
 50 		this.onDataReady.notify();	
 51 }
 52 
 53 /**
 54  * Parses the response from the server.
 55  * @private
 56  */
 57 nitobi.calendar.EventsManager.prototype.getComplete = function(eventArgs)
 58 {
 59 	var data = eventArgs.result;
 60 	var dm = nitobi.base.DateMath;
 61 	var root = data.documentElement;
 62 	var dates = nitobi.xml.getChildNodes(root);
 63 	for (var i = 0; i < dates.length; i++)
 64 	{
 65 		var dateNode = dates[i];
 66 		var type = dateNode.getAttribute("e");
 67 		var parsedDate = {};
 68 		if (type == "event")
 69 		{
 70 			var startDate = dateNode.getAttribute("a");
 71 			startDate = dm.parseIso8601(startDate);
 72 			parsedDate.startDate = startDate;
 73 			var endDate = dateNode.getAttribute("b");
 74 			if (endDate)
 75 				endDate = dm.parseIso8601(endDate);
 76 			else
 77 				endDate = null;
 78 			parsedDate.endDate = endDate;
 79 			parsedDate.location = dateNode.getAttribute("c");
 80 			parsedDate.description = dateNode.getAttribute("d");
 81 			var datesObj = this.dates.events[dm.resetTime(dm.clone(startDate)).valueOf()];
 82 			if (datesObj)
 83 			{
 84 				datesObj.push(parsedDate);
 85 			}
 86 			else
 87 			{
 88 				datesObj = [parsedDate];
 89 				this.dates.events[dm.resetTime(dm.clone(startDate)).valueOf()] = datesObj;
 90 			}
 91 			
 92 			this.addEventDate(startDate, endDate);
 93 		}
 94 		else
 95 		{
 96 			var startDate = dm.parseIso8601(dateNode.getAttribute("a"))
 97 			parsedDate.date = startDate;
 98 			this.addDisabledDate(dm.clone(startDate));
 99 		}
100 	}
101 	this.onDataReady.notify();
102 }
103 
104 /**
105  * Adds event information about a date.
106  * @private
107  */
108 nitobi.calendar.EventsManager.prototype.addEventDate = function(start, end)
109 {
110 	var dm = nitobi.base.DateMath;
111 	var startDate = dm.clone(start);
112 	startDate = dm.resetTime(startDate);
113 	if (!end)
114 		return this.eventsCache[startDate.valueOf()] = start;
115 	end = dm.clone(end);
116 	end = dm.resetTime(end);
117 	
118 	while (startDate.valueOf() <= end.valueOf())
119 	{
120 		this.eventsCache[startDate.valueOf()] = start;
121 		startDate = dm.add(startDate, 'd', 1);
122 	}
123 }
124 
125 /**
126  * Adds a date to disable.
127  * @private
128  */
129 nitobi.calendar.EventsManager.prototype.addDisabledDate = function(date)
130 {
131 	date = nitobi.base.DateMath.resetTime(date);
132 	return this.disabledCache[date.valueOf()] = true;
133 }
134 
135 /**
136  * Returns information about an event for some date.
137  * @param {Date} date The date for which we'd like some event information.
138  */
139 nitobi.calendar.EventsManager.prototype.getEventInfo = function(date)
140 {
141 	var dm = nitobi.base.DateMath;
142 	var events = this.dates.events;
143 	return events[date.valueOf()];
