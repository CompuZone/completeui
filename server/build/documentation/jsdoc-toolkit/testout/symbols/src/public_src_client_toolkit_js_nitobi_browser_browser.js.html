 57 /**
 58  * Detects which browser is being used. <BR>
 59  * This function sets up the booleans found above.  It is called automatically when this class is 
 60  * included in your page.
 61  */
 62 nitobi.browser.detect = function ()
 63 {
 64 	var data = [
 65 		{
 66 			string: navigator.vendor,
 67 			subString: "Adobe",
 68 			identity: "AIR"
 69 		},
 70 		{
 71 			string: navigator.vendor,
 72 			subString: "Apple",
 73 			identity: "Safari"
 74 		},
 75 		{
 76 			prop: window.opera,
 77 			identity: "Opera"
 78 		},
 79 		{
 80 			string: navigator.vendor,
 81 			subString: "iCab",
 82 			identity: "iCab"
 83 		},
 84 		{
 85 			string: navigator.vendor,
 86 			subString: "KDE",
 87 			identity: "Konqueror"
 88 		},
 89 		{
 90 			string: navigator.userAgent,
 91 			subString: "Firefox",
 92 			identity: "Firefox"
 93 		},
 94 		{	// for newer Netscapes (6+)
 95 			string: navigator.userAgent,
 96 			subString: "Netscape",
 97 			identity: "Netscape"
 98 		},
 99 		{
100 			string: navigator.userAgent,
101 			subString: "MSIE",
102 			identity: "Explorer",
103 			versionSearch: "MSIE"
104 		},
105 		{
106 			string: navigator.userAgent,
107 			subString: "Gecko",
108 			identity: "Mozilla",
109 			versionSearch: "rv"
110 		},
111 		{ 	// for older Netscapes (4-)
112 			string: navigator.userAgent,
113 			subString: "Mozilla",
114 			identity: "Netscape",
115 			versionSearch: "Mozilla"
116 		},
117 		{	
118 			string: navigator.vendor,
119 			subString: "Camino",
120 			identity: "Camino"
121 		}
122 	];
123 
124 	var browser = "Unknown";
125 	for (var i=0;i<data.length;i++)	{
126 		var dataString = data[i].string;
127 		var dataProp = data[i].prop;
128 		if (dataString) {
129 			if (dataString.indexOf(data[i].subString) != -1)
130 			{
131 				browser = data[i].identity;
132 				break;
133 			}
134 		}
135 		else if (dataProp)
136 		{
137 			browser = data[i].identity;
138 			break;
139 		}
140 	}
141 	nitobi.browser.IE = (browser == "Explorer");
142 	nitobi.browser.IE6 = (nitobi.browser.IE && !window.XMLHttpRequest);
143 	nitobi.browser.IE7 = (nitobi.browser.IE && window.XMLHttpRequest);
144 
145 	nitobi.browser.MOZ = (browser == "Netscape" || browser == "Firefox" || browser == "Camino");
146 	nitobi.browser.SAFARI = (browser == "Safari");
147 	nitobi.browser.OPERA = (browser == "Opera");
148 	nitobi.browser.AIR = (browser == "AIR");
149 
150 if (nitobi.browser.SAFARI)
151 	nitobi.browser.OPERA = true;
152 if (nitobi.browser.AIR)
153 	nitobi.browser.SAFARI = true;
154 
155 	nitobi.browser.XHR_ENABLED = nitobi.browser.OPERA || nitobi.browser.SAFARI || nitobi.browser.MOZ || nitobi.browser.IE;
156 
157 	// This is a pretty liberal usage of the word unknown - we just don't care about other browsers.
158 	nitobi.browser.UNKNOWN = !(nitobi.browser.IE || nitobi.browser.MOZ || nitobi.browser.SAFARI);
159 };

166 	} catch (e)
