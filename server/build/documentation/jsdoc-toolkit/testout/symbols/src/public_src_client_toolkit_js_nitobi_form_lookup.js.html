165 nitobi.form.Lookup.prototype.bindComplete = function(searchString)
166 {
167  	var datasourceXmlDoc = this.dataTable.getXmlDoc(); 
168 
169 	// By now the DisplayFields parameter contains the mapped field names.
170 	var listboxXsl = nitobi.form.listboxXslProc;
171 	listboxXsl.addParameter('DisplayFields', this.displayFields, '');
172 	listboxXsl.addParameter('ValueField', this.valueField, '');
173 	listboxXsl.addParameter('val', nitobi.xml.constructValidXpathQuery(this.cell.getValue(),false), '');
174 
175 	// IE 6 standards mode layout problem - need to specify the size of the select to make it have any height
176 	if (nitobi.browser.IE && document.compatMode == "CSS1Compat")
177 		listboxXsl.addParameter('size', 6, '');
178 
179 	//Workaround for not being able to set the innerHTML of a select element in nitobi.browser.IE ...
180 	//http://support.microsoft.com/default.aspx?scid=kb;en-us;276228
181 
182 	//	TODO: this should not be creating an XML doc from the xml string!!!
183 	this.listXml = nitobi.xml.transformToXml(nitobi.xml.createXmlDoc(datasourceXmlDoc.xml), nitobi.form.listboxXslProc);
184 	this.listXmlLower = nitobi.xml.createXmlDoc(this.listXml.xml.toLowerCase());
185 
186 	// Clear the size attribute so that it doesnt affect the Listbox control
187 	if (nitobi.browser.IE && document.compatMode == "CSS1Compat")
188 		listboxXsl.addParameter('size', '', '');
189 
190 	this.selectPlaceholder.innerHTML = nitobi.xml.serialize(this.listXml);
191 
192 	var tc = this.textControl;
193 	// Setup the select control for the lookup
194 	var sc = this.selectControl = nitobi.html.getFirstChild(this.selectPlaceholder);
195 	sc.setAttribute("id","ntb-lookup-options");
196 	sc.setAttribute("size", this.size);
197 	sc.style.display = "none";
198 	// Bug in IE6 improperly renders the select.
199 	if (nitobi.browser.IE6 && document.compatMode != "CSS1Compat")
200 		sc.style.height = "100%";
201 
202 	// TODO: this is a pain since we re-call the blurable interface and so destroy any state like lastFocus in IBlurable
203 	// Setup the blurable interface to capture blur events on multiple different elements
204 	nitobi.form.IBlurable.call(this, [tc, sc], this.deactivate);
205 
206 	this.selectClicked = false;
207 	this.bVisible = false;
208 	
209 	this.align();
210 
211 	// Now we need to adjust the editor width for padding / borders etc
212 	nitobi.html.fitWidth(this.placeholder, tc);
213 
214 	if( this.autoComplete )
215 	{
216 		var rn = this.search(searchString);
217 	
218 		if (rn > 0)
219 		{
220 			sc.selectedIndex = rn-1;
221 			tc.value = sc[sc.selectedIndex].text;
222 			nitobi.html.highlight(tc, tc.value.length - (tc.value.length - searchString.length));
223 			this.autocompleted = true;
224 		}
225 		// Check to see if there is a selection from the possible rows.
226 		else
227 		{
228 			var row = datasourceXmlDoc.selectSingleNode('//'+nitobi.xml.nsPrefix+'e[@' + this.valueField + '=\'' + searchString + '\']');
229 			if (row != null)
230 			{
231 				//	TODO: displayFields can possibly be | separated values.
232 				tc.value = row.getAttribute(this.displayFields);
233 				var rn = this.search(tc.value);
234 				sc.selectedIndex = parseInt(rn)-1;
235 			}
236 			else
237 			{
238 				// If this is a custom value, just use the custom value.
239 			tc.value = searchString;
240 			sc.selectedIndex=-1;
241 			}
242 		}
243 	}
244     else
245     {
246         // If this is a custom value, just use the custom value.
247         tc.value = searchString;
248         sc.selectedIndex=-1;
249     }	
250 
251 // this is silly - doesnt even work in FF
252 //	this.textControl.className = this.cell.getDomNode().className;
253 
254 	tc.parentNode.style.height = nitobi.html.getHeight(this.cell.getDomNode()) + "px";
255 
256 	sc.style.display = 'inline';
257 
258 	tc.focus();
259 }

284 /**
285  * @ignore
286  */
287 nitobi.form.Lookup.prototype.deactivate = function(evt)
288 {
289 	if (nitobi.form.Lookup.base.deactivate.apply(this, arguments) == false) return;
290 
291 	var sc = this.selectControl;
292 	var tc = this.textControl;
293 
294 	var text = "", value = "";
295 
296 	// If there was a match in the list, then select the match, otherwise 
297 	// the value is a custom value.
298 	if (evt != null && evt != false) 
299 	{
300 		// If an item from the list is selected, pick that one, otherwise pick the custom text.
301 		if (sc.selectedIndex >= 0)
302 		{
303 			value = sc.options[sc.selectedIndex].value;
304 			text = sc.options[sc.selectedIndex].text;
305 		}
306 		else
307 		{
308 			// ForceValidOption determines whether to restrict the cell's value to only items
309 			// retrieved from the server
310 			if (this.column.getModel().getAttribute("ForceValidOption") != "true")
311 			{
312 				value = tc.value;
313 				text = value;
314 			}
315 			else if (this.autoClear)
316 			{
317 				value = "";
318 				text = "";
319 			}
320 			else
321 			{
322 				value = this.cell.getValue();
323 				var len = sc.options.length;
324 				for (var i=0; i<len; i++)
325 				{
326 					if (sc.options[i].value == value)
327 						text = sc.options[i].text;
328 				}
329 			}
330 		}
331 	} else {
332 		value = this.cell.getValue();
333 		var len = sc.options.length;
334         var found = false;
335         for (var i=0; i<len; i++)
336         {
337             if (sc.options[i].value == value)
338             {
339                 text = sc.options[i].text;
340                 found = true;
341                 break;
342             }
343         }
344 
345         if( !found && this.autoClear)
346         {
347             value = '';
348             text = '';
349         }
350 	}
351 
352 	nitobi.html.detachEvents(sc, this.textEvents);
353 
354 	// Clear any pending timeout to send a request to the server
355 	window.clearTimeout(this.timeoutId);
356 
357 	return this.afterDeactivate(nitobi.html.encode(text), value);
358 }

360 /**
361  * HandleKey deals with keys that we want to mask out. For example, esc and enter are masked to 
362  * cause the lookup to deactivate.
363  * @param {nitobi.html.Event} evt The event object.
364  * @param {HtmlElement} element The HTML element that the event handler was registered on.
365  */
366 nitobi.form.Lookup.prototype.handleKey = function(evt, element)
367 {
368 	
369 	var k = evt.keyCode;
370 	// If the user pressed up or down, don't blur the select
371 	if (k != 40 && k != 38)
372 		nitobi.form.Lookup.base.handleKey.call(this, evt);
373 }

378 nitobi.form.Lookup.prototype.search = function(searchValue)
379 {
380 	searchValue = nitobi.xml.constructValidXpathQuery(searchValue,false);
381 
382 	this.searchXslProc.addParameter('searchValue', searchValue.toLowerCase(), '');
383 
384 	var lineno = nitobi.xml.transformToString(this.listXmlLower, this.searchXslProc);
385 
386 	if ("" == lineno)
387 		lineno=0;
388 	else
389 		lineno=parseInt(lineno);
390 
391 	return lineno;
392 }

399  */
400 nitobi.form.Lookup.prototype.filter = function(evt, o)
401 {
402 	var k=evt.keyCode;
403 	
404 	if (this.onKeyUp.notify(evt) == false) return;
405 
406 	if (!this.firstKeyup  && k != 38 && k != 40)
407 	{
408 		this.firstKeyup = true;
409 		return;
410 	}
411 	
412 	var tc = this.textControl;
413 	var sc = this.selectControl;
414 
415 	switch (k) {
416 		case 38:
417 			if (sc.selectedIndex == -1) sc.selectedIndex=0;
418 			if (sc.selectedIndex > 0) sc.selectedIndex--;
419 
420 			tc.value = sc.options[sc.selectedIndex].text;
421 			nitobi.html.highlight(tc, tc.value.length);
422 			tc.select();
423 			break;
424 		case 40:
425 			if (sc.selectedIndex < (sc.length -1)) {
426 				sc.selectedIndex++;
427 			}
428 			tc.value = sc.options[sc.selectedIndex].text;
429 			nitobi.html.highlight(tc, tc.value.length);
430 			tc.select();
431 			break;
432 
433 		default:
434 			if ((!this.getOnEnter && ((k<193 && k>46) || k==8 || k==32)) // All keys, backspace, del
435 				|| (this.getOnEnter && k== 13))						  	 // enter
436 			{
437 				var searchValue = tc.value;
438 				this.get(searchValue);
439 			}
440 /*
441 			//search
442 			lineno = this.search(searchValue);
443 
444 			//8 = backspace 46 = delete
445 			// Don't highlight anything if delete or backspace was pressed.
446 			if (lineno > 0 && k != 8 && k!=46 && (k<188 && k>46))
447 			{
448 				sc.selectedIndex = lineno-1;
449 				tc.value = sc[sc.selectedIndex].text;
450 				nitobi.html.highlight(tc, tc.value.length - (tc.value.length - searchValue.length));

459 /**
460  * Gets the data from the server using a SYNCHRONOUS get request.
461  * @ignore
462  */
463 nitobi.form.Lookup.prototype.get = function(searchString, force)
464 {
465 	if (this.getHandler != null && this.getHandler != ''/*&& !oColumn.fire("BeforeLookup")*/)
466 	{		
467 		if (force || !this.delay)
468 		{
469 			this.doGet(searchString);
470 		}
471 		else
472 		{
473 			if (this.timeoutId)
474 			{
475 				// cancel the pending request.
476 				window.clearTimeout(this.timeoutId);
477 				this.timeoutId = null;
478 			}
479 			this.timeoutId = window.setTimeout(nitobi.lang.close(this, this.doGet, [searchString]), this.delay);
480 		}
481 	}else{
482 		this.bindComplete(searchString);
483 	}
484 }

486 /**
487  * @ignore
488  */
489 nitobi.form.Lookup.prototype.doGet = function(searchString)
490 {
491 	// We are using a remote datasource
492 	if (searchString)
493 	{
494 		this.dataTable.setGetHandlerParameter("SearchString",searchString);
495 	}
496 	if(this.referenceColumn != null && this.referenceColumn != '')
497 	{
498 		//get value
499 		var referenceValue = this.owner.getCellValue(this.cell.row,this.referenceColumn);
500 		this.dataTable.setGetHandlerParameter("ReferenceColumn",referenceValue);
501 	}
502 	//, nitobi.lang.close(this, this.bindComplete, [searchString])
503 	this.dataTable.get(null, this.pageSize, this);
504 
505 	this.timeoutId = null;
506 
507 	this.bindComplete(searchString);
508 }
