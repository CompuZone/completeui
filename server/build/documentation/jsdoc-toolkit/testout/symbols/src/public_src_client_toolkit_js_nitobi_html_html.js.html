 15 /**
 16  * Creates a new HTML element with the specified tag name, attributes and styles.
 17  * @param {String} tagName The name of the element to create such as "div" or "span".
 18  * @param {Map} attrs A hash of name/value pairs of attributes to set such as {"id":"myDiv","class":"myClass"}.
 19  * @param {Map} styles A hash of name/value pair of styles to set such as {"color":"red"}.
 20  */
 21 nitobi.html.createElement = function(tagName, attrs, styles)
 22 {
 23 	var elem = document.createElement(tagName);
 24 	for (var attr in attrs)
 25 	{
 26 		if (attr.toLowerCase().substring(0,5) == "class")
 27 			elem.className = attrs[attr];
 28 		else
 29 			elem.setAttribute(attr, attrs[attr]);
 30 	}
 31 	for (var style in styles)
 32 	{
 33 		elem.style[style] = styles[style];
 34 	}
 35 	return elem;
 36 }

 69 /**
 70  * Adjusts the <code>child</code> element to have the same width as the <code>parent</code> element
 71  * taking into account the browser doctype as well as any borders or padding.
 72  * @param {HTMLElement} parent The element that the <code>child</code> element width should fill.
 73  * @param {HTMLElement} child The element that should have the width changed on to fit inside the <code>parent</code> element.
 74  */
 75 nitobi.html.fitWidth = function(parent, child) 
 76 {
 77 	var w;
 78 	var C = nitobi.html.Css;
 79 	if (nitobi.browser.IE && !nitobi.lang.isStandards())
 80 		w = (parseInt(C.getStyle(parent, "width")) - parseInt(C.getStyle(parent, "paddingLeft")) - parseInt(C.getStyle(parent, "paddingRight")) - parseInt(C.getStyle(parent, "borderLeftWidth")) - parseInt(C.getStyle(parent, "borderRightWidth"))) + "px";
 81 	else if (nitobi.lang.isStandards())
 82 		w = (parseInt(C.getStyle(parent, "width")) - (child.offsetWidth - parseInt(C.getStyle(child, "width")))) + "px";
 83 	else
 84 		w = parseInt(C.getStyle(parent, "width")) + "px";
 85 	child.style.width = w;
 86 }

628        for (var lx=0,ly=0;element!=null;
629            lx+=element.offsetLeft,ly+=element.offsetTop,element=element.offsetParent);

631            lx-=originalElement.scrollLeft,ly-=originalElement.scrollTop,originalElement=originalElement.parentNode);

817 /**
818  * Gets the outer HTML for the node as per Internet Explorers outerHTML element property.
819  * Note: Comment nodes are ignored.
820  * @param {HTMLElement} node The element what one wants the outer HTML of.
821  * @type String
822  */
823 nitobi.html.getOuterHtml = function (node) 
824 {
825 	if (nitobi.browser.IE)
826 	{
827 		return node.outerHTML;
828 	}
829 	else
830 	{
831 		var html = '';
832 		  switch (node.nodeType) {
833 		case Node.ELEMENT_NODE:
834 		html += '<';
835 		html += node.nodeName.toLowerCase();
836 		if (!nitobi.html.specialElements[node.nodeName]) {
837 	      for (var a = 0; a < node.attributes.length; a++)
838 	        {
839 	        	if (node.attributes[a].nodeName.toLowerCase() != "_moz-userdefined")
840 	        	{
841 		          html += ' ' + node.attributes[a].nodeName.toLowerCase() +
842 		                  '="' + node.attributes[a].nodeValue + '"';
843 	            }
844          	}
845 	        html += '>'; 
846 	        if (!nitobi.html.emptyElements[node.nodeName]) {
847 	          html += node.innerHTML;
848 	          html += '<\/' + node.nodeName.toLowerCase() + '>';
849 	        }
850 	      }
851 	      else switch (node.nodeName) {
852 	        case 'TEXTAREA':
853 	          for (var a = 0; a < node.attributes.length; a++)
854 	            if (node.attributes[a].nodeName.toLowerCase() != 'value')
855 	              html += ' ' + node.attributes[a].nodeName.toUpperCase() +
856 	                      '="' + node.attributes[a].nodeValue + '"';
857 	            else 
858 	              var content = node.attributes[a].nodeValue;
859 	          html += '>'; 
860 	          html += content;
861 	          html += '<\/' + node.nodeName + '>';
862 	          break; 
863 	      }
864 	      break;
865 	    case Node.TEXT_NODE:
866 	      html += node.nodeValue;
867 	      break;
868 	    case Node.COMMENT_NODE:
869 	      html += '<!' + '--' + node.nodeValue + '--' + '>';
870 	      break;
871 	  }
872 	  return html;
873 	}

968 	{

975 				this.parentNode.insertBefore(df, this);
976 				break;
977 			case "afterbegin":
978 				r.selectNodeContents(this);
979 				r.collapse(true);
980 				df = r.createContextualFragment(sHTML);
981 				this.insertBefore(df, this.firstChild);
982 				break;
983 			case "beforeend":
984 				// workaround==true:
985 				// we lose parsing feature of createContextualFragment but this seems
986 				// to fix the visual bug that occurs when the original code is used
987 				if(workaround==true){
988 					this.innerHTML = this.innerHTML + sHTML;
989 				} else {
990 					r.selectNodeContents(this);
991 					r.collapse(false);
992 					df = r.createContextualFragment(sHTML);
993 					this.appendChild(df);
994 				}
995 				break;
996 			case "afterend":
997 				r.setStartAfter(this);
998 				df = r.createContextualFragment(sHTML);
999 				this.parentNode.insertBefore(df, this.nextSibling);
