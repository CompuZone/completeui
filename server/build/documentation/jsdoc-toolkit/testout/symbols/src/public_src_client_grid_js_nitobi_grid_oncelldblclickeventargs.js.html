  3 /**
  4  * Constructs a OnCellDblClickEventArgs object.
  5  * @class When you subscribe to Grid events through the declaration, you
  6  * can optionally pass information about the event to the function
  7  * registered to handle it.  You do this by using the eventArgs keyword.
  8  * Is fired when a Cell is double clicked.
  9  * <p>
 10  * <b>Example</b>
 11  * </p>
 12  * <div class="code">
 13  * <pre><code class="html">
 14  * <ntb:grid id="grid1" mode="livescrolling" oncelldblclickevent="clickHandler(eventArgs)"></ntb:grid>
 15  * </code></pre>
 16  * </div>
 17  * <p>
 18  * The handler function might look like this:
 19  * </p>
 20  * <div class="code">
 21  * <pre><code class="javascript">
 22  * function clickHandler(event)
 23  * {
 24  * 	var cell = event.getCell();
 25  * 	cell.getDomNode().style.backgroundColor = "red";
 26  * }
 27  * </code></pre>
 28  * </div>
 29  * @constructor
 30  * @param {nitobi.grid} source The object which is firing the event.
 31  * @param {nitobi.grid.Cell} cell The Cell object of the cell that was clicked.
 32  * @extends nitobi.grid.CellEventArgs
 33  * @private
 34  */
 35 nitobi.grid.OnCellDblClickEventArgs = function(source, cell)
 36 {
 37 	nitobi.grid.OnCellDblClickEventArgs.baseConstructor.call(this, source, cell);
 38 }
