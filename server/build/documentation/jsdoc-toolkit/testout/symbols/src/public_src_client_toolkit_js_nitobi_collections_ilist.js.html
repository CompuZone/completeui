  3 /**
  4 	Creates a IList.
  5 	@class A more feature rich IList than a javascript array. If all the items in the IList
  6 	implement ISerializable, then the entire IList will be serializable.
  7 	@constructor
  8 	@extends nitobi.Object
  9 	@implements nitobi.base.ISerializable
 10 	@implements nitobi.collections.IEnumerable	
 11  */
 12 nitobi.collections.IList = function() 
 13 {
 14 	nitobi.base.ISerializable.call(this);
 15 	nitobi.collections.IEnumerable.call(this);
 16 }
 17 
 18 nitobi.lang.implement(nitobi.collections.IList,nitobi.base.ISerializable);
 19 nitobi.lang.implement(nitobi.collections.IList,nitobi.collections.IEnumerable);
 20 
 21 /**
 22  Indicates that the class implements this interface. This is copied onto the class itself
 23  when it is implemented. Useful to determine if all IList functions are available.
 24  @type Boolean
 25  */
 26 nitobi.collections.IList.prototype.IListImplemented = true;

 29 /**
 30 	Adds a item to the end of the IList.
 31 	@param {Object} obj An object or primitive to add to the IList.
 32  */
 33 nitobi.collections.IList.prototype.add = function(obj)
 34 {
 35 	nitobi.collections.IEnumerable.prototype.add.call(this,obj);
 36 	if (obj.ISerializableImplemented == true && obj.profile != null)
 37 	{
 38 		this.addChildObject(obj);		
 39 	}
 40 
 41 }
 42 
 43 /**
 44  * Inserts an item at the specified index.  The object at that index and subsequent objects
 45  * will be moved one place to the right (one index higher).
 46  * @param {Number} index the index of the object once it is inserted
 47  * @param {Object} obj an object or primitive to insert in the list
 48  */
 49 nitobi.collections.IList.prototype.insert = function(index, obj)
 50 {
 51 	var oldObj = this.get(index);
 52 	nitobi.collections.IEnumerable.prototype.insert.call(this,index, obj);
 53 	if (obj.ISerializableImplemented == true && obj.profile != null)
 54 	{
 55 		this.insertBeforeChildObject(obj,oldObj);		
 56 	}
 57 };

 59 /**
 60  * @private
 61  */
 62 nitobi.collections.IList.prototype.addToCache = function(obj, index)
 63 {
 64 	nitobi.base.ISerializable.prototype.addToCache.call(this,obj);
 65 	this.list[index] = obj;
 66 }
 67 
 68 /**
 69  * @private
 70  */
 71 nitobi.collections.IList.prototype.removeFromCache = function(index)
 72 {
 73 	nitobi.base.ISerializable.prototype.removeFromCache.call(this,this.list[index].getId());
 74 }

 76 /**
 77  * @private
 78  */
 79 nitobi.collections.IList.prototype.flushCache = function()
 80 {
 81 	nitobi.base.ISerializable.prototype.flushCache.call(this);
 82 	this.list = new Array();
 83 }

 85 /**
 86  * Returns an item from the IList.
 87  * @param {Number} index The index of the item you want. Indexed from zero, and throws an error if the index is out of bounds. If index is an object, the same
 88  * object is returned.
 89  * @returns {Object} The object you wanted.
 90  */
 91 nitobi.collections.IList.prototype.get = function(index)
 92 {
 93 	if (typeof(index) == "object")
 94 	{
 95 		return index;
 96 	}
 97 	if (index < 0 || index >= this.getLength())
 98 	{
 99 		nitobi.lang.throwError(nitobi.error.OutOfBounds);
100 	}
101 	var obj = null;
102 	if (this.list[index] != null)
103 	{
104 		obj = this.list[index];
105 	} 
106 	if (obj == null)
107 	{
108 		var xmlNode = nitobi.xml.getChildNodes(this.xmlNode)[index]; 
109 	
110 		if (xmlNode==null)
111 		{
112 			return null;
113 		}
114 		else
115 		{
116 			obj = this.factory.createByNode(xmlNode);
117 			this.onCreateObject.notify(obj);
118 			nitobi.collections.IList.prototype.addToCache.call(this,obj,index);
119 		}
120 		obj.setParentObject(this);
121 	}
122 	
123 	return obj;
124 }

126 /**
127  * @private
128  */
129 nitobi.collections.IList.prototype.getById = function(id)
130 {
131 	var node = this.xmlNode.selectSingleNode("*[@id='"+id+"']");
132 	var index = nitobi.xml.indexOfChildNode(node.parentNode,node);
133 	return this.get(index);
134 }

136 /**
137 	Sets an item in the IList.
138 	@param {Number} index The index for where you want to add the item. Throws an error if out of bounds.
139 	@param {Object} value The object you want to store.
140  */
141 nitobi.collections.IList.prototype.set = function(index,value)
142 {
143 	if (index < 0 || index >= this.getLength())
144 	{
145 		nitobi.lang.throwError(nitobi.error.OutOfBounds);
146 	}
147 	try
148 	{
149 		if (value.ISerializableImplemented == true)
150 		{
151 			var obj = this.get(index);
152 			
153 			// No need to reset the object if the same object is already in the IList.
154 			if (obj.getXmlNode() != value.getXmlNode())
155 			{
156 				var newNode = this.xmlNode.insertBefore(value.getXmlNode(),obj.getXmlNode());
157 				this.xmlNode.removeChild(obj.getXmlNode());
158 				obj.setXmlNode(newNode);	
159 			}
160 		}
161 		value.setParentObject(this);
162 		// Always update cache last.
163 		nitobi.collections.IList.prototype.addToCache.call(this,value,index);
164 	}
165 	catch(err)
166 	{
167 		nitobi.lang.throwError(nitobi.error.Unexpected,err);
168 	}
169 }

171 /**
172 	Removes an item from the IList. An error is thrown if the item does not exists
173 	or the index is out of bounds.
174 	@param {Number/Object} value An index or object you want removed from the IList.
175 */
176 nitobi.collections.IList.prototype.remove = function(value)
177 {
178 	var i;
179 	if (typeof(value) != "number")
180 	{
181 		i = this.indexOf(value);
182 	}
183 	else
184 	{
185 		i = value;
186 	}
187 
188 	var obj = this.get(i);
189 
190 	nitobi.collections.IEnumerable.prototype.remove.call(this,value);
191 	
192 	this.xmlNode.removeChild(obj.getXmlNode());
193 }

195 /**
196 	Returns the number of items in the IList.
197 	@returns {Number} The number of items.
198 */
199 nitobi.collections.IList.prototype.getLength = function()
200 {
201 	return nitobi.xml.getChildNodes(this.xmlNode).length;
202 }
