 31 /**
 32  * Creates an event arguments object based on the given parameters.
 33  * @class Arguments that are passed when nitobi.error.onError fires.
 34  * @constructor
 35  * @param {Object} source The object that fired the event.
 36  * @param {String} description A human-readable description of the error.
 37  * @param {String} type The type of object that threw the error.
 38  * @extends nitobi.base.EventArgs
 39  */
 40 nitobi.error.ErrorEventArgs = function(source, description, type)
 41 {
 42 	nitobi.error.ErrorEventArgs.baseConstructor.call(this,source);	
 43 
 44 	/**
 45 	 * The error description.
 46 	 * @type String
 47 	 */
 48 	this.description = description;
 49 	/**
 50 	 * The type of object that threw the error.
 51 	 * @type String
 52 	 */
 53 	this.type = type;
 54 }
