  3 /**
  4  * @class nitobi.collections.CellSet is a contiguous set of cells. The endpoints can be specified in any fashion, but the set will be addressed from top left to botom right.
  5  * @constructor
  6  * @param owner {nitobi.grid.Grid} The nitobi.grid.Grid object on whose cells the set sits. (Note: owner must implement getCellObject(x,y))
  7  * @param startRow {int} The row index of the start cell of the set.
  8  * @param startColumn {int} The column index of the start column of the set.
  9  * @param endRow {int} The row index of the end cell of the set.
 10  * @param endColumn {int} The column index of the end cell of the set.
 11  */
 12 nitobi.collections.CellSet = function(owner, startRow, startColumn, endRow, endColumn)
 13 {
 14 	this.owner = owner;
 15 	if (startRow != null && startColumn != null && endRow != null && endColumn != null)
 16 	{
 17 		this.setRange(startRow, startColumn, endRow, endColumn);
 18 	}
 19 	else
 20 	{
 21 		this.setRange(0,0,0,0);
 22 	}
 23 };
 24 
 25 nitobi.collections.CellSet.prototype.toString = function()
 26 {
 27 	var str = "";
 28 	for (var i = this._topRow; i <= this._bottomRow; i++)
 29 	{
 30 		str += "["
 31 		for (var j = this._leftColumn; j <= this._rightColumn; j++)
 32 		{
 33 			str += "("+i+","+j+")";
 34 		}
 35 		str += "]";
 36 	}
 37 	return str;
 38 };

 40 /**
 41  * Sets the endpoints of the set of cells on the grid to which the cellset refers.
 42  * @param startRow {int} The row index of the start cell of the set.
 43  * @param startColumn {int} The column index of the start column of the set.
 44  * @param endRow {int} The row index of the end cell of the set.
 45  * @param endColumn {int} The column index of the end cell of the set.
 46  */
 47 nitobi.collections.CellSet.prototype.setRange = function(startRow, startColumn, endRow, endColumn)
 48 {
 49 	ntbAssert(startRow != null && startColumn != null && endRow != null && endColumn != null, "nitobi.collections.CellSet.setRange requires startRow, startColumn, endRow, endColumn as integers",null,EBA_THROW);
 50 	this._startRow = startRow;
 51 	this._startColumn = startColumn;
 52 	this._endRow = endRow;
 53 	this._endColumn = endColumn;
 54 	
 55 	this._leftColumn = Math.min(startColumn, endColumn);
 56 	this._rightColumn = Math.max(startColumn, endColumn);
 57 	this._topRow = Math.min(startRow, endRow);
 58 	this._bottomRow = Math.max(startRow, endRow);
 59 };

 61 /**
 62  * Increases or decreases the size of the CellSet by changing the location of the starting cell.
 63  * @param startRow {int} The row index of the start cell of the set.
 64  * @param startColumn {int} The column index of the start cell of the set.
 65  */
 66 nitobi.collections.CellSet.prototype.changeStartCell = function(startRow, startColumn)
 67 {
 68 	this._startRow = startRow;
 69 	this._startColumn = startColumn;
 70 	
 71 	this._leftColumn = Math.min(startColumn, this._endColumn);
 72 	this._rightColumn = Math.max(startColumn, this._endColumn);
 73 	this._topRow = Math.min(startRow, this._endRow);
 74 	this._bottomRow = Math.max(startRow, this._endRow);
 75 };
 76 
 77 /**
 78  * Increases or decreases the size of the CellSet by changing the location of the ending cell.
 79  * @param startRow {int} The row index of the end cell of the set.
 80  * @param startColumn {int} The column index of the end cell of the set.
 81  */
 82 nitobi.collections.CellSet.prototype.changeEndCell = function(endRow, endColumn)
 83 {
 84 	this._endRow = endRow;
 85 	this._endColumn = endColumn;
 86 	
 87 	this._leftColumn = Math.min(endColumn, this._startColumn);
 88 	this._rightColumn = Math.max(endColumn, this._startColumn);
 89 	this._topRow = Math.min(endRow, this._startRow);
 90 	this._bottomRow = Math.max(endRow, this._startRow);
 91 };

 94 /**
 95  * Returns the number of rows in the CellSet
 96  * @type int
 97  */
 98 nitobi.collections.CellSet.prototype.getRowCount = function()
 99 {
100 	return this._bottomRow - this._topRow + 1;
101 };
102 
103 /**
104  * Returns the number of columns in the CellSet
105  * @type int
106  */
107 nitobi.collections.CellSet.prototype.getColumnCount = function()
108 {
109 	return this._rightColumn - this._leftColumn +1 ;
110 };

112 /**
113  * An inline object where object.top is the nitobi.drawing.Point object for the top left cell and object.bottom 
114  * is the nitobi.drawing.Point object for the bottom right cell
115  * @type Object
116  * @see nitobi.drawing.Point
117  */
118 nitobi.collections.CellSet.prototype.getCoords = function()
119 {
120 	return {'top':new nitobi.drawing.Point(this._leftColumn,this._topRow), 'bottom':new nitobi.drawing.Point(this._rightColumn,this._bottomRow)};
121 };

123 /**
124  * Returns the cell object at a position in the container offset from the top left cell in the CellSet
125  * @param relRow {int} The row index (indexed from 0 at the top of the CellSet) of the desired cell.
126  * @param relColumn {int} The column index (indexed from 0 at the leftmost column in the CellSet) of the desired cell.
127  * @type nitobi.grid.Cell
128  */
129 nitobi.collections.CellSet.prototype.getCellObjectByOffset = function(relRow, relColumn)
130 {
131 	return this.owner.getCellObject(this._topRow+relRow, this._leftColumn + relColumn);
132 };
