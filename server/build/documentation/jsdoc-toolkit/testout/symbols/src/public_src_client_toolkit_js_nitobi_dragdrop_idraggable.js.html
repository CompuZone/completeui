1 /**
  2  * Defines an interface for Nitobi UI classes that are to be draggable.
  3  * @param {HTMLElement} element The HTML element that is draggable.
  4  */
  5 nitobi.ui.IDraggable = function nitobi_ui_IDraggable(element)
  6 {
  7 	this.DragElement = element;
  8 }
  9 
 10 /**
 11  * Returns a reference to the HTML element that is draggable.
 12  */
 13 nitobi.ui.IDraggable.prototype.getDragElement = function()
 14 {
 15 	return this.DragElement;
 16 }
 17 
 18 /**
 19  * Makes the specified object draggable.
 20  * @param {Object} obj A nitobi object like an instance of a nitobi.ui.Panel.
 21  */
 22 nitobi.ui.IDraggable.makeDraggable = function(obj, element)
 23 {
 24 	obj.DragElement = element;
 25 	obj.getDragElement = createFunction(obj, nitobi.ui.IDraggable.prototype.getDragElement);
 26 }
 27 
 28 function createFunction(obj, func)
 29 {
 30 	return function() {return func.apply(obj, arguments)};
 31 }
