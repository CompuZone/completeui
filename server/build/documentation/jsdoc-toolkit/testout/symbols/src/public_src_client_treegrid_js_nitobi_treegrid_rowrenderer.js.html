  4 nitobi.grid.RowRenderer = function(xmlDataSource, columnDefinitions, firstColumn, uniqueId, surfaceKey, showHeaders, rowHeight, rowHover)
  5 {
  6 	this.rowHeight = rowHeight;
  7 	this.xmlDataSource = xmlDataSource;
  8 	this.dataTableId = '';
  9 	this.definitions = columnDefinitions;
 10 	this.columnSet = "";
 11 	
 12 	this.surfaceKey = surfaceKey;
 13 	//this.xslTemplate = xslTemplate;
 14 
 15 	this.firstColumn = firstColumn;
 16 
 17 	this.uniqueId = uniqueId;
 18 	
 19 	this.showHeaders = showHeaders;
 20 	this.rowHover = rowHover;
 21 
 22 	this.mergeDoc = nitobi.xml.createXmlDoc("<ntb:root xmlns:ntb=\"http://www.nitobi.com\"><ntb:columns><ntb:stub/></ntb:columns><ntb:data><ntb:stub/></ntb:data></ntb:root>");
 23 	this.mergeDocCols = this.mergeDoc.selectSingleNode("//ntb:columns");

 27 /**
 28  * @private
 29  */
 30 nitobi.grid.RowRenderer.prototype.render = function(firstRow,rows, activeColumn, activeRow, sortColumn, sortDir)
 31 {
 32 //	if (this.xslTemplate == null)
 33 //		return "";
 34 	var firstRow = Number(firstRow) || 0;
 35 	var rows = Number(rows) || 0;
 36 
 37 	var xt = nitobi.grid.rowXslProc;
 38 	xt.addParameter("start", firstRow, "");
 39 	xt.addParameter("end", firstRow + rows, "");
 40 	xt.addParameter('sortColumn', sortColumn, '');
 41 	xt.addParameter('sortDirection', sortDir, '');
 42 	xt.addParameter('dataTableId', this.dataTableId, '');
 43 	xt.addParameter('columnSet', this.columnSet, '');
 44 
 45 	// Do +0 to cast the bool to an int for Safari
 46 	xt.addParameter("showHeaders", this.showHeaders+0, "");
 47 	xt.addParameter("firstColumn", this.firstColumn, "");
 48 	this.lastColumn = this.definitions.childNodes.length;
 49 	xt.addParameter("lastColumn", this.lastColumn, "");
 50 	xt.addParameter("uniqueId", this.uniqueId, "");
 51 	xt.addParameter("rowHover", this.rowHover, "");
 52 	// No support for frozen columns in grouping grid
 53 	//xt.addParameter("frozenColumnId", this.frozenColumnId, "");
 54 	xt.addParameter("surfaceKey", this.surfaceKey, "");
 55 
 56 	var data = this.xmlDataSource.xmlDoc();
 57 	// TODO: This may not be applicable in all cases ... just the first render. THIS MAY BREAK STUFF!
 58 	if (data.documentElement.firstChild == null)
 59 		return "";
 60 
 61 	var root = this.mergeDoc;
 62 	this.mergeDocCols.replaceChild((!nitobi.browser.IE?root.importNode(this.definitions, true):this.definitions.cloneNode(true)), this.mergeDocCols.firstChild);
 63 	this.mergeDocData.replaceChild((!nitobi.browser.IE?root.importNode(data.documentElement, true):data.documentElement.cloneNode(true)), this.mergeDocData.firstChild);
 64 
 65 	s2 = nitobi.xml.transformToString(root, xt, "xml");
 66 	s2 = s2.replace(/ATOKENTOREPLACE/g," ");
 67 	s2 = s2.replace(/\#\<\;\#/g,"<").replace(/\#\>\;\#/g,">").replace(/\#\&eq\;\#/g,"=").replace(/\#\"\;\#/g,"\"").replace(/\#\&\;\#/g,"&");
 68 
 69 	return s2;
 70 }
