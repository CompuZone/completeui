  6 /**
  7  * nitobi.collections.CacheMap represents a doubly linked list of CacheNode objects.
  8  * It is used to keep track of ranges. The next and prev properties provide
  9  * access to the respective nodes in the linked list.
 10  */
 11 nitobi.collections.CacheMap = function()
 12 {
 13 	this.tail = null;
 14 	this.debug = new Array();

 17 /**
 18  * Insert will create and insert a new CacheNode into the CacheMap at the appropriate position.
 19  * If the range specified by the low and high overlaps any existing CacheNodes those
 20  * nodes will be expanded and/or merged.
 21  * @param {int} low The low argument is the low end of the range to insert (inclusive)
 22  * @param {int} high The high argument is the high end of the range to insert (inclusive)
 23  */
 24 nitobi.collections.CacheMap.prototype.insert = function(low,high)
 25 {
 26 	low = Number(low);
 27 	high = Number(high);
 28 	this.debug.push("insert("+low+","+high+")");
 29 	var newNode = new nitobi.collections.CacheNode(low,high);
 30 	if (this.head==null){
 31 		this.debug.push("empty cache, adding first node");
 32 		this.head = newNode;
 33 		this.tail = newNode;
 34 	}
 35 	else
 36 	{
 37 		// Handle adding at beginning or end as special
 38 		// case due to the fact that ranges are quite
 39 		// often prepended or appended to the list.
 40 		var n = this.head;
 41 		// find the node that will come after the new one.
 42 		while (n != null && low>n.high+1)
 43 		{
 44 			n = n.next;
 45 		}
 46 		if (n==null) {
 47 			this.debug.push("appending node to end");
 48 			this.tail.next = newNode;
 49 			newNode.prev = this.tail;
 50 			this.tail = newNode;
 51 		}
 52 		else
 53 		{
 54 			this.debug.push("inserting new node before " + n.toString());
 55 			if (n.prev != null)
 56 			{
 57 				newNode.prev = n.prev;
 58 				n.prev.next = newNode;
 59 			}
 60 			newNode.next = n;
 61 			n.prev = newNode;
 62 
 63 			while(newNode.mergeNext())
 64 			{
 65 			}
 66 
 67 			if (newNode.prev==null)
 68 			{
 69 				this.head = newNode;
 70 			}
 71 			if (newNode.next==null)
 72 			{
 73 				this.tail = newNode;
 74 			}
 75 		}
 76 	}
 77 }

 79 /**
 80  * Remove will remove a the specified range from the CacheMap.
 81  * If the range specified by the low and high arguments overlaps any existing CacheNodes those
 82  * nodes will be reduced in size or removed completely.
 83  * @param {int} low The low argument is the low end of the range to remove (inclusive)
 84  * @param {int} high The high argument is the high end of the range to remove (inclusive)
 85  */
 86 nitobi.collections.CacheMap.prototype.remove = function(low, high)
 87 {
 88 	low = Number(low);
 89 	high = Number(high);
 90 	this.debug.push("insert("+low+","+high+")");
 91 	if (this.head==null){
 92 	}
 93 	else
 94 	{	
 95 	    if (high < this.head.low || low > this.tail.high)
 96 	        return;
 97 
 98 		// Handle adding at beginning or end as special
 99 		// case due to the fact that ranges are quite
100 		// often prepended or appended to the list.
101 		var start = this.head;
102 
103 		while (start != null && low > start.high)
104 		{
105 			start = start.next;
106 		}
107 
108 		if (start==null)
109 		{
110 			this.debug.push("the range was not found");
111 		}
112 		else
113 		{
114 			var end = start;
115 			var temp = null;
116 
117 			while (end != null && high > end.high)
118     		{
119     		    if ((end.next != null && high < end.next.low) || end.next == null)
120     		        break;
121 				temp = end.next;
122 				if (end != start)
123 			    {
124 					this.removeNode(end);
125 				}
126 				end = temp;
127 			}
128 
129             if (start != end)
130             {
131                 if (high >= end.high)
132                 {
133                     //    that means the entire end node is to be removed
134                     this.removeNode(end);
135                 }
136 
137                 if (low <= start.low)
138                 {
139                     //    that means the entire start node is to be removed
140                     this.removeNode(start);
141                 }
142             }
143             //    start and end are the same ...
144             //    only thing left here is if we are removing the exact node
145             else if (start.low >= low && start.high <= high)
146             {
147                 this.removeNode(start);
148                 return;
149             }
150             //    otherwise we need to create a node cause we are removing a range
151             //    that is smaller than the single node.
152             else if (low > start.low && high < start.high)
153             {
154                 var origLow = start.low;
155                 var origHigh = start.high;
156                 this.removeNode(start);
157                 this.insert(origLow, low-1);
158                 this.insert(high+1, origHigh);
159                 return;
160             }
161 
162             if (end != null && high < end.high)
163             {
164                 //    that means that the end node is not the start and it will have to made smaller
165                 end.low = high+1;
166             }
167 
168             if (start != null && low > start.low)
169             {
170                 //    that means we have to shorten the start range since low is somewhere in it
171                 start.high = low-1;
172             }
173 		}
174 	}		
175 }

177 /**
178  * The gaps method will return an array containing the gaps in the CacheMap 
179  * that exist in the specified range.
180  * @param {int} low The low argument is the low end of the range to check for gaps in (inclusive)
181  * @param {int} high The high argument is the high end of the range to check for gaps in (inclusive)
182  * @returns {Array} Returns an array of Range objects.
183  */
184 nitobi.collections.CacheMap.prototype.gaps = function(low,high)
185 {
186 	// Could this search be executed faster in XPath???
187 	var g = new Array();
188 	var n = this.head;
189 
190 	if (n==null || n.low>high || this.tail.high<low)
191 	{
192 		// our search range lies entirely outside our cache
193 		g.push(new nitobi.collections.Range(low,high));
194 		return g;
195 	}
196 
197 	//	This loops through all the nodes in our cacheMap until
198 	//	a node high value is greater than the low end of our range
199 
200 	//TODO: change this to a binary search???
201 	var minLow = 0;
202 	while (n != null && n.high < low) // shouldn't overlap???
203 	{
204 		minLow = n.high+1;
205 		n = n.next;
206 	}
207 
208 	if (n!=null)
209 	{
210 		do
211 		{
212 			if (g.length == 0) // if this is the first gap.
213 			{
214 				if (low < n.low) {
215 //						g.push(new nitobi.collections.Range(low,Math.min(n.low-1,high)));
216 					g.push(new nitobi.collections.Range(Math.max(low,minLow),Math.min(n.low-1,high))); // Need to consider the case where the previous high overlaps the inserted range's low
217 				}
218 			}
219 			if (high > n.high)
220 			{
221 				if (n.next == null || n.next.low > high) // Need to consider the case where the inserted range's high overlaps the next range's low
222 				{
223 					g.push(new nitobi.collections.Range(n.high+1, high));
224 				}
225 				else
226 				{
227 					g.push(new nitobi.collections.Range(n.high+1, n.next.low-1));
228 				}
229 			}
230 			n = n.next;
231 		} while (n != null && n.high < high)
232 	}
233 	else
234 	{
235 		g.push(new nitobi.collections.Range(this.tail.high+1,high));
236 	}
237 	return g;
238 }

289 nitobi.collections.CacheMap.prototype.removeNode = function(node)
290 {
291 	if (node.prev != null)
292 	{
293 		node.prev.next = node.next;
294 	}
295 	else
296 	{
297 		this.head = node.next;
298 	}
299 
300 	if (node.next != null)
301 	{
302 		node.next.prev = node.prev;
303 	}
304 	else
305 	{
306 		this.tail = node.prev;
307 	}
308 
309 	node = null;
310 }

331 nitobi.collections.CacheMap.prototype.flush = function()
332 {
333 	var node = this.head;
334 	while(Boolean(node))
335 	{
336 		var next = node.next;
337 		delete(node);
338 		node = next;
339 	}
340 	this.head=null;
341 	this.tail=null;
342 }

344 /**
345 * Inserts a single entry into the map. If the entry exists within a
346 * range, then the ranges are updated. If the entry does not exist within
347 * any range, then the entry is added.
348 * @param {Number} index The index of the entry to add.
349 */
350 nitobi.collections.CacheMap.prototype.insertIntoRange = function(index)
351 {
352 	var n = this.head;
353 	var inc = 0;
354 	while (n != null) 
355 	{
356 		if (index >= n.low && index <= n.high)
357 		{
358 			inc = 1;
359 			n.high += inc;
360 		}
361 		else
362 		{
363 			n.low += inc;
364 			n.high += inc;
365 		}
366 		n = n.next;
367 	}
368 	// Check to see if we found anything at all.
369 	if (inc == 0)
370 	{
371 		this.insert(index,index);
372 	}
373 }

375 /**
376 * Deletes a single entry from the map. If the range in which the 
377 * entry resides is of length == 1, then the whole entry is removed.
378 * All subsequent ranges are updated.
379 * @param {Number} index The index of the entry to delete.
380 */
381 nitobi.collections.CacheMap.prototype.removeFromRange = function(index)
382 {
383 	var n = this.head;
384 	var inc = 0;
385 	while (n != null) 
386 	{
387 		if (index >= n.low && index <= n.high)
388 		{
389 			inc = -1;
390 			if (n.low == n.high)
391 			{
392 				this.remove(index,index);	
393 			}
394 			else
395 			{
396 				n.high += inc;
397 			}
398 		}
399 		else
400 		{
401 			n.low += inc;
402 			n.high += inc;
403 		}
404 		n = n.next;
405 	}
406 	ntbAssert(inc!=0,"Tried to remove something from a range where the range does not exist");
