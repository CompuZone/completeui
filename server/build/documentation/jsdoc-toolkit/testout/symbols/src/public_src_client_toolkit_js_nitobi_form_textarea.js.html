  6 nitobi.form.TextArea = function()

 12 			"<table border='0' cellpadding='0' cellspacing='0' class='ntb-input-border'><tr><td></td></table>";
 13

 15 	ph.style.top="-3000px";
 16 	ph.style.left="-3000px";
 17 
 18 	// We need this initial width so that we can fit the textarea to it's parent node width
 19 	this.control = nitobi.html.createElement('textarea', {}, {width:"100px"});
 20 }
