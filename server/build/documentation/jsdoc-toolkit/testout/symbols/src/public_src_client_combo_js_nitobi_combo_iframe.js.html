 16 // NOTE: DOUBLE SLASHED COMMENTS SO THAT CSHARP GEN DOESN'T CREATE TWO CTORS
 17 // <function name='Iframe' access="private">
 18 // <summary>Constructor for Iframe, a (cross-browser) object that encapsulates an absolutely positioned IFRAME.</summary>
 19 // <returns type="void"></returns></function>
 20 // createAttribute(), createDocumentFragment(), createElement(),
 21 // createTextNode(), getElementById(), etc.: REMEMBER to call these from
 22 // Iframe.document vs. document because this IFRAME is a different (new) document
 23 // parameters:
 24 // - attachee (required) - HTMLElement to attach this IFRAME to
 25 // - h (optional) - height of IFRAME, defaults to 0 if not specified
 26 // - w (optional) - width of IFRAME, defaults to offsetWidth of attachee if not specified
 27 /**
 28  * @private
 29  * @ignore
 30  */
 31 function Iframe(attachee,h,w, ComboHtmlTagObject)
 32 {
 33 	if(!attachee){
 34 		var msg="Iframe constructor: attachee is null!";
 35 		alert(msg);
 36 		throw msg;
 37 	}
 38 	var d=document;
 39 	var oIF=d.createElement("IFRAME");
 40 	var s=oIF.style;
 41 	// attach() depends on this.oIFStyle so set this before calling
 42 	// attach() depends on this.attachee so set this before calling
 43 	this.oIFStyle=s;
 44 	this.attachee=attachee;
 45 	this.attach();
 46 	s.position="absolute";
 47 	w = w || attachee.offsetWidth;
 48 	s.width=w;
 49 	s.height = h || 0;
 50 	s.display="none";
 51 	s.overflow="hidden";
 52 	// .name is needed to use frames[name]; .id is not enough
 53 	var name="IFRAME"+oIF.uniqueID;
 54 	oIF.name=name;
 55 	oIF.id=name;
 56 	oIF.frameBorder=0;
 57 	// Keeps IE from complaining about unsecure items when in secure site
 58 	oIF.src="javascript:true";
 59 	
 60 	// Insert the iframe just before the form end if possible.
 61 	// This is done to keep the iframe in any form the combo might be in
 62 	// If this is not done, the list causes the webpage to resize.
 63 	var hostTag = Browser_GetParentElementByTagName(ComboHtmlTagObject,"form");
 64 	if (null==hostTag)
 65 	{
 66 		hostTag = d.body;
 67 	}
 68 	hostTag.appendChild(oIF);
 69 	
 70 	var oF=window.frames[name];
 71 	var oD=oF.document;
 72 	oD.open();
 73 	// - oD.write(...) is needed to create a body for oD; remember to access oD.body AFTER oD.write(...);
 74 	// - margin-* all need to be set to 0 or we'd get some default margin space;
 75 	// - background-color:white; needed for Mozilla to show w/o bleeding as in IE;
 76 	// - float:left; needed for Mozilla or else SPAN won't show;
 77 	// - need to put border in SPAN (vs. in IFRAME) or we'd get minor bleeding w/ SELECT elements in IE;
 78 	// - bodySpan's width and height are reduced by 2 pixels for Mozilla to show the same as in IE
 79 	//		because IE's width/height include borders whereas Mozilla's don't; so if user needs to change
 80 	//		border width to 0 (i.e. Iframe.bodySpan.style.border) then please adjust the width/height
 81 	//		according for Moz (i.e. add border pixels back); similarly, adjustments to width/height should
 82 	//		take into account the borders for Moz (i.e. minus border pixels);
 83 	// - note that inline styles below will not be overriden by css (i.e. assignment to .className);
 84 	//		IF it's necessary to override, then do it through DOM (i.e. Iframe.bodySpan.style.*);
 85 	//		user may want to override the border color, width, style (see previous note on this for Moz)
 86 	// BUG: TEXTAREA's cursor still bleeds through IFRAME in IE;
 87 	oD.write("<html><head></head><body style=\"margin:0;background-color:white;\"><span id=\"bodySpan\" class=\"ntb-combobox-list-outer-border\" style=\"overflow:hidden;float:left;border-width:1px;border-style:solid;width:"+(w-(nitobi.browser.MOZ?2:0))+";height:"+(h-(nitobi.browser.MOZ?2:0))+";\"></span></body></html>");
 88 //	oD.write("<html><head></head><body style=\"margin:0;background-color:white;\"><span id=\"bodySpan\" style=\"overflow:hidden;float:left;border:0px solid black;width:"+w+";height:0;\"></span></body></html>");
 89 	oD.close();
 90 	// because this IFRAME is a new document, need to import window.document's
 91 	// stylesheets or else they won't be available to this IFRAME
 92 	var dss=d.styleSheets;
 93 	var ss=oD.createElement("LINK");
 94 	for(var i=0,n=dss.length; i<n; i++){
 95 		// cloneNode() is faster than createElement()
 96 		var ss2=ss.cloneNode(true);
 97 		ss2.rel=(nitobi.browser.IE ? dss[i].owningElement.rel : dss[i].ownerNode.rel);
 98 		ss2.type="text/css";
 99 		ss2.href=dss[i].href;
100 		ss2.title=dss[i].title;
101 		oD.body.appendChild(ss2);
102 	}
103 	// only one HEAD tag written above so [0] is our desired element
104 	var head=oD.getElementsByTagName("head")[0];
105 	// - because this IFRAME is a new document, need to import window.document's
106 	//		scripts or else they won't be available to this IFRAME;
107 	// - document.scripts is only in IE; need to grab SCRIPT tags in Moz;
108 	var ds=(d.scripts?d.scripts:d.getElementsByTagName("script"));
109 	var st=oD.createElement("SCRIPT");
110 	var src=null;
111 	for(var i=0,n=ds.length; i<n; i++){
112 		src=ds[i].src;
113 		if(""!=src)
114 		{
115 			// cloneNode() is faster than createElement()
116 			var st2=st.cloneNode(true);
117 			st2.language=ds[i].language;
118 			st2.src=src;
119 			head.appendChild(st2);
120 		}
121 	}
122 	this.oIF=oIF;
123 	this.oF=oF;
124 	this.d=oD;
125 	this.bodySpan=oD.getElementById("bodySpan");
126 	this.bodySpanStyle=this.bodySpan.style;
127 	if(window.addEventListener){
128 		// Moz way of moving IFRAME on body resize; onresize in Moz is called
129 		// once after the user releases the mouse after resizing
130 		window.addEventListener('resize',this,false);
131 	}else if(window.attachEvent){
132 		if(!window.g_Iframe_oIFs){
133 			window.g_Iframe_oIFs=new Array;
134 			window.g_Iframe_onresize=window.onresize;
135 			Iframe_oResize();
136 			window.onresize=window.oResize.check1;
137 		}
138 		window.g_Iframe_oIFs[name]=this;
139 	}
140 }

181 function Iframe_onafterresize(){
182 	for(var f in window.g_Iframe_oIFs){
183 		var oIF=window.g_Iframe_oIFs[f];
184 		oIF.attach();
185 	}
186 	if(window.g_Iframe_onresize)

189 /**
190  * @private
191  * @ignore
192  */
193 function Iframe_dfxWinXY(w){
194 	var b,d,x,y;
195 	x=y=0;
196 	var d=window.document;
197 	if(d.body){
198 		b = d.documentElement.clientWidth ? d.documentElement : d.body;
199 		x=b.clientWidth||0;
200 		y=b.clientHeight||0;
201 	}
202 	return {x:x,y:y};
203 }
204 /**
205  * @private
206  * @ignore
207  */
208 function Iframe_oResize(){
209 	window.oResize={
210 		CHECKTIME:500,
211 		oldXY:Iframe_dfxWinXY(window),
212 		timerId:0,
213 		check1:function(){window.oResize.check2()},
214 		check2:function(){
215 			if(this.timerId)
216 				window.clearTimeout(this.timerId);
217 			this.timerId = setTimeout("window.oResize.check3()",this.CHECKTIME);
218 		},
219 		check3:function(){
220 			var newXY = Iframe_dfxWinXY(window);
221 			this.timerId = 0;
222 			if((newXY.x != this.oldXY.x) || (newXY.y != this.oldXY.y)){
223 				this.oldXY = newXY;
224 				Iframe_onafterresize();
225 			}
226 		}
227 	}
228 }

275 function Iframe_offset(o, attr, a){
276 // for absolutely positioned 'o', only take its offset* and add
277 // table border pixels back in;
278 // for relatively positioned 'o', also add in its ancestor's offset*
279 	var x = (a ? o[attr] : 0);
280 	var _o = o;
281 	while(o){
282 		x += (a ? 0 : o[attr]);
283 
284 		// seems to fix a bug in IE where if the element is inside a TABLE
285 		// with a border != 0, the offset* is off by one
286 		if(nitobi.browser.IE && "TABLE"==o.tagName && "0"!=o.border && ""!=o.border){
287 			x++;
288 		}
289 		o=o.offsetParent;
290 	}
291 // no longer necessary because we've fixed the textbox+button position issues;
292 // but leave here just in case; comments get stripped out at build-time anyway
293 //	// IE hack-like code to deal w/ TR's w/ VALIGN of 'middle' and 'bottom',

301 //				// the difference is odd); most of the time, this is the desired
302 //				// behavior; some other times, the top is off by one (i.e. didn't

334 function Iframe_setHeight(h,workaround){
335 	h=parseInt(h);
336 	this.oIFStyle.height=h;
337 	// bodySpan's height is reduced by ? border pixels for Mozilla to show the same as in IE
338 	//	because IE's width/height include borders whereas Mozilla's don't
339 	if(workaround!=true){
340 		this.bodySpanStyle.height =
341 			(h-(nitobi.browser.MOZ ? parseInt(this.bodySpanStyle.borderTopWidth) + parseInt(this.bodySpanStyle.borderBottomWidth) : 0));
342 	}
343 }

384 function Iframe_show(){
385 // in Moz & IE, resizing window when IFRAME is not shown and then
386 // showing the IFRAME can cause the IFRAME to be off, position-wise; solve by

388 	this.attach();

452 // *****************************************************************************
453 // * attach
454 // *****************************************************************************
455 // <function name="attach" access="private">
456 // <summary>Attaches this IFRAME to the bottom of 'attachee', matching its left border.</summary>
457 // </function>
458 // attach() depends on this.oIFSytle so set this before calling attach()
459 // attach() depends on this.attachee so set this before calling attach()
460 /**
461  * @private
462  * @ignore
463  */
464 Iframe.prototype.attach=Iframe_attach;
465 /**
466  * @private
467  * @ignore
468  */
469 function Iframe_attach(){
470 	var attachee=this.attachee;
471 	var a = (attachee.offsetParent && "absolute"==attachee.offsetParent.style.position);
472 	// oIF.style.top: minus 1 to match f's top border w/ attachee's bottom border
473 	// attachees w/ absolutely positioned parents need to be handled differently
474 	this.oIFStyle.top = this.offset(attachee, "offsetTop", a) + attachee.offsetHeight - 1 + (a ? parseInt(attachee.offsetParent.style.top) : 0);
475 	this.oIFStyle.left = this.offset(attachee, "offsetLeft", a) + (a ? parseInt(attachee.offsetParent.style.left) : 0);
476 }
