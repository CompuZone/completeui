 14 /**

 16  * @param {HTMLElement} domElement the HTML DOM element whose class attribute will be changed
 17  * @param {String} class1 the name of the class to replace with <CODE>class2</CODE>
 18  * @param {String} class2 the name of the class that will replace <CODE>class1</CODE>

 20 nitobi.html.Css.swapClass = function(domElement, class1, class2)
 21 {
 22 	if (domElement.className)
 23 	{
 24 		var reg = new RegExp('(\\s|^)'+class1+'(\\s|$)');

 29 /**
 30  * Replace one CSS class with another in an <CODE>Element</CODE>'s class list.  If class1 
 31  * is not present, then append class2 to the class list.
 32  * @param {HTMLElement} domElement the HTML DOM element whose class attribute will be changed
 33  * @param {String} class1 the name of the class to replace with <CODE>class2</CODE>
 34  * @param {String} class2 the name of the class that will replace <CODE>class1</CODE>

 36 nitobi.html.Css.replaceOrAppend = function(domElement, class1, class2)
 37 {
 38 	if (nitobi.html.Css.hasClass(domElement,class1))
 39 	{
 40 		nitobi.html.Css.swapClass(domElement,class1,class2);
 41 	}
 42 	else
 43 	{
 44 		nitobi.html.Css.addClass(domElement,class2);
 45 	}

 92 /**
 93  * Adds a CSS rule to the specified stylesheet object.
 94  * @param {String} selector The CSS rule selector string such as <code>.myClass</code>.
 95  * @param {String} style The CSS rule style string such as <code>color:red;border:1px solid black;</code>.
 96  */
 97 nitobi.html.Css.addRule = function(styleSheet, selector, style)
 98 {
 99 	if (styleSheet.cssRules) {
100 		var index = styleSheet.insertRule(selector+"{"+(style||"")+"}", styleSheet.cssRules.length);
101 		return styleSheet.cssRules[index];
102 	} else {
103 		// sets style nitobi:placeholder since IE doesnt like the empty steez
104 		styleSheet.addRule(selector, style||"nitobi:placeholder;");
105 		return styleSheet.rules[styleSheet.rules.length-1];
106 	}

112  * @type Array

114 nitobi.html.Css.getRules = function(styleSheetIndex)
115 {
116 	var sheet = null;
117 	if (typeof(styleSheetIndex) == 'number')
118 	{
119 		sheet = document.styleSheets[styleSheetIndex];

131 			return sheet.cssRules;
132 		}
133 		if (sheet.rules)
134 		{
135 			return sheet.rules;
136 		}
137 	} catch (e) {/*error is due to cross domain CSS*/}
138 	return null;

231 /**
232  * Loads images from a stylesheet.  If no stylesheet is specified, loads every image from every
233  * stylesheet in the page.  After an image is pre-cached, the browser will not need to load the 
234  * image from the server.
235  * @param {StyleSheet} sheet the stylesheet to parse for image names
236  */
237 
238 nitobi.html.Css.precacheImages = function(sheet)
239 {
240 	nitobi.html.Css.isPrecaching = true;
241 	if (!sheet)
242 	{
243 		var ss = document.styleSheets;
244 		for (var i = 0; i < ss.length; i++)
245 		{
246 			nitobi.html.Css.precacheImages(ss[i]);
247 		}
248 		//this.onPrecached.notify();
249 		//this.onPrecached.subscribe();
250 		return;
251 	}
252 
253 	var regex = /.*?url\((.*?)\).*?/;
254 	var rules = nitobi.html.Css.getRules(sheet);
255 	var url = nitobi.html.Css.getPath(sheet);
256 	for (var i = 0; i < rules.length; i++)
257 	{
258 		var rule = rules[i];
259 		if (rule.styleSheet)
260 		{
261 			nitobi.html.Css.precacheImages(rule.styleSheet);
262 		}
263 		else
264 		{
265 			var s = rule.style;
266 			var bgImage = s ? s.backgroundImage : null;
267 			if (bgImage)
268 			{
269

272 
273 				if (!nitobi.html.Css.imageCache[bgImage])
274 				{
275 					var image = new Image();
276 					image.src = bgImage;
277 					//console.log(bgImage);
278 
279 					nitobi.html.Css.precacheArray[bgImage] = false;
280 
281 					// Create a memory leak safe closure that calls the global function
282 					var closure = nitobi.lang.close({}, nitobi.html.Css.trackPrecache, [bgImage]);	
283 					image.onload = closure;
284 					image.onerror = closure;
285 					image.onabort = closure;
286 
287 					nitobi.html.Css.imageCache[bgImage] = image;
288 					try {
289 					if (image.width > 0) 
290 						nitobi.html.Css.precacheArray[bgImage] = true;					
291 					} catch(e) {}
292 				
293 				}
294 			}
295 		}
296 	}
297 	// IE6 imports bug - its actually a problem with using imports on dynamic stylesheets
298 	if (sheet.href != "" && sheet.imports)
299 	{
300 		for (var i = 0; i < sheet.imports.length; i++)
301 		{
302 			nitobi.html.Css.precacheImages(sheet.imports[i]);
303 		}
304 	}

424 			else

609     	}
610     }

688 {
689 	var ss;
690 	if (nitobi.browser.IE)
691 	{
692 		ss = document.createStyleSheet();
693 	}
694 	else
695 	{
696 		ss = document.createElement('style');
697 		ss.setAttribute("type", "text/css");
698 		document.body.appendChild(ss);
699 		ss.appendChild(document.createTextNode(""));
700 	}
701 	if (cssText != null)
702 		nitobi.html.Css.setStyleSheetValue(ss, cssText);
703 	return ss;

708 	if (nitobi.browser.IE)
709 		ss.cssText = cssText;
