 14 nitobi.base.DateMath = {

 18 	YEAR:'y',

 21 /**
 22  * Common function for add and subtract
 23  * @param {Date} date The javascript date object
 24  * @param {String} unit The unit 
 25  * @param {Number} amount The amount of unit
 26  * @return the result
 27  * @type Date
 28  * @private
 29  */
 30 nitobi.base.DateMath._add = function(date,unit,amount){	
 31 	if(unit == this.DAY)	date.setDate(date.getDate()+amount);
 32 	else if(unit==this.WEEK) date.setDate(date.getDate()+7*amount);		
 33 	else if(unit==this.MONTH) date.setMonth(date.getMonth()+amount);
 34 	else if(unit==this.YEAR) date.setFullYear(date.getFullYear()+amount);
 35 	return date;
 36 }
 37 /**
 38  * Adds a date by an amount
 39  * @param {Date} date The javascript date object
 40  * @param {String} unit The unit
 41  * @param {Number} amount The amount of unit
 42  * @return 
 43  * @type Date
 44  * @static
 45  */
 46 nitobi.base.DateMath.add = function(date,unit,amount){
 47 	return this._add(date,unit,amount);
 48 }
 49 /**
 50  * Subtracts a date by an amount

 54  * @return the resulting Date
 55  * @type Date
 56  * @static
 57  */
 58 nitobi.base.DateMath.subtract = function(date,unit,amount){

107 /**
108  * Determines whether the year in the Date object is a leap year.
109  * @param {Date} date The javascript Date object
110  * @return Object
111  * @type Boolean
112  * @static
113  */
114 nitobi.base.DateMath.isLeapYear = function(date){
115 	var y = date.getFullYear();
116 	var	_1 = String(y/4).indexOf('.') == -1; 
117 	var _2 = String(y/100).indexOf('.') == -1;
118 	var _3 = String(y/400).indexOf('.') == -1;
119 	return (_3)?true:(_1 && !_2)?true:false;
120 }

151 /**
152  * Checks if the date is today
153  * @param {Date} date The javascript Date object
154  * @return Object
155  * @type Boolean
156  * @static
157  */
158 nitobi.base.DateMath.isToday = function(date){
159 	var start = this.resetTime(new Date());
160 	var end = this.add(this.clone(start),this.DAY,1);
161 	return this.between(date,start,end);
162 }

166  * @param {Date} date The first date in the comparison.
167  * @param {Date} compare The second date in the comparison.
168  */
169 nitobi.base.DateMath.isSameDay = function(date, compare)
170 {
171 	date = this.resetTime(this.clone(date));
172 	compare = this.resetTime(this.clone(compare));
173 	return date.valueOf() == compare.valueOf();
174 }

187 /**
188  * Calculates the week number of the given date in that year.
189  * Week numbering starts from 0
190  * @param {Date} date The javascript date object
191  * @return Object
192  * @type Number
193  * @static
194  */
195 nitobi.base.DateMath.getWeekNumber = function(date){
196 	var january = this.getJanuary1st(date);
197 	return Math.ceil(this.getNumberOfDays(january,date) / 7);
198 }

200 /**
201  * Calculates the number of days from start date to end date.
202  * Both start and end date are inclusive.
203  * @param {Date} date The javascript date object
204  * @return Object
205  * @type Number
206  * @static
207  */
208 nitobi.base.DateMath.getNumberOfDays = function(start,end){
209 	var duration = this.resetTime(this.clone(end)).getTime() - this.resetTime(this.clone(start)).getTime();
210 //	alert(duration/this.ONE_DAY_MS+1);
211 	return Math.round(duration/this.ONE_DAY_MS)+1;
212 }

225 /**
226  * Resets the time
227  * @param {Date} date The javascript Date object
228  * @return Object
229  * @type Date
230  * @static
231  */
232 nitobi.base.DateMath.resetTime = function(date){
233 	if (nitobi.base.DateMath.invalid(date))
234 		return date;
235 	date.setHours(0);
236 	date.setMinutes(0);
237 	date.setSeconds(0);		
238 	date.setMilliseconds(0);		
239 	return date;
240 }

242 /**
243  * Returns a Javascript <code>Date</code> object from a given ISO 8601 date string.
244  * @param {String} date an ISO 8601 date string eg. 1980-12-21 08:30:12
245  * @type Date
246  */
247 nitobi.base.DateMath.parseIso8601 = function(date) {
248 	return new Date(date.replace(/^(....).(..).(..)(.*)$/, "$1/$2/$3$4"));
249 };

251 /**
252  * Returns an ISO 8601 formatted date string from a javascript Date object.
253  * @param {Date} date a Javascript date object
254  * @type String
255  */
256 nitobi.base.DateMath.toIso8601 = function(date) {
257 	if (nitobi.base.DateMath.invalid(date)) return "";
258 	var pz = nitobi.lang.padZeros;	
259 	return date.getFullYear()+"-"+pz(date.getMonth()+1)+"-"+pz(date.getDate())+" "+pz(date.getHours())+":"+pz(date.getMinutes())+":"+pz(date.getSeconds());	
260 };
