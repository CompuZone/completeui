216 /**
217  * @private
218  */
219 nitobi.grid.Column.prototype.xGET = function()
220 {
221 	var node = null, xpath = "@"+arguments[0], val = "";
222 	var cachedNode = this.modelNodes[xpath];
223 	if (cachedNode!=null)
224 		node = cachedNode;
225 	else
226 		node = this.modelNodes[xpath] = this.getModel().selectSingleNode(xpath);
227 
228 	if (node!=null)
229 		val = node.nodeValue;
230 	return val;
231 }

253 /**
254  * @private
255  */
256 nitobi.grid.Column.prototype.eSET = function(name, arguments)
257 {
258 	var oFunction = arguments[0];
259 	var funcRef = oFunction;
260 
261 	var subName = name.substr(2);
262 	subName = subName.substr(0,subName.length-5);
263 
264 	// TODO: this should look like eSET in Grid.
265 	if (typeof(oFunction) == 'string')
266 	{
267 		funcRef = function(eventArgs) {return eval(oFunction)};
268 	}
269 
270 	if (typeof(this[name]) != 'undefined')
271 	{
272 		alert('unsubscribe');
273 		this.unsubscribe(subName, this[name]);
274 	}
275 
276 	// name should be OnCellClickEvent but we just expect CellClick for firing
277 	var guid = this.subscribe(subName,funcRef);
278 	this.jSET(name, [guid]);
279 }
