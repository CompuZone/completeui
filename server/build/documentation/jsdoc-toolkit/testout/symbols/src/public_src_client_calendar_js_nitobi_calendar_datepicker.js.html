230 /**
231  * Sets the selected date.
232  * @param {Date} date The date to set as the selected date.
233  * @type Boolean
234  * @private
235  */
236 nitobi.calendar.DatePicker.prototype._setSelectedDate = function(date, forceRender)
237 {
238 	this.setDateAttribute("selecteddate", date); 
239 	
240 	var hidden = this.getHtmlNode('value');
241 	if (hidden) hidden.value = this.formatDate(date, this.getSubmitMask());
242 	
243 	var input = this.getInput();
244 	if (input)
245 	{
246 		var inputMask = input.getDisplayMask();
247 		var inputValue = this.formatDate(date, inputMask);
248 		input.setValue(inputValue);
249 		input.setInvalidStyle(false);
250 	}
251 	
252 	// Set the highlighted date for the calendar now
253 	var calendar = this.getCalendar();
254 	if (calendar)
255 	{
256 		calendar.clearHighlight(date);
257 		var dm = nitobi.base.DateMath;
258 		var startDate = dm.getMonthStart(this.getStartDate());
259 		var months = calendar.getMonthColumns() * calendar.getMonthRows() - 1;
260 		var endDate = dm.getMonthEnd(dm.add(dm.clone(startDate), 'm', months));
261 		if (dm.between(date, startDate, endDate))
262 		{
263 			calendar.highlight(date);
264 		}
265 		if (forceRender)
266 		{
267 			this.setStartDate(dm.getMonthStart(dm.clone(date)));
268 			calendar.render();
269 		}
270 	}
271 	
272 	var eventsManager = this.getEventsManager();
273 	if (eventsManager.isEvent(date))
274 	{
275 		var startDate = eventsManager.eventsCache[date.valueOf()];
276 		var eventInfo = this.eventsManager.getEventInfo(startDate);
277 		this.onEventDateSelected.notify({events: eventInfo});
278 	}
279 	this.onDateSelected.notify(new nitobi.ui.ElementEventArgs(this, this.onDateSelected));
280 }

282 /**
283  * Validates the date and fires the appropriate events if it isn't valid.  Returns true if the date is valid,
284  * false otherwise.
285  * @param {Date} date The date to validate against.
286  * @type Boolean
287  * @private
288  */
289 nitobi.calendar.DatePicker.prototype.validate = function(newDate)
290 {
291 	var E = nitobi.ui.ElementEventArgs;
292 	if (nitobi.base.DateMath.invalid(newDate))
293 	{
294 		this.onSetInvalidDate.notify(new E(this, this.onSetInvalidDate));
295 		return false;
296 	}
297 	if (this.isOutOfRange(newDate))
298 	{
299 		this.onSetOutOfRangeDate.notify(new E(this, this.onSetOutOfRangeDate));
300 		return false;
301 	}
302 	if (this.isDisabled(newDate))
303 	{
304 		this.onSetDisabledDate.notify(new E(this, this.onSetDisabledDate));
305 		return false;
306 	}
307 	return true;
308 }

347 /**
348  * Returns true if the given date is within the range specified by the mindate and maxdate attributes.
349  * @param {Date} date The date to check.
350  */
351 nitobi.calendar.DatePicker.prototype.isOutOfRange = function(date)
352 {
353 	var dm = nitobi.base.DateMath;
354 	var minDate = this.getMinDate();
355 	var maxDate = this.getMaxDate();
356 	var isOutOfRange = false;
357 	if (minDate && maxDate) isOutOfRange = !dm.between(date, minDate, maxDate);
358 	else if (minDate && maxDate == null) isOutOfRange = dm.before(date, minDate);
359 	else if (minDate == null && maxDate) isOutOfRange = dm.after(date, maxDate);
360 	return isOutOfRange;
361 }

498 /**
499  * Parses a natural language expression and returns a the corresponding date.
500  * Allowable expressions include: yesterday, today, tomorrow, lastweek, nextweek, lastmonth, nextmonth, lastyear, and nextyear.
501  * @type Date
502  * @private
503  */
504 nitobi.calendar.DatePicker.prototype.parseLanguage = function(date)
505 {
506 	var dm = nitobi.base.DateMath;
507 	var parsedDate = Date.parse(date);
508 	if (parsedDate && typeof(parsedDate) == "object" && !isNaN(parsedDate) && !dm.invalid(parsedDate))
509 	{
510 		// Then there is some sort of date lib included in the page.
511 		return parsedDate;
512 	}
513 	if (date == "" || date == null)
514 		return null;
515 
516 	date = date.toLowerCase();
517 	var today = dm.resetTime(new Date());
518 	switch (date)
519 	{
520 		case "today":
521 			date = today;
522 			break;
523 		case "tomorrow":
524 			date = dm.add(today, 'd', 1);
525 			break;
526 		case "yesterday":
527 			date = dm.subtract(today, 'd', 1);
528 			break;
529 		case "last week":
530 			date = dm.subtract(today, 'd', 7);
531 			break;
532 		case "next week":
533 			date = dm.add(today, 'd', 7);
534 			break;
535 		case "last year":
536 			date = dm.subtract(today, 'y', 1);
537 			break;
538 		case "last month":
539 			date = dm.subtract(today, 'm', 1);
540 			break;
541 		case "next month":
542 			date = dm.add(today, 'm', 1);
543 			break;
544 		case "next year":
545 			date = dm.add(today, 'y', 1);
546 			break;
547 		default:
548 			date = dm.resetTime(new Date(date));
549 			break;
550 	}
551 	if (dm.invalid(date))
552 		return null;
553 	else
554 		return date;
555 }

700 nitobi.calendar.DatePicker.prototype.parseDate = function(date, mask)
701 {
702 	var parsedValues = {};
703 	while (mask.length > 0)
704 	{
705 		var c = mask.charAt(0);
706 		var testExp = new RegExp(c + "+");
707 		var format = testExp.exec(mask)[0];
708 		if (c != "d" && c != "y" && c != "M" && c != "N" && c != "E")
709 		{
710 			mask = mask.substring(format.length);
711 			date = date.substring(format.length);
712 		}
713 		else
714 		{
715 			var separator = mask.charAt(format.length);
716 			var currentValue = (separator == ""?date:date.substring(0, date.indexOf(separator)));
717 			var validatedDate = this.validateFormat(currentValue, format);
718 			if (validatedDate.valid)
719 			{
720 				parsedValues[validatedDate.unit] = validatedDate.value;
721 			}
722 			else
723 			{
724 				return null;
725 			}
726 			mask = mask.substring(format.length);
727 			date = date.substring(currentValue.length);
728 		}
729 	}
730 	var date = new Date(parsedValues.y, parsedValues.m, parsedValues.d);
731 	return date;
732 }

734 /**
735  * @private
736  */
737 nitobi.calendar.DatePicker.prototype.validateFormat = function(value, format)
738 {
739 	var validated = {valid: false, unit: "", value: ""};
740 	switch (format)
741 	{
742 		case "d":
743 		case "dd":
744 			var parsedValue = parseInt(value);
745 			var isValid;
746 			if (format == "d") isValid = !isNaN(value) && value.charAt(0) != "0" && value.length <= 2;
747 			else isValid = !isNaN(value) && value.length == 2
748 			if (isValid)
749 			{
750 				validated.valid = true;
751 				validated.unit = 'd';
752 				validated.value = value;
753 			}
754 			else
755 			{
756 				validated.valid = false;
757 			}
758 			break;
759 		case "y":
760 		case "yyyy":
761 			if (isNaN(value))
762 			{
763 				validated.valid = false;
764 			}
765 			else
766 			{
767 				validated.valid = true;
768 				validated.unit = 'y';
769 				validated.value = value;
770 			}
771 			break;
772 		case "M":
773 		case "MM":
774 			var parsedValue = parseInt(value, 10);
775 			var isValid;
776 			if (format == "M") isValid = !isNaN(value) && value.charAt(0) != "0" && value.length <= 2 && parsedValue >= 1 && parsedValue <= 12;
777 			else isValid = !isNaN(value) & value.length == 2 && parsedValue >= 1 && parsedValue <= 12;
778 			
779 			if (isValid)
780 			{
781 				validated.valid = true;
782 				validated.unit = 'm';
783 				validated.value = parsedValue - 1;
784 			}
785 			else
786 			{
787 				validated.valid = false;
788 			}
789 			break;
790 		case "MMM":
791 		case "NNN":
792 		case "E":
793 		case "EE":
794 			var names;
795 			if (format == "MMM") names = this.getLongMonthNames();
796 			else if (format == "NNN") names = this.getShortMonthNames();
797 			else if (format == "E") names = this.getShortDayNames();
798 			else names = this.getLongDayNames();
799 			var i;
800 			for (i = 0; i < names.length; i++)
801 			{
802 				var dateName = names[i];
803 				if (value.toLowerCase() == dateName.toLowerCase())
804 					break;
805 			}
806 			if (i < names.length)
807 			{
808 				validated.valid = true;
809 				if (format == "MMM" || format == "NNN") validated.unit = 'm';
810 				else validated.unit = 'dl';
811 				validated.value = i;
812 			}
813 			else
814 			{
815 				validated.valid = false;
816 			}
817 			break;
818 	}
819 	return validated;
820 }

825 nitobi.calendar.DatePicker.prototype.formatDate = function(date, mask)
826 {
827 	var parsedMask = {};	
828 	var year = date.getFullYear() + "";
829 	var month = date.getMonth() + 1 + "";
830 	var numDate = date.getDate() + "";
831 	var day = date.getDay();
832 	
833 	parsedMask["y"] = parsedMask["yyyy"] = year;
834 	parsedMask["yy"] = year.substring(2,4);
835 	
836 	parsedMask["M"] = month + "";
837 	parsedMask["MM"] = nitobi.lang.padZeros(month,2);
838 	parsedMask["MMM"] = this.getLongMonthNames()[month - 1];
839 	parsedMask["NNN"] = this.getShortMonthNames()[month - 1];
840 	
841 	parsedMask["d"] = numDate;
842 	parsedMask["dd"] = nitobi.lang.padZeros(numDate,2);
843 	
844 	parsedMask["EE"] =  this.getLongDayNames()[day];
845 	parsedMask["E"] = this.getShortDayNames()[day];
846 	
847 	var formattedDate = "";
848 	while (mask.length > 0)
849 	{
850 		var c = mask.charAt(0);
851 		var testExp = new RegExp(c + "+");
852 		// No check because we're guaranteed to get a result
853 		var currentMask = testExp.exec(mask)[0];
854 		formattedDate += parsedMask[currentMask] || currentMask;
855 		mask = mask.substring(currentMask.length);
856 	}
857 	return formattedDate;
858 }
