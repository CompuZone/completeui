 26 nitobi.lang.defineNs("nitobi.debug");
 27 
 28 nitobi.debug.PerfMon = {
 29 		funcs:[],
 30 		objects:[],
 31 		function_index:[],
 32 		report:"",
 33 		inline:[],
 34 		currentLevel:0,
 35 		callerFunc:[],
 36 
 37 		register: function(funcName, objRef)
 38 		{
 39 			if (objRef[funcName] instanceof Function)
 40 			{
 41 				objRef[funcName+'_base'] = objRef[funcName];
 42 				objRef[funcName] = function()
 43 				{
 44 					var args = [];
 45 					for (var i=0; i<arguments.length; i++)
 46 					{
 47 						args[i] = arguments[i];
 48 					}
 49 					
 50 					// This is the first call ...
 51 					// Get the current level as the previous plus 1
 52 					var callingFunc = nitobi.debug.PerfMon.callerFunc[nitobi.debug.PerfMon.callerFunc.length-1];
 53 					//nitobi.debug.PerfMon.currentLevel = nitobi.debug.PerfMon.currentLevel + 1;
 54 
 55 					// Set the caller method name for the next method.
 56 					nitobi.debug.PerfMon.callerFunc.push(funcName);
 57 
 58 var tabs = '';
 59 for (var i=0; i<nitobi.debug.PerfMon.currentLevel; i++)
 60 {
 61 	tabs += '\t';
 62 }
 63 
 64 nitobi.debug.PerfMon.report += "Calling " + funcName + "\tFrom " + callingFunc + "\t" + nitobi.debug.PerfMon.callerFunc.length + "\t" + tabs + "\t\n";
 65 
 66 					var s =new Date().getTime();
 67 
 68 					var retVal = objRef[funcName+'_base'].apply(objRef, args);
 69 					//	need to ave the results of the bench
 70 					//	check the caller ... somehow
 71 					nitobi.debug.PerfMon.callerFunc.pop();
 72 
 73 nitobi.debug.PerfMon.report += "Done " + funcName + "\tFrom " + callingFunc + "\t" + nitobi.debug.PerfMon.callerFunc.length + "\t" + tabs + "\t" + (new Date().getTime() - s) + "\n";
 74 					return retVal;
 75 				};
 76 			}
 77 		},
 78 		
 79 		registerAll: function(klass)
 80 		{
 81 			for (var item in klass)
 82 			{
 83 				if (item.indexOf('_base') > 0)
 84 					continue;
 85 				this.register(item, klass);
 86 			}
 87 		},
 88 
 89 		dispatch: function(args, funcName)
 90 		{
 91 
 92 		},
 93 
 94 		unregister: function(funcName, objRef)
 95 		{
 96 			objRef[funcName] = null;
 97 			objRef[funcName] = objRef[funcName+'_base'];
 98 			objRef[funcName+'_base'] = null;
 99 		},
100 
101 		start: function(id)
102 		{
103 			this.inline[id] = new Date().getTime();
104 		},
105 
106 		stop: function(id)
107 		{
108 			this.report += "Inline monitor: " + id + " - Duration: " + (new Date().getTime() - this.inline[id]) + "\n";
109 		}
110 }
