  3 /**
  4  * The textbox portion of the ComboBox
  5  * @class The textbox portion of the combo-box.
  6  * @constructor
  7  * @param {HTMLElement} userTag The HTML object that represents TextBox
  8  * @param {nitobi.combo.Combo} comboObject The combo object that is the owner of the textbox
  9  * @param {Boolean} hasButton true if the textbox has a button associated with it, false otherwise
 10  * @see nitobi.combo.Combo
 11  */
 12 nitobi.combo.TextBox = function(userTag, comboObject, hasButton)
 13 {
 14 	var DEFAULTCLASSNAME = "";
 15 	if (nitobi.browser.IE)
 16 	{
 17 		DEFAULTCLASSNAME="ntb-combobox-text-ie";
 18 	}
 19 	else
 20 	{
 21 		DEFAULTCLASSNAME="ntb-combobox-text-moz";
 22 	}
 23 	var DEFAULTWIDTH="100px";
 24 	var DEFAULTHEIGHT="";
 25 	var DEFAULTEDITABLE=true;
 26 	var DEFAULTVALUE="";
 27 	var DEFAULTDATAFIELDINDEX=0;
 28 	var DEFAULTSEARCHTERM="";
 29 	var DEFAULTONEDITKEYUP="";
 30 	this.SetCombo(comboObject);
 31 	
 32 	var oeku=(userTag ? userTag.getAttribute("OnEditKeyUpEvent") : null);
 33 	((null == oeku) || ("" == oeku))
 34 		? this.SetOnEditKeyUpEvent(DEFAULTONEDITKEYUP)
 35 		: this.SetOnEditKeyUpEvent(oeku);
 36 
 37 	var width=(userTag ? userTag.getAttribute("Width") : null);
 38 	((null == width) || ("" == width))
 39 		? this.SetWidth(DEFAULTWIDTH)
 40 		: this.SetWidth(width);
 41 
 42 	var height=(userTag ? userTag.getAttribute("Height") : null);
 43 	((null == height) || ("" == height))
 44 		? this.SetHeight(DEFAULTHEIGHT)
 45 		: this.SetHeight(height);
 46 
 47 	var ccn=(userTag ? userTag.getAttribute("CSSClassName") : null);
 48 	((null == ccn) || ("" == ccn))
 49 		? this.SetCSSClassName(DEFAULTCLASSNAME)
 50 		: this.SetCSSClassName(ccn);
 51 
 52 	var editable=(userTag ? userTag.getAttribute("Editable") : null);
 53 	((null == editable) || ("" == editable))
 54 		? this.SetEditable(DEFAULTEDITABLE)
 55 		: this.SetEditable(editable);
 56 
 57 	var value=(userTag ? userTag.getAttribute("Value") : null);
 58 	((null == value) || ("" == value))
 59 		? this.SetValue(DEFAULTVALUE)
 60 		: this.SetValue(value);
 61 
 62 	var dataTextField = comboObject.GetDataTextField();
 63 	if (dataTextField != null)
 64 		this.SetDataFieldIndex(comboObject.GetList().GetXmlDataSource().GetColumnIndex(dataTextField));
 65 	else{
 66 		var dfi=(userTag ? userTag.getAttribute("DataFieldIndex") : null);
 67 		((null == dfi) || ("" == dfi))
 68 			? this.SetDataFieldIndex(DEFAULTDATAFIELDINDEX)
 69 			: this.SetDataFieldIndex(dfi);
 70 	}
 71 
 72 	var st=(userTag ? userTag.getAttribute("SearchTerm") : null);
 73 	if ((null == st) || ("" == st)){
 74 		this.SetSearchTerm(DEFAULTSEARCHTERM);
 75 		this.SetIndexSearchTerm(DEFAULTSEARCHTERM);
 76 	}else{
 77 		this.SetSearchTerm(st);
 78 		this.SetIndexSearchTerm(st);
 79 	}
 80 	/**
 81 	 * @private
 82 	 * @ignore
 83 	 */
 84 	this.hasButton = hasButton;
 85 	/**
 86 	 * @private
 87 	 * @ignore
 88 	 */
 89 	this.m_userTag = userTag;
 90 }

389 nitobi.combo.TextBox.prototype.SmartSetValue = TextBox_SmartSetValue;
390 /**
391  * @private
392  */
393 function TextBox_SmartSetValue(Value, PutSmartListSeparator){
394 	var t = this.m_HTMLTagObject;
395 	var combo = this.GetCombo();
396 	var lio = t.value.lastIndexOf(combo.SmartListSeparator);
397 	if(lio > -1)
398 	{
399 		Value = t.value.substring(0,lio) + combo.SmartListSeparator + " " + Value;
400 	}
401 	if(PutSmartListSeparator)
402 	{
403 		Value += combo.SmartListSeparator + " ";
404 	}
405 	t.value = Value;
406 	this.m_TextValueTag.value = Value;
407 }

1011 /// <function name='OnKeyOperation' access='private'>
1012 /// <summary>Called when some kind of key event is fired.</summary>
1013 /// <param name='e' type='object' >Either an event object in Mozilla or null in IE, in which case window.event will be used.</param>

1019  */
1020 function TextBox_OnKeyOperation(e,EventType)

1023 	{
1024 		return;
1025 	}
1026 	// in Moz, e!=null, in IE e==null
1027 	e = e ? e : window.event;
1028 
1029 	// Types of events this function is linked to.
1030 	var EVENT_KEYUP=0;
1031 	var EVENT_KEYPRESS=1;
1032 	var EVENT_KEYDOWN=2;
1033 	// Various keys on the keyboard. Not ASCII.
1034 	var KEY_ENTER = 13;
1035 	var KEY_ESC = 27;
1036 	var KEY_TAB = 9;
1037 	var KEY_a = 65;
1038 	var KEY_z = 90;

1042 	var KEY_UP = 38;

1053 	var KEY_F12 = 123;

1055 	var KEY_CTRL = 17;

1060 	var t=this.m_HTMLTagObject;
1061 	var combo = this.GetCombo();
1062 	var list = combo.GetList();
1063 
1064 	var keyCode = e.keyCode;
1065 	combo.SetEventObject(e);
1066 	var dfi = this.GetDataFieldIndex();
1067 
1068 	// Separate the events. We'll deal with keys in one specific event only.
1069 	switch(EventType){
1070 		case(EVENT_KEYUP):
1071 		{
1072 			if(KEY_ENTER != keyCode && KEY_ESC != keyCode && KEY_TAB != keyCode
1073 				&& (keyCode < KEY_PGUP || keyCode > KEY_DOWN)
1074 				&& (keyCode < KEY_F1 || keyCode > KEY_F12)
1075 				&& (keyCode < KEY_SHIFT || keyCode > KEY_ALT))
1076 			{
1077 				// In smartsearch, the cache is always considered dirty because
1078 				// it searches anywhere inside a string.
1079 				// IS THIS TRUE? I don't think so!. TODO: look at this issue. it seems to
1080 				// me that the same widening rules apply. If you search for dog anywhere in
1081 				// the string, then hits for og and do will be in the cache. The one problem
1082 				// is that the page get may not be big enough. hmmmm...Does smartsearch support
1083 				// paging? We may be able to optimise here.
1084 				if (combo.mode=="smartsearch" || combo.mode=="smartlist" || combo.mode=="filter" || combo.mode=="compact")
1085 				{
1086 					list.GetXmlDataSource().m_Dirty=true;
1087 				}
1088 				this.OnChanged(e);
1089 				eval(this.GetOnEditKeyUpEvent());
1090 			}
1091 			if(keyCode==KEY_UP || keyCode==KEY_DOWN || keyCode==KEY_PGUP || keyCode==KEY_PGDN || keyCode==KEY_ENTER){
1092 				if(this.smartlistWA==true)
1093 					this.smartlistWA=false;
1094 				else
1095 				{
1096 				// 2005.04.26
1097 				// code below doesn't work for IE; but since we're going to be at the end of the line
1098 				// anyways, let's use the t.value=t.value hack (moves cursor to the end of the line)

1100 						t.value=t.value;
1101 					else
1102 						nitobi.html.setCursor(t,t.value.length);
1103 				}
1104 			}
1105 			if(combo.mode=="smartlist" && keyCode==KEY_ENTER && list.GetActiveRow()!=null)
1106 			{
1107 				this.SetValue(list.GetSelectedRowValues()[this.GetDataFieldIndex()], true);
1108 				list.SetActiveRow(null);
1109 			}
1110 // 2005.04.27
1111 // due to time constraints, we had to settle w/ this workaround - please fix if you think it's necessary
1112 			if(combo.mode=="smartlist"){
1113 				var lio = t.value.lastIndexOf(combo.SmartListSeparator);
1114 				if(this.lio != lio)
1115 					list.Hide();
1116 				this.lio = lio;
1117 			}
1118 			break;
1119 		}
1120 		case(EVENT_KEYDOWN):
1121 		{
1122 			switch(keyCode){
1123 				case(KEY_ENTER):
1124 				{
1125 // 2005.04.27
1126 // due to time constraints, we had to settle w/ this workaround - please fix if you think it's necessary
1127 					if(combo.mode=="smartlist"){

1129 						if(lio != this.lio){
1130 							list.Hide();
1131 							break;
1132 						}
1133 					}
1134 					this.m_skipBlur=true;
1135 					list.SetActiveRowAsSelected();
1136

1138 					t.focus();
1139 					
1140 					// Always call this when the user has pressed enter.
1141 					eval(combo.GetOnSelectEvent());
1142 					// Since this function is called by multiple different events,
1143 					// only trigger this code on keydown.
1144 					nitobi.html.cancelEvent(e);
1145 					this.m_skipBlur=false;
1146 					break;
1147 				}
1148 				case(KEY_TAB):
1149 				{
1150 					list.Hide();
1151 					eval(combo.GetOnTabEvent());
1152 					
1153 					
1154 					// Check to see if the conditions that prevent blurring 
1155 					// are true. If so, set them to false.
1156 					// Otherwise, in some cases ontab is called but onblur is not.
1157 					// This is most noticeable when the mouse is over the list and tab is pressed.
1158 					if (this.m_skipBlur || combo.m_Over)
1159 					{
1160 						this.m_skipBlur = false;
1161 						combo.m_Over = false;
1162 					}

1164 					eval(combo.GetOnSelectEvent());

1166 				}

1169 					// Since this function is called by multiple different events,
1170 					// only trigger this code on keydown.
1171 					list.Hide();
1172 					break;
1173 				}
1174 				case(KEY_UP):
1175 				{
