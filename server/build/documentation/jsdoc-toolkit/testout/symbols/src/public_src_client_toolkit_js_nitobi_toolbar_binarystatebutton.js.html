 88 /**
 89  * A button that has a binary state. It is other switched on or off.
 90  * @constructor
 91  * @extends nitobi.ui.Button
 92  * @param xml {string} XML that defines the button.
 93  * @param id {string} The id of the button.
 94  */
 95 nitobi.ui.BinaryStateButton = function (xml, id)
 96 {
 97 	this.initialize(xml,nitobi.ui.BinaryStateButtonXsl,id);
 98 	/**
 99 	 * @ignore
100 	 */
101 	this.m_Checked=false;
102 }
103 
104 nitobi.lang.extend(nitobi.ui.BinaryStateButton, nitobi.ui.Button);
105 
106 /**
107  * Checks to see if the button is checked or not.
108  * @return {bool} true if the button is checked and false otherwise.
109  */
110 nitobi.ui.BinaryStateButton.prototype.isChecked = function()
111 {
112 	return this.m_Checked;
113 }
114 
115 /**
116  * Puts the button in the checked state.
117  */
118 nitobi.ui.BinaryStateButton.prototype.check = function()
119 {
120 	var button = this.getHtmlElementHandle();
121 	button.className = "ntb-button-checked";
122 	this.m_Checked=true;
123 }
124 
125 /**
126  * Puts the button in the unchecked state.
127  */
128 nitobi.ui.BinaryStateButton.prototype.uncheck = function()
129 {
130 	var button = this.getHtmlElementHandle();
131 	button.className = "ntb-button";
132 	this.m_Checked=false;
133 }
134 
135 /**
136  * Toggles the state of the button.
137  */
138 nitobi.ui.BinaryStateButton.prototype.toggle = function()
139 {
140 	var button = this.getHtmlElementHandle();
141 	if (button.className == "ntb-button-checked")
142 	{
143 		this.uncheck();
144 	}
145 	else
146 	{
147 		this.check();
148 	};
149 }
