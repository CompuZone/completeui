1 /**
  2  * @class Just-in-time paging as the user scrolls through data. 
  3  * <p>
  4  * <b>Fixed Size</b><br/>
  5  * 	Height of grid specified explicitly<br/>
  6  * 	Vertical scrollbar allows user to scroll through rendered rows<br/>
  7  * 	If there are not enough rows to fill the grid, a blank area will be present, the grid will not shrink.
  8  * </p> 
  9  * <p>
 10  * <b>On-Demand Data</b><br/>
 11  * 	Data in the client-side datasource is a subset of server-side datasource records.<br/>
 12  * 	The grid renders visible rows present in the client-side datasource that match the specified 
 13  *  filter criteria.<br/>
 14  * 	Sorting is performed server-side - client-side data is purged.<br/>
 15  * 	Grid rows are rendered in multiple render operations (as the user scrolls).<br/>
 16  * 	Data is loaded on-demand as it is needed. 
 17  * </p>
 18  * <p>
 19  * <b>Virtual Paging</b> (Paging is hidden from user)<br/> 
 20  * 	Paging operations request data from server if it is not present in the client-side datasource.<br/> 
 21  * 	Client-side data is kept across paging requests.<br/>
 22  * 	Rendered rows are kept across paging requests.<br/>
 23  * </p>
 24  * @constructor
 25  * @param {String} uid The unique ID of the Grid.
 26  */
 27 nitobi.grid.GridLiveScrolling = function(uid) {
 28 	nitobi.grid.GridLiveScrolling.baseConstructor.call(this, uid);
 29 	this.mode = "livescrolling";
 30 }
 31 nitobi.lang.extend(nitobi.grid.GridLiveScrolling, nitobi.grid.TreeGrid);
 32 
 33 nitobi.grid.GridLiveScrolling.prototype.createChildren=function() {
 34 	var args = arguments;
 35 
 36 	nitobi.grid.GridLiveScrolling.base.createChildren.call(this,args);
 37 
 38 	nitobi.grid.GridLiveScrolling.base.createToolbars.call(this,nitobi.ui.Toolbars.VisibleToolbars.STANDARD);	
 39 }
 40 
 41 nitobi.grid.GridLiveScrolling.prototype.bind=function() 
 42 {
 43 	nitobi.grid.GridStandard.base.bind.call(this);
 44 
 45 	if (this.getGetHandler()!='') {
 46 		this.ensureConnected();
 47 		var rows = this.getRowsPerPage();
 48 		if (this.datatable.mode == 'local')
 49 			rows = null
 50 		this.datatable.get(0, rows, this, this.getComplete);
 51 	} else {
 52 		// TODO: if we have created the datasource and loaded it with data BEFORE connecting to it
 53 		// the rowcount changed events etc will not have propagated to the Grid and so we will
 54 		// have an incorrect row count.
 55 		this.finalizeRowCount(this.datatable.getRemoteRowCount());
 56 		this.bindComplete();
 57 	}
 58 }

 63 nitobi.grid.GridLiveScrolling.prototype.getComplete=function(evtArgs)
 64 {
 65 	nitobi.grid.GridLiveScrolling.base.getComplete.call(this, evtArgs);
 66 
 67 	// No need to set the row count here since dataTable events take care of it.
 68 
 69 	// TODO: This is not needed since the connected datatable will fire all the
 70 	// right events on the Grid when it gets the data back.
 71 	if (!this.columnsDefined)
 72 	{
 73 		this.defineColumnsFinalize();
 74 	}
 75 
 76 	this.bindComplete();
 77 }
 78 
 79 nitobi.grid.GridLiveScrolling.prototype.pageSelect= function(dir)
 80 {
 81 	var rowRange = this.Scroller.getUnrenderedBlocks();
 82 	var rows = rowRange.last - rowRange.first;
 83 	this.reselect(0, rows * dir);
 84 }
 85 nitobi.grid.GridLiveScrolling.prototype.page= function(dir)
 86 {
 87 	//	page needs to retrieve and render the next page of data 
 88 	//	and then set the active cell and the selection box to the 
 89 	//	corresponding cell in the next page of data.
 90 	var rowRange = this.Scroller.getUnrenderedBlocks();
 91 	var rows = rowRange.last - rowRange.first;
 92 	this.move(0, rows * dir);
 93 }
