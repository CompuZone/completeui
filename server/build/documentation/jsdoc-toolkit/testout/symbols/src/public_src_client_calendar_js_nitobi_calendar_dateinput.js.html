  3 /**
  4  * Creates the input portion of the DatePicker.
  5  * @class The input portion of the DatePicker allows the user the input dates directly.
  6  * @constructor
  7  * @param {Node} element
  8  * @extends nitobi.ui.Element
  9  * @see nitobi.calendar.DatePicker
 10  */
 11 nitobi.calendar.DateInput = function(element) 
 12 {
 13 	nitobi.calendar.DateInput.baseConstructor.call(this, element);
 14 	
 15 	/**
 16 	 * Fires when the input if blurred.
 17 	 * @type nitobi.base.Event
 18 	 */
 19 	this.onBlur = new nitobi.base.Event();
 20 	this.eventMap["blur"] = this.onBlur;
 21 	
 22 	/**
 23 	 * Fires when the input if focussed.
 24 	 * @type nitobi.base.Event
 25 	 */
 26 	this.onFocus = new nitobi.base.Event();
 27 	this.eventMap["focus"] = this.onFocus;
 28 	
 29 	/**
 30 	 * Used to attach/detach html events.
 31 	 * @private
 32 	 */
 33 	this.htmlEvents = [];
 34 	
 35 	this.subscribeDeclarationEvents();
 36 }

 38 nitobi.lang.extend(nitobi.calendar.DateInput, nitobi.ui.Element);

 40 nitobi.calendar.DateInput.profile = new nitobi.base.Profile("nitobi.calendar.DateInput",null,false,"ntb:dateinput");
 41 nitobi.base.Registry.getInstance().register(nitobi.calendar.DateInput.profile);

 43 /**
 44  * Attachs the necessary events to the input.
 45  * @private
 46  */
 47 nitobi.calendar.DateInput.prototype.attachEvents = function()
 48 {
 49 	var he = this.htmlEvents;
 50 	he.push({type: "focus", handler: this.handleOnFocus});
 51 	he.push({type: "blur", handler: this.handleOnBlur});
 52 	he.push({type: "keydown", handler: this.handleOnKeyDown});
 53 	nitobi.html.attachEvents(this.getHtmlNode("input"), he, this);
 54 }
 55 
 56 nitobi.calendar.DateInput.prototype.detachEvents = function()
 57 {
 58 	nitobi.html.detachEvents(this.getHtmlNode("input"), this.htmlEvents);
 59 }

 82 /**
 83  * @private
 84  */
 85 nitobi.calendar.DateInput.prototype.handleOnFocus = function()
 86 {
 87 	var inputMask = this.getEditMask();
 88 	var datePicker = this.getParentObject();
 89 	var selectedDate = datePicker.getSelectedDate();
 90 	if (selectedDate)
 91 	{
 92 		var value = datePicker.formatDate(selectedDate, inputMask);
 93 		this.setValue(value);
 94 		var datePicker = this.getParentObject();
 95 		datePicker.blurInput = true;
 96 	}
 97 	this.onFocus.notify(new nitobi.ui.ElementEventArgs(this, this.onFocus));
 98 }
 99 
100 /**
101  * @private
102  */
103 nitobi.calendar.DateInput.prototype.handleOnBlur = function()
104 {
105 	var datePicker = this.getParentObject();
106 	var calendar = datePicker.getCalendar();
107 	if (datePicker.blurInput)
108 	{
109 		var editMask = this.getEditMask();
110 		var newDate = this.getValue();
111 		newDate = datePicker.parseDate(newDate, editMask);
112 		if (datePicker.validate(newDate))
113 		{
114 			datePicker._setSelectedDate(newDate, true);
115 			if (calendar)
116 				calendar.hide();
117 		}
118 		else
119 		{
120 			if (calendar)
121 				calendar.clearHighlight();
122 			datePicker.clear();
123 			this.setInvalidStyle(true);
124 		}
125 	}
126 	this.onBlur.notify(new nitobi.ui.ElementEventArgs(this, this.onBlur));
127 }

129 /**
130  * Just to allow the user to hit "enter" to set the new date.
131  * @private
132  */
133 nitobi.calendar.DateInput.prototype.handleOnKeyDown = function(event)
134 {
135 	var key = event.keyCode;
136 	if (key == 13)
137 	{
138 		this.getHtmlNode("input").blur();
139 	}
140 }

142 /**
143  * Sets the input to the invalid style.
144  * @param {Boolean} isInvalid true to set the input to the invalid style, false to return the input to the normal style.
145  * @private
146  */
147 nitobi.calendar.DateInput.prototype.setInvalidStyle = function(isInvalid)
148 {
149 	var Css = nitobi.html.Css;
150 	var container = this.getHtmlNode("container");
151 	if (isInvalid)
152 		Css.swapClass(container, "ntb-inputcontainer", "ntb-invalid");
153 	else
154 		Css.swapClass(this.getHtmlNode("container"), "ntb-invalid", "ntb-inputcontainer");
155 
156 	var bgColor = Css.getStyle(container, "backgroundColor");
157 	var input = this.getHtmlNode("input");
158 	Css.setStyle(input, "backgroundColor", bgColor);
159 }
