370 	var mS = nitobi.callout.Callout.getScrollPosition();
371 	
372 	
373 	var newScrollY = mS.scrollTop;
374 	var newScrollX = mS.scrollLeft;
375 	if (y1<mS.scrollTop)
376 		var newScrollY = y1-(mS.clientHeight/4.5);
377 
378 	if (x1<mS.scrollLeft)
379 		var newScrollX = x1-(mS.clientWidth/4.5);
380 		
381 	if (y2>(mS.scrollTop+mS.clientHeight))
382 		var newScrollY = y2-mS.clientHeight+(mS.clientHeight/4.5);		
383 
384 	if (x2>(mS.scrollLeft+mS.clientWidth))
385 		var newScrollX = x2-mS.clientWidth+(mS.clientWidth/4.5);
386 		
387 	if (newScrollY < 0)
388 		newScrollY = 0;
389 		
390 	if (newScrollX < 0)
391 		newScrollX = 0;
392 		
393 	if (newScrollY > (mS.bodyHeight-mS.clientHeight))
394 		newScrollY = (mS.bodyHeight-mS.clientHeight+10);
395 
396 	if (newScrollX > (this.oS.bodyWidth-this.oS.clientWidth))
397 		newScrollX = (this.oS.bodyWidth-this.oS.clientWidth);	
398 				
399 
400 	this.targetScrollX = newScrollX; this.targetScrollY = newScrollY;
401 	if (!this.inScroll) {
402 		this.startingScrollX = mS.scrollLeft;
403 		this.startingScrollY = mS.scrollTop;
404 		this.inScroll = true;
405 	}
406 	//console.log('a:' + this.scrollX);
407 	this.scrollX = mS.scrollLeft;
408 	this.scrollY = mS.scrollTop;
409 	this.iterateScroller();
410 	//window.scrollTo(newScrollX,newScrollY);
411 	} else {
412 		try {
413 		this.onfindscroll.call();	
414 		} catch(e) {}
415 	}

421 nitobi.spotlight.Spotlight.prototype.iterateScroller = function() {
422 
423 	var scrollxVelocity = 0;
424 	var scrollyVelocity = 0;
425 	var ids = this;
426 	var scrollxAmplitude = 10;
427 	var scrollyAmplitude = 10;
428 	var scrollxIterator = 0;
429 	var scrollyIterator = 0;
430 	
431 	if ((this.targetScrollX-this.scrollX) > 0)
432 		scrollxVelocity = 1.1;
433 	if ((this.targetScrollX-this.scrollX) < 0)
434 		scrollxVelocity = -1.1;
435 
436 	if ((this.targetScrollY-this.scrollY) > 0)
437 		scrollyVelocity = 1.1;
438 	if ((this.targetScrollY-this.scrollY) < 0)
439 		scrollyVelocity = -1.1;
440 	
441 	scrollxAmplitude = (Math.abs(this.targetScrollX-this.startingScrollX)/20)*0.8;
442 	scrollyAmplitude = (Math.abs(this.targetScrollY-this.startingScrollY)/20)*0.8;
443 	scrollxIterator = (this.scrollX-this.startingScrollX)/(this.targetScrollX-this.startingScrollX+0.0001);
444 	scrollyIterator = (this.scrollY-this.startingScrollY)/(this.targetScrollY-this.startingScrollY+0.0001);
445 	
446 	
447 	scrollxVelocity = (this.easeInEaseOut(scrollxAmplitude, 1, scrollxIterator)*scrollxVelocity);
448 	scrollyVelocity = (this.easeInEaseOut(scrollyAmplitude, 1, scrollyIterator)*scrollyVelocity);
449 	
450 	//console.log(mS.scrollTop + ':' + scrollyVelocity);
451 	this.scrollX += scrollxVelocity;
452 	this.scrollY += scrollyVelocity;
453 	
454 	if (this.scrollY < 0)
455 		this.scrollY = 0;
456 		
457 	if (this.scrollX < 0)
458 		this.scrollX = 0;	
459 	
460 	window.scrollTo(this.scrollX,this.scrollY);
461 	
462 	
463 	if ((Math.abs(this.targetScrollY-this.scrollY) > 3) || (Math.abs(this.targetScrollX-this.scrollX) > 3)) {
464 //if ((scrollyIterator < 0.90) || (scrollxIterator < 0.90)) {
465 		this.scrollTimer = setTimeout(function(){ids.iterateScroller()}, 15);
466 	} else {this.inScroll = false; try {this.onfindscroll.call();}catch(e){}}
467 //this.scrollTimer

512 /**
513  * @private
514  */
515 nitobi.spotlight.Spotlight.prototype.lensMagnify = function(objC) {
516 	var resObj;
517 	resObj = objC;
518 	if (this.useLens) {
519 		resObj.x -= parseInt(objC.w*this.lensOversize);
520 		resObj.y -= parseInt(objC.h*this.lensOversize);
521 		resObj.w += parseInt(objC.w*(this.lensOversize*2));
522 		resObj.h += parseInt(objC.h*(this.lensOversize*2)); 
523 	}	
524 	return resObj;
525 }
