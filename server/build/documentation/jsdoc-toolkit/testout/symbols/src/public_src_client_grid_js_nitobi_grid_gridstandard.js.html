1 /**
  2  * @class Fixed-height grid with remote paging that does not scroll 
  3  * 
  4  * <p>
  5  * <b>Stretch</b><br/> 
  6  * 	Miniumum height of grid specified explicitly.<br/>
  7  * 	Maximum height of grid specified explicitly.<br/> 
  8  * 	Grid height stretches and shrinks to accomodate rows rendered.   
  9  * 	If there are not enough rows to fill the grid (based on min height), a blank area will be present.<br/> 
 10  * 	Vertical scrollbar appears once max height has been exceeded.
 11  * </p>
 12  * <p>
 13  * <b>Server-side and Client-Side Data</b><br/> 
 14  * 	Data in the client-side datasource is a subset of server-side datasource records.   
 15  * 	The grid renders all records present in the client-side datasource that match the 
 16  *  specified filter criteria.<br/>
 17  * 	Sorting is performed server-side - client-side data is purged 
 18  * </p>
 19  * <p>
 20  * <b>Server-side Paging</b><br/> 
 21  * 	Paging operations request data from the server.<br/>
 22  * 	Single page of rows is rendered for each paging request.<br/> 
 23  * 	Client-side data is purged before each paging request.<br/> 
 24  * 	Rendered rows are purged before each paging request. 
 25  * </p>
 26  * @constructor
 27  * @param {String} uid The unique ID of the Grid.
 28  */
 29 nitobi.grid.GridStandard = function(uid) {
 30 	nitobi.grid.GridStandard.baseConstructor.call(this, uid);
 31 	this.mode="standard";
 32 //	this.PagingMode="standard";	//0 - None | 1 - Standard | 2 - LiveScrolling
 33 	this.setPagingMode(nitobi.grid.PAGINGMODE_STANDARD);
 34 //	this.DataMode="paging";		//0 - Unbound | 1 - Static | 2 - Paging | 3 - Caching
 35 // TODO: ensure that properties with setters are using the setters
 36 	this.setDataMode(nitobi.data.DATAMODE_PAGING);		//0 - Unbound | 1 - Static | 2 - Paging | 3 - Caching
 37 }
 38 nitobi.lang.extend(nitobi.grid.GridStandard, nitobi.grid.Grid);

 40 nitobi.grid.GridStandard.prototype.createChildren=function() {
 41 	var args = arguments;
 42 
 43 	nitobi.grid.GridStandard.base.createChildren.call(this,args);
 44 
 45 	nitobi.grid.GridStandard.base.createToolbars.call(this, nitobi.ui.Toolbars.VisibleToolbars.STANDARD | nitobi.ui.Toolbars.VisibleToolbars.PAGING);
 46 
 47 	// Attach events
 48 	this.toolbars.subscribe("NextPage",nitobi.lang.close(this,this.pageNext));
 49 	this.toolbars.subscribe("PreviousPage",nitobi.lang.close(this,this.pagePrevious));
 50 	this.subscribe("EndOfData",this.disableNextPage); 
 51 	this.subscribe("TopOfData",this.disablePreviousPage);
 52 	this.subscribe("NotTopOfData",this.enablePreviousPage); 
 53 	this.subscribe("NotEndOfData",this.enableNextPage);
 54 	this.subscribe("TableConnected", nitobi.lang.close(this, this.subscribeToRowCountReady));
 55 }

 57 nitobi.grid.GridStandard.prototype.connectToTable = function(table)
 58 {
 59 	if (nitobi.grid.GridStandard.base.connectToTable.call(this, table) != false)
 60 	{
 61 		this.datatable.subscribe("RowInserted",nitobi.lang.close(this,this.incrementDisplayedRowCount));
 62 		this.datatable.subscribe("RowDeleted",nitobi.lang.close(this,this.decrementDisplayedRowCount));	
 63 	}
 64 }

122 nitobi.grid.GridStandard.prototype.pagePrevious=function() {
123 	this.fire("BeforeLoadPreviousPage");
124 	this.loadDataPage(Math.max(this.getCurrentPageIndex()-1,0));
125 	this.fire("AfterLoadPreviousPage");
126 }

128 nitobi.grid.GridStandard.prototype.pageNext=function() {
129 	this.fire("BeforeLoadNextPage");
130 	this.loadDataPage(this.getCurrentPageIndex()+1);
131 	this.fire("AfterLoadNextPage");
132 }
133 nitobi.grid.GridStandard.prototype.loadDataPage = function(newPageNumber) 
134 {
135 	// Clear the selection if there is one.
136 	this.fire('BeforeLoadDataPage');
137 
138 	//	Check if the nePageNumber is greater than -1
139 	if (newPageNumber > -1) 
140 	{
141 		if (this.sortColumn)
142 		{
143 			if (this.datatable.sortColumn)
144 			{
145 				for (var i = 0; i < this.getColumnCount(); i++)
146 				{
147 					var colObj = this.getColumnObject(i);
148 					if (colObj.getColumnName() == this.datatable.sortColumn)
149 					{
150 						this.setSortStyle(i,this.datatable.sortDir);
151 						break;
152 					}
153 				}
154 			}
155 			else
156 			{
157 				this.setSortStyle(this.sortColumn.column, "", true);
158 			}
159 		}
160 		this.setCurrentPageIndex(newPageNumber);
161 		var	startRow = this.getCurrentPageIndex()*this.getRowsPerPage(); //freezeTop not supported in standard paging (cuz how would that work? what rows would be frozen?)
162 		var rows = this.getRowsPerPage()-this.getfreezetop();
163 //		var endRow = startRow+rows-1;
164 //		this.datatable.get(startRow, endRow, this, this.afterLoadDataPage);
165 		this.datatable.flush();
166 		this.datatable.get(startRow, rows, this, this.afterLoadDataPage);
167 	}
168 	// Resize the grid if necessary, show/hide scrollbars
169 	// Set focus to topleft cell
170 	this.fire('AfterLoadDataPage');
171 }
172 nitobi.grid.GridStandard.prototype.afterLoadDataPage=function(eventArgs) 
173 {
174 	this.setDisplayedRowCount(eventArgs.numRowsReturned);
175 	this.setRowCount(eventArgs.numRowsReturned)
176 	if(eventArgs.numRowsReturned != this.getRowsPerPage()) {
177 		this.fire("EndOfData");
178 	} else {
179 		this.fire("NotEndOfData");
180 	}
181 	if(this.getCurrentPageIndex() == 0) {
182 		this.fire("TopOfData");
183 	} else {
184 		this.fire("NotTopOfData");
185 	}
186 	this.clearSurfaces();
187 	this.updateCellRanges();
188 	this.scrollVertical(0);
189 
190 //	this.Scroller.render();
191 
192 // There is no need to explicitly call render here as the render is taken care of by the scroller...
193 //	this.Scroller.view.midleft.renderGap(startRow, endRow, false);
194 //	this.Scroller.view.midcenter.renderGap(startRow, endRow, false);
195 }
