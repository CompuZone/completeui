225 /**
226  * @ignore
227  */
228 nitobi.tree.Tree.prototype.handleEvent = function(eventArgs)
229 {
230 	var event = eventArgs.htmlEvent;
231 	var idProperties = nitobi.ui.Element.parseId(eventArgs.targetId);
232 	
233 	// Cut right to the specified object for a click event:
234 	if (event.type === 'mousedown')
235 	{
236 		var targetNode = this.find('id',idProperties.id)[0];
237 		if (idProperties.localName === 'chooser')
238 		{
239 			this.setSelected([targetNode]);
240 			targetNode.notify(event,eventArgs.targetId);
241 		}
242 		else if (idProperties.localName === 'junction')
243 		{
244 			targetNode.toggleChildren();
245 		}
246 		else
247 		{
248 			targetNode.notify(event,eventArgs.targetId);
249 		}
250 	}
251 	else if (event.type === 'mouseover')
252 	{
253 		if (this.getHoverHighlight())
254 		{
255 			var el = $(idProperties.id+'.css');
256 			if (el)
257 			{
258 				nitobi.html.Css.addClass(el,'hover');
259 			}
260 		}
261 	}
262 	else if (event.type === 'mouseout')
263 	{
264 		if (this.getHoverHighlight())
265 		{
266 			var el = $(idProperties.id+'.css');
267 			if (el)
268 			{
269 				nitobi.html.Css.removeClass(el,'hover');
270 			}
271 		}
272 	}
273 	else if (event.type === 'keydown')
274 	{
275 		var key = event.keyCode;
276 		var keyMap = nitobi.tree.Tree.keyMap;
277 		var selected = this.getSelected();
278 		if (selected.length)
279 		{
280 			var currentNode = selected[0];
281 			switch(key)
282 			{
283 				case keyMap.LEFT:
284 					nitobi.html.cancelEvent(event);
285 					var c = currentNode.getChildren();
286 					if (c && c.isVisible())
287 					{
288 						currentNode.toggleChildren();
289 					}
290 					else
291 					{
292 						var p = currentNode.getParent(); 
293 						if (p)
294 						{
295 							this.setSelected([p]);
296 						}
297 					}
298 					break;
299 				case keyMap.RIGHT:
300 					nitobi.html.cancelEvent(event);
301 					var c = currentNode.getChildren();
302 					if (c && c.getLength() && c.isVisible())
303 					{
304 						this.setSelected([c.get(0)]);
305 					}
306 					else
307 					{
308 						currentNode.toggleChildren();
309 					}
310 					break;
311 				case keyMap.UP:
312 					nitobi.html.cancelEvent(event);
313 					var p = currentNode.getParent();
314 					var c = currentNode.getParentObject();
315 					var i = c.indexOf(currentNode);
316 					if (i)
317 					{
318 						this.setSelected([c.get(i-1).getFurthestVisibleDescendent()]);
319 					}
320 					else if (p)
321 					{
322 						this.setSelected([p]);
323 					}
324 					break;
325 				case keyMap.DOWN:
326 					nitobi.html.cancelEvent(event);
327 					var c = currentNode.getChildren();
328 					
329 					if (c && c.getLength() && c.isVisible())
330 					{
331 						this.setSelected([c.get(0)]);
332 					}
333 					else
334 					{
335 						var c = null;
336 						while(currentNode && (c = currentNode.getParentObject()))
337 						{
338 							var i = c.indexOf(currentNode);
339 							if (i < c.getLength()-1)
340 							{
341 								this.setSelected([c.get(i+1)])
342 								break;
343 							}
344 							currentNode = currentNode.getParent();
345 						}
346 					}
347 					break;
348 			}
349 		}
350 		else
351 		{
352 			var root = this.getRoot();
353 			if (root)
354 			{
355 				this.setSelected([root]);
356 			}
357 		}
358 	}

463 /**
464  * Sets the currently selected nodes in the tree.
465  * @param {Array} selected an array of {@link nitobi.tree.Node} objects.
466  * @private
467  */
468 nitobi.tree.Tree.prototype.setSelected = function(selected)
469 {
470 	var p = null;
471 	
472 	while (p = this.selected.pop())
473 	{
474 		try {
475 			p.deselect();
476 			this.onDeselect.notify(new nitobi.ui.ElementEventArgs(this,this.onDeselect,p.getId()));
477 		}
478 		catch (e)
479 		{
480 			// These objects may or may not still exist.
481 		}		
482 	}
483 	
484 	p = null;
485 	var tabindexSet = false; 
486 	while (p = selected.pop())
487 	{
488 		if (typeof(p) === 'string' )
489 		{
490 			p = this.find('id', p)[0];
491 		}
492 		this.selected.push(p);
493 
494 		// Frame Targeting
495 		var targetFrame = this.getTargetFrame();
496 		var url = p.getUrl();
497 		if (targetFrame && targetFrame != "" && url && url != "")
498 		{
499 			var frame = eval('parent.'+targetFrame);
500 			if (!frame)
501 			{
502 				frame = $(targetFrame);
503 				if (frame)
504 				{
505 					frame.src = url;
506 				}
507 			}
508 			else
509 			{
510 				frame.location.href = url;
511 			}
512 		}
513 		// Set the tab index
514 		if (tabindexSet == false)
515 		{
516 			p.select(this.getTabIndex());
517 			var scrollerNode = this.getHtmlNode("scroller");
518 			if (scrollerNode.tabIndex != -1)
519 			{
520 				scrollerNode.tabIndex = -1;
521 			}
522 			tabIndexSet = true;  
523 		}
524 		else
525 		{
526 			p.select();
527 		}
528 		this.onSelect.notify(new nitobi.ui.ElementEventArgs(this,this.onSelect,p.getId()));
529 	}	
530 };

532 /**
533  * Returns an <CODE>Array</CODE> of the nodes matching the given key/value pair.
534  * Typical usage would be to find a node with a known id:
535  * <div class="code">
536  * <pre><CODE>
537  * 	var myNode = tree.find('id','myNodeId')[0];
538  * </CODE></pre>
539  * </div>
540  * Or to find all the nodes of a specific type:
541  * <div clas="code">
542  * <pre><CODE>
543  * 	var fooNodes = tree.find('type','foo');
544  * </CODE></pre>
545  * </div>
546  * Note: The running time for this method increases linearly with the number of nodes found.
547  * @param {String} key
548  * @param {String} value
549  */
550 nitobi.tree.Tree.prototype.find = function(key, value)
551 {
552 	if (typeof(value) === 'string')
553 	{
554 		value = "'" + value + "'";
555 	}
556 	var found = new Array();
557 	var c = this.getChildren();
558 	if (c)
559 	{
560 		var xmlNode = c.getXmlNode();
561 
562 		var located = xmlNode.selectNodes('//*[@'+key+'='+value+']');
563 		for (var i = 0; i < located.length; i++)
564 		{
565 			var node = located[i];
566 			var locator = '';
567 			while (node)
568 			{
569 				var name = node.nodeName;
570 				if (name.indexOf('node') > -1)
571 				{
572 					locator = ".getById('"+node.getAttribute("id")+"')" + locator;
573 				}
574 				else if (name.indexOf('children') > -1)
575 				{
576 					locator = ".getChildren()" + locator;
577 				}
578 				else if (name.indexOf('tree') > -1)
579 				{
580 					locator = "this" + locator;
581 					break;
582 				}
583 				else
584 				{
585 					nitobi.lang.throwError("Unexpected and Invalid XML Structure");
586 				}
587 				node = node.parentNode;
588 			}
589 			if (locator !== '')
590 			{
591 				found.push(eval(locator));
592 			}
593 		}
594 	}
595 	return found;
596 };

758 /**
759  * Translates data from the old backend format to the <CODE>children->node</CODE> format
760  * used by Tree.  This is used by the tree to transform data returned from the server into the
761  * XML format used by the tree (ntb:children -> ntb:node ...).  If this function is re-implemented it can serve
762  * as a translator for any kind of XML data as long as it returns a document in the tree's native XML format. 
763  * @param {XMLDocument} xmlDocument the old-backend formatted XML document.
764  * @return the translated XMLDocument
765  * @type XMLDocument 
766  */
767 nitobi.tree.Tree.translateData = function(xmlDocument)
768 {
769 	if (xmlDocument.documentElement.nodeName == 'root')
770 	{
771 		// Add a <fields> node that allows attribute-name lookup for the attributes. 
772 		
773 		var fieldNames = xmlDocument.documentElement.getAttribute('fields');
774 		fieldNames = fieldNames.split("|");
775 		fieldKeys = new Array();
776 		for (var i = 0; i < fieldNames.length; i++)
777 		{
778 			fieldKeys[i] = nitobi.lang.numToAlpha(i);
779 		}
780 		var fields = xmlDocument.createElement('fields');
781 		for (var i = 0; i < fieldNames.length; i++)
782 		{
783 			fields.setAttribute(fieldKeys[i],fieldNames[i]);
784 		}
785 		xmlDocument.documentElement.appendChild(fields);
786 		
787 		// Now do the transformation
788 		return nitobi.xml.transformToXml(xmlDocument,nitobi.tree.dataTranslator,'xml');
789 	}
790 	else
791 	{
792 		return xmlDocument;
793 	}
794 };

796 /**
797  * Precaches images found in the tree's stylesheets.  Calling this function before loading
798  * a tree component will ensure that all the tree's images are loaded before the tree is rendered.
799  * @param {String} url an optional url for your own tree css file (just the filename)
800  */
