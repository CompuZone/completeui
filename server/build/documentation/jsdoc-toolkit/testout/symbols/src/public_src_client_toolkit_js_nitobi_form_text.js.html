1 /**
  2  * Text is a text editor class that is implemented using an HTML input tag
  3  * @class
  4  * @constructor
  5  */
  6 nitobi.form.Text = function()
  7 {
  8 	nitobi.form.Text.baseConstructor.call(this);
  9 
 10 	var ph = this.placeholder;
 11 	ph.setAttribute("id","text_span");
 12 	// TODO: this is required for proper aligment of the editor
 13 	ph.style.top="-3000px";
 14 	ph.style.left="-3000px";
 15 
 16 	// width is required to be able to fit the textbox to its parent width the first time
 17 	var tc = this.control = nitobi.html.createElement('input', {"id":"ntb-textbox"}, {width:"100px"});
 18 	//align
 19 	//maxlength
 20 	tc.setAttribute("maxlength", 255);
 21 
 22 	this.events = [{type:"keydown", handler:this.handleKey},
 23 					{type:"keyup", handler:this.handleKeyUp},
 24 					{type:"keypress", handler:this.handleKeyPress},
 25 					{type:"change", handler:this.handleChange},
 26 					{type:"blur", handler:this.deactivate}];
 27 }
 28 
 29 nitobi.lang.extend(nitobi.form.Text, nitobi.form.Control);
 30 
 31 /**
 32  * @private
 33  */
 34 nitobi.form.Text.prototype.initialize = function()
 35 {
 36 	// The text control needs to be attached here such that the password editor can change the input field type in the constructor
 37 	var container = this.placeholder.rows[0].cells[0];
 38 	container.appendChild(this.control);
 39 	nitobi.html.attachEvents(this.control, this.events, this);
 40 }
 41 
 42 //cell, xModel, xData, initialKeyChar
 43 /**
 44  * @private
 45  */
 46 nitobi.form.Text.prototype.bind = function(owner, cell, initialKeyChar)
 47 {	
 48 	nitobi.form.Text.base.bind.apply(this, arguments);
 49 
 50 	// All these events are complicated in the different browsers ... understand this better when we have time.
 51 	if(initialKeyChar != null && initialKeyChar != '') {
 52 		this.control.value = initialKeyChar;
 53 	} else {
 54 		this.control.value = cell.getValue();
 55 	}
 56 
 57 	var columnModel = this.cell.getColumnObject().getModel();
 58 
 59 	this.eSET("onKeyPress", [columnModel.getAttribute('OnKeyPressEvent')]);
 60 	this.eSET("onKeyDown", [columnModel.getAttribute('OnKeyDownEvent')]);
 61 	this.eSET("onKeyUp", [columnModel.getAttribute('OnKeyUpEvent')]);
 62 	this.eSET("onChange", [columnModel.getAttribute('OnChangeEvent')]);
 63 
 64 	this.control.setAttribute("maxlength", columnModel.getAttribute('MaxLength'));
 65 
 66 	// Add the column specific styles to the editor
 67 	// Remember to remove it when we deactivate
 68 	nitobi.html.Css.addClass(this.control, "ntb-column-data"+this.owner.uid+"_"+(this.cell.getColumn()+1));
 69 }
 70 
 71 /**
 72  * Mimic is responsible for placing and sizing the editor control
 73  * to the desired position on the screen.
 74  */
 75 nitobi.form.Text.prototype.mimic = function()
 76 {
 77 	this.align();
 78 
 79 	// Now we need to adjust the editor width for padding / borders etc
 80 	nitobi.html.fitWidth(this.placeholder, this.control);
 81 
 82 	this.selectText();
 83 }
 84 
 85 /**
 86  * Sets the focus to the editor.
 87  */
 88 nitobi.form.Text.prototype.focus = function() {
 89 	this.control.focus();
 90 }
 91 
 92 /**
 93  * @private
 94  */
 95 nitobi.form.Text.prototype.deactivate = function(evt)
 96 {
 97 	if (nitobi.form.Text.base.deactivate.apply(this, arguments) == false) return;
 98 
 99 	nitobi.html.Css.removeClass(this.control, "ntb-column-data"+this.owner.uid+"_"+(this.cell.getColumn()+1));
100 
101 	return this.afterDeactivate(this.control.value);
102 }

105 /**
106  * @private
107  */
108 nitobi.form.Text.prototype.dispose = function()
109 {
110 	nitobi.html.detachEvents(this.control, this.events);
111 	var parent = this.placeholder.parentNode;
112 	parent.removeChild(this.placeholder);
113 	this.control = null;
114 	this.owner = null;
115 	this.cell = null;
116 }
