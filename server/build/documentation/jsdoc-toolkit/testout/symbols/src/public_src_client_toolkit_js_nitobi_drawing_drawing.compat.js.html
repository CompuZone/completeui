 22 /**
 23  * Aligns two DOM nodes on in the web browser.
 24  * @param {HtmlElement} source
 25  * @param {HtmlElement} target
 26  * @param {BitMask} align
 27  * @param {Number} oh The height offset for the target HtmlElement.
 28  * @param {Number} ow The width offset for the target HtmlElement.
 29  * @param {Number} oy The left offset for the target HtmlElement.
 30  * @param {Number} ox The top offset for the target HtmlElement.
 31  */
 32 nitobi.drawing.align = function(source,target,AlignBit_HWTBLRCM,oh,ow,oy,ox)
 33 {
 34 	oh=oh || 0;
 35 	ow=ow || 0;
 36 	oy=oy || 0;
 37 	ox=ox || 0;
 38 	var a=AlignBit_HWTBLRCM;
 39 	var td,sd,tt,tb,tl,tr,th,tw,st,sb,sl,sr,sh,sw;
 40 
 41 	if (nitobi.browser.IE)
 42 	{
 43 		//	this is for IE
 44 		td=target.getBoundingClientRect();
 45 		sd=source.getBoundingClientRect();
 46 		tt=td.top;
 47 		tb=td.bottom;
 48 		tl=td.left;
 49 		tr=td.right;
 50 		th=Math.abs(tb-tt);
 51 		tw=Math.abs(tr-tl);
 52 		st=sd.top;
 53 		sb=sd.bottom;
 54 		sl=sd.left;
 55 		sr=sd.right;
 56 		sh=Math.abs(sb-st);
 57 		sw=Math.abs(sr-sl);
 58 	}
 59 	else if (nitobi.browser.MOZ)
 60 	{
 61 		//	this is for Mozilla
 62 		td = document.getBoxObjectFor(target);
 63 		sd = document.getBoxObjectFor(source);
 64 
 65 		tt = td.y;
 66 		tl = td.x;
 67 		tw = td.width;
 68 		th = td.height;
 69 
 70 		st = sd.y;
 71 		sl = sd.x;
 72 		sw = sd.width;
 73 		sh = sd.height;
 74 	}
 75 	else
 76 	{
 77 		td = nitobi.html.getCoords(target);
 78 		sd = nitobi.html.getCoords(source);
 79 
 80 		tt = td.y;
 81 		tl = td.x;
 82 		tw = td.width;
 83 		th = td.height;
 84 
 85 		st = sd.y;
 86 		sl = sd.x;
 87 		sw = sd.width;
 88 		sh = sd.height;
 89 	}
 90 	var s = source.style;
 91 
 92 	if (a&0x10000000) s.height = (th+oh)+'px'; // make same height
 93 	if (a&0x01000000) s.width = (tw+ow)+'px'; // make same width
 94 	if (a&0x00100000) s.top = (nitobi.html.getStyleTop(source)+tt-st+oy)+'px'; // align top
 95 	if (a&0x00010000) s.top = (nitobi.html.getStyleTop(source)+tt-st+th-sh+oy)+'px'; // align bottom
 96 	if (a&0x00001000) s.left = (nitobi.html.getStyleLeft(source)-sl+tl+ox)+'px'; // align left
 97 	if (a&0x00000100) s.left = (nitobi.html.getStyleLeft(source)-sl+tl+tw-sw+ox)+'px'; // align right
 98 	if (a&0x00000010) s.top = (nitobi.html.getStyleTop(source)+tt-st+oy+Math.floor((th-sh)/2))+'px'; // align middle vertically
 99 	if (a&0x00000001) s.left = (nitobi.html.getStyleLeft(source)-sl+tl+ox+Math.floor((tw-sw)/2))+'px'; // align middle horizontally
100 }

132 nitobi.drawing.alignOuterBox = function(source,target,AlignBit_HWTBLRCM,oh,ow,oy,ox,show)
133 {
134 	oh=oh || 0;
135 	ow=ow || 0;
136 	oy=oy || 0;
137 	ox=ox || 0;
138 
139 	//CROSS BROWSER
140 	//if (target.getBoundingClientRect)
141 	//{
142 		//	this is for IE
143 	//}
144 	if (nitobi.browser.moz)
145 	{
146 		//	this is for Mozilla
147 		td = document.getBoxObjectFor(target);
148 		sd = document.getBoxObjectFor(source);
149 
150 		var borderLeftTarget = parseInt(document.defaultView.getComputedStyle(target, '').getPropertyValue('border-left-width'));
151 		var borderTopTarget = parseInt(document.defaultView.getComputedStyle(target, '').getPropertyValue('border-top-width'));
152 
153 		var borderTop = parseInt(document.defaultView.getComputedStyle(source, '').getPropertyValue('border-top-width'));
154 		var borderBottom = parseInt(document.defaultView.getComputedStyle(source, '').getPropertyValue('border-bottom-width'));
155 
156 		var borderLeft = parseInt(document.defaultView.getComputedStyle(source, '').getPropertyValue('border-left-width'));
157 		var borderRight = parseInt(document.defaultView.getComputedStyle(source, '').getPropertyValue('border-right-width'));
158 
159 		oy = oy + borderTop - borderTopTarget;
160 		ox = ox + borderLeft - borderLeftTarget;
161 	}
162 
163 	nitobi.drawing.align(source,target,AlignBit_HWTBLRCM,oh,ow,oy,ox,show);
164 }
