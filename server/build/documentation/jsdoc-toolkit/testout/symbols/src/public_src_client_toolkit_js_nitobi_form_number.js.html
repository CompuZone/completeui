1 /**
  2  * Number is a number editor class that is implemented using an HTML <input> tag.
  3  * @class
  4  * @constructor
  5  */
  6 nitobi.form.Number = function()
  7 {
  8 	nitobi.form.Number.baseConstructor.call(this);
  9 	this.defaultValue = 0;
 10 }
 11 
 12 nitobi.lang.extend(nitobi.form.Number, nitobi.form.Text);
 13 
 14 nitobi.form.Number.prototype.handleKey = function(evt)
 15 {
 16 	nitobi.form.Number.base.handleKey.call(this, evt);
 17 
 18 	var k = evt.keyCode;
 19 	if (!this.isValidKey(k))
 20 	{
 21 		nitobi.html.cancelEvent(evt);
 22 		return false;
 23 	}
 24 }
 25 
 26 
 27 nitobi.form.Number.prototype.isValidKey = function (k)
 28 {
 29 	
 30 	if ((k < 48 || k > 57) && (k < 37 || k > 40) && (k < 96 || k > 105) && k != 190 && k != 110 && k != 189 && k != 109 && k != 9 && k != 45 && k != 46 && k != 8)
 31 	{
 32 		return false;
 33 	}
 34 	return true;
 35 }
 36 
 37 nitobi.form.Number.prototype.bind = function(owner, cell, initialKeyChar)
 38 {
 39 	var charCode = initialKeyChar.charCodeAt(0);
 40 	if (charCode >= 97) charCode = charCode - 32;
 41 	var k = this.isValidKey(charCode) ? initialKeyChar : "";
 42 	nitobi.form.Number.base.bind.call(this, owner, cell, k);
 43 }
