1 /**
  2  * Event arguments for after a save event.
  3  * @param {nitobi.data.DataTable} source
  4  * @param {XMLDocument} responseData
  5  * @param {Boolean} success
  6  */
  7 nitobi.data.OnAfterSaveEventArgs = function(source, responseData, success)
  8 {
  9 	nitobi.data.OnAfterSaveEventArgs.baseConstructor.call(this,source);
 10 	this.success = success;
 11 	this.responseData = responseData;
 12 };
 13 
 14 nitobi.lang.extend(nitobi.data.OnAfterSaveEventArgs, nitobi.data.DataTableEventArgs);
 15 
 16 /**
 17  * Returns the XML Document that was returned by the save handler.
 18  * @type XMLDocument
 19  */
 20 nitobi.data.OnAfterSaveEventArgs.prototype.getResponseData = function()
 21 {
 22 	return this.responseData;
 23 };
 24 
 25 /**
 26  * Returns a boolean signifying whether or not the save was successful.
 27  * @type Boolean 
 28  */
 29 nitobi.data.OnAfterSaveEventArgs.prototype.getSuccess = function()
 30 {
 31 	return this.success;
 32 };
