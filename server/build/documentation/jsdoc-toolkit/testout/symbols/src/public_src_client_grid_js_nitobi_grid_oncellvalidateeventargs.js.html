  3 /**
  4  * Constructs a OnCellValidateEventArgs object.
  5  * @class This object may also be returned from the OnCellValidateEvent 
  6  * handler with new values for the newValue, displayValue and status 
  7  * properties. To invalidate the value you can return false from the function
  8  * set to handle the event.
  9  * <div class="code">
 10  * <pre><code class="javascript">
 11  * function validate(event)
 12  * {
 13  * 	if (event.getNewValue() < event.getOldValue())
 14  * 	{
 15  * 		// By returning false, we revert the value of the cell back to the old value
 16  * 		return false;
 17  * 	}
 18  * }
 19  * </code></pre>
 20  * </div>
 21  * @constructor
 22  * @param {nitobi.grid.Grid} source The object which is firing the event.
 23  * @param {nitobi.grid.Cell} cell The Cell object of the cell that received focus. 
 24  * @param {String} newValue The value that the user has entered into the cell.
 25  * @param {String} oldValue The previous value that the was entered into the cell.
 26  * @extends nitobi.grid.CellEventArgs
 27  * @private
 28  */
 29 nitobi.grid.OnCellValidateEventArgs = function(source, cell, newValue, oldValue)
 30 {
 31 	nitobi.grid.OnCellValidateEventArgs.baseConstructor.call(this, source, cell);
 32 	/**
 33 	 * The previous value that the was entered into the cell.
 34 	 * @private
 35 	 */
 36 	this.oldValue = oldValue;
 37 	/**
 38 	 * The value that the user has entered into the cell.
 39 	 * @private
 40 	 */
 41 	this.newValue = newValue;
 42 }
 43 
 44 nitobi.lang.extend(nitobi.grid.OnCellValidateEventArgs, nitobi.grid.CellEventArgs);
