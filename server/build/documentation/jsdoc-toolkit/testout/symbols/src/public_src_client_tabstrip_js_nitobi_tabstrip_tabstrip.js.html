 16 /**
 17  * Creates a nitobi tabstrip.  Optionally, creates a tabstrip at the given declaration.
 18  * @class A nitobi tabstrip.  A tabstrip declaration takes this form:
 19  * <div class="code">
 20  * <pre><code>
 21  * <ntb:tabstrip id="SimpleTabstrip" width="800px" height="600px">
 22  * 	<ntb:tabs height="" align="center" overlap="15">
 23  * 		<ntb:tab width="190px" tooltip="Welcome." label="IFrame Tab" source="http://www.nitobi.com" containertype="iframe"></ntb:tab>
 24  * 		<ntb:tab width="190px" tooltip="Welcome." label="DOM Tab" source="tab2"></ntb:tab>
 25  * 		<ntb:tab width="190px" tooltip="Welcome." label="Ajax Tab" source="tab3.html" loadondemandenabled="true"></ntb:tab>
 26  * 	</ntb:tabs>
 27  * </ntb:tabstrip>
 28  * <div id="tab2">
 29  * 	<h1>DOM Tab</h1>
 30  * 	<img src="images/nitobi.jpg" />
 31  * </div>
 32  * </code></pre>
 33  * </div>
 34  * Calling <code>nitobi.loadComponent('SimpleTabstrip');</code> will render a tabstrip at the 
 35  * location of that XML declaration in your web page. To access this API you get the tabstrip object from 
 36  * its node in the HTML document. We can, for example, change the source of the first tab:
 37  * <div class="code">
 38  * <pre><code>
 39  * var tabstrip = $('tabstrip1').jsObject;
 40  * var tab = tabstrip.getTabs().get(0);
 41  * tab.setSource('http://www.google.com');
 42  * </code></pre>
 43  * </div>
 44  * Note: some operations require that you call {@link nitobi.tabstrip.TabStrip#render} after changing 
 45  * some properties.  This is due to the expense of changing them and rendering them individually.
 46  * @constructor
 47  * @see nitobi.tabstrip.Tabs
 48  * @see nitobi.tabstrip.Tab
 49  * @param {String} id Optional. The id of the control. If you do not specify an id, one is created for you.
 50  * @extends nitobi.ui.Element
 51  */
 52 nitobi.tabstrip.TabStrip = function(id) 
 53 {
 54 	nitobi.prepare();
 55 	nitobi.tabstrip.TabStrip.baseConstructor.call(this,id);
 56 	this.renderer.setTemplate(nitobi.tabstrip.tabstripProc);
 57 	
 58 	
 59 	/**
 60 	 * Fired on click. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 61 	 * @type nitobi.base.Event
 62 	 */
 63 	this.onClick = new nitobi.base.Event("click");
 64 	this.eventMap["click"] = this.onClick;
 65 	/**
 66 	 * Fired on mouse out. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 67 	 * @see nitobi.ui.ElementEventArgs
 68 	 * @type nitobi.base.Event
 69 	 */
 70 	this.onMouseOut = new nitobi.base.Event("mouseout");
 71 	this.eventMap["mouseout"] = this.onMouseOut;
 72 	/**
 73 	 * Fired on mouse over. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 74 	 * @see nitobi.ui.ElementEventArgs
 75 	 * @type nitobi.base.Event
 76 	 */
 77 	this.onMouseOver = new nitobi.base.Event("mouseover");
 78 	this.eventMap["mouseover"] = this.onMouseOver;
 79 	
 80 	this.subscribeDeclarationEvents();
 81 	
 82 	/**
 83 	 * @ignore
 84 	 */
 85 	this.renderTimes = 0;
 86 	/**
 87 	 * The version number.
 88 	 * @type String
 89 	 */
 90 	this.version = "0.8";
 91 	this.onCreated.notify(new nitobi.ui.ElementEventArgs(this));
 92 }

136 /**
137  * Correctly fits the inside containers to the outside container. If 
138  * you perform an operation on the outside container that modifies
139  * its dimensions, you may need to call this function.
140  * @see #setWidth
141  * @see #setHeight
142  * @see #render
143  */
144 nitobi.tabstrip.TabStrip.prototype.fitContainers = function()
145 {
146 	try
147 	{
148 		var primary = this.getHtmlNode();
149 		if (primary)
150 		{
151 			var secondary = this.getHtmlNode("secondarycontainer");
152 			if (secondary)
153 			{
154 				
155 				var box = nitobi.html.getBox(primary);
156 				secondary.style.height = box.height + "px";
157 				secondary.style.width = box.width + "px";
158 			}
159 		}
160 	}
161 	catch(err)
162 	{
163 		//	nitobi.error.throwError(nitobi.error.Unexpected + " fitContainers failed",err)
164 	}
165 }

167 /**
168  * Renders the tabstrip.  The first time the tabstrip is rendered, both the tabs and their 
169  * body containers are rendered. On subsequent render calls, unless tabs are added and removed,
170  * only the tabs are rendered. This prevents the user losing data in a tab body.
171  */
172 nitobi.tabstrip.TabStrip.prototype.render = function()
173 {
174 	this.onBeforeRender.notify(new nitobi.ui.ElementEventArgs(this,null,this.getId()));
175 	if (this.renderTimes==0)
176 	{
177 		nitobi.tabstrip.TabStrip.base.render.call(this);
178 		var tabs = this.getTabs();
179 		this.onRender.subscribe(tabs.handleRender,tabs)
180 		tabs.loadTabs();
181 		var node = this.getHtmlNode();
182 		if (nitobi.browser.IE)
183 		{
184 			nitobi.html.attachEvent(node,"resize",this.fitContainers,this);
185 			nitobi.html.attachEvent(node,"resize",tabs.handleResize,tabs);
186 		}
187 		else
188 		{
189 			nitobi.html.attachEvent(window,"resize",this.fitContainers,this);
190 			nitobi.html.attachEvent(window,"resize",tabs.handleResize,tabs);
191 		}
192 	}
193 	else
194 	{
195 		var tabs = this.getTabs();
196 		tabs.render();
197 	}
198 	this.renderTimes++;
199 	this.onRender.notify(new nitobi.ui.ElementEventArgs(this,null,this.getId()));
200 	this.fitContainers();
201 	if (node) node.jsObject = this;
202 }

349 /**
350  * @private
351  */
352 nitobi.tabstrip.TabStrip.handleEvent = function(id, event, targetId, cancelBubble)
353 {
354 	try
355 	{
356 		var tabstrip = $(id);
357 		if (tabstrip == null)
358 		{
359 			nitobi.lang.throwError("The tabstrip event could not find the component object.  The element with the specified id could not be found on the page.");
360 		}
361 		tabstrip = tabstrip.jsObject;
362 	    tabstrip.notify(event,targetId,null,cancelBubble);	
363 	}
364 	catch(err)
365 	{
366 		nitobi.lang.throwError(nitobi.error.Unexpected,err);
367 	}
368 }
