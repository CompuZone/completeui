 14  /**
 15  * Creates a Declaration object.
 16  * @class This class is used to retrieve and set the defaults from a Nitobi component declaration.
 17  * @ignore
 18  * @constructor
 19  * @extends	nitobi.Object
 20  */
 21 nitobi.base.Declaration = function() 
 22 {
 23 	nitobi.base.Declaration.baseConstructor.call(this);
 24 	/**
 25 	 * @private
 26 	 */
 27 	this.xmlDoc=null;
 28 }
 29 
 30 nitobi.lang.extend(nitobi.base.Declaration, nitobi.Object);
 31 
 32 /**
 33  * Given an HTML element, loads the HTML and creates an XML document from it.
 34  * If the HTML cannot be loaded into a XML document an error is thrown. Consult the
 35  * error string for information on what caused the parse error.
 36  * @param element {HTMLElement/String} The id of an element, or the element itself. 
 37  * @returns {void}
 38  */
 39 nitobi.base.Declaration.prototype.loadHtml = function(element)
 40 {
 41 	try
 42 	{
 43 		element = $(element);
 44 		this.xmlDoc = nitobi.xml.parseHtml(element);
 45 		return this.xmlDoc;
 46 	}
 47 	catch(err)
 48 	{
 49 		nitobi.lang.throwError(nitobi.error.DeclarationParseError,err);
 50 	}
 51 }
 52 
 53 /**
 54  * Returns the XML Document.
 55  * @type XMLDocument
 56  */
 57 nitobi.base.Declaration.prototype.getXmlDoc = function()
 58 {
 59 	return this.xmlDoc;
 60 }
 61 
 62 /**
 63  * Returns XML.
 64  * @type String
 65  */
 66 nitobi.base.Declaration.prototype.serializeToXml = function()
 67 {
 68 	return nitobi.xml.serialize(this.xmlDoc);	
 69 }
