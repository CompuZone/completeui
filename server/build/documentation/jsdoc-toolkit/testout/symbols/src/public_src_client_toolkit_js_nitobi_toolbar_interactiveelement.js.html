1 /**
  2 * An interactive ui element is defined by the fact it responds to user activity as
  3 * opposed to being a static element. The HTML object that represents EBAInteractiveUIElement.
  4 * @constructor
  5 * @extends nitobi.ui.UiElement
  6 * @param htmlObject {object} 
  7 */
  8 nitobi.ui.InteractiveUiElement  = function(htmlObject)
  9 {
 10 	this.enable();
 11 }
 12 
 13 nitobi.lang.extend(nitobi.ui.InteractiveUiElement,nitobi.ui.UiElement);
 14 
 15 /**
 16  * Puts the ui element in the enabled state.
 17  */
 18 nitobi.ui.InteractiveUiElement.prototype.enable = function()
 19 {
 20 	this.m_Enabled=true;
 21 }
 22 
 23 /**
 24  * Puts the ui element in the disabled state.
 25  */
 26 nitobi.ui.InteractiveUiElement.prototype.disable = function ()
 27 {
 28 	this.m_Enabled=false;
 29 }
