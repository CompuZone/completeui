1 nitobi.lang.defineNs('nitobi.effects');

  3 /**
  4  * Creates a scaling effect.  After the effect is created it can be started by calling 
  5  * <code>start()</code>.
  6  * @class A class that facilitates animated scaling.  Each instance is a different 
  7  * scaling effect on a different element with its own set of parameters.
  8  * @constructor
  9  * @param {HTMLElement} element the HTML element that will be affected by this effect
 10  * @param {Map} params initial values for the effect's fields - ie 
 11  * <code>{@link nitobi.effects.Effect#duration} = params.duration</code> 
 12  * @param {Number} percent the percentage (0-100) of size to which the effect should scale <code>element</code>
 13  * @extends nitobi.effects.Effect
 14  */
 15 nitobi.effects.Scale = function(element, params, percent)
 16 {
 17 	nitobi.effects.Scale.baseConstructor.call(this,element,params);
 18 	/**
 19 	 * Whether or not to scale the width of the element. Default: <code>true</code>
 20 	 * @type Boolean
 21 	 */
 22 	this.scaleX = typeof(params.scaleX) == 'boolean' ? params.scaleX : true;
 23 	/**
 24 	 * Whether or not to scale the height of the element. Default: <code>true</code>
 25 	 * @type Boolean
 26 	 */
 27 	this.scaleY = typeof(params.scaleY) == 'boolean' ? params.scaleY : true;
 28 	/**
 29 	 * The percentage (0-100) size to start the animation at. Default: <code>100.0</code>
 30 	 * @type Number
 31 	 */
 32 	this.scaleFrom = typeof(params.scaleFrom) == 'number' ? params.scaleFrom : 100.0;
 33 	/**
 34 	 * @ignore
 35 	 */
 36 	this.scaleTo = percent;
 37 };
 38 
 39 nitobi.lang.extend(nitobi.effects.Scale, nitobi.effects.Effect);
 40 
 41 /**
 42  * @private
 43  */
 44 nitobi.effects.Scale.prototype.setup = function()
 45 {
 46 	var style = this.element.style;
 47 	var Css = nitobi.html.Css;
 48 	this.originalStyle = {
 49 		'top': style.top,
 50 		'left': style.left,
 51 		'width': style.width,
 52 		'height': style.height,
 53 		'overflow': Css.getStyle(this.element, "overflow")
 54 	};
 55 	this.factor = (this.scaleTo - this.scaleFrom) / 100.0;
 56 	this.dims = [this.element.scrollWidth, this.element.scrollHeight];
 57 	style.width = this.dims[0]+"px";
 58 	style.height = this.dims[1]+"px";
 59 	Css.setStyle(this.element, "overflow", "hidden");
 60 };
 61 
 62 /**
 63  * @private
 64  */
 65 nitobi.effects.Scale.prototype.finish = function()
 66 {
 67 	for (var s in this.originalStyle)
 68 		this.element.style[s] = this.originalStyle[s]; 
 69 };
 70 
 71 /**
 72  * @private
 73  */
 74 nitobi.effects.Scale.prototype.update = function( pos )
 75 {
 76 	var currentScale = (this.scaleFrom / 100.0) + (this.factor * pos);
 77 	this.setDimensions(Math.floor(currentScale * this.dims[0]) || 1, Math.floor(currentScale * this.dims[1]) || 1);
 78 };	
 79 
 80 /**
 81  * @private
 82  */
 83 nitobi.effects.Scale.prototype.setDimensions = function( x, y )
 84 {
 85 	if (this.scaleX) this.element.style.width = x + 'px';
 86 	if (this.scaleY) this.element.style.height = y + 'px';
 87 };
