 12 /**
 13  * Returns a cookie with the given ID.
 14  * @param {String} id The ID of the cookie to retrieve.
 15  * @returns {Cookie}
 16  */
 17 nitobi.browser.Cookies.get = function(id)
 18 {
 19 	var begin, end;
 20 	if (document.cookie.length > 0)
 21 	{
 22 		begin = document.cookie.indexOf(id+"=");
 23 		if (begin != -1)
 24 		{
 25 			begin += id.length+1;
 26 			end = document.cookie.indexOf(";", begin);
 27 			if (end == -1) end = document.cookie.length;
 28 			return unescape(document.cookie.substring(begin, end)); 
 29 		}
 30 	}
 31 	return null;
 32 };
 33 
 34 /**
 35  * Sets the
 36  * @param {String} id The ID of the cookie.
 37  * @param {String} value The value to be stored in the cookie.
 38  * @param {String} expireDays The number of days before the cookie should expire.
 39  */
 40 nitobi.browser.Cookies.set = function(id, value, expireDays)
 41 {
 42 	var expiry = new Date ();
 43 	expiry.setTime(expiry.getTime() + (expireDays * 24 * 3600 * 1000));
 44 
 45 	document.cookie = id + "=" + escape(value) +
 46 	((expireDays == null) ? "" : "; expires=" + expiry.toGMTString());
 47 };
 48 
 49 /**
 50  * Destroys cookie.
 51  * @param {String} id The ID of the cookie to destroy.
 52  */
 53 nitobi.browser.Cookies.remove = function(id)
 54 {
 55 	if (nitobi.browser.Cookies.get(id)) 
 56 	{
 57 		document.cookie = id + "=" + "; expires=Thu, 01-Jan-70 00:00:01 GMT";
 58 	}
 59 };
