  3 /**
  4  * Creates a tree Node.
  5  * @class A tree node to be used with a {@link nitobi.tree.Tree} object.  Tree nodes can be accessed
  6  * through their parent {@link nitobi.tree.Children} object.  In this example we set the variable
  7  * <code>myNode</code> to the second child of the first node in a tree:
  8  * <div class="code">
  9  * <pre><code>
 10  * var tree = $('myTreeId').jsObject;
 11  * var firstNode = tree.getChildren().get(0);
 12  * var myNode = firstNode.getChildren().get(1);
 13  * </code></pre>
 14  * </div>
 15  * @constructor
 16  * @param {String} id Optional. The id of the control. If you do not specify an id, one is created for you.
 17  * @extends nitobi.ui.Element
 18  * @see nitobi.tree.Tree 
 19  * @see nitobi.tree.Children
 20  */
 21 nitobi.tree.Node = function(element, onLoadCallback)
 22 {
 23 	nitobi.tree.Node.baseConstructor.apply(this, arguments);
 24 	nitobi.base.ISerializable.call(this,{className:"nitobi.tree.Node"});
 25 	this.subscribeToChildrenEvents();
 26 	this.renderer.setTemplate(nitobi.tree.xslTemplate);
 27 	this.renderer.setParameters(
 28 		{
 29 			'apply-id': this.getId()
 30 		}
 31 	);
 32 	
 33 	/**
 34 	 * @ignore
 35 	 */
 36 	this.onLoadCallback = onLoadCallback || null;
 37 	//API Events
 38 	/**
 39 	 * Fires when this node's <CODE>children</CODE> object is set.
 40 	 * @type nitobi.base.Event 
 41 	 */
 42 	this.onSetChildren = new nitobi.base.Event();
 43 	this.onSetChildren.subscribe(this.subscribeToChildrenEvents, this);
 44 
 45 	/**
 46 	 * Fires when the node is selected.
 47 	 * The node is passed as an argument to each subscribed function.
 48 	 * @type nitobi.base.Event
 49 	 */
 50 	this.onSelect = new nitobi.base.Event();
 51 	this.eventMap["select"] = this.onSelect;
 52 
 53 	/**
 54 	 * Fires when the node is deselected.
 55 	 * The node is passed as an argument to each subscribed function.
 56 	 * @type nitobi.base.Event
 57 	 */
 58 	this.onDeselect = new nitobi.base.Event();
 59 	this.eventMap["deselect"] = this.onDeselect;
 60 	
 61 	/**
 62 	 * Fires when a node is clicked on.  The clickable area includes the icon and the label of the node.
 63 	 * @type nitobi.base.Event
 64 	 */
 65 	this.onClick = new nitobi.base.Event();
 66 	this.eventMap["mousedown"] = this.onClick;
 67 	this.eventMap["click"] = this.onClick;
 68 	
 69 	this.subscribeDeclarationEvents();
 70 		
 71 	// Set up the UrlConnector in an unintuitive way:
 72 	var gethandler = this.getGetHandler();
 73 	if (gethandler)
 74 	{
 75 		this.setGetHandler(gethandler);
 76 	}
 77 };

 85 /**
 86  * Subscribes to the events that need to be subscribed to when the children are changed.
 87  * @private
 88  */
 89 nitobi.tree.Node.prototype.subscribeToChildrenEvents = function()
 90 {
 91 	var c = this.getChildren();
 92 	if (c) 
 93 	{
 94 		c.onBeforeSetVisible.subscribe(this.handleChildrenBeforeVisibility, this);
 95 		c.onSetVisible.subscribe(this.handleChildrenVisibility, this);
 96 	}	
 97 };

115 		this.setBoolAttribute('expanded',false);

127 /**
128  * @private
129  */
130 nitobi.tree.Node.prototype.toggleChildren = function(effect)
131 {
132 	if (!this.isLeaf())
133 	{
134 		var c = this.getChildren();
135 		if (c == null)
136 		{
137 			var el = this.getHtmlNode('expander');
138 			if (el)
139 			{
140 				nitobi.html.Css.swapClass(el,'collapsed', 'working');
141 			}
142 			this.loadData();
143 		}
144 		else
145 		{
146 			var tree = this.getTree(); 
147 			var visible = !c.isVisible();
148 			effect = typeof(effect) == 'undefined' ? (visible ? tree.showEffect : tree.hideEffect) : effect;
149 			c.setVisible(!c.isVisible(), effect, nitobi.lang.close(this, this.handleToggle));
150 		}
151 	}
152 };

158 /**
159  * Load data from the gethandler.  The tree's gethandler will be used, and all attributes of this
160  * node will be sent as parameters on the URL.
161  * @param {Map} params a Map of parameters to add to the URL for this request   
162  */
163 nitobi.tree.Node.prototype.loadData = function(params)
164 {
165 	params = params || {};
166 
167 	var attributes = this.getXmlNode().attributes;
168 	
169 	for (var i = 0; i < attributes.length; i++)
170 	{
171 		var key = attributes[i].nodeName;
172 		var value = attributes[i].nodeValue;
173 		if (!params[key] && key != 'xmlns:ntb')
174 		{
175 			params[key] = value;
176 		}
177 	}
178 	params['treeId'] = this.getTree().getId();
179 	var tree = this.getDataboundAncestor();
180 	tree.urlConnector.get(params, nitobi.lang.close(this, this.handleDataReady));	
181 };

183 /**
184  * @ignore
185  */
186 nitobi.tree.Node.prototype.handleDataReady = function(eventArgs)
187 {
188 	//fire onBeforeDataReady
189 	var result = eventArgs.result;
190 	var newChildren = new nitobi.tree.Children(result.documentElement);
191 	var oldChildren = this.getChildren();
192 	if (!oldChildren || !oldChildren.isVisible() || !oldChildren.getLength())
193 	{ 
194 		this.setBoolAttribute('expanded',false);
195 	}
196 	else
197 	{
198 		this.setBoolAttribute('expanded',true);
199 	}
200 	this.setChildren(newChildren);
201 	var hasChildren = this.getChildren().getLength() != 0;
202 	if (!hasChildren)
203 	{
204 		this.setBoolAttribute('haschildren',false);
205 	}
206 	this.render();
207 	if (hasChildren && !this.getBoolAttribute('expanded'))
208 	{
209 		newChildren.setVisible(true, this.getTree().showEffect, nitobi.lang.close(this, this.handleToggle));	
210 	}
211 	//fire onDataReady
212 };

255 {
256 	var parentChildren = this.getParentObject();
257 	if (parentChildren)
258 	{

260 		if (p)
261 		{

267 	}

271 /**
272  * Makes visible each of this node's ancestor nodes.
273  * @param {nitobi.effects.Effect} effect an animation effect to use when opening the topmost ancestor.
274  */
275 nitobi.tree.Node.prototype.showAncestors = function(effect)
276 {
277 	var container = this.getParentObject();
278 	var toShow = new Array();
279 	
280 	while (container)
281 	{
282 		if (!container.isVisible()) 
283 		{
284 			toShow.push(container);
285 		}
286 		container = container.getParentObject(); // container is now a Node or a Tree
287 		if (container)
288 		{
289 			container = container.getParentObject(); 
290 			// if container was a Node it is now that Node's Parent's Children object.
291 		}
292 	}
293 
294 	if (toShow.length)
295 	{
296 		for (var i = 0; i < toShow.length-1; i++)
297 		{
298 			toShow[i].show(null);
299 		}
300 		toShow[toShow.length-1].show(effect);
301 	}
302 };

304 /**
305  * Returns the closest ancestor of this node with a gethandler specified.
306  * @type nitobi.tree.Node|nitobi.tree.Tree
307  */
308 nitobi.tree.Node.prototype.getDataboundAncestor = function()
309 {
310 	var gethandler = this.getAttribute('gethandler');
311 	var p = this;
312 	var retVal = p;
313 	while (!gethandler && (p = p.getParentObject()))
314 	{
315 		gethandler = p.getAttribute('gethandler');
316 		retVal = p;	
317 	}
318 	return retVal.getAttribute('gethandler') ? retVal : null;
319 };

407 /**
408  * Render the current state of the node and delete any previously rendered version.
409  */
410 nitobi.tree.Node.prototype.render = function()
411 {
412 	var oldNode = this.getHtmlNode();
413 	var parentNode = this.getParent();
414 	if (parentNode)
415 	{
416 		var hierarchy = parentNode.getHtmlNode('hierarchy');
417 //		if (nitobi.browser.IE)
418 //		{
419 			var doc = nitobi.xml.parseHtml(hierarchy);
420 			hierarchy = doc.documentElement;
421 			var lastOne = doc.createElement('div');
422 			parentNode.getHtmlNode('junction').className === 'tee' ? lastOne.setAttribute('class','pipe') : lastOne.setAttribute('class','spacer');
423 			hierarchy.appendChild(lastOne);
424 //		}
425 		
426 		this.renderer.setParameters(
427 			{
428 				'hierarchy': hierarchy  
429 			}
430 		);
431 	}
432 	
433 	if (oldNode)
434 	{
435 		this.flushHtmlNodeCache();
436 		oldNode.style.display = 'none';
437 		this.htmlNode = this.renderer.renderBefore(oldNode, this.getState().ownerDocument.documentElement)[0];
438 		oldNode.parentNode.removeChild(oldNode);
439 	}
440 	else
441 	{
442 		var beforeNode = null;
443 		var p = this.getParentObject();
444 		var i = p.indexOf(this);
445 		var l = p.getLength();
446 		if (i == l - 1 && i > 0)
447 		{
448 			p.get(i-1).render();
449 		} 
450 		if (++i < l)
451 		{
452 			while ((i < l) && !(beforeNode = p.get(i++).getHtmlNode())){/* find the closest rendered sibling */}			
453 		}
454 		// parent_doc = this.getState().ownerDocument.documentElement;
455 		parent_doc = this.parentNode.ownerDocument.documentElement;
456 		this.htmlNode = this.renderer._renderBefore(p.getHtmlNode('container'), beforeNode, parent_doc)[0];
457 	}
458 	var tabIndex = this.getIntAttribute('tabindex', -1);
459 	if (tabIndex > -1)
460 	{
461 		this.getHtmlNode('selector').focus();
462 	}
463 };

518 /**

520  * @param {Number} tabIndex the tabIndex to set for this node. (Optional)

576 /**
577  * Returns <code>true</code> if the node is a leaf node.
578  */
579 nitobi.tree.Node.prototype.isLeaf = function()
580 {
581 	var test;
582 	if (test = this.getAttribute('nodetype'))
583 	{
584 		return (test == 'leaf');
585 	}
586 	else if (test = this.getAttribute('haschildren'))
587 	{
588 		return (test == 'false');
589 	}
590 	else 
591 	{
592 		return this.getChildren() ? false : true;
593 	}
594 };

596 /**
597  * Returns the tree to which this node belongs.
598  * @type nitobi.tree.Tree
599  */
600 nitobi.tree.Node.prototype.getTree = function()
601 {
602 	if (this.tree) return this.tree;
603 	var p = this;
604 	var r = null;
605 	while (r = p.getParentObject())
606 	{
607 		p = r;
608 	}
609 	this.tree = p;
610 	return p;
