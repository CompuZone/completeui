1 nitobi.lang.defineNs("nitobi.collections");
  2 /*
  3  * @class nitobi.collections.BlockMap is similar to nitobi.collections.CacheMap except that adjacent blocks are not collapsed into a single block
  4  * @constructor
  5  */
  6 nitobi.collections.BlockMap = function() {
  7 	this.head = null;
  8 	this.tail = null;
  9 	this.debug = new Array();
 10 }

 12 nitobi.lang.extend(nitobi.collections.BlockMap, nitobi.collections.CacheMap);

 14 /*
 15  * Inserts a block into the blockmap.
 16  * @param low {int} The low end of the block range.
 17  * @param high {int} The high end of the block range.
 18  */
 19 nitobi.collections.BlockMap.prototype.insert = function(low,high) 
 20 {
 21 	low = Number(low);
 22 	high = Number(high);
 23 	this.debug.push("insert("+low+","+high+")");
 24 	if (this.head==null){
 25 		var newNode = new nitobi.collections.CacheNode(low,high);
 26 		this.debug.push("empty cache, adding first node");
 27 		this.head = newNode;
 28 		this.tail = newNode;
 29 	}
 30 	else
 31 	{
 32 		// Handle adding at beginning or end as special
 33 		// case due to the fact that ranges are quite
 34 		// often prepended or appended to the list.
 35 		var n = this.head;
 36 		// find the node that will come after the new one.
 37 		// 1.              [44,46] 
 38 		// 2.                      [54,66] 
 39 		// 3.                           [64,76] 
 40 		//    [10,10] [30,40] ^ [50,60]    ^
 41 		// Slide right comparing the inserted low to each existing high until the inserted low is greater than the exisiting high
 42 		// This will give you ...
 43 		// Case 1:(Inserted low is in a node)    - the node that would contain the low - this will become the "nextNode" for the inserted node
 44 		// Case 2:(Inserted low is in a gap)     - the node that would follow the gap  - this will become the "nextNode" for the inserted node
 45 		// Case 3:(Inserted low higher than all) - NULL                                - Inserted node takes over as the new "tail"
 46 		// Case 4:(No existing nodes-see above)  - NULL - impossible cuz head!=null    - Inserted node becomes the "head" (and "tail")
 47 		while (n != null && low>n.high) 
 48 		{
 49 			n = n.next;
 50 		}
 51 		if (n==null) {
 52 			var newNode = new nitobi.collections.CacheNode(low,high);
 53 			this.debug.push("appending node to end");
 54 			this.tail.next = newNode;
 55 			newNode.prev = this.tail;
 56 			this.tail = newNode;
 57 		}
 58 		else
 59 		{
 60 			this.debug.push("inserting new node into or before " + n.toString());
 61 			// Case 1 or 2
 62 			// Case 1:(Inserted low is in a node) 
 63 			//         - the newNode will follow the insertion point node "n" (assuming its not completely contained)
 64 			//         - the newNode's low is n.high+1
 65 			//         - the newNode's high ... is min(newNode.high,next.low-1) ( high could overlap several blocks - chop at first one)
 66 			// Case 2:(Inserted low is in a gap) 
 67 			//         - the newNode will preceed the insertion point node "n"
 68 			//         - the newNode's low is unchanged
 69 			//         - the newNode's high ... is min(newNode.high,next.low-1) ( high could overlap several blocks - chop at first one)
 70 			if (low<n.low || high>n.high) // inserting into a gap
 71 			{
 72 				if (low<n.low)
 73 				{
 74 					var newNode = new nitobi.collections.CacheNode(low,high);
 75 					newNode.prev = n.prev;
 76 					newNode.next = n;
 77 					if (n.prev != null) {
 78 						n.prev.next = newNode;
 79 					}
 80 					n.prev = newNode;
 81 					newNode.high = Math.min(newNode.high,n.low-1);
 82 				} else {
 83 					var newNode = new nitobi.collections.CacheNode(n.high+1,high);
 84 					newNode.prev = n;
 85 					newNode.next = n.next;
 86 					if (n.next!=null) {
 87 						n.next.prev = newNode;
 88 						newNode.high = Math.min(high,newNode.next.low-1);
 89 					}
 90 					n.next = newNode;
 91 				}
 92 				if (newNode.prev==null)
 93 				{
 94 					this.head = newNode;
 95 				}
 96 				if (newNode.next==null)
 97 				{
 98 					this.tail = newNode;
 99 				}
100 			}
101 		}
102 	}
103 }

106 /*
107  * Searches the blockmap for all blocks that intersect the specified range.
108  * @param low {int} The low end of the block range.
109  * @param high {int} The high end of the block range.
110  */
111 nitobi.collections.BlockMap.prototype.blocks = function(low,high) 
112 {
113 	var g = new Array();
114 	var n = this.head;
115 
116 	if (n==null || n.low>high || this.tail.high<low)
117 	{
118 		// our search range lies entirely outside our cache
119 		g.push(new nitobi.collections.Range(low,high));
120 		return g;
121 	}
122 
123 	//	This loops through all the nodes in our cacheMap until
124 	//	a node high value is greater than the low end of our range
125 
126 	//TODO: change this to a binary search???
127 	var minLow = 0;
128 	while (n != null && n.high < low) // shouldn't overlap???
129 	{
130 		minLow = n.high+1;
131 		n = n.next;
132 	}
133 
134 	if (n!=null)
135 	{
136 		do
137 		{
138 			if (g.length == 0) // if this is the first gap.
139 			{
140 				if (low < n.low) {
141 					g.push(new nitobi.collections.Range(Math.max(low,minLow),Math.min(n.low-1,high))); // Need to consider the case where the previous high overlaps the inserted range's low
142 				}
143 			}
144 			if (high > n.high)
145 			{
146 				if (n.next == null || n.next.low > high) // Need to consider the case where the inserted range's high overlaps the next range's low
147 				{
148 					g.push(new nitobi.collections.Range(n.high+1, high));
149 				}
150 				else
151 				{
152 					g.push(new nitobi.collections.Range(n.high+1, n.next.low-1));
153 				}
154 			}
155 			n = n.next;
156 		} while (n != null && n.high < high)
157 	}
158 	else
159 	{
160 		g.push(new nitobi.collections.Range(this.tail.high+1,high));
161 	}
162 	return g;
163 }
