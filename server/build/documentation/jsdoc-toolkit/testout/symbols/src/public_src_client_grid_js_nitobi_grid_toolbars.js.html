 78 nitobi.ui.Toolbars.prototype.makeToolbar= function()
 79 {
 80 	var imgDir = this.findCssUrl();
 81 	this.toolbarXml.documentElement.setAttribute("id","toolbar"+this.uid);	
 82 	
 83 	this.toolbarXml.documentElement.setAttribute("image_directory",imgDir);
 84 
 85 	//TODO: This is begging for a cleaner solution.
 86 	var nodes = this.toolbarXml.selectNodes('/toolbar/items/*');
 87 	for (var i = 0; i < nodes.length; i++)
 88 	{
 89 		if (nodes[i].nodeType != 8)
 90 		{
 91 			nodes[i].setAttribute('id',nodes[i].getAttribute('id')+this.uid);
 92 		}
 93 	}
 94 
 95 	this.standardToolbar = new nitobi.ui.Toolbar(this.toolbarXml,"toolbar"+this.uid);
 96 	this.toolbarPagingXml.documentElement.setAttribute("id","toolbarpaging"+this.uid);	
 97 	this.toolbarPagingXml.documentElement.setAttribute("image_directory",imgDir);
 98 	
 99 	nodes = (this.toolbarPagingXml.selectNodes('/toolbar/items/*'));
100 	for (var i = 0; i < nodes.length; i++)
101 	{
102 		if (nodes[i].nodeType != 8)
103 		{
104 			nodes[i].setAttribute('id',nodes[i].getAttribute('id')+this.uid);
105 		}
106 	}
107 	
108 	this.pagingToolbar = new nitobi.ui.Toolbar(this.toolbarPagingXml,"toolbarpaging"+this.uid);
109 
110 }

118 /**
119  * Find the URL of the stylesheet that contains the toolbar classes.
120  * @private
121  * @return String The URL of the toolbar stylesheet.
122  */
123 nitobi.ui.Toolbars.prototype.findCssUrl = function()
124 {
125 	var sheet = nitobi.html.Css.findParentStylesheet(".ntb-toolbar");
126 	if (sheet==null)
127 	{
128 		sheet = nitobi.html.Css.findParentStylesheet(".ntb-grid");
129 		if (sheet==null)
130 		{	
131 			nitobi.lang.throwError("The CSS for the toolbar could not be found.  Try moving the nitobi.grid.css file to a location accessible to the browser's javascript or moving it to the top of the stylesheet list. findParentStylesheet returned " + sheet);
132 		}
133 	}
134 	return nitobi.html.Css.getPath(sheet);
135 }

143 nitobi.ui.Toolbars.prototype.render= function() 
144 {
145 	var toolbarDiv = this.container;
146 	toolbarDiv.style.visibility="hidden";
147 
148 	var xsl = nitobi.ui.ToolbarXsl;
149 	if (xsl.indexOf("xsl:stylesheet") == -1)
150 	{
151 		xsl = "<xsl:stylesheet version=\"1.0\" xmlns:xsl=\"http://www.w3.org/1999/XSL/Transform\"><xsl:output method=\"xml\" version=\"4.0\" />" + xsl
152 		+ "</xsl:stylesheet>";
153 	}
154 	var xslDoc = nitobi.xml.createXslDoc(xsl);
155 
156 	var toolbarHtml = nitobi.xml.transformToString(this.standardToolbar.getXml(), xslDoc,"xml");
157 	
158 	toolbarDiv.innerHTML = toolbarHtml;
159 	toolbarDiv.style.zIndex="1000";
160 
161 	var toolbarPagingHtml = nitobi.xml.transformToString(this.pagingToolbar.getXml(), xslDoc,"xml");
162 	toolbarDiv.innerHTML += toolbarPagingHtml;
163 
164 	xslDoc = null;
165 	xmlDoc = null;
166 
167 	this.standardToolbar.attachToTag();
168 	this.pagingToolbar.attachToTag();
169 
170 	this.resize();
171 
172 	var _this = this;
173 	var buttons = this.standardToolbar.getUiElements()
174 	
175 	// using a foreach loop and a switch statement allows users to create toolbars without some buttons
176 	// was previously causing an error with custom toolbars
177 	for (eachbutton in buttons) {
178 		switch (eachbutton) {
179 			case "save"+this.uid:
180 				buttons[eachbutton].onClick = 
181 					function()
182 					{
183 						_this.fire("Save");
184 					};
185 			break;
186 			case "newRecord"+this.uid:
187 				buttons[eachbutton].onClick = 
188 					function()
189 					{
190 						_this.fire("InsertRow");
191 					};
192 				// disable button i	 row insert is not allowed
193 				if(!this.isRowInsertEnabled())
194 				{
195 					buttons[eachbutton].disable();
196 				}
197 			break;
198 			case "deleteRecord"+this.uid:
199 				buttons[eachbutton].onClick = 
200 					function()
201 					{
202 						_this.fire("DeleteRow");
203 					};
204 				// disable button if row delete is not allowed
205 				if(!this.isRowDeleteEnabled())
206 				{
207 					buttons[eachbutton].disable();
208 				}
209 			break;
210 			case "refresh"+this.uid:
211 				buttons[eachbutton].onClick = 
212 					function()
213 					{
214 						var refreshOk=confirm("Refreshing will discard any changes you have made. Is it OK to refresh?");
215 						if (refreshOk)
216 						{
217 							_this.fire("Refresh");			
218 						}
219 					};
220 			break;
221 			default:
222 		}
223 	}
224 	
225 	// likewise for paging buttons
226 	var buttonsPaging = this.pagingToolbar.getUiElements();
227 	var _this = this;
228 
229 	for (eachPbutton in buttonsPaging) {
230 		switch (eachPbutton) {
231 			case "previousPage"+this.uid:
232 				buttonsPaging[eachPbutton].onClick = 
233 					function()
234 					{
235 						_this.fire("PreviousPage");			
236 					};
237 				buttonsPaging[eachPbutton].disable();
238 			break;
239 			case "nextPage"+this.uid:
240 				buttonsPaging[eachPbutton].onClick = 
241 					function()
242 					{
243 						_this.fire("NextPage");
244 					};
245 			break;
246 			default:
247 		}
248 	}
249 
250 	if (this.visibleToolbars & nitobi.ui.Toolbars.VisibleToolbars.STANDARD)
251 	{
252 		this.standardToolbar.show();
253 	}
254 	else
255 	{
256 		this.standardToolbar.hide();
257 	}
258 	if (this.visibleToolbars & nitobi.ui.Toolbars.VisibleToolbars.PAGING)
259 	{
260 		this.pagingToolbar.show();
261 	}
262 	else
263 	{
264 		this.pagingToolbar.hide();
265 	}
266 	toolbarDiv.style.visibility="visible";		
267 }

269 nitobi.ui.Toolbars.prototype.resize = function()
270 {
271 	var standardWidth = this.getWidth();
272 	if (this.visibleToolbars & nitobi.ui.Toolbars.VisibleToolbars.PAGING) {
273 		//standardWidth = standardWidth - 2 - parseInt(this.pagingToolbar.getWidth());
274 		standardWidth = standardWidth - parseInt(this.pagingToolbar.getWidth());
275 		this.standardToolbar.setHeight(this.getHeight());
276 	}
277 	if (this.visibleToolbars & nitobi.ui.Toolbars.VisibleToolbars.STANDARD) {
278 		this.standardToolbar.setWidth(standardWidth)
279 		this.standardToolbar.setHeight(this.getHeight());
280 	}
281 }

288 nitobi.ui.Toolbars.prototype.subscribe= function(evt,func,context)

291 	return nitobi.event.subscribe(evt+this.uid,nitobi.lang.close(context, func));

294 nitobi.ui.Toolbars.prototype.dispose= function()  
295 {
296 	this.toolbarXml = null;
297 	this.toolbarPagingXml = null;
298 	//	Manually dispose of important objects belonging to Grid
299 	if(this.toolbar && this.toolbar.dispose)
300 	{
301 	 	this.toolbar.dispose();
302 		this.toolbar = null;
303 	}
304 
305 	if(this.toolbarPaging && this.toolbarPaging.dispose)
306 	{
307 	 	this.toolbarPaging.dispose();
308 		this.toolbarPaging = null;
309 	}
