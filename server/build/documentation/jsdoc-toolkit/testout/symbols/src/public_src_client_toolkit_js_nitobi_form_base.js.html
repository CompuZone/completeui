159 	nitobi.drawing.align(this.placeholder, this.cell.getDomNode(), 0x11101000, 1, 1, -oY, -oX);

190 /**
191  * @ignore
192  */
193 nitobi.form.Control.prototype.handleKey = function(evt)
194 {
195 	var k = evt.keyCode;
196 
197 	if (this.onKeyDown.notify(evt) == false) return;
198 
199 	var K = nitobi.form.Keys;
200 
201 	var y=0;
202 	var x=0;
203 	if (k==K.UP){//up
204 		y=-1;
205 	}else if(k==K.DOWN){//down
206 		y=1;
207 	}else if(k==K.TAB){//tab
208 		x=1;
209 		if (evt.shiftKey) x=-1;
210 
211 		// tab is a special case and the keyCode needs to be clear for nitobi.browser.IE.
212 		// in nitobi.browser.MOZ this causes the following error:
213 		//Error ``setting a property that has only a getter'' 
214 		if(nitobi.browser.IE)
215 			evt.keyCode="";
216 	}else if(k==K.ENTER){//enter
217 		y=1;
218 	}else{
219 		if (k==K.ESC){//esc
220 			// TODO: Figure this out once and for all!
221 			// nitobi.html.detachEvent(this.control, "blur", this.deactivate);
222 			this.ignoreBlur = true;
223 			this.hide();
224 			this.owner.focus();
225 			this.onCancel.notify(this);
226 		}
227 		return;
228 	}
229 
230 	if (!this.checkValidity(evt)) return;
231 
232 	this.owner.move(x,y);
233 	nitobi.html.cancelBubble(evt);
234 
235 	//this.onEnter.notify(this);
236 	//this.onKeyPress.notify();
237 };

262 /**
263  * @private
264  * This is only for declaration use. Any subscription through this should have a name like "OnSomethingEvent".
265  * Then this.Something becomes the GUID of the event for unsubscribing if need by.
266  */
267 nitobi.form.Control.prototype.eSET=function(name, args)
268 {
269 	var oFunction = args[0];
270 	var funcRef = oFunction;
271 
272 	var subName = name.substr(2);
273 	subName = subName.substr(0,subName.length-5);
274 
275 	if (typeof(oFunction) == 'string')
276 	{
277 		funcRef = function() {return nitobi.event.evaluate(oFunction,arguments[0])};
278 	}
279 	
280 	// Get rid of the subscribed function if it is already there ... this is only for declarations
281 	if (this[subName] != null)
282 	{
283 		this[name].unSubscribe(this[subName]);
284 	}
285 
286 	// name should be OnCellClickEvent but we just expect CellClick for firing
287 	var guid = this[name].subscribe(funcRef);
288 	this.jSET(subName, [guid]);
289 	return guid;
290 }

292 /**
293  * @private
294  */
295 nitobi.form.Control.prototype.afterDeactivate = function(text, value) {
296 	// Accept either one or two params, if one param then text and value are assumed the same.
297 	value = value || text;
298 	if (this.editCompleteHandler != null) {
299 		var eventArgs = new nitobi.grid.EditCompleteEventArgs(this, text, value, this.cell);
300 		var result =  this.editCompleteHandler.call(this.owner, eventArgs);
301 		if(!result){
302 			this.ignoreBlur = false;
303 		}
304 		return result;
305 	}
306 }
