1880 	if (true == nitobi.Browser.GetVerticalScrollBarStatus(list))
1881 	{
1882 		if (nitobi.Browser.GetMeasurementUnitType(this.GetWidth()) != "%")
1883 		{

1885 			// TODO: Why is this being assigned to a global?

1892 		}
1893

1895 		var header = this.GetSectionHTMLTagObject(EBAComboBoxListHeader);
1896 		var footer = this.GetSectionHTMLTagObject(EBAComboBoxListFooter);
1897 		if (header != null) header.style.width = listWidth;
1898 		if (footer != null) footer.style.width = listWidth;

1908 		// called over and over - when looking at the above code initially

1911 		// - even though EBAComboBoxList grows (because of style:visible), its style.width
1912 		// does NOT grow (EBAComboBoxList's width adjusts but we're not changing its style.width)

1919 		{

1921 			iframeStyle.width=listStyle.width;
