  6 /**
  7  * Builds a well formed UTF-8 eba XML document based on the input parameters.  Extra column data will be truncated.
  8  * @param {String} htmlTable - A set of data that is formated in TABLE, TR, TD HTML elements.
  9  * @param {Array} columns - a set of contigious columns that describe the table data
 10  * @startRow {int} the first zero-based row to add to the table results.  
 11  * @return {xmlDoc}correctly formated well formed eba xml based on paramters.  Extra column data will be truncated.   
 12  *
 13  * For now merged table data are out of scope. The eba xml created will only have attributes for columns that are in column param and has a column in the table.  The eba xml created will not have an datasourcestructure tag
 14  */
 15 nitobi.data.FormatConverter.convertHtmlTableToEbaXml = function(htmlTable, columns, startRow)
 16 {
 17 	var s='<xsl:stylesheet version="1.0" xmlns:ntb="http://www.nitobi.com" xmlns:xsl="http://www.w3.org/1999/XSL/Transform"><xsl:output encoding="UTF-8" method="xml" omit-xml-declaration="no" />';
 18 	s+='<xsl:template match="//TABLE"><ntb:data id="_default">';
 19     s+='<xsl:apply-templates /></ntb:data> </xsl:template>';
 20    	s+='<xsl:template match = "//TR">  <xsl:element name="ntb:e"> <xsl:attribute name="xi"><xsl:value-of select="position()-1+' + parseInt(startRow) + '"/></xsl:attribute>';	
 21 	for (var index=0; index < columns.length; index++) 
 22 	{
 23 		s+='<xsl:attribute name="'+ columns[index] + '\" ><xsl:value-of select="TD['+parseInt(index+1)+']"/></xsl:attribute>';
 24 	}
 25 	s+='</xsl:element></xsl:template>';
 26 	s+='</xsl:stylesheet>';
 27 
 28 	var tableDataXmlDoc = nitobi.xml.createXmlDoc(htmlTable);
 29 	var xsltProcessor 	= nitobi.xml.createXslProcessor(s);
 30 	var result = nitobi.xml.transformToXml(tableDataXmlDoc, xsltProcessor);
 31 	return result;
 32 }

 34 /**
 35  * Builds a valid well formed UTF-8 XML document based on the input parameters.  Extra column data will be truncated.
 36  * @note For now lookup fields are out of scope
 37  * @param tsv {String}- tab seperated list of data to be converted to EBA xml
 38  * @param columns {Array} - a set of contigious columns that describe the table data
 39  * @startRow {int} the first zero-based row to add to the table results.  
 40  * @return {xmlDoc}correctly formated well formed eba xml based on paramters.  Extra column data will be truncated.   
 41  */
 42 nitobi.data.FormatConverter.convertTsvToEbaXml = function(tsv, columns, startRow)
 43 {
 44 	if (!nitobi.browser.IE && tsv[tsv.length-1]!="\n") {
 45 		tsv =tsv+'\n';
 46 	}
 47 	var htmlTableFormattedData = "<TABLE><TBODY>"
 48 	+tsv.replace(/[\&\r]/g,"")
 49    .replace(/([^\t\n]*)[\t]/g,"<TD>$1</TD>")
 50    .replace(/([^\n]*?)\n/g,"<TR>$1</TR>")
 51    .replace(/\>([^\<]*)\<\/TR/g,"><TD>$1</TD></TR")+"</TBODY></TABLE>";
 52 
 53 	// This will handle single-cell paste.
 54 	if (htmlTableFormattedData.indexOf("<TBODY><TR>") == -1)
 55 	{
 56 		htmlTableFormattedData = htmlTableFormattedData.replace(/TBODY\>(.*)\<\/TBODY/,"TBODY><TR><TD>$1</TD></TR></TBODY");
 57 	}
 58 
 59 	return nitobi.data.FormatConverter.convertHtmlTableToEbaXml(htmlTableFormattedData, columns, startRow);
 60 }

 63 /**
 64  * Builds 2*2 JavaScript array from the tab separated data. Extra column data will be truncated.
 65  * @param tsv {String}- tab seperated list of data to be converted to EBA xml
 66  * @param columns {Array} - a set of contigious columns that describe the table data
 67  * @startRow {int} the first zero-based row to add to the table results.  
 68  * @return {xmlDoc}correctly formated well formed eba xml based on paramters.  Extra column data will be truncated.   
 69  */
 70 nitobi.data.FormatConverter.convertTsvToJs = function(tsv)
 71 {
 72 	var jsFormattedData = "["
 73 	+tsv.replace(/[\&\r]/g,"")
 74 	.replace(/([^\t\n]*)[\t]/g,"$1\",\"")
 75 	.replace(/([^\n]*?)\n/g,"[\"$1\"],") + "]";
 76 
 77 	return jsFormattedData;
 78 }

 81 /**
 82  * Builds a data set that is formatted using HTML Elements TABLE, TR, TD based on the parameters. 
 83  * A best effort will be made to add the correct data.  
 84  * For now lookup fields are out of scope
 85  *
 86  * @param ebaXmlDocument {xmlDoc} the set of data from the grid (do we need a grid id?)
 87  * @param columns {Array} a array of columns to be added to the table
 88  * @param startRow {int}  the first zero-based row to be added to the table results.  
 89  * @param endRow {int}    the last zero-based row to be added to the table results.  
 90  * @return {String} well formed HTML table with UTF-8 encocding
 91  */
 92 nitobi.data.FormatConverter.convertEbaXmlToHtmlTable = function(ebaXmlDocument, columns, startRow, endRow)
 93 {
 94 	var s='<xsl:stylesheet version="1.0" xmlns:ntb="http://www.nitobi.com" xmlns:xsl="http://www.w3.org/1999/XSL/Transform"><xsl:output encoding="UTF-8" method="html" omit-xml-declaration="yes" /><xsl:template match = "*"><xsl:apply-templates /></xsl:template><xsl:template match = "/">';
 95 	s+='<TABLE><TBODY><xsl:for-each select="//ntb:e[@xi>' + parseInt(startRow - 1) + ' and @xi < ' + parseInt(endRow + 1) + ']" ><TR>';
 96 	for (var index=0; index < columns.length; index++) 
 97 	{
 98 		s+='<TD><xsl:value-of select=\"@' + columns[index] + '\" /></TD>';
 99 	}
100 	s+='</TR></xsl:for-each></TBODY></TABLE></xsl:template></xsl:stylesheet>'
101 
102 	var xsltProcessor = nitobi.xml.createXslProcessor(s);
103     return nitobi.xml.transformToXml(ebaXmlDocument, xsltProcessor).xml.replace(/xmlns:ntb="http:\/\/www.nitobi.com"/,"");
104 }

106 nitobi.data.FormatConverter.convertEbaXmlToTsv = function(ebaXmlDocument, columns, startRow, endRow)
107 {
108 	var s='<xsl:stylesheet version="1.0" xmlns:ntb="http://www.nitobi.com" xmlns:xsl="http://www.w3.org/1999/XSL/Transform"><xsl:output encoding="UTF-8" method="text" omit-xml-declaration="yes" /><xsl:template match = "*"><xsl:apply-templates /></xsl:template><xsl:template match = "/">';
109 	s+='<xsl:for-each select="//ntb:e[@xi>' + parseInt(startRow - 1) + ' and @xi < ' + parseInt(endRow + 1) + ']" >\n';
110 	for (var index=0; index < columns.length; index++) 
111 	{
112 		s+='<xsl:value-of select=\"@' + columns[index] + '\" />';
113 		if (index < columns.length-1)
114 		{
115 			s+='<xsl:text>	</xsl:text>';
116 		}
117 	}
118 	s+='<xsl:text>
</xsl:text></xsl:for-each></xsl:template></xsl:stylesheet>'
119 
120 	var xsltProcessor = nitobi.xml.createXslProcessor(s);
121     return nitobi.xml.transformToString(ebaXmlDocument, xsltProcessor).replace(/xmlns:ntb="http:\/\/www.nitobi.com"/,"");
122 }

124 /*
125  * Returns the number of columns in some TSV or TABLE based data.
126  */
127 nitobi.data.FormatConverter.getDataColumns = function(data)
128 {
129 	var retVal = 0;
130 	if (data != null && data != '')
131 	{
132 		if (data.substr(0,1) == '<')
133 		{
134 			retVal = data.toLowerCase().substr(0, data.toLowerCase().indexOf('</tr>')).split('</td>').length-1;
135 		}
136 		else
137 		{
138 			retVal = data.substr(0, data.indexOf('\n')).split('\t').length;
139 		}
140 	}
141 	else
142 	{
143 		retVal = 0;
144 	}
145 	return retVal;
146 }

148 /*
149  * Returns the number of rows in some TSV or TABLE based data.
150  */
151 nitobi.data.FormatConverter.getDataRows = function(data)
152 {
153 	var retVal = 0;
154 	if (data != null && data != '')
155 	{
156 		if (data.substr(0,1) == '<')
157 		{
158 			retVal = data.toLowerCase().split('</tr>').length-1;
159 		}
160 		else
161 		{
162 			retValArray = data.split('\n');
163 			retVal = retValArray.length;
164 			if (retValArray[retValArray.length-1] == "")
165 			{
166 				retVal--;
167 			}
168 		}
169 	}
170 	else
171 	{
172 		retVal = 0;
173 	}
174 	return retVal;
175 }
