  3 /**
  4  * The combo's button.
  5  * @class The button that is rendered for the ComboBox.
  6  * @constructor
  7  * @param {Object} userTag The HTML object that represents Button
  8  * @param {Object} comboObject
  9  * @see nitobi.combo.Combo
 10  */
 11 nitobi.combo.Button = function(userTag, comboObject)
 12 {
 13 	try{
 14 
 15 		var DEFAULTCLASSNAME="ntb-combobox-button";
 16 		var DEFAULTPRESSEDCLASSNAME="ntb-combobox-button-pressed";
 17 		var DEFAULTWIDTH="";
 18 		var DEFAULTHEIGHT="";
 19 		this.SetCombo(comboObject);
 20 
 21 		var width=(userTag ? userTag.getAttribute("Width") : null);
 22 		((null == width) || (width == ""))
 23 			? this.SetWidth(DEFAULTWIDTH)
 24 			: this.SetWidth(width);
 25 
 26 		var height=(userTag ? userTag.getAttribute("Height") : null);
 27 		((null == height) || (height == ""))
 28 			? this.SetHeight(DEFAULTHEIGHT)
 29 			: this.SetHeight(height);
 30 
 31 		var dccn=(userTag ? userTag.getAttribute("DefaultCSSClassName") : null);
 32 		((null == dccn) || (dccn == ""))
 33 			? this.SetDefaultCSSClassName(DEFAULTCLASSNAME)
 34 			: this.SetDefaultCSSClassName(dccn);
 35 
 36 		var pccn=(userTag ? userTag.getAttribute("PressedCSSClassName") : null);
 37 		((null == pccn) || (pccn == ""))
 38 			? this.SetPressedCSSClassName(DEFAULTPRESSEDCLASSNAME)
 39 			: this.SetPressedCSSClassName(pccn);
 40 
 41 		this.SetCSSClassName(this.GetDefaultCSSClassName());
 42 		/**
 43 		 * @private
 44 		 */
 45 		this.m_userTag = userTag;
 46 		/**
 47 		 * @private
 48 		 */
 49 		this.m_prevImgClass = "ntb-combobox-button-img";
 50 
 51 	}catch(err){
 52 
 53 
 54 	}
 55 }

363 nitobi.combo.Button.prototype.OnClick = Button_OnClick;
364 /**
365  * Handles the mouse click event for the button
366  * @private
367  */
368 function Button_OnClick(){
369 	var combo = this.GetCombo();
370 	// hide dropdowns for all other lists
371 	var allcombos = document.getElementsByTagName((nitobi.browser.MOZ||nitobi.browser.SAFARI)?"ntb:Combo":"combo");
372 	for (var i=0; i<allcombos.length; i++){
373 		var other = allcombos[i].object;
374 		try
375 		{
376 			if (combo.GetId() != other.GetId())
377 				other.GetList().Hide();
378 		}
379 		catch(err)
380 		{
381 			// Do nothing. The list may not have been rendered yet.
382 		}
383 	}
384 	var l = combo.GetList();
385 	l.Toggle();
386 	var t = combo.GetTextBox();
387 	var tb = t.GetHTMLTagObject();
388 	if(t.focused)
389 		t.m_skipFocusOnce=true;
390 	tb.focus();
391 }

397 nitobi.combo.Button.prototype.GetHTMLRenderString = Button_GetHTMLRenderString;
398 /**
399  * Returns the HTML that is used to render the object.
400  * @private
401  */
402 function Button_GetHTMLRenderString(){
403 	var comboId = this.GetCombo().GetId();
404 	var uid = this.GetCombo().GetUniqueId();
405 	// three notes:
406 	// - align=nitobi.browser.MOZ?"absbottom":"absmiddle" is very important to lining up the IMG w/ the textbox
407 	// - keeping <input type="text"></input><img></img> on the same line w/o any spaces, breaks, etc.
408 	// is important to ensuring that no gap appears between the two elements so do NOT put an
409 	// "\n" before/after this html bit
410 	// - of course, the <nobr></nobr> is necessary to keep the two together also (nobr is not
411 	// written in this context... see parent context)
412 	var w = this.GetWidth();
413 	var h = this.GetHeight();
414 
415 	// In IE, if the ComboBox is on a page using SSL, it will prompt the user with a "This page is using
416 	// both secure and unsecure items" alert.  The cause is the src attribute of the image tag.
417 	if( (nitobi.browser.MOZ||nitobi.browser.SAFARI) ) {
418 		var html =	"<span id='EBAComboBoxButton" + uid + "' " +
419 				"class='" + this.GetDefaultCSSClassName() + "' " +
420 				"style='" + (null!=w && ""!=w ? "width:"+w+";" : "") + (null!=h && ""!=h ? "height:"+h+";" : "") + "'>" +
421 				// Note: the img tag has a default src that may not exist; we replace it in initialize.
422 				"<img src='javascript:void(0);' class='ntb-combobox-button-img' id='EBAComboBoxButtonImg" + uid + "' " +
423 				"onmouseover='$(\"" + comboId + "\").object.GetButton().OnMouseOver(this, true)' "+
424 				"onmouseout='$(\"" + comboId + "\").object.GetButton().OnMouseOut(this, true)' "+
425 				// use onmousedown now: solves button image dragging bug (i.e. list stays open on blur)
426 				"onmousedown='$(\"" + comboId + "\").object.GetButton().OnMouseDown(this);return false;' "+
427 				"onmouseup='$(\"" + comboId + "\").object.GetButton().OnMouseUp(this)' "+
428 				"onmousemove='return false;' "+
429 				"></img></span>";
430 	} else {
431 		var html =	"<span id='EBAComboBoxButton" + uid + "' " +
432 					"class='" + this.GetDefaultCSSClassName() + "' " +
433 					"style='" + (null!=w && ""!=w ? "width:"+w+";" : "") + (null!=h && ""!=h ? "height:"+h+";" : "") + "'>" +
434 					// Note: the img tag has a default src that may not exist; we replace it in initialize.
435 					"<img class='ntb-combobox-button-img' id='EBAComboBoxButtonImg" + uid + "' " +
436 					"onmouseover='$(\"" + comboId + "\").object.GetButton().OnMouseOver(this, true)' "+
437 					"onmouseout='$(\"" + comboId + "\").object.GetButton().OnMouseOut(this, true)' "+
438 					// use onmousedown now: solves button image dragging bug (i.e. list stays open on blur)
439 					"onmousedown='$(\"" + comboId + "\").object.GetButton().OnMouseDown(this);return false;' "+
440 					"onmouseup='$(\"" + comboId + "\").object.GetButton().OnMouseUp(this)' "+
441 					"onmousemove='return false;' "+
442 					"></img></span>";
443 	}
444 	return html;
445 }

447 // *****************************************************************************
448 // * Initialize
449 // *****************************************************************************
450 nitobi.combo.Button.prototype.Initialize = Button_Initialize;
451 /**
452  * Initializes the object after creation.
453  * @private
454  */
455 function Button_Initialize(){
456 	var combo = this.GetCombo();
457 	var uid = combo.GetUniqueId();
458 	this.SetHTMLTagObject($("EBAComboBoxButton" + uid));
459 	var img = $("EBAComboBoxButtonImg" + uid);
460 
461 	// The browser requires that every img tag has a src.  We use the backgound-image
462 	// in the CSS to display the button.gif. However, this causes an image not found
463 	// icon in the browser because img.src is null. Here we replace img.src with
464 	// blank.gif, found in the same directory as button.gif
465 	// Start by getting the button.gif url.
466 	var blankImgPath = nitobi.html.Css.getStyle(img,"background-image");
467 	// Replace button.gif with blank.gif, and remove url,(", and ") chars.
468 	blankImgPath = blankImgPath.replace(/button\.gif/g,"blank.gif");
469 	if (nitobi.browser.IE)
470 	{
471 		blankImgPath = blankImgPath.substr(5,blankImgPath.length-7);
472 	}
473 	else
474 	{
475 		blankImgPath = blankImgPath.substr(4,blankImgPath.length-5);
476 		// Replace \( with (. Moz adds the \ for some reason.
477 		blankImgPath = blankImgPath.replace(/\\\(/g,"(");
478 		// Replace \) with )
479 		blankImgPath = blankImgPath.replace(/\\\)/g,")");
480 	}
481 	img.src=blankImgPath;
482 
483 	this.m_Img = img;
484 	this._onmouseover=img.onmouseover;
485 	this._onmouseout=img.onmouseout;
486 	this._onclick=img.onclick;
487 	this._onmousedown=img.onmousedown;
488 	this._onmouseup=img.onmouseup;
489 	if(!this.GetCombo().GetEnabled())
490 		this.Disable();
491 	// Don't user this after init. Use the proper accessors.
492 	this.m_userTag = null;
493 }
