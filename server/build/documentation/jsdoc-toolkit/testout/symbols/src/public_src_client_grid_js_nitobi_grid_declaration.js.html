 78 nitobi.grid.Declaration.loadDataSources = function(xDeclaration, grid)
 79 {
 80 	var declarationDatasources = new Array();
 81 	if (xDeclaration["datasources"])
 82 	{
 83 		declarationDatasources = xDeclaration.datasources.selectNodes("//"+nitobi.xml.nsPrefix+"datasources/*");
 84 	}
 85 	if (declarationDatasources.length > 0)
 86 	{
 87 		for (var i = 0; i < declarationDatasources.length; i++)
 88 		{
 89 			var id = declarationDatasources[i].getAttribute('id');
 90 			if (id != "_default")
 91 			{
 92 				var sData = declarationDatasources[i].xml.replace(/fieldnames=/g,"FieldNames=").replace(/keys=/g,"Keys=");
 93 	
 94 				sData = '<ntb:grid xmlns:ntb="http://www.nitobi.com"><ntb:datasources>'+sData+'</ntb:datasources></ntb:grid>'
 95 				var newDataTable = new nitobi.data.DataTable('local',grid.getPagingMode() != nitobi.grid.PAGINGMODE_NONE,{GridId:grid.getID()},{GridId:grid.getID()},grid.isAutoKeyEnabled());
 96 				newDataTable.initialize(id, sData);
 97 				newDataTable.initializeXml(sData);
 98 				grid.data.add(newDataTable);
 99 				
100 				// The datasource we have loaded could be that of the Grid as well ...
101 				var columns = grid.model.selectNodes("//nitobi.grid.Column[@DatasourceId='"+id+"']");
102 				for (var j = 0; j < columns.length; j++)
103 				{
104 					//	Now that we have created the datasource we need to set the ValueField and DisplayFields etc
105 					grid.editorDataReady(columns[j]);
106 				}
107 			}
108 		}
109 	}
110 }
