  5 nitobi.data.GetCompleteEventArgs = function(firstRow, lastRow, startXi, pageSize, ajaxCallback, dataSource, obj, callback)
  6 {
  7 	//	This is the first row to be rendered by the RowRenderer
  8 	this.firstRow = firstRow;
  9 	//	This is the last row to be rendered by the RowRenderer
 10 	this.lastRow = lastRow;
 11 	//	This is the callback to be called once the data is all ready
 12 	this.callback = callback;
 13 	//	This is the datasource on which the get was requested
 14 	this.dataSource = dataSource;
 15 	//	This is the context (ie the object) of the callback method
 16 	this.context = obj;
 17 	//	This is the nitobi.ajax.HttpRequest object handling the request
 18 	this.ajaxCallback = ajaxCallback;
 19 	//	This is the xi of the first record that is returned such that we can integrate returned data that has no xi values
 20 	this.startXi = startXi;
 21 	this.pageSize = pageSize;
 22 	/**
 23 	 * True if the request page was the last page.
 24 	 */
 25 	this.lastPage=false;
 26 	this.status = "success";
 27 }
