 14 /**
 15  * Loads and renders a component from a declaration. It returns the javascript object.
 16  * @param {HTMLElement/String} el Either the id of the declaration or the HTMLElement itself.
 17  * @type Object
 18  */
 19 nitobi.loadComponent = function(el)
 20 {
 21 	var id = el;
 22 	el = $(el);
 23 	if (el == null)
 24 	{
 25 		nitobi.lang.throwError("nitobi.loadComponent could not load the component because it could not be found on the page. The component may not have a declaration, node, or it may have a duplicated id. Id: " + id);
 26 	}
 27 	if (el.jsObject != null) {
 28 		return el.jsObject;
 29 	}
 30 	var component;
 31 	var tagName = nitobi.html.getTagName(el);
 32 	
 33 	// If the component is grid or combo then use their special init procedures.
 34 	if (tagName == "ntb:grid") 
 35 	{
 36 		component = nitobi.initGrid(el.id);
 37 	} 
 38 	else if (tagName === "ntb:combo")
 39 	{
 40 		component = nitobi.initCombo(el.id);
 41 	}
 42 	else if (tagName == "ntb:treegrid")
 43 	{
 44 		component = nitobi.initTreeGrid(el.id);
 45 	}
 46 	else 
 47 	{
 48 		// The component is a new-style component that doesn't require any special init process.
 49 		if (el.jsObject == null)
 50 		{
 51 			component = nitobi.base.Factory.getInstance().createByTag(tagName,el.id,nitobi.component.renderComponent);
 52 			if (component.render && !component.onLoadCallback)
 53 			{
 54 				component.render();
 55 			}
 56 		}
 57 		else
 58 		{
 59 			component=el.jsObject;	
 60 		}
 61 	}			
 62 	return component;
 63 }

 73 /**
 74  * Returns the javascript object for a component with a particular id. The 
 75  * component must have been already rendered.
 76  * @param {String} id The id of the component.
 77  * @type Object
 78  */
 79 nitobi.getComponent = function(id)
 80 {
 81 	var el = $(id);
 82 	if (el == null) return null;
 83 	return el.jsObject;
 84 }

101 /**
102  * Finds all the child nodes of the supplied node that are Nitobi components
103  * @param {HTMLNode} rootNode The node to start the search on
104  * @param {Array} foundNodes An array to store the nodes
105  * @private
106  */
107 nitobi.getComponents = function(rootNode, foundNodes)
108 {
109 	if (foundNodes == null)
110 		foundNodes = [];
111 
112 	if (nitobi.component.isNitobiElement(rootNode)) {
113 		foundNodes.push(rootNode);
114 		return;
115 	}
116 	
117 	var nodes = rootNode.childNodes;
118 	for (var i = 0; i < nodes.length; i++) {
119 		nitobi.getComponents(nodes[i], foundNodes);
120 	}
121 	
122 	return foundNodes;
123 }
124 
125 /**
126  * @private
127  */
128 nitobi.component.isNitobiElement = function(rootNode)
129 {
130 	var rootNodeName = nitobi.html.getTagName(rootNode);
131 	if (rootNodeName.substr(0, 3) == "ntb") {
132 		return true;
133 	} else {
134 		return false;
135 	}
136 }

138 /**
139  * @private
140  */
141 nitobi.component.loadComponentsFromNode = function(rootNode)
142 {
143 	var ntbElements = new Array();
144 	nitobi.getComponents(rootNode, ntbElements);
145 	for (var i = 0; i < ntbElements.length; i++) {
146 		nitobi.loadComponent(ntbElements[i].getAttribute('id'));
147 	}
148 }
