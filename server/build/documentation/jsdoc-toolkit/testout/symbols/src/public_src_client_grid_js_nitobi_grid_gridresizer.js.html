1 /**
  2  * Responsible for doing grid resize animation.
  3  * @private
  4  */
  5 nitobi.grid.GridResizer = function(grid)
  6 {
  7 	this.grid = grid;
  8 
  9 	this.widthFixed = false;
 10 	this.heightFixed = false;
 11 
 12 	this.minHeight = 0;
 13 	this.minWidth = 0;
 14 
 15 	this.box = document.getElementById("ntb-grid-resizebox" + grid.uid);
 16 
 17 	/**
 18 	 * @private
 19 	 * Event that fires after the resize of the object is complete.
 20 	 */
 21 	this.onAfterResize = new nitobi.base.Event();
 22 }

 71 
 72 /**
 73  * @private
 74  */
 75 nitobi.grid.GridResizer.prototype.resize = function()
 76 {
 77 	var x = this.dd.x;
 78 	var y = this.dd.y;
 79 
 80 	var rect = this.grid.UiContainer.getBoundingClientRect();
 81 	var L = rect.left;
 82 	var T = rect.top;
 83 
 84 	this.box.style.display = "block";
 85 	if ((x-L) > this.minWidth) this.box.style.width=(x-L) + "px";
 86 	if ((y-T) > this.minHeight) this.box.style.height=(y-T) + "px";
 87 }
 88 
 89 /**
 90  * @private
 91  */
 92 nitobi.grid.GridResizer.prototype.endResize = function()
 93 {
 94 	var x = this.dd.x;
 95 	var y = this.dd.y;
 96 
 97 	this.box.style.display = "none";
 98 	var prevWidth = this.grid.getWidth();
 99 	var prevHeight = this.grid.getHeight();
100 	this.newWidth = Math.max(parseInt(prevWidth) + (x - this.x), this.minWidth);
101 	this.newHeight = Math.max(parseInt(prevHeight) + (y - this.y), this.minHeight);
102 	if (isNaN(this.newWidth) || isNaN(this.newHeight))
103 		return;
104 
105 	this.onAfterResize.notify(this)
106 }

108 /**
109  * @private
110  */
111 nitobi.grid.GridResizer.prototype.dispose = function()
112 {
113 	this.grid = null;
114 }
