150 /**
151  * Returns a row range of what should currently be rendered in the grid. When paging is not being used,
152  * it simply returns the total number of rows in the Grid. For LiveScrolling it calculates the rows 
153  * to show based on the scroll position. Finally, for Standard paging it returns the rowsPerPage offset
154  * by the index of the current page. 
155  * @return {Pair} A struct range with first and last values which are the start and end of the row range.
156  */
157 nitobi.grid.Scroller3x3.prototype.getUnrenderedBlocks = function()
158 {
159 	var pair = {first : this.freezetop, last: this.rows-1-this.freezetop};
160 	if (!this.implementsShowAll()) {
161 		var scrollTop = this.getScrollTop()+ this.getTop() - this.headerHeight;
162 
163 		var MC = this.view.midcenter;
164 		var b0=MC.findBlockAtCoord(scrollTop);
165 		var b1=MC.findBlockAtCoord(scrollTop+this.height);
166 		var firstVisibleRow=null;
167 		var lastVisibleRow=null;
168 		if (b0 == null) {
169 			return;
170 		}
171 		firstVisibleRow = b0.top+Math.floor((scrollTop-b0.offsetTop)/this.rowHeight);
172 		if (b1) {
173 			lastVisibleRow = b1.top+Math.floor((scrollTop+this.height-b1.offsetTop)/this.rowHeight);
174 		} else {
175 			lastVisibleRow=firstVisibleRow+Math.floor(this.height/this.rowHeight);
176 		}
177 		// TODO: this used to be this.rows-1 since lastVisibleRow is zero based.
178 		lastVisibleRow=Math.min(lastVisibleRow,this.rows);
179 
180 		// We check if standard paging is being used, and if so apply an offset 
181 		// of the page size * the current page to the first visible row.
182 		if (this.implementsStandardPaging()) 
183 		{
184 			var topOffset = 0;
185 			//topOffset = this.getRowsPerPage() * this.getCurrentPageIndex();
186 
187 			if (this.owner.getRenderMode() == nitobi.grid.RENDERMODE_ONDEMAND)
188 			{
189 				// ** Live Scrolling + Standard Paging ...
190 				// TODO: this will make standard paging look like livescrolling on the page level
191 				// however, it causes the value in the editor to be "undefined" when you edit a cell.
192 				var first = firstVisibleRow + topOffset;
193 				// We use getDisplayedRowCount to take into account changes to do with insertion and deletion.
194 				var last = Math.min(lastVisibleRow+topOffset, topOffset+this.getDisplayedRowCount()-1); 
195 				pair = {first : first, last: last};
196 			}
197 			else
198 			{
199 				//Purely standard paging
200 				// TODO: although this should be correct it is causing double gets at startup if we include it.
201 
202 				var first = topOffset;
203 				//var first = this.getRowsPerPage() * this.getCurrentPageIndex();
204 				// We use getDisplayedRowCount to take into account changes to do with insertion and deletion.
205 				var last = first + this.getDisplayedRowCount() - 1;
206 				pair = {first : first, last: last};
207 			}
208 		}
209 		else
210 		{
211 			pair = {first : firstVisibleRow, last: lastVisibleRow};
212 		}
213 
214 		this.onRangeUpdate.notify(pair);
215 	}
216 	return pair;
217 }
218 /**
219  * Initiatiate the render operation. Because the data may not yet be available, the actual render doesn't occur until renderReady()
220  */
221 nitobi.grid.Scroller3x3.prototype.render=function(force)
222 {
223 	// Check if the grid is bound to the datasource yet and if the scroll has changed
224 	if (this.owner.isBound() && (this.getScrollTop() != this.lastScrollTop || force || this.scrollTopPercent>.9))
225 	{
226 		// Wait a little bit before rendering, just in case the user has scrolled past this position.
227 		// This enables quickly scrolling through large data sets without rendering every record.
228 		var funcRef = nitobi.lang.close(this, this.performRender, []);
229 		window.clearTimeout(this.lastTimeoutId);
230 		this.lastTimeoutId = window.setTimeout(funcRef, EBAScroller_RENDERTIMEOUT);
231 	}
232 }
233 /*
234  * The actual render operation
235  */
236 nitobi.grid.Scroller3x3.prototype.performRender=function()
237 {
238   	var visibleRows = this.getUnrenderedBlocks();
239 
240 	if (visibleRows==null) {return; }
241 
242   	var scrollTop = this.getScrollTop();
243 
244 	var mc = this.view.midcenter;
245 	var ml = this.view.midleft;
246 	// Logic for increasing gap size moved into scroller
247 	var datatable = this.getDataTable();
248 	var first = visibleRows.first;
249 	var last = visibleRows.last;
250 	if (last>=datatable.remoteRowCount-1 && !datatable.rowCountKnown) {
251 		last+=2;
252 	}
253 	var gaps = this.cacheMap.gaps(first,last);
254 
255 	// Behaviour is different on an empty grid when in livescrolling vs. other
256 	var noRows = (this.owner.mode=="livescrolling"?(first+last<=0):(first+last<=-1));
257 
258 	if (noRows) {
259 		// If there are no rows to render then HTML is ready
260 		this.onHtmlReady.notify();
261 	} else if (gaps[0] != null) {
262 		var low = gaps[0].low;
263 		var high = gaps[0].high;
264 
265 		var rows = high - low + 1;
266 
267 		// If not all data is in cache we must issue a get
268 		if (!datatable.inCache(low, rows)) {
269 			if (low==null || rows==null) {
270 				alert("low or rows =null")
271 			}
272 			if (this.implementsStandardPaging())
273 			{
274 				var firstRow = this.getCurrentPageIndex() * this.getRowsPerPage();
275 				var lastRow = firstRow + this.getRowsPerPage();
276 				datatable.get(firstRow, lastRow);
277 			}
278 			else
279 			{
280 				datatable.get(low, rows);
281 			}
282 			// We may already have at least some of the data ... so render what we got
283 			var cached = datatable.cachedRanges(low,high);
284 			for (var i=0;i<cached.length;i++) {
285 				var subGaps = this.cacheMap.gaps(cached[i].low,cached[i].high);
286 				for (var j=0;j<subGaps.length;j++) {
287 					visibleRows.first = subGaps[j].low;
288 					visibleRows.last = subGaps[j].high;
289 
290 					this.renderGap(subGaps[j].low,subGaps[j].high);
291 				}
292 			}
293 			return false;
294 		} else {
295 			//this.cacheMap.insert(low, high);
296 			this.renderGap(low, high);
297 		}
298 	}
299 
300 	this.onRenderComplete.notify();
301 }

306 nitobi.grid.Scroller3x3.prototype.renderGap = function(low,high)
307 {
308 	var gaps = this.cacheMap.gaps(low,high);
309 	var mc = this.view.midcenter;
310 	var ml = this.view.midleft;
311 
312 	if (gaps[0] != null)
313 	{
314 		var low = gaps[0].low;
315 		var high = gaps[0].high;
316 
317 		var rows = high - low + 1;
318 		this.cacheMap.insert(low, high);
319 
320 		mc.renderGap(low,high);
321 		ml.renderGap(low,high);
322 	}
323 }

338 nitobi.grid.Scroller3x3.prototype.getViewportByCoords=function(row, column) 
339 {
340 	var topOffset = 0;
341 	// Top Left
342 	if (row>=topOffset && row<this.owner.getfreezetop() && column>=0 && column<this.owner.frozenLeftColumnCount())
343 		return this.view.topleft;
344 	// Top Center
345 	if (row>=topOffset && row<this.owner.getfreezetop() && column>=this.owner.getFrozenLeftColumnCount() && column<this.owner.getColumnCount())
346 		return this.view.topcenter;
347 	// Mid Left
348 	if (row>=this.owner.getfreezetop()+topOffset && row<this.owner.getDisplayedRowCount()+topOffset && column>=0 && column<this.owner.getFrozenLeftColumnCount())
349 		return this.view.midleft;
350 	// Mid Center
351 	if (row>=this.owner.getfreezetop()+topOffset && row<this.owner.getDisplayedRowCount()+topOffset && column>=this.owner.getFrozenLeftColumnCount() && column<this.owner.getColumnCount())
352 		return this.view.midcenter;
353 }

420 nitobi.grid.Scroller3x3.prototype.dispose = function()
421 {
422 	try
423 	{
424 		(this.cacheMap!= null?this.cacheMap.flush():'');
425 		this.cacheMap = null;
426 
427 		var disposalLength = this.disposal.length;
428 		for (var i=0; i<disposalLength; i++)
429 		{
430 			if (typeof(this.disposal[i]) == "function")
431 			{
432 				this.disposal[i].call(this);
433 			}
434 			this.disposal[i] = null;
435 		}
436 
437 		for (var v in this.view)
438 		{
439 			this.view[v].dispose();
440 		}
441 
442 		for (var item in this)
443 		{
444 			if (this[item] != null && this[item].dispose instanceof Function)
445 			{
446 				this[item].dispose();
447 			}
448 		}
449 	}
450 	catch(e)
451 	{
452 //		ntbAssert();
453 	}
