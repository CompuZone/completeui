181 /**
182  * @private
183  * @ignore
184  */
185 function Debug_ShowCallStack(){
186 	var s="";
187 	var tabs="\t";
188 	for (var i=0; i<this.m_CallStackMarker; i++)
189 	{
190 		s += tabs + this.m_CallStack[i] + "\n";
191 		tabs+="\t";
192 	}
193 	return s;
194 }

209 /**
210  * @private
211  * @ignore
212  */
213 function Debug_SetWatch(VariableName, Variable)
214 {
215 	this.EnterFunction('SetWatch');
216 	try
217 	{
218 		_ebaWatch.value = VariableName + " = " + Variable + "\n" + _ebaWatch.value;
219 		this.ExitFunction();
220 	}catch(err)
221 	{
222 		//Debug.Assert(false,err.message);
223 		this.ExitFunction();
224 	}
225 }
