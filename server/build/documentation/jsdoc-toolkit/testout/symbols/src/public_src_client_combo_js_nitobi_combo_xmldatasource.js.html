  6  * client side data. The XML stored in this object is compressed EBA XML in the form:
  7  * <div class="code">
  8  * <pre><code class="xml">
  9  * <root>
 10  * 	<e xk="1" a="The Clash" b="London Calling"/>

 12  * 	<e xk="3" a="XTC" b="Black Sea"/>
 13  * </root>
 14  * </code></pre>
 15  * </div>
 16  * @class The client side data source for the List
 17  * @constructor
 18  * @param {HTMLElement} userTag The HTML object that represents ListColumnDefinition
 19  * @param {Boolean} clip Whether or not to clip list
 20  * @param {Number} clipLength The clip length to use to clip the list
 21  * @param {nitobi.combo.Combo} comboObject The owner combo object
 22  * @see nitobi.combo.Combo
 23  */

 96 nitobi.combo.XmlDataSource.prototype.SetXmlObject = XmlDataSource_SetXmlObject;
 97 /**
 98  * Sets the XML DOM object.
 99  * The type of this object changes depending on the type of the browser.
100  * In Internet Explorer it is the latest version (up to version 4) of the Microsoft XML DOM, e.g.,
101  * Msxml4.DOMDocument. For Mozilla based browsers, this is the standard XML doc document. Note: Some
102  * text in the document is escaped.
103  * @param {XMLDocument} XmlObject The XML document
104  * @param {Number} clip Whether or not to clip list
105  * @param {Boolean} clipLength The clip length to use to clip the list
106  * @private
107  */
108 function XmlDataSource_SetXmlObject(XmlObject, clip, clipLength){
109 	if(null==XmlObject.documentElement) return;
110 	if(clip==true)
111 		XmlObject = xbClipXml(XmlObject, "root", "e", clipLength);
112 	this.m_XmlObject = XmlObject;
113 	// Since the whole object was blown away, treat this like a getpage.
114 	// Set the page size to number of rows we receive initially.
115 	// Here we assume that the initial data we receive is like a page
116 	// get.  We don't actually do a page get because we don't want
117 	// to copy our initial xml datasource.
118 	this.SetLastPageSize(this.GetNumberRows());
119 
120 	var fields = XmlObject.documentElement.getAttribute("fields");
121 	if (null==fields){
122 		// TODO: How are we going to deal with exceptions in dontnet.
123 //			alert( "You must have at least one field in your dataset.  The root's 'fields' attribute was not set.");
124 	}else{
125 		var colNames = fields.split("|");
126 		this.SetColumnNames(colNames);
127 		this.SetNumberColumns(colNames.length);
128 	}
129 }

216 /**
217  * Searches the list and returns the ordinal value of the first matched row. Searches the local cache only.
218  * @param {String} Value The value to match.
219  * @param {Int} ColumnIndex The column to search on.
220  * @param {Boolean} Smart Whether or not this is a "smart" mode search.
221  */
222 nitobi.combo.XmlDataSource.prototype.Search = function(Value, ColumnIndex, Smart){
223 
224 	Value = Value.toLowerCase();
225 	Value=nitobi.xml.constructValidXpathQuery(Value,true);
226 
227 	// ** A few notes on this Search mechanism. **//
228 	// ** There is only one central xml datasource object. IE searches this datasource object using XSL. 
229 	// ** The XSL calls the js toLowerCase() function in order to do case-insensitive search.
230 	// ** Netscape has no such facility. For netscape, we must mirror the xml object and make a lower case version of it.
231 	// ** This is not done for IE because it is more error prone. I'm trying to minimize the risk here. Sometimes, people
232 	// ** modify the xml directly.
233 	var xsl="<xsl:stylesheet xmlns:xsl=\"http://www.w3.org/1999/XSL/Transform\" version=\"1.0\">";
234 	xsl+="<xsl:output method=\"text\" />";
235 	xsl+="<xsl:template match=\"/\"><xsl:apply-templates select=\"//e[" + (Smart==true ? "contains" : "starts-with") + "(@"+String.fromCharCode(97 + parseInt(ColumnIndex))+"," + Value + ")][1]\"/></xsl:template>";
236 	xsl+="<xsl:template match=\"e\">";
237 	// Note this needs to be indexing from 0 not 1.
238 	xsl+="<xsl:value-of select=\"count(preceding-sibling::e)\" />";
239 	xsl+="</xsl:template>";
240 	xsl+="</xsl:stylesheet>";
241 
242 	var oXSL=nitobi.xml.createXslProcessor(xsl);
243 	var searchXml = nitobi.xml.createXmlDoc(this.GetXmlObject().xml.replace(/>\s+</g,"><").toLowerCase());
244 	var index = nitobi.xml.transformToString(searchXml, oXSL);
245 
246 	if("" == index)
247 		index = -1;
248 	return parseInt(index);
249 }

252 // * AddPage
253 // *****************************************************************************
254 /// <function name='AddPage' access='private'>
255 /// <summary>Adds a page of XML to the local datasource.</summary>
256 /// <param name='XML' type='string' >The xml page including the root tags.</param>
257 /// </function>
258 nitobi.combo.XmlDataSource.prototype.AddPage = XmlDataSource_AddPage;
259 /**
260  * @private
261  */
262 function XmlDataSource_AddPage(XML){
263 	// Got rid of the lower case XML crap for Netscape
264 	
265 	// Copy the XML into our local datasource.
266 	var tmp=nitobi.xml.createXmlDoc(XML);
267 	var newNodes = tmp.selectNodes("//e");
268 	var root = this.GetXmlObject().documentElement;
269 	// In the future we may support different page sizes. For now
270 	// this should be the same as the specified preset page size.
271 	this.SetLastPageSize(tmp.selectNodes("//e").length);
272 	for(var i=0; i<newNodes.length; i++)
273 	{
274 		// removed some lower case XML crap here
275   		root.appendChild(newNodes[i].cloneNode(true));
276   	}
277   	
278   	this.m_Dirty=false;
279 }

313 nitobi.combo.XmlDataSource.prototype.GetRow = XmlDataSource_GetRow;
314 /**
315  * Returns a row from the dataset.
316  * This returns a row from the XML cache. The ordinal position of the row
317  * corresponds to the ordinal position of the row displayed in the list. If you want 
318  * to get all rows, i.e., the entire dataset, you should use {@link #GetXmlObject}.
319  * <p>
320  * <b>Example</b>
321  * </p>
322  * <div class="code">
323  * <pre><code class="javascript">
324  * var datasource = nitobi.getComponent('combo1').GetList().GetXmlDataSource();
325  * var row = datasource.GetRow(0);
326  * alert('The 3rd column of data: ' + row[2]);
327  * </code></pre>
328  * </div>
329  * @param {Number} Index The index of the row you want to retrieve. Indexed from zero
330  * @type Array
331  */
332 function XmlDataSource_GetRow(Index){
333 	Index = parseInt(Index);
334 	var row = this.GetXmlObject().documentElement.childNodes.item(Index);
335 	var values = new Array;
336 	for (var i = 0; i < this.GetNumberColumns(); i++){
337 		values[i] = row.getAttribute(String.fromCharCode(97+i));
338 	}
339 	return values;
340 }

362 // *****************************************************************************
363 /// <function name='GetColumnIndex' access='private'>
364 /// <summary>Returns a column index given its name.</summary>
365 /// <param name='Name' type='string' >The name of the column whose index you want to find.</param>
366 /// </function>
367 nitobi.combo.XmlDataSource.prototype.GetColumnIndex = XmlDataSource_GetColumnIndex;
368 /**
369  * @private
370  */
371 function XmlDataSource_GetColumnIndex(Name){
372 	// better to default to col 0 (i.e. GetColumnIndex(null) now returns 0) than to fail and throw a fit
373 	if(Name==null)
374 		return 0;
375 		
376 	Name=Name.toLowerCase();
377 	var colNames = this.GetColumnNames();
378 	
379 	if (colNames != null){
380 		for (var i = 0 ; i < colNames.length ; i++){
381 			if (Name == colNames[i].toLowerCase()){
382 				return parseInt(i);
383 			}
384 		}
385 	}
386 	return -1;
