  4 /**
  5  * Creates a HttpRequest object.  After calling the constructor you can set up the object by setting
  6  * particular fields to the values you desire. 
  7  * @class Makes a cross browser XMLHttpRequest object and processes responses from the server.<br><br>
  8  * <code>var xhr = new nitobi.ajax.HttpRequest();<br>
  9  * xhr.handler = path + "data.xml";<br>
 10  * xhr.async = true; // async is true by default<br>
 11  * xhr.responseType = "xml"; // by default the appropriate responseType will be used - ie XML if the data is valid XML otherwise "text".<br>
 12  * xhr.onGetComplete.subscribeOnce(function(evtArgs) {alert("In onGetComplete:\n" + nitobi.xml.serialize(evtArgs.response))});<br>
 13  * xhr.completeCallback = function(evtArgs) {alert("In completeCallback:\n" + nitobi.xml.serialize(evtArgs.response))};<br>
 14  * xhr.get();</code>
 15  * @constructor
 16  */
 17 nitobi.ajax.HttpRequest = function()
 18 {
 19 	/**
 20 	 * This is the URL of the server resource that data is being posted to or
 21 	 * retrieved from.
 22 	 * @type String
 23 	 */
 24   	this.handler 		= '';
 25   	/**
 26   	 * This specifies if the requests should be made in an asynchronous or synchronous manner.
 27   	 * @type Boolean
 28   	 */
 29 	this.async 	= true;
 30 	/**
 31 	 * This specifies if XML data or general text data is being retrieved from the server. 
 32 	 * Valid values are "xml", "text" or "". Using "text" can be used for returning JSON data or any other
 33 	 * data format including XML. The data passed to the callback method will be an XMLDocument if "xml" is
 34 	 * specified as the responseType, it will be a String if "text" is specified.
 35 	 * @type String
 36 	 */
 37 	this.responseType	= null;
 38 	/**
 39 	 * This is the underlying XMLHttpRequest object.
 40 	 * @type XMLHttpRequest
 41 	 * @private
 42 	 */
 43 	this.httpObj      	= nitobi.ajax.createXmlHttp();
 44 	/**
 45 	 * This is the Event that is fired when a POST request returns from the server.
 46 	 * @type nitobi.base.Event
 47 	 */
 48 	this.onPostComplete	= new nitobi.base.Event();
 49 	/**
 50 	 * This is the Event that is fired when a GET request returns from the server.
 51 	 * @type nitobi.base.Event
 52 	 */
 53 	this.onGetComplete	= new nitobi.base.Event();
 54 	/**
 55 	 * This is the Event that is fired when any request returns from the server.
 56 	 * @type nitobi.base.Event
 57 	 */
 58 	this.onRequestComplete	= new nitobi.base.Event();
 59 	/**
 60 	 * This is a function reference to a function that is executed if there is an error during or after the request.
 61 	 * @type nitobi.base.Event
 62 	 */	
 63 	this.onError		= new nitobi.base.Event();
 64 	/**
 65 	 * The length of time that can pass before the request is cancelled. A value of 0 indicates there is no timeout.
 66 	 * @type Number
 67 	 */
 68 	this.timeout = 0;
 69 	/**
 70 	 * The ID of the window.setTimeout registration.
 71 	 * @type Number
 72 	 */
 73 	this.timeoutId = null;
 74 	/**
 75 	 * This is a parameter object that can contain any information needed to be passed
 76 	 * to the postComplete or getComplete functions after an asynchronous request has returned.
 77 	 * This can be used to maintain state across an asynchronous request.
 78 	 * @type Object
 79 	 */
 80 	this.params			= null;
 81 	/**
 82 	 * A copy of the data that is sent to the server in a post
 83 	 * @type String
 84 	 */
 85 	this.data		= "";
 86 	/**
 87 	 * A function to execute when the request completes.
 88 	 * @type Function
 89 	 */
 90 	this.completeCallback = null;
 91 	/**
 92 	 * The status. Either 'pending' or 'complete'.
 93 	 * @type String
 94 	 */
 95 	this.status = "complete";
 96 	/**
 97 	 * Specifies if a unique querystring value should be sent with GET requests to prevent page caching.
 98 	 * Default value is true.
 99 	 */
100 	this.preventCache = true;
101 
102 	/**
103 	 * The username to be sent to the server with the HTTP request.
104 	 */
105 	this.username = "";
106 
107 	/**
108 	 * The password to be sent to the server with the HTTP request.
109 	 */
110 	this.password = "";
111 
112 	/**
113 	 * This is used for keeping track of the method used when the HttpRequest open and send methods are used.
114 	 * @private
115 	 */
116 	this.requestMethod = "get";
117 
118 	/**
119 	 * @private
120 	 */
121 	this.requestHeaders = {};
122 }

132 /**
133  * This method takes the response from the server and checks the HTTP status 
134  * to ensure that the server did not have any errors and returned valid XML.
135  * @type {String | XMLDocument}
136  * @private
137  */
138 nitobi.ajax.HttpRequest.prototype.handleResponse = function()
139 {
140   	var result = null;
141   	var error = null;
142 
143    
144 	if ((this.httpObj.responseXML != null && this.httpObj.responseXML.documentElement != null) && this.responseType != "text")
145 	{
146 		result = this.httpObj.responseXML;
147 	}
148 	else if(this.responseType == "xml")
149     {
150     	result = nitobi.xml.createXmlDoc(this.httpObj.responseText);
151     } 
152     else
153     {
154         result = this.httpObj.responseText;
155     }
156     
157 	if (this.httpObj.status != 200)
158 	{
159 		this.onError.notify({"source":this,"status":this.httpObj.status,"message":"An error occured retrieving the data from the server. " +
160 				"Expected response type was '"+this.responseType+"'."});
161 	}
162 
163 	return result;
164 }

166 /**
167  * This method posts some data to the given url using the XMLHttpRequest object.
168  * Various parameters such as asynchronous and handler should have already been set.
169  * @param {String} data The only argument passed to this method is some data to post
170  * to the server. It can be in any format such as XML or JSON.
171  * @param {String} url The URL where the data is to be posted to. This is option and can also be specified using the handler property.
172  * @return if the request is synchronous, the content of the server response is 
173  * returned as either XML data or plain text, depending on the response type
174  * @type XMLDocument|String
175  */
176 nitobi.ajax.HttpRequest.prototype.post = function(data, url)
177 {
178 	this.data = data;
179 	return this._send("POST", url, data, this.postComplete);
180 }

182 /**
183  * This is used to retrieve data from the server.
184  * It will retrieve data from the URL specified by the handler property of the Callback object.
185  * @return if the request is synchronous, the content of the server response is 
186  * returned as either XML data or plain text, depending on the response type
187  * @param {String} url The URL to request the data from. This is an alternative to setting the handler parameter.
188  * @type String|XMLDocument
189  */
190 nitobi.ajax.HttpRequest.prototype.get = function(url)
191 {
192 	return this._send("GET", url, null, this.getComplete)
193 }

196  * @ignore
197  */
198 nitobi.ajax.HttpRequest.prototype.postComplete = function()
199 {
200     if(this.httpObj.readyState==4)
201     {
202     	this.status = "complete";
203     	var callbackParams = {'response':this.handleResponse(),'params':this.params};
204     	this.responseXml = this.responseText = callbackParams.response;
205     	this.onPostComplete.notify(callbackParams);
206     	this.onRequestComplete.notify(callbackParams);
207     	if (this.completeCallback)
208     	{
209     		this.completeCallback.call(this, callbackParams);
210     	}
211 	}
212 }

214 /**
215  * This does various assertions on XMLData then calls postData.
216  * If xmlData is has no valid child nodes the just return
217  * @param {XmlNode} xmlData The single argument passed to this method is a valid XML DOM object.
218  * @return if the request is synchronous, the content of the server response is 
219  * returned as either XML data or plain text, depending on the response type
220  * @type XMLDocument|String
221  */
222 nitobi.ajax.HttpRequest.prototype.postXml = function(xmlData)
223 {
224 	this.setTimeout();
225 
226     // validate the XML data parameter
227     if(("undefined" == typeof(xmlData.documentElement)) || 
228        (null == xmlData.documentElement) || 
229        ("undefined" == typeof(xmlData.documentElement.childNodes)) ||
230        (1 > xmlData.documentElement.childNodes.length))
231     {
232         ebaErrorReport("updategram is empty. No request sent. xmlData[" + xmlData + "]\nxmlData.xml[" + xmlData.xml + "]");
233         // not sure that return is best here.
234         return; 
235     }
236 
237     //  get data from xml (null == xmlData.xml)
238     if(null == xmlData.xml)
239 	{
240             // looks like we are not running nitobi.browser.IE
241             // lets try to get xml data            
242             var xmlSerializer = new XMLSerializer();
243             xmlData.xml     = xmlSerializer.serializeToString(xmlData);
244 	}
245 	return this.post(xmlData.xml);
246 }

248 /**
249  * @private
250  * Actually makes the Ajax request whether it is called through 
251  * <code>post</code>, <code>get</code>, or <code>send</code>.
252  */
253 nitobi.ajax.HttpRequest.prototype._send = function(method, url, data, completeHandler)
254 {
255 	this.handler = url || this.handler;
256 	this.setTimeout();
257 
258 	this.status = "pending";
259 
260 	this.httpObj.open(method, (this.preventCache?this.cacheBust(this.handler):this.handler), this.async, this.username, this.password);
261 
262     if (this.async)
263 	    this.httpObj.onreadystatechange = nitobi.lang.close(this, completeHandler);
264 
265 	for (var item in this.requestHeaders) {
266 		this.httpObj.setRequestHeader(item, this.requestHeaders[item]);
267 	}
268 
269 	if (this.responseType == "xml")
270 	    this.httpObj.setRequestHeader("Content-Type","text/xml");
271 	else if (method.toLowerCase() == "post")
272 		this.httpObj.setRequestHeader("Content-Type","application/x-www-form-urlencoded");
273 
274 	// TODO: check if the data is a form. if so then serialize it
275 	/*
276 	if (data.tagName == "form") {
277 		var s = "";
278 		for (var i=0; i<data.elements; i++) {
279 			var e = data.elements[i];
280 			s += e.name + "=" + e.value + "&";
281 		}
282 		data = s;
283 	}
284 	 */
285 
286     this.httpObj.send(data);
287 
288 	if (!this.async)
289 		return this.handleResponse();
290 	return this.httpObj;
291 }

293 /**
294  * Compatibility with the native XMLHttpRequest object. Opens the connection to the server.
295  * After calling open headers can be set and the request can be sent using the <code>send</code> method.
296  */
297 nitobi.ajax.HttpRequest.prototype.open = function(method, url, async, username, password) {
298 	this.requestMethod = method;
299 	this.async = async;
300 	this.username = username;
301 	this.password = password;
302 	this.handler = url;
303 }

305 /**
306  * Compatibility with the native XMLHttpRequest object. Sends the request to the server.
307  */
308 nitobi.ajax.HttpRequest.prototype.send = function(data) {
309 	// TODO: Do we want to ensure that the url is specified?
310 	var response = null;
311 	switch (this.requestMethod.toUpperCase())
312 	{
313 		case "POST":
314 			response = this.post(data);
315 			break;
316 		default:
317 			response = this.get();
318 			break
319 	}
320 	this.responseXml = this.responseText = response;
321 }

326 nitobi.ajax.HttpRequest.prototype.setTimeout = function()
327 {
328 	if (this.timeout > 0)
329 	{
330 		this.timeoutId = window.setTimeout(nitobi.lang.close(this, this.abort), this.timeout);
331 	}
332 }

334 /**
335  * The onReadyStateChange event is used internally for monitoring the 
336  * readyState of the XMLHttpRequest when data is being retrieved from the server.
337  * When the readyState is 4 then it will call the function pointer onGetComplete.
338  * @private
339  */
340 nitobi.ajax.HttpRequest.prototype.getComplete = function()
341 {
342     if(this.httpObj.readyState==4)
343     {
344     	this.status = "complete";
345 
346     	var callbackParams = {'response':this.handleResponse(),'params':this.params,'status':this.httpObj.status,'statusText':this.httpObj.statusText};
347     	this.responseXml = this.responseText = callbackParams.response;
348     	this.onGetComplete.notify(callbackParams);
349     	this.onRequestComplete.notify(callbackParams);
350     	if (this.completeCallback)
351     	{
352     		this.completeCallback.call(this, callbackParams);
353     	}
354 	}
355 }

399 /**
400  * Adds a cache-busting querystring parameter to the request.
401  * @private
402  */
403 nitobi.ajax.HttpRequest.prototype.cacheBust = function(url)
404 {
405 	var urlArray = url.split('?');
406 	var param = 'nitobi_cachebust=' + (new Date().getTime());
407 	if (urlArray.length == 1)
408 		url += '?' + param;
409 	else
410 		url += '&' + param;
411 	return url;
