  3 /**
  4  * The base hin t constructor. Use this to create a new callout as a tooltip on an object.<br><Br>
  5  * @param {integer} timerLength How long to wait before showing the callout (on a mouseover). Optional.
  6  * @param {integer} expireTimeout How long to wait getting rid of the tooltip. Optional.
  7  * @param {string} stylesheet The callout stylesheet to use for the hints (XP by default). Optional.
  8  */
  9 nitobi.callout.Hint = function(timerLength, expireTimeout, stylesheet)
 10 {
 11 	/**
 12 	 * @private
 13 	 */
 14 	this.uid = Math.random().toString().replace('.', '').replace('0', '');
 15 	this.ver = '0.9';
 16 	/**
 17 	 * @private
 18 	 */
 19 	this.activeHints = Array();
 20 	/**
 21 	 * @private
 22 	 */
 23 	this.timerObj = null;
 24 	/**
 25 	 * @private
 26 	 */
 27 	this.timerLength = 1000;
 28 	if (timerLength != null) 
 29 	{
 30 		this.timerLength = timerLength
 31 	}
 32 	if (expireTimeout != null) 
 33 	{
 34 		this.expireTimeout = expireTimeout
 35 	}
 36 	this.stylesheet = 'xp';
 37 	if (stylesheet != null)
 38 	{
 39 		this.stylesheet = stylesheet;
 40 	}
 41 }

 44 /**
 45  * @private
 46  */
 47 nitobi.callout.Hint.prototype.runHint = function(objID, hintTitle, hintText) 
 48 {
 49 	/**
 50 	 * @private
 51 	 */
 52 	var ff = this;
 53 	clearTimeout(this.timerObj);
 54 	
 55 	if (this.activeHints[objID] != true) 
 56 	{
 57 		this.timerObj = setTimeout(function(){ff.execHint(objID,hintTitle,hintText)}, this.timerLength);
 58 	}
 59 
 60 }
 61 /**
 62  * @private
 63  */
 64 nitobi.callout.Hint.prototype.execHint = function(objID, hintTitle, hintText) 
 65 {
 66 	var TimeoutLength = 0;
 67 	var athis = this;
 68 	if (this.activeHints[objID] != true) 
 69 	{
 70 		this.activeHints[objID] = true;
 71 		var myNtbCalloutObj = new nitobi.callout.Callout(this.stylesheet); 
 72 		myNtbCalloutObj.attachToElement(objID); 
 73 		myNtbCalloutObj.setTitle(hintTitle); 
 74 		myNtbCalloutObj.setBody(hintText); 
 75 		if (this.expireTimeout != null) 
 76 		{
 77 			TimeoutLength = this.expireTimeout;
 78 		} 
 79 		else 
 80 		{
 81 			TimeoutLength = 3000 + (hintText.split(" ").length*220);
 82 		}
 83 		myNtbCalloutObj.setExpire(TimeoutLength);
 84 		myNtbCalloutObj.ondestroy = function() {athis.activeHints[objID] = false;}
 85 		myNtbCalloutObj.show();
 86 		myNtbCalloutObj = null;
 87 	}
 88 }

 90 /**

 92  * @param {string} objID The ID of the DOM element.
 93  * @param {string} hintTitle The title of the tooltip. Optional.
 94  * @param {string} hintText The body of the tooltip. Optional. Need either a hintTitle, hintText, or both.
 95  */
 96 nitobi.callout.Hint.prototype.addHint = function(objID, hintTitle, hintText) 
 97 {
 98 	var abd = this;
 99 	try 
100 	{
101 		document.getElementById(objID).onmouseover = function() { abd.runHint(this.id, hintTitle, hintText);};
102 		document.getElementById(objID).onmouseout = function() { clearTimeout(abd.timerObj);}
103 		document.getElementById(objID).onmousedown = function() { abd.execHint(this.id, hintTitle, hintText);};
104 		document.getElementById(objID).id = objID;
105 	} 
106 	catch(e) 
107 	{
108 		for(var i=0; i<document.forms.length; i++)
109 		{
110 			for(var b=0; b<document.forms[i].elements.length; b++)
111 			{
112 				if (document.forms[i].elements[b].name.toUpperCase() == objID.toUpperCase()) 
113 				{
114 					document.forms[i].elements[b].onmouseover = function() { abd.runHint(this.id, hintTitle, hintText);};
115 					document.forms[i].elements[b].onmouseout = function() { clearTimeout(abd.timerObj);}
116 					document.forms[i].elements[b].onmousedown = function() { abd.execHint(this.id, hintTitle, hintText);};
117 					document.forms[i].elements[b].id = objID;
118 				}
119 			
120 			}
121 		}
122 	}
