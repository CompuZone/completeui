110 /**
111  * @private
112  */
113 nitobi.grid.Row.getRowElements = function(grid, row, surfacePath)
114 {
115 	surfacePath = surfacePath || "";
116 	var C = nitobi.grid.Cell;
117 	// TODO: refactor this offset stuff into a method in grid.
118 	
119 	var midCol = grid.getFrozenLeftColumnCount();
120 	var cell = C.getCellElement(grid, row, midCol, surfacePath);
121 	var midRow = (cell?cell.parentNode:null);
122 	if (!midCol)
123 		return {left: null, mid: midRow};
124 	
125 	var rows = {};
126 	rows.left = C.getCellElement(grid, row, 0).parentNode;
127 	rows.mid = midRow;
128 	return rows;
129 }
