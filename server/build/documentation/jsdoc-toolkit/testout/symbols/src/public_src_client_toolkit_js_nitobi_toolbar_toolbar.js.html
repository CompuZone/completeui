 56 /// <function name='nitobi.ui.Toolbar' access="private">
 57 /// <summary>Constructor for nitobi.ui.Toolbar</summary>

 60 nitobi.ui.Toolbar = function (xml,id)
 61 {
 62 	nitobi.ui.Toolbar.baseConstructor.call(this);

 73 nitobi.ui.Toolbar.prototype.getUiElements = function ()

 86 /**
 87  * Attaches all the elements that to toolbar has to the HTML elements
 88  * and the toolbar object.
 89  * @private
 90  */
 91 nitobi.ui.Toolbar.prototype.attachButtonObjects = function ()
 92 {
 93 	if (!this.m_UiElements)
 94 	{
 95 		this.m_UiElements = new Array();
 96 		var tag = this.getHtmlElementHandle();
 97 		var children = tag.childNodes;
 98 
 99 		for (var i = 0; i < children.length; i++)
100 		{
101 			var child = children[i];
102 			// Don't process white space nodes.
103 			if (child.nodeType!=3)
104 			{
105 				var newElement;
106 				switch(child.className)
107 				{
108 					case("ntb-button"):
109 					{
110 						newElement = new nitobi.ui.Button(null,child.id);	
111 					
112 						break;
113 					}
114 					case("ntb-binarybutton"):
115 					{
116 						newElement = new nitobi.ui.BinaryStateButton(null,child.id);	
117 						break;
118 					}
119 					default:
120 					{
121 						newElement = new nitobi.ui.UiElement(null,null,child.id);	
122 						break;
123 					}
124 				}
125 				newElement.attachToTag();
126 				this.m_UiElements[child.id]=newElement;
127 			}
128 		}
129 	}
130 }

132 /**
133  * Renders the toolbar inside of the specified container. If the container is
134  * null, then it renders at the end of the page.
135  * @param htmlElement {object} The html element inside which the toolbar is rendered.
136  */
137 nitobi.ui.Toolbar.prototype.render = function (htmlElement)
138 {
139 	nitobi.ui.Toolbar.base.base.render.call(this, htmlElement);
140 	this.attachButtonObjects();
141 }
142 
143 /**
144  * Disables each element in the toolbar.
145  */
146 nitobi.ui.Toolbar.prototype.disableAllElements = function ()
147 {
148 	for (var i in this.m_UiElements) 
149 	{
150 		if (this.m_UiElements[i].disable)
151 		{
152 			this.m_UiElements[i].disable();
153 		}
154 	}
155 }
156 
157 /**
158  * Enables all elements in the toolbar.
159  */
160 nitobi.ui.Toolbar.prototype.enableAllElements = function()
161 {
162 	for (var i in this.m_UiElements) 
163 	{
164 		if (this.m_UiElements[i].enable)
165 		{
166 			this.m_UiElements[i].enable();
167 		}
168 	}
169 }
170 
171 
172 /**
173  * Attaches the toolbar object to the toolbar. You can then access the object using
174  * the elements javascriptObject property.
175  */
176 nitobi.ui.Toolbar.prototype.attachToTag = function ()
177 {
178 	nitobi.ui.Toolbar.base.base.attachToTag.call(this);
179 	this.attachButtonObjects();
180 }
181 
182 nitobi.ui.Toolbar.prototype.dispose = function ()
183 {
184 	if (typeof(this.m_UiElements) != "undefined")
185 	{
186 		for (var button in this.m_UiElements)
187 		{
188 			this.m_UiElements[button].dispose();
189 		}
190 		this.m_UiElements = null;
191 	}
192 	nitobi.ui.Toolbar.base.dispose.call(this);
193 }
