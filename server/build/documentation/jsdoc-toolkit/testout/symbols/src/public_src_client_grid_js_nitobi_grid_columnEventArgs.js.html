  3 /**
  4  * Constructs a ColumnEventArgs object.
  5  * @class When you subscribe to Grid events through the declaration, you
  6  * can optionally pass information about the event to the function
  7  * registered to handle it.  You do this by using the eventArgs keyword.
  8  * @constructor
  9  * @param {nitobi.grid} source The object which is firing the event.
 10  * @param {nitobi.grid.Cell} cell The Cell object of the cell that received focus.
 11  */
 12 nitobi.grid.ColumnEventArgs = function(source, column)
 13 {
 14 	/**
 15 	 * @private
 16 	 */
 17 	this.grid = source;
 18 	/**
 19 	 * @private
 20 	 */
 21 	this.column = column;
 22 	/**
 23 	 * @private
 24 	 */
 25 	this.event = nitobi.html.Event;
 26 }

 28 /**

 37  * Gets the Cell on which the event was fired.
