  3 /**
  4  * When calling ISerializable, the class must identify itself
  5  * using a partial profile and it must be registered.  If the object is created by a node using the factory, then
  6  * the profile is deduced from the nodename.<p>
  7  * E.g.<br>
  8  * <code>nitobi.ui.Button.baseConstructor.call(this,{className:"nitobi.ui.Button"},xmlNode);</code><p>
  9  * @class A generalized form of a wrapper around an XML node that allows for serialization to xml.
 10  * This class allows for the creation of a javascript object from an XML node. 
 11  * This means that any Xml node can quickly be transformed into its corresponding javascript object and treated accordingly. 
 12  * The node is used for data storage.  This interface is used to correctly serialize and deserialize javascript objects.<p>
 13  * 
 14  * @constructor
 15  * @extends nitobi.Object
 16  * @param {object} xmlNode Optional. The node that specifies the object's data. Optional. If this is null, a new node is created.
 17  * @param {string} id  The Id of this object. Optional.
 18  * @param {string} xml  The XML string that defines the object and its children. Used to deserialize. Optional.
 19  * @param {nitobi.base.ISerializable} parent The object that is the parent or container of this object.
 20  */
 21 nitobi.base.ISerializable = function(xmlNode, id, xml, parent) 
 22 {
 23 	nitobi.Object.call(this);
 24 	if (typeof(this.ISerializableInitialized) == "undefined")
 25 	{
 26 		/**
 27 		 * When <code>true</code>, this object has been initialized from an xml node or 
 28 		 * in-page XML declaration.
 29 		 * @type Boolean
 30 		 */
 31 		this.ISerializableInitialized = true;
 32 	}
 33 	else
 34 	{
 35 		return;
 36 	}
 37 	
 38 	/**
 39 	 * The XML Node to which this object is attached.
 40 	 * @type Node
 41 	 * @private
 42 	 */
 43 	this.xmlNode = null;
 44 	this.setXmlNode(xmlNode);
 45 	
 46 	// Fill out as much info as we can about this instance.
 47 	if (xmlNode != null)
 48 	{
 49 		/**
 50 		 * Describes this object.
 51 		 * @type {nitobi.base.Profile}
 52 		 * @private
 53 		 */
 54 		this.profile = nitobi.base.Registry.getInstance().getCompleteProfile({idField:null,tagName:xmlNode.nodeName});
 55 	}
 56 	else
 57 	{
 58 		this.profile = nitobi.base.Registry.getInstance().getProfileByInstance(this);
 59 	}
 60 		
 61 	/**
 62 	 * Fires after deserialization.
 63 	 * @type nitobi.base.Event
 64 	 */
 65 	this.onDeserialize = new nitobi.base.Event();
 66 	
 67 	/**
 68 	 * Fires after the node knows it's parent.
 69 	 * @private
 70 	 * @type nitobi.base.Event;
 71 	 */
 72 	this.onSetParentObject = new nitobi.base.Event();
 73 	
 74 	/**
 75 	 * The factory used to create this object and its children.
 76 	 * @type Factory
 77 	 * @private
 78 	 */
 79 	this.factory = nitobi.base.Factory.getInstance();
 80 	/**
 81 	 * A cache of objects already created from nodes.
 82 	 * @type HashMap
 83 	 * @private
 84 	 */
 85 	this.objectHash = {};
 86 	
 87 	
 88 	/**
 89 	 * Fired when an object is newly created.
 90 	 * @type nitobi.base.Event
 91 	 */
 92 	this.onCreateObject = new nitobi.base.Event();
 93 
 94 	if (xmlNode != null)
 95 	{
 96 		this.deserializeFromXmlNode(this.getXmlNode());
 97 	}
 98 	else if (this.factory!=null && this.profile.tagName != null)
 99 	{
100 		this.createByProfile(this.profile, this.getXmlNode());
101 	}
102 	else
103 	{
104 		if (xml != null && xmlNode != null)
105 		{
106 			this.createByXml(xml);
107 		}
108 	}
109 
110 	this.disposal.push(this.xmlNode);
111 }

194 	this.addToCache(child);

205 	child.setXmlNode(this.xmlNode.appendChild(nitobi.xml.importNode(this.xmlNode.ownerDocument, xmlNode, true)));

233  * @return A new Xml node/element.
234  * @type XMLNode
235  * @param name {String} name The name of the node including the ns prefix.

240 	var xmlDoc;
241 	if (this.xmlNode == null || this.xmlNode.ownerDocument == null)
242 	{
243 		xmlDoc = nitobi.xml.createXmlDoc();	
244 	}
245 	else

248 	}
249 	
250 	
251 	if (nitobi.browser.IE)
252 	{
253 		return xmlDoc.createNode(1, name, nitobi.base.XmlNamespace.uri);
254 	}
255 	else if (xmlDoc.createElementNS)
256 	{
257 		return xmlDoc.createElementNS(nitobi.base.XmlNamespace.uri, name);
258 	} else
259 	{
260 		nitobi.lang.throwError("Unable to create a new xml node on this browser.");
261 	}

311 /**
312  * @private
313  */
314 nitobi.base.ISerializable.prototype.areGuidsGenerated = function(xmlNode)
315 {
316 	if (xmlNode == null || xmlNode.ownerDocument == null) return false;
317 	if (nitobi.browser.IE)
318 	{
319 		var node = xmlNode.ownerDocument.documentElement;
320 		if (node == null)
321 		{
322 			return false;
323 		}
324 		else
325 		{
326 			var id = node.getAttribute("id");
327 			if (id == null || id == "")
328 			{
329 				return false;
330 			} 
331 			else
332 			{
333 				return (nitobi.base.ISerializable.guidMap[id] != null);
334 			}
335 		}
336 	}
337 	else
338 	{
339 		return (xmlNode.ownerDocument.generatedGuids == true);
340 	}
341 }

344  * @private
345  */
346 nitobi.base.ISerializable.prototype.setGuidsGenerated = function(xmlNode,value)
347 {
348 	if (xmlNode == null || xmlNode.ownerDocument == null) return;
349 	if (nitobi.browser.IE)
350 	{
351 		var node = xmlNode.ownerDocument.documentElement;
352 		if (node != null)
353 		{
354 			var id = node.getAttribute("id");
355 			if (id != null && id != "")
356 			{
357 				nitobi.base.ISerializable.guidMap[id] = true;
358 			} 
359 		}
360 	}
361 	else
362 	{
363 		xmlNode.ownerDocument.generatedGuids = true;
364 	}

387 /**
388  * Deserializes the object from an XML node. All references to the properties of the 
389  * object are destroyed or invalidated.  The cache is also emptied.  The XML Node
390  * should be of the same type as the object, that is, don't deserialize a node such as 
391  * <foo> if the class is not Foo.
392  * @param {XMLNode} xmlNode The XML Node from which this object will be deserialized.
393  */
394 nitobi.base.ISerializable.prototype.deserializeFromXmlNode = function(xmlNode)
395 {
396 	if (!this.areGuidsGenerated(xmlNode))
397 	{
398 		xmlNode = this.generateGuids(xmlNode);
399 	}
400 	this.setXmlNode(xmlNode);
401 	this.flushCache();
402 		
403 	if (this.profile == null)
404 	{
405 		this.profile = nitobi.base.Registry.getInstance().getCompleteProfile({idField:null,tagName:xmlNode.nodeName});
406 	}
407 
408 	this.onDeserialize.notify();
409 }

707  * @param {nitobi.base.Profile/nitobi.base.ISerializable} locator Either the profile of the object, or the object itself. 
708  * If the locator is an ISerializable object, it may or may not be the same object that is currently being stored.
709  * @param {String} If there is more than one type of this object being stored, then you must include the instance name. 
710  * If only one type of this object is stored, the name is not required.
711  * @type String
712  * @private
713  */
714 nitobi.base.ISerializable.prototype.getChildObjectId = function(locator, instanceName) 
715 {
716 	var tagname = (typeof(locator.className) == "string" ? locator.tagName : locator.getXmlNode().nodeName);
717 	var xpath = tagname;
718 	if (instanceName)
719 	{
720 		xpath += "[@instancename='"+instanceName+"']";
721 	}
722 	
723 	var node = this.getXmlNode().selectSingleNode(xpath);
724 	if (null == node)
725 	{
726 		return null;
727 	}
728 	else
729 	{
730 		return node.getAttribute("id");
731 	}
732 }

734 /**
735  * Sets an object value as a sub element of the object.  The object must implement 
736  * ISerializable otherwise an error is thrown.  In order to uniquely identify child objects of the same type,
737  * an id is used. If getId() is not defined on the object, it is added and the name
738  * is used as an id.
739  * @param {String} instanceName The name of the object. This must be a valid XML attribute name.  If no name is specified
740  * it is assumed that there is only one type of this object.
741  * @param {Object} value The value. Must implement ISerializable.
742  */
743 nitobi.base.ISerializable.prototype.setObject = function(value, instanceName) 
744 {
745 	if (value.ISerializableImplemented != true)
746 	{
747 		nitobi.lang.throwError(nitobi.error.ExpectedInterfaceNotFound + " ISerializable");
748 	}
749 	
750 	var id = this.getChildObjectId(value,instanceName);
751 	if (null != id)
752 	{
753 		this.deleteChildObject(id);
754 	}
755 	if (instanceName)
756 	{
757 		value.setAttribute("instancename",instanceName);
758 	}
759 	this.addChildObject(value);
760 }

762 /**
763  * Returns a child object of the current object. The class must be 
764  * registered with the factory this object is using.  If the current object contains
765  * more that one type of the child, the name of the child is used as an id.
766  * @param {String} instanceName The name of the child. Optional. If this is not supplied, then
767  * the first child that matches the classname is retrieved.
768  * @param {nitobi.base.Profile} profile The profile of the child you want to retrieve.
769  * @returns {Object} The child object.
770  */
771 nitobi.base.ISerializable.prototype.getObject = function(profile, instanceName) 
772 {
773 	var id = this.getChildObjectId(profile,instanceName);
774 	if (null == id)
775 	{
776 		return id;
777 	}
778 	return this.getChildObject(id);
779 }

801 /**
802  * Returns an array of xmlNodes that describes the path from this node
803  * to a descendant node.
804  * @param {String} id The id of the descendant.
805  * @type Array
806  * @private
807  */
808 nitobi.base.ISerializable.prototype.getPathToLeaf = function(id)
809 {
810 	var node = this.getXmlNode();
811 	var child = node.selectSingleNode("//*[@id='" + id + "']");
812 	if (nitobi.browser.IE)
813 	{
814 		child.ownerDocument.setProperty("SelectionLanguage","XPath");
815 	}
816 	var pathNodes = child.selectNodes("./ancestor-or-self::*");
817 	var thisId = this.getId();
818 	var start=0;
819 	for (var i=0;i<pathNodes.length;i++)
820 	{
821 		if (pathNodes[i].getAttribute("id") == thisId)
822 		{
823 			start = i+1;
824 			break;
825 		}
826 	}
827 	var arr = nitobi.collections.IEnumerable.createNewArray(pathNodes,start);
828 	return arr.reverse();
829 }

832 /**
833  * @private
834  */
835 nitobi.base.ISerializable.prototype.isDescendantInstantiated = function(id)
836 {
837 	var node = this.getXmlNode();
838 	var child = node.selectSingleNode("//*[@id='" + id + "']");
839 	if (nitobi.browser.IE)
840 	{
841 		child.ownerDocument.setProperty("SelectionLanguage","XPath");
842 	}
843 	var pathNodes = child.selectNodes("ancestor::*");
844 	var startPushing = false;
845 	var obj = this;
846 	for (var i=0;i<pathNodes.length;i++)
847 	{
848 		if (startPushing)
849 		{
850 			var pathId = pathNodes[i].getAttribute("id");
851 			instantiated = obj.inCache(pathId);
852 			if (!instantiated)
853 			{
854 				return false;
855 			}
856 			obj = this.getObjectById(pathId)
857 		}
858 		if (pathNodes[i].getAttribute("id") == this.getId())
859 		{
860 			startPushing = true;
861 		}
862 	}
863 	return obj.inCache(id);
864 }
