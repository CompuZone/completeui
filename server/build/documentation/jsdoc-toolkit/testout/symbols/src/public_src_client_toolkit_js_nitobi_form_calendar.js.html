 31 /**
 32  * This is called each time the editor is ready to be used.
 33  * @private
 34  */
 35 nitobi.form.Calendar.prototype.initialize = function()
 36 {
 37 	this.datePicker = new nitobi.calendar.DatePicker(nitobi.component.getUniqueId());
 38 	this.datePicker.setContainer(this.pickerDiv);
 39 	this.datePicker.onSetDate.subscribe(this.handlePick,this);
 40 
 41 	var tc = this.control;
 42 	var H = nitobi.html;
 43 	H.attachEvent(tc, 'keydown', this.handleKey, this, false);
 44 	H.attachEvent(tc, 'blur', this.deactivate, this, false);
 45 
 46 	H.attachEvent(this.pickerDiv, 'mousedown', this.handleCalendarMouseDown, this);
 47 	H.attachEvent(this.pickerDiv, 'mouseup', this.handleCalendarMouseUp, this);
 48 
 49 	var a = this.placeholder.getElementsByTagName('a')[0];
 50 	H.attachEvent(a, 'mousedown', this.handleClick, this);
 51 	H.attachEvent(a, 'mouseup', this.handleMouseUp, this);
 52 }

 55 /**
 56  * @private
 57  */
 58 nitobi.form.Calendar.prototype.bind = function(owner, cell, initialKeyChar)
 59 {	
 60 	this.isPickerVisible = false;
 61 	nitobi.html.Css.addClass(this.pickerDiv, NTB_CSS_HIDE);
 62 
 63 	nitobi.form.Calendar.base.bind.apply(this, arguments);
 64 
 65 	if(initialKeyChar != null && initialKeyChar != '')
 66 		this.control.value = initialKeyChar;
 67 	else
 68 		this.control.value = cell.getValue();
 69 
 70 	this.column = this.cell.getColumnObject();
 71 	this.control.maxlength = this.column.getModel().getAttribute('MaxLength');
 72 }

103 /**
104  * @private
105  */
106 nitobi.form.Calendar.prototype.handleClick = function(evt)
107 {
108 	if (!this.isPickerVisible)
109 	{
110 		this.datePicker.setDate(nitobi.base.DateMath.parseIso8601(this.control.value));
111 		this.datePicker.render();
112 	}
113 
114 	this.ignoreBlur = true;
115 	//node.innerHTML = !this.isPickerVisible ? "▲" : "▼";
116 	nitobi.ui.Effects.setVisible(this.pickerDiv, !this.isPickerVisible, "none", this.setVisibleComplete, this);
117 };

150 nitobi.form.Calendar.prototype.setVisibleComplete = function()

155 nitobi.form.Calendar.prototype.handlePick = function()
156 {
157 	var date = this.datePicker.getDate();
158 	var fDate = nitobi.base.DateMath.toIso8601(date);
159 	this.control.value = fDate;
160 	this.datePicker.hide();
161 };

164 /**
165  * @private
166  */
167 nitobi.form.Calendar.prototype.dispose = function()
168 {
169 	//this.control.onkeydown = null;
170 	nitobi.html.detachEvent(this.control, 'keydown', this.handleKey);
171 	//this.control.onblur = null;
172 	nitobi.html.detachEvent(this.control, 'blur', this.deactivate);
173 	var parent = this.placeholder.parentNode;
174 	parent.removeChild(this.placeholder);
175 	this.control = null;
176 	this.placeholder = null;
177 	this.owner = null;
178 	this.cell = null;
179 }
