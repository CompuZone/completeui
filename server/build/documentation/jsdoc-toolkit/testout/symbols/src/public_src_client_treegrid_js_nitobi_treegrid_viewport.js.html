 53 /**
 54  * Creates an empty block if this viewport is in the middle region (i.e. if this.region is
 55  * between 2 and 5 exclusively).
 56  * @param {Number} low The bottom range of the block
 57  * @param {Number} high The high range of the block
 58  * @private
 59  */
 60 nitobi.grid.Viewport.prototype.makeLastBlock = function(low,high) {
 61 	if (this.lastEmptyBlock == null && this.grid && this.region>2 && this.region<5 && this.container) {
 62 		if (this.container.lastChild) {
 63 //			alert(low+":"+high)
 64 			low = Math.max(low,this.container.lastChild.bottom);
 65 		}
 66 		this.lastEmptyBlock = this.renderEmptyBlock(low,high);
 67 	}
 68 }

 73 nitobi.grid.Viewport.prototype.setCellRanges = function(startRow,rows,startColumn,columns)
 74 {
 75 	this.startRow=startRow;
 76 	this.rows=rows;
 77 	this.startColumn=startColumn;
 78 	this.columns=columns;
 79 
 80 	// Setup the initial empty block.
 81 	// TODO: THIS CAME FROM GROUPING GRID
 82 	this.makeLastBlock(this.startRow,this.startRow+rows-1);
 83 	if (this.lastEmptyBlock!=null  && this.region>2 && this.region<5 && this.rows>0) {
 84 		var bottom=this.startRow+this.rows-1;
 85 		if (this.lastEmptyBlock.top>bottom) {
 86 			this.container.removeChild(this.lastEmptyBlock);
 87 			this.lastEmptyBlock=null;
 88 		} else {
 89 			this.lastEmptyBlock.bottom=bottom;
 90 			this.lastEmptyBlock.style.height = (this.rowHeight*(this.lastEmptyBlock.bottom-this.lastEmptyBlock.top+1))+"px";
 91 			if (this.lastEmptyBlock.bottom<this.lastEmptyBlock.top)
 92 				throw "blocks are miss aligned.";
 93 		}
 94 	}
 95 	// TODO: END - THIS CAME FROM GROUPING GRID
 96 }

115 	}
116 	
117 	// After we clear, we need to ensure there is an empty block
118 	// otherwise the Grid won't be able to determine whether to
119 	// get more data from the server.
120 	this.makeLastBlock(0, (rows != null?rows - 1:0));

189 /**
190  *
191  */
192 nitobi.grid.Viewport.prototype.getBlocks = function(startRow, endRow)
193 {
194 	var blocks = [];
195 	var startBlock = this.findBlock(startRow);
196 	var endBlock = startBlock;
197 	blocks.push(startBlock);
198 	while (endRow > endBlock.bottom)
199 	{
200 		var nextSibling = endBlock.nextSibling;
201 		if (nextSibling != null)
202 			endBlock = nextSibling;
203 		else
204 			break;
205 		blocks.push(endBlock);
206 	}
207 	return blocks;
208 }

229 /**
230  * Splits the block in two at the given row number.
231  * Returns the two new blocks in an array
232  * @param {String|Number} row The row index to split the block at.
233  * @type Array
234  * @private
235  */
236 nitobi.grid.Viewport.prototype.splitBlock = function(row)
237 {
238 	row = parseInt(row);
239 	var block = this.getBlocks(row, row)[0];
240 
241 	var top = block.top, bottom = block.bottom;
242 
243 	// re-render the rows for the new half block.
244 	// delete the rows from the original block.
245 
246 	var origTable = nitobi.html.getFirstChild(nitobi.html.getFirstChild(block));
247 	// We need to get the group (if there is one) because it will have to go into
248 	// the bottom block
249 	var group;
250 	// TODO: Need a reliable way to get a block's group
251 	var i = block.childNodes.length - 1;
252 	while (i > 0 && block.childNodes[i].nodeType == 3 && block.childNodes[i].tagName.toLowerCase() != "table")
253 	{
254 		i--;
255 	}
256 	if (i > 0)
257 	{
258 		group = block.childNodes[i];
259 	}
260 	var rowOffset = row - top;
261 
262 	// Remove the bottom rows from the top table
263 	var topTable = origTable.cloneNode(true);
264 	for (var i=origTable.rows.length-1; i>rowOffset; i--)
265 		topTable.deleteRow(i);
266 
267 	// Update the details on the block node
268 	this.updateBlock(block, top, row);
269 	
270 	// Clone the original table as the bottom table now and remove the top rows from that table
271 	var bottomTable = origTable.cloneNode(true);
272 	for (var i=0; i<rowOffset+1; i++)
273 		bottomTable.deleteRow(0);
274 
275 	var newBlock = this.renderBlock(parseInt(row) + 1, bottom, block.nextSibling);
276 	var contents = document.createElement("div");
277 	contents.appendChild(bottomTable);
278 	newBlock.appendChild(contents);
279 
280 	// Replace the current block with this new smaller one.
281 	origTable.parentNode.replaceChild(topTable, origTable);
282 
283 	// Insert the new bottom part of the table after the split
284 	block.insertAdjacentElement("afterEnd", newBlock);
285 	
286 	// Now move the group to the bottom block if need be.
287 	if (group)
288 	{
289 		newBlock.appendChild(group);
290 	}
291 	// Now we have split the block up in two... can we do this any better???
292 	return {top:block, bottom:newBlock};
293 }

295 /**
296  * Renders a range of rows inside an empty block.  
297  */
298 nitobi.grid.Viewport.prototype.renderInsideEmptyBlock=function(top, bottom, Empty) {
299 	if (Empty==null) {
300 		return this.renderBlock(top,bottom);
301 	}
302 	
303 	// Case - Completely Replace
304 	if (top==Empty.top && bottom>=Empty.bottom) {
305 		var newBlock = this.renderBlock(top,bottom,Empty)
306 		this.container.replaceChild(newBlock,Empty);
307 		if (Empty.bottom<Empty.top)
308 			throw "Render error";
309 		return newBlock;
310 	}
311 
312 	// Case - Insert at Beginning - (Move top and insert before)
313 	if (top==Empty.top && bottom<Empty.bottom) {
314 		Empty.top = bottom+1;
315 		Empty.style.height = (this.rowHeight*(Empty.bottom-Empty.top+1))+"px";
316 		Empty.rows = Empty.bottom-Empty.top+1
317 		if (Empty.bottom<Empty.top)
318 			throw "Render error";
319 		return this.renderBlock(top,bottom,Empty);
320 	}
321 
322 	// Case - Insert at End - (Move bottom and insert after)
323 	if (top>Empty.top && bottom>=Empty.bottom) {
324 		Empty.bottom = top-1;
325 		Empty.style.height = (this.rowHeight*(Empty.bottom-Empty.top+1))+"px";
326 		if (Empty.bottom<Empty.top)
327 			throw "Render error";
328 		return this.renderBlock(top,bottom,Empty.nextSibling);
329 	}
330 
331 	// Case - Insert in the Middle (Move end insert after followed by a new empty block)
332 	if (top>Empty.top && bottom<Empty.bottom) {
333 		// Original emptyblock becomes the end emptyblock and a new start emptyblock is created
334 		var startBlock=this.renderEmptyBlock(Empty.top,top-1,Empty);
335 		Empty.top = bottom+1;
336 		Empty.style.height = (this.rowHeight*(Empty.bottom-Empty.top+1))+"px";
337 		if (Empty.bottom<Empty.top)
338 			throw "Render error";
339 		return this.renderBlock(top,bottom,Empty);
340 	}
341 
342 	throw "Could not insert " + top + "-" + bottom+Empty.outerHTML;
343 }

365  */
366 nitobi.grid.Viewport.prototype.renderBlock=function(top, bottom,nextSibling) {
367 //	if (bottom<top) return null;
368 	var o = document.createElement("div");
369 	// This will be reset by the calling function if called from renderEmptyBlock
370 	o.setAttribute('id','eba_grid_block_'+this.region+'_'+top+'_'+bottom+'_'+this.grid.uid);
371 	o.top=top;
372 	o.bottom=bottom;
373 	o.left = this.startColumn;
374 	o.right = this.startColumn + this.columns;
375 	o.rows=bottom-top+1;
376 	o.columns = this.columns;
377 	if (nextSibling) {
378 		this.container.insertBefore(o,nextSibling);
379 	} else {
380 		this.container.insertBefore(o,null);
381 	}
382 //	if (o.bottom<o.top) {
383 //	crash
384 //	}
385 	return o;
386 }
