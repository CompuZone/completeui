159 /**
160  * Returns a class's profile given a partially completed profile.
161  * @param {nitobi.base.Profile} A profile with one or more fields completed. 
162  * @returns {nitobi.base.Profile} The complete profile.
163  * 
164  */
165 nitobi.base.Registry.prototype.getCompleteProfile = function(incompleteProfile)
166 {
167 	if (nitobi.lang.isDefined(incompleteProfile.className) && incompleteProfile.className != null)
168 	{
169 		return this.classMap[incompleteProfile.className];
170 	}
171 	if (nitobi.lang.isDefined(incompleteProfile.tagName) && incompleteProfile.tagName != null)
172 	{
173 		return this.tagMap[incompleteProfile.tagName];
174 	}	
175 	nitobi.lang.throwError("A complete class profile could not be found. Insufficient information was provided.");
176 }

178 /**
179  * Register a class with the registry.  Registering a class enables it's creation by factory.
180  * @param {nitobi.base.Profile} profile A complete profile.
181  * @returns {void}
182  */
183 nitobi.base.Registry.prototype.register = function(profile)
184 {
185 	//if (!nitobi.lang.isDefined(profile.schema) || null==profile.schema) nitobi.lang.throwError("Illegal to register a class without a schema.");
186 	if (!nitobi.lang.isDefined(profile.tagName) || null==profile.tagName) nitobi.lang.throwError("Illegal to register a class without a tagName.");
187 	if (!nitobi.lang.isDefined(profile.className) || null==profile.className) nitobi.lang.throwError("Illegal to register a class without a className.");
188 	this.tagMap[profile.tagName] = profile;
189 	this.classMap[profile.className] = profile;
190 }
