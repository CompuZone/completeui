172 /**
173  * Initiatiate the render operation. Because the data may not yet be available, the actual render 
174  * doesn't occur until renderReady()
175  */
176 nitobi.grid.Scroller3x3.prototype.render=function(force)
177 {
178 	// Check if the grid is bound to the datasource yet and if the scroll has changed
179 	if (this.owner.isBound() && (this.getScrollTop() != this.lastScrollTop || force || this.scrollTopPercent>.9))
180 	{
181 		// Wait a little bit before rendering, just in case the user has scrolled past this position.
182 		// This enables quickly scrolling through large data sets without rendering every record.
183 		var funcRef = nitobi.lang.close(this, this.performRender, []);
184 		window.clearTimeout(this.lastTimeoutId);
185 		this.lastTimeoutId = window.setTimeout(funcRef, EBAScroller_RENDERTIMEOUT);
186 		this.resetHeaders();
187 	}
188 }

194 	/*var visibleRows = this.getUnrenderedBlocks(pagingMode, renderMode);

197 	*/
198 	var scrollTop = this.getScrollTop();
199 	this.surface.renderAtScrollPosition(scrollTop);
200 	this.onRenderComplete.notify();
201 }

300 /**
301  * Returns the Surface identified by the given path.
302  * @param Array path The path to the Surface.  For example, given the array [0,2,1,4], the Surface belonging
303  * to the 4th row of the 1st group of the 2nd group of the root surface.
304  * @type nitobi.grid.Surface
305  */
306 nitobi.grid.Scroller3x3.prototype.getSurface = function(path)
307 {
308 	if (this.surfaceMap[path] != null)
309 	{
310 		return this.surfaceMap[path];
311 	}
312 	if (typeof path == "string")
313 	{
314 		path = path.split("_");
315 	}
316 	var surface = this.surface;
317 	// We start at 1 because we ignore the root surface
318 	for (var i = 1; i < path.length; i++)
319 	{
320 		surface = surface.surfaces[path[i]];
321 	}
322 	return (surface?this.surfaceMap[path] = surface:null);
323 }

335 nitobi.grid.Scroller3x3.prototype.dispose = function()
336 {
337 	try
338 	{
339 		(this.cacheMap!= null?this.cacheMap.flush():'');
340 		this.cacheMap = null;
341 
342 		var disposalLength = this.disposal.length;
343 		for (var i=0; i<disposalLength; i++)
344 		{
345 			if (typeof(this.disposal[i]) == "function")
346 			{
347 				this.disposal[i].call(this);
348 			}
349 			this.disposal[i] = null;
350 		}
351 
352 		for (var v in this.view)
353 		{
354 			this.view[v].dispose();
355 		}
356 
357 		for (var item in this)
358 		{
359 			if (this[item] != null && this[item].dispose instanceof Function)
360 			{
361 				this[item].dispose();
362 			}
363 		}
364 	}
365 	catch(e)
366 	{
367 //		ntbAssert();
368 	}
