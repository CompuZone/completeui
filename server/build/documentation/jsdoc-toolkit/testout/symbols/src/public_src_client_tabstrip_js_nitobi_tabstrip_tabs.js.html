  3 /**
  4  * @class Creates a list of tabs. This class allows for modifications of certain shared tab attributes. 
  5  * It allows you to re-order, add, and remove the tabs from the collection. If you modify the tab 
  6  * collection by calling add and remove, call {@link nitobi.tabstrip.TabStrip#render} for
  7  * the changes to take effect.
  8  * <p>
  9  * To obtain a reference to a specific tab in the collection use 
 10  * {@link nitobi.ui.Container#get}
 11  * </p>
 12  * @constructor
 13  * @see nitobi.tabstrip.TabStrip
 14  * @see nitobi.tabstrip.Tab
 15  * @param {XmlNode} node Optional. If you want to create a tabs list and deserialize it from the node. 
 16  * @extends nitobi.ui.Container
 17  * @implements nitobi.ui.IScrollable
 18  */
 19 nitobi.tabstrip.Tabs = function(node) 
 20 {
 21 	nitobi.tabstrip.Tabs.baseConstructor.call(this,node);
 22 	nitobi.ui.IScrollable.call(this);
 23 	
 24 	/**
 25 	 * Fired on click. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 26 	 * @type nitobi.base.Event
 27 	 */
 28 	this.onClick = new nitobi.base.Event("click");
 29 	this.eventMap["click"] = this.onClick;
 30 	
 31 	/**
 32 	 * Fired on mouse out. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 33 	 * @type nitobi.base.Event
 34 	 */
 35 	this.onMouseOut = new nitobi.base.Event("mouseout");
 36 	this.eventMap["mouseout"] = this.onMouseOut;
 37 	
 38 	/**
 39 	 * Fired on mouse over. {@link nitobi.ui.ElementEventArgs} are passed to the event handler.
 40 	 * @type nitobi.base.Event
 41 	 */
 42 	this.onMouseOver = new nitobi.base.Event("mouseover");
 43 	this.eventMap["mouseover"] = this.onMouseOver;
 44 	
 45 	
 46 	this.onEventNotify.subscribe(this.handleEventNotify,this);
 47 	this.onBeforeEventNotify.subscribe(this.handleBeforeEventNotify,this);
 48 	
 49 	/**
 50 	 * Fired before the tabs change. If the event returns false, nothing will happen.
 51 	 * {@link nitobi.tabstrip.TabChangeEventArgs} are passed to the event handler.
 52 	 * @type nitobi.base.Event
 53 	 */
 54 	this.onBeforeTabChange = new nitobi.base.Event("beforetabchange");
 55 	this.eventMap["beforetabchange"] = this.onBeforeTabChange;
 56 
 57 	/**
 58 	 * Fired after the tabs have changed. {@link nitobi.tabstrip.TabChangeEventArgs} are passed to the event handler.
 59 	 * @type nitobi.base.Event
 60 	 */
 61 	this.onTabChange = new nitobi.base.Event("tabchange");
 62 	this.eventMap["tabchange"] = this.onTabChange;
 63 	
 64 	this.subscribeDeclarationEvents();
 65 	
 66 	this.renderer.setTemplate(nitobi.tabstrip.tabstripProc);
 67 	this.onCreated.notify(new nitobi.ui.ElementEventArgs(this));
 68 }

 80 /**
 81  * @private
 82  * Load all the tabs that don't use IFRAMEs.
 83  */
 84 nitobi.tabstrip.Tabs.prototype.loadTabs = function()
 85 {	
 86 	// We load the tab set as active, regardless of its loadondemandenabled attribute
 87 	var tab = this.get(this.getActiveTabIndex());
 88 	if (tab == null) {
 89 		return;
 90 	} else {
 91 		tab.load();
 92 	}
 93 
 94 	// Now go through the rest of the tabs and load only the ones that have loadondemandenabled=false
 95 	var nodes = this.getXmlNode().selectNodes("ntb:tab[@loadondemandenabled!='true' or not(@loadondemandenabled) and not(@id='"+tab.getId()+"')]");
 96 	for (var i = 0; i < nodes.length; i++ ) {
 97 		var index = nitobi.xml.indexOfChildNode(this.getXmlNode(), nodes[i]);
 98 		var tab = this.get(index);
 99 		tab.load();
100 	} 
101 }

174 /**
175  * Renders the tabs.
176  * @private
177  */
178 nitobi.tabstrip.Tabs.prototype.render = function()
179 {
180 	this.onBeforeRender.notify(new nitobi.ui.ElementEventArgs(this,null,this.getId()));
181 	this.setContainer(this.getHtmlNode().parentNode);
182 	this.renderer.setParameters({'apply-template':'tabs'});
183 	nitobi.tabstrip.Tabs.base.render.call(this,null,this.getXmlNode().ownerDocument);	
184 	
185 	// If we render again, we don't want to destroy all the containers.
186 	// Loop through them, and add ones that we don't have.
187 	var lastTab = null;
188 	var len = this.getLength();
189 	for (var i=0;i < len;i++)
190 	{
191 		var tab = this.get(i);
192 		
193 		// The bodypulse div is set to width: 100% from the xsl transform.
194 		// This is corrected when tab.load() is called, but we need to do it
195 		// here because not all tabs will have load() called on them.
196 		var box = nitobi.html.getBox(tab.getHtmlNode("labeltable"));
197 		tab.getHtmlNode("bodypulse").style.width = box.width + "px";
198 		
199 		if (!tab.isBodyHtmlNodeAvail())
200 		{
201 			this.renderer.setParameters({'apply-template':'body'});
202 			this.renderer.setParameters({'apply-id':(i+1)});
203 			if (lastTab == null)
204 			{
205 				this.renderer.renderIn(this.getBodiesContainerHtmlNode(),this.getState().ownerDocument);
206 			}
207 			else
208 			{
209 				this.renderer.renderAfter(lastTab,this.getState().ownerDocument);
210 			}
211 
212 			tab.load();
213 
214 		}
215 		else
216 		{
217 			tab.autoSetActivityIndicator();
218 			lastTab = tab.getBodyHtmlNode();
219 		}
220 		
221 	}
222 	if (len > 0) 
223 	{
224 		this.getActiveTab().show();
225 	}
226 	this.onRender.notify(new nitobi.ui.ElementEventArgs(this,null,this.getId()));
227 }

263 /**
264  * Sets the active tab. The tab must be a part of the tabstrip's collection
265  * of tabs.
266  * @param {nitobi.tabstrip.Tab/Number} tab The tab to make active or the index of the tab to make active.
267  */
268 nitobi.tabstrip.Tabs.prototype.setActiveTab = function(tab)
269 {
270 	if (null == tab)
271 	{
272 		nitobi.lang.throwError(nitobi.error.BadArgType);
273 	}
274 	try
275 	{
276 		var index;
277 		var activeTab = this.getActiveTab();
278 	
279 		if (typeof(tab) == "object")
280 		{
281 			index = this.indexOf(tab);
282 		}
283 		else
284 		{
285 			index = tab;
286 			tab = this.get(index);
287 		}
288 		if (index == this.getActiveTabIndex())
289 		{
290 			return;
291 		}
292 		var args = new nitobi.tabstrip.TabChangeEventArgs(this, null, this.getId(), activeTab, tab);
293 		if(this.onBeforeTabChange.notify(args))
294 		{
295 			if (this.getActivateEffect() == "fade")
296 			{
297 				nitobi.tabstrip.Tabs.transition(this,activeTab,tab);			
298 			}
299 			else
300 			{
301 				activeTab.hide();
302 				if (tab.getContentLoaded() == false) {
303 					tab.load();
304 				} 
305 				tab.show();	
306 			}
307 			this.setActiveTabIndex(index);
308 			this.onTabChange.notify(args);
309 		}
310 	}
311 	catch(err)
312 	{
313 		nitobi.lang.throwError(nitobi.error.Unexpected + " The active tab could not be set.",err);	
314 	}
315 }

317 /**
318  * @private
319  */
320 nitobi.tabstrip.Tabs.transition = function(tabs,tabToHide,tabToShow)
321 {
322 	var hideEl = tabToHide.getBodyHtmlNode();
323 	var showEl = tabToShow.getBodyHtmlNode();
324 	var hideIndex = nitobi.html.indexOfChildNode(hideEl.parentNode,hideEl);
325 	var showIndex = nitobi.html.indexOfChildNode(showEl.parentNode,showEl);
326 
327 	var elToMove, elToKeep;
328 	
329 	if (showIndex > hideIndex)
330 	{
331 		elToMove = showEl;
332 		elToKeep = hideEl;
333 	}
334 	else
335 	{
336 		elToMove = hideEl;
337 		elToKeep = showEl;
338 	}
339 	
340 	nitobi.html.Css.setOpacity(hideEl, 100);
341 	
342 	var index = nitobi.html.indexOfChildNode(hideEl.parentNode,hideEl);
343 	var box = nitobi.html.getBox(hideEl);
344 
345 	if (hideIndex < showIndex)
346 	{
347 		//	TODO: The "effect" argument is here is a place holder for a real effect.
348 		tabToShow.show("effect");
349 	}
350 	
351 	var container = tabs.getParentObject().getHtmlNode("tabbodiesdivcontainer");
352 	container.style.height = nitobi.html.getBox(container).height + "px";
353 	var top = -1 * nitobi.html.getBox(hideEl).height;
354 	var disp = elToMove.style.display;
355 	elToMove.style.display = "none";
356 	elToMove.style.top = top + "px";
357 	elToMove.style.left="0px";
358 	elToMove.style.left = "0px";
359 	elToMove.style.position = "relative";
360 	elToMove.style.height = box.height + "px";
361 	elToKeep.style.height = box.height + "px";
362 	
363 	container.style.height = "100%";
364 
365 	elToMove.style.display = disp;	
366 	tabToHide.hide("effect");
367 	if (hideIndex > showIndex)
368 	{
369 		tabToShow.show("effect");
370 	}
371 	nitobi.ui.Effects.fade(showEl,100,400,nitobi.lang.close(tabToShow, tabToShow.show));
372 }

459 /**
460  * Removes a tab. Call {@link nitobi.tabstrip.TabStrip#render} after adding or removing tabs.
461  * @param {nitobi.tabstrip.Tab} The tab or the index of the tab to remove.
462  */
463 nitobi.tabstrip.Tabs.prototype.remove = function(value)
464 {
465 	if (value == null)
466 	{
467 		nitobi.lang.throwError(nitobi.error.BadArg);
468 	}
469 	var i;
470 	// TODO: Wrap this indexing stuff up somewhere else.
471 	if (typeof(value) != "number")
472 	{
473 		i = this.indexOf(value);
474 	}
475 	else
476 	{
477 		i = value;
478 	}
479 	if (i == -1)
480 	{
481 		nitobi.lang.throwError(nitobi.error.BadArg + " The tab could not be found.");
482 	}
483 
484 	var tab = this.get(i);
485 	var activeI = this.getActiveTabIndex();
486 	if (this.getLength() == 1)
487 	{
488 		activeI = -1;
489 	}
490 	else if (activeI > i)
491 	{
492 		activeI--;	
493 	} 
494 	else if (activeI == i)
495 	{
496 		if (!(activeI == 0 && i == 0))
497 		{
498 			activeI--;	
499 		}
500 	}
501 	
502 	this.setActiveTabIndex(activeI);
503 	tab.destroyHtml();
504 	nitobi.tabstrip.Tabs.base.remove.call(this,value);
505 }

508 /**
509  * @private
510  */
511 nitobi.tabstrip.Tabs.prototype.handleBeforeEventNotify = function(eventArgs)
512 {
513 	var event = eventArgs.htmlEvent;
514 	var idInfo = nitobi.ui.Element.parseId(eventArgs.targetId);
515 	if (event.type == "click")
516 	{
517 		// Ignore the click if the tab is disabled.
518 		var tab = this.getById(idInfo.id);
519 		if (null == tab)
520 		{
521 			return false;
522 		}
523 		else
524 		{
525 			return tab.isEnabled();
526 		}
527 	}
528 }

530 /**
531  * @private
532  */
533 nitobi.tabstrip.Tabs.prototype.handleEventNotify = function(eventArgs)
534 {
535 	var event = eventArgs.htmlEvent;
536 	var idInfo = nitobi.ui.Element.parseId(eventArgs.targetId);
537 	switch(event.type)
538 	{
539 		case "click":
540 		{
541 			try
542 			{
543 				if (idInfo.localName!="scrollleft" && idInfo.localName!="scrollright")
544 				{
545 					var tab = this.getById(idInfo.id);
546 					this.setActiveTab(tab);
547 				}
548 			}
549 			catch(err)
550 			{
551 				nitobi.lang.throwError("The Tabs object encountered an error handling the click event.",err);
552 			}
553 			break;
554 		}
555 		case "mousedown":
556 		{
557 			var closure;
558 			switch(idInfo.localName)
559 			{
560 				case "scrollleft":
561 				{
562 					this.scrollLeft();
563 					closure = nitobi.lang.close(this,this.scrollLeft,[]);		
564 					break;
565 				}
566 				case "scrollright":
567 				{
568 					this.scrollRight();
569 					closure = nitobi.lang.close(this,this.scrollRight,[]);
570 					break;
571 				}
572 			}
573 			this.stopScrolling();
574 			this.scrollerEventId = window.setInterval(closure,100);
575 			nitobi.html.attachEvent(document.body,"mouseup",this.stopScrolling,this);
576 			break;
577 		}
578 		case "mouseup":
579 		{
580 			this.stopScrolling();
581 			break;
582 		}
583 		
584 	}
585 }
