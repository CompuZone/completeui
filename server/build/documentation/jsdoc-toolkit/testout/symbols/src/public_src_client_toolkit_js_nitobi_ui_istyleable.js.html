  3 /**
  4  * Initializes the IStyleable interface.
  5  * @class Allows for easy manipulation of styles on an HTML node.
  6  * @constructor
  7  * @param {HTMLElement} htmlNode The HTML element that will have styles applied to it.
  8  */
  9 nitobi.ui.IStyleable = function(htmlNode) 
 10 {
 11 	/** @ignore */
 12 	this.htmlNode = htmlNode || null;
 13 	/**
 14 		Fires when a style is updated and passes the style name and value.
 15 		@type nitobi.base.Event 
 16 	*/
 17 	this.onBeforeSetStyle = new nitobi.base.Event();
 18 //	this.eventMap['beforesetstyle'] = this.onBeforeSetStyle;
 19 	
 20 	this.onSetStyle = new nitobi.base.Event();
 21 //	this.eventMap['setstyle'] = this.onSetStyle;
 22 }
 23 
 24 /**
 25 	Returns the HTML node that this interface manipluates.
 26 */
 27 nitobi.ui.IStyleable.prototype.getHtmlNode = function()
 28 {
 29 	return this.htmlNode;
 30 }
 31 
 32 /**
 33 	Sets the HTML node that this interface manipluates.
 34 	@param {HTMLElement} node The element to which styles will be applied.
 35 */
 36 nitobi.ui.IStyleable.prototype.setHtmlNode = function(node)
 37 {
 38 	this.htmlNode = node;
 39 }
 40 
 41 /**
 42 	Sets a style on the element.  
 43 	@param {String} name The name of the style. This can be either all lowercase using dashes, or camel-case
 44 			e.g., either background-color or backgroundColor.
 45 	@param {String} value The value to which you want the style set.
 46 */
 47 nitobi.ui.IStyleable.prototype.setStyle = function(name, value)
 48 {
 49 	if (this.onBeforeSetStyle.notify(new nitobi.ui.StyleEventArgs(this,this.onBeforeSetStyle,name,value)) && this.getHtmlNode() != null)
 50 	{
 51 		nitobi.html.Css.setStyle(this.getHtmlNode(), name, value);
 52 		this.onSetStyle.notify(new nitobi.ui.StyleEventArgs(this,this.onSetStyle,name,value));
 53 	}
 54 }
 55 
 56 /**
 57  * Gets the computer style of the element with the specified name.
 58  * @param {String} name The name of the style whose value you want.
 59  * @returns {String} The value of the style.  This may return values in different formats depending on the browser. 
 60  */
 61 nitobi.ui.IStyleable.prototype.getStyle = function(name)
 62 {
 63 	return nitobi.html.Css.getStyle(this.getHtmlNode(), name);
 64 }
